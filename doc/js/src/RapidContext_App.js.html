 32 /**
 33  * Initializes the platform, API:s and RapidContext UI. If an app
 34  * identifier is provided, the default platform UI will not be
 35  * created. Instead the app will be launched with the root document
 36  * as its UI container.
 37  *
 38  * @param {String/Object} [app] the app id or class name to start
 39  *
 40  * @return {Deferred} a `MochiKit.Async.Deferred` object that will
 41  *         callback when the initialization has completed
 42  */
 43 RapidContext.App.init = function (app) {
 44     // Setup libraries
 45     RapidContext.Log.context("RapidContext.App.init()");
 46     RapidContext.Log.info("Initializing RapidContext");
 47     RapidContext.Util.registerFunctionNames(RapidContext, "RapidContext");
 48
 49     // Setup UI
 50     RapidContext.Util.registerSizeConstraints(document.body, "100%-20", "100%-20");
 51     var resizer = MochiKit.Base.partial(RapidContext.Util.resizeElements, document.body);
 52     MochiKit.Signal.connect(window, "onresize", resizer);
 53     RapidContext.Util.resizeElements(document.body);
 54     var overlay = new RapidContext.Widget.Overlay({ message: "Loading..." });
 55     MochiKit.DOM.replaceChildNodes(document.body, overlay);
 56
 57     // Load platform data
 58     var list = [ RapidContext.App.callProc("System.Status"),
 59                  RapidContext.App.callProc("System.Session.Current"),
 60                  RapidContext.App.callProc("System.App.List") ];
 61     var d = MochiKit.Async.gatherResults(list);
 62
 63     // Launch app
 64     d.addBoth(function () {
 65         RapidContext.Log.context(null);
 66     });
 67     d.addCallback(function () {
 68         try {
 69             return RapidContext.App.startApp(app || "start");
 70         } catch (e) {
 71             RapidContext.UI.showError(e);
 72             return RapidContext.App.startApp("start");
 73         }
 74     });
 75     d.addErrback(RapidContext.UI.showError);
 76     return d;
 77 };

115 /**
116  * Returns an array with running app instances.
117  *
118  * @return {Array} the array of app instances
119  */
120 RapidContext.App._instances = function () {
121     var res = [];
122     var apps = RapidContext.App.apps();
123     for (var i = 0; i < apps.length; i++) {
124         res = res.concat(apps[i].instances || []);
125     }
126     return res;

129 /**
130  * Finds the app launcher from an app instance, class name or
131  * launcher. In the last case, the matching cached launcher will be
132  * returned.
133  *
134  * @param {String/Object} app the app id, instance, class name or
135  *        launcher
136  *
137  * @return {Object} the read-only app launcher, or
138  *         `null` if not found
139  */
140 RapidContext.App.findApp = function (app) {
141     if (app == null) {
142         return null;
143     }
144     var apps = RapidContext.App.apps();
145     for (var i = 0; i < apps.length; i++) {
146         var l = apps[i];
147         if (l.className == null) {
148             RapidContext.Log.error("Launcher does not have 'className' property", l);
149         } else if (l.id == app || l.className == app || l.id == app.id) {
150             return l;
151         }
152     }
153     return null;
154 };

163 /**
164  * Creates and starts an app instance. Normally apps are started in the default
165  * app tab container or in a provided widget DOM node. By specifying a newly
166  * created window object as the parent container, apps can also be launched
167  * into separate windows.
168  *
169  * Note that when a new window is used, the returned deferred will callback
170  * immediately with a `null` app instance (normal app communication is not
171  * possible cross-window).
172  *
173  * @param {String/Object} app the app id, class name or launcher
174  * @param {Widget/Window} [container] the app container widget or
175  *            window, defaults to create a new pane in the app tab
176  *            container
177  *
178  * @return {Deferred} a `MochiKit.Async.Deferred` object that will
179  *         callback with the app instance (or `null` if not available)
180  *
181  * @example
182  * // Starts the help app in a new tab
183  * RapidContext.App.startApp('help');
184  *
185  * // Starts the help app in a new window
186  * RapidContext.App.startApp('help', window.open());
187  */
188 RapidContext.App.startApp = function (app, container) {
189
190     // Setup variables and find app launcher
191     var launcher = RapidContext.App.findApp(app);
192     if (launcher == null) {
193         RapidContext.Log.error("No matching app launcher found", app);
194         throw new Error("No matching app launcher found: " + app);
195     }
196     var instance = null;
197     var instances = RapidContext.App._instances();
198     var startApp = RapidContext.App.findApp("start");
199     var d = MochiKit.Async.wait(0);
200     var ui = null;
201
202     // Initialize app UI container
203     launcher.instances = launcher.instances || [];
204     if ($.isWindow(container)) {
205         var url = "rapidcontext/app/" + launcher.id;
206         container.location.href = RapidContext.Util.resolveURI(url);
207         return d;
208     } else if (startApp && startApp.instances && startApp.instances.length) {
209         var opts = { title: launcher.name, closeable: (launcher.launch != "once") };
210         ui = startApp.instances[0].initAppPane(container, opts);
211     } else if (instances.length == 1 && launcher.id != "start") {
212         // Switch from single-app to multi-app mode
213         d = RapidContext.App.startApp("start");
214         d.addCallback(function (instance) {
215             return RapidContext.App.startApp(app, container);
216         });
217         return d;
218     } else {
219         var ui = { root: document.body, overlay: document.body.childNodes[0] };
220     }
221     MochiKit.Signal.connect(ui.root, "onclose", d, "cancel");
222
223     // Load app resources
224     var logCtx = "RapidContext.App.startApp(" + launcher.id + ")";
225     RapidContext.Log.context(logCtx);
226     if (launcher.creator == null) {
227         RapidContext.Log.info("Loading app/" + launcher.id + " resources", launcher);
228         launcher.resource = {};
229         for (var i = 0; i < launcher.resources.length; i++) {
230             var res = launcher.resources[i];
231             var url = RapidContext.App._rebaseUrl(res.url);
232             if (res.type == "code" || res.type == "js" || res.type == "javascript") {
233                 d.addCallback(MochiKit.Base.partial(RapidContext.App.loadScript, url));
234             } else if (res.type == "style" || res.type == "css") {
235                 d.addCallback(MochiKit.Base.partial(RapidContext.App.loadStyles, url));
236             } else if (res.type == "ui") {
237                 d.addCallback(MochiKit.Base.partial(RapidContext.App.loadXML, url));
238                 d.addCallback(RapidContext.App._cbAssign(launcher, "ui"));
239             } else if (res.type == "json" && res.id != null) {
240                 d.addCallback(MochiKit.Base.partial(RapidContext.App.loadJSON, url, null, null));
241                 d.addCallback(RapidContext.App._cbAssign(launcher.resource, res.id));
242             } else if (res.id != null) {
243                 launcher.resource[res.id] = url;
244             }
245         }
246         d.addCallback(function () {
247             launcher.creator = this[launcher.className] || window[launcher.className];
248             if (launcher.creator == null) {
249                 RapidContext.Log.error("App constructor " + launcher.className + " not defined", launcher);
250                 throw new Error("App constructor " + launcher.className + " not defined");
251             }
252             RapidContext.Util.registerFunctionNames(launcher.creator, launcher.className);
253         });
254     }
255
256     // Create app instance, build UI and start app
257     d.addCallback(function () {
258         RapidContext.Log.info("Starting app/" + launcher.id, launcher);
259         var fun = launcher.creator;
260         instance = new fun();
261         launcher.instances.push(instance);
262         var props = MochiKit.Base.setdefault({ ui: ui }, launcher);
263         delete props.creator;
264         delete props.instances;
265         MochiKit.Base.setdefault(instance, props);
266         MochiKit.Signal.disconnectAll(ui.root, "onclose");
267         MochiKit.Signal.connect(ui.root, "onclose",
268                                 MochiKit.Base.partial(RapidContext.App.stopApp, instance));
269         if (launcher.ui != null) {
270             var widgets = RapidContext.UI.buildUI(launcher.ui, ui);
271             MochiKit.DOM.appendChildNodes(ui.root, widgets);
272             RapidContext.Util.resizeElements(ui.root);
273         }
274         ui.overlay.hide();
275         ui.overlay.setAttrs({ message: "Working..." });
276         return RapidContext.App.callApp(instance, "start");
277     });
278
279     // Convert to start app UI (if previously in single-app mode)
280     if (launcher.id == "start" && instances.length == 1) {
281         var elems = MochiKit.Base.extend([], document.body.childNodes);
282         var opts = { title: instances[0].name, closeable: false, background: true };
283         d.addCallback(function () {
284             return RapidContext.App.callApp(instance, "initAppPane", null, opts);
285         });
286         d.addCallback(function (ui) {
287             ui.root.removeAll();
288             ui.root.addAll(elems);
289             RapidContext.Util.resizeElements(ui.root);
290         });
291     }
292
293     // Report errors and return app instance
294     d.addErrback(function (err) {
295         if (err instanceof MochiKit.Async.CancelledError) {
296             // Ignore cancellation errors
297         } else {
298             RapidContext.Log.error("Failed to start app", err);
299             MochiKit.Signal.disconnectAll(ui.root, "onclose");
300             var label = MochiKit.DOM.STRONG(null, "Error: ");
301             MochiKit.DOM.appendChildNodes(ui.root, label, err.message);
302             ui.overlay.hide();
303         }
304         RapidContext.Log.context(null);
305         return err;
306     });
307     d.addCallback(function () {
308         RapidContext.Log.context(null);
309         return instance;
310     });
311     RapidContext.App._addErrbackLogger(d, logCtx);
312     return d;
313 };

315 /**
316  * Stops an app instance. If only the class name or launcher is specified, the
317  * most recently created instance will be stopped.
318  *
319  * @param {String/Object} app the app id, instance, class name or launcher
320  *
321  * @return {Deferred} a `MochiKit.Async.Deferred` object that will
322  *         callback when the app has been stopped
323  */
324 RapidContext.App.stopApp = function (app) {
325     var launcher = RapidContext.App.findApp(app);
326     if (!(launcher && launcher.instances && launcher.instances.length)) {
327         RapidContext.Log.error("No running app instance found", app);
328         throw new Error("No running app instance found");
329     }
330     RapidContext.Log.info("Stopping app " + launcher.name);
331     var pos = MochiKit.Base.findIdentical(launcher.instances, app);
332     if (pos < 0) {
333         app = launcher.instances.pop();
334     } else {
335         launcher.instances.splice(pos, 1);
336     }
337     app.stop();
338     if (app.ui.root != null) {
339         RapidContext.Widget.destroyWidget(app.ui.root);
340     }
341     for (var n in app.ui) {
342         delete app.ui[n];
343     }
344     for (var n in app) {
345         delete app[n];
346     }
347     MochiKit.Signal.disconnectAllTo(app);
348     return MochiKit.Async.wait(0);
349 };

351 /**
352  * Performs an asynchronous call to a method in an app. If only the class name
353  * or launcher is specified, the most recently created instance will be used.
354  * If no instance is running, one will be started. Also, before calling the app
355  * method, the app UI will be focused.
356  *
357  * @param {String/Object} app the app id, instance, class name or launcher
358  * @param {String} method the app method name
359  * @param {Mixed} [args] additional parameters sent to method
360  *
361  * @return {Deferred} a `MochiKit.Async.Deferred` object that will
362  *         callback with the result of the call on success
363  */
364 RapidContext.App.callApp = function (app, method) {
365     var args = MochiKit.Base.extend([], arguments, 2);
366     var launcher = RapidContext.App.findApp(app);
367     var d;
368     if (launcher == null) {
369         RapidContext.Log.error("No matching app launcher found", app);
370         throw new Error("No matching app launcher found");
371     }
372     var logCtx = "RapidContext.App.callApp(" + launcher.id + "," + method + ")";
373     if (!(launcher.instances && launcher.instances.length)) {
374         d = RapidContext.App.startApp(app);
375     } else {
376         var pos = MochiKit.Base.findIdentical(launcher.instances, app);
377         var instance = (pos >= 0) ? app : launcher.instances[launcher.instances.length - 1];
378         d = MochiKit.Async.wait(0, instance);
379         RapidContext.App._addErrbackLogger(d, logCtx);
380     }
381     d.addCallback(function (instance) {
382         RapidContext.Log.context(logCtx);
383         var child = instance.ui.root;
384         var parent = MochiKit.DOM.getFirstParentByTagAndClassName(child, null, "widget");
385         if (parent != null && typeof(parent.selectChild) == "function") {
386             parent.selectChild(child);
387         }
388         var methodName = launcher.className + "." + method;
389         if (instance == null || instance[method] == null) {
390             var msg = "No app method " + methodName + " found";
391             RapidContext.Log.error(msg);
392             RapidContext.Log.context(null);
393             throw new Error(msg);
394         }
395         RapidContext.Log.log("Calling app method " + methodName, args);
396         try {
397             return instance[method].apply(instance, args);
398         } catch (e) {
399             var msg = "Caught error in " + methodName;
400             RapidContext.Log.error(msg, e);
401             RapidContext.Log.context(null);
402             throw new Error(msg + ": " + e.toString());
403         }
404         RapidContext.Log.context(null);
405     });
406     return d;
407 };

409 /**
410  * Performs an asynchronous procedure call. This function returns a deferred
411  * object that will produce either a `callback` or an `errback` depending on
412  * the server response.
413  *
414  * @param {String} name the procedure name
415  * @param {Array} [args] the array of arguments, or `null`
416  *
417  * @return {Deferred} a `MochiKit.Async.Deferred` object that will
418  *         callback with the response data on success
419  */
420 RapidContext.App.callProc = function (name, args) {
421     var params = {};
422     var options = { timeout: 60 };
423
424     // TODO: remove this legacy name conversion
425     if (name.indexOf("RapidContext.") == 0) {
426         name = "System" + name.substring(8);
427     }
428     RapidContext.Log.log("Call request " + name, args);
429     for (var i = 0; args != null && i < args.length; i++) {
430         if (args[i] == null) {
431             params["arg" + i] = "null";
432         } else {
433             params["arg" + i] = JSON.stringify(args[i]);
434         }
435     }
436     var logLevel = RapidContext.Log.level();
437     if (logLevel == "log" || logLevel == "all") {
438         params["system:trace"] = 1;
439     }
440     var d = RapidContext.App.loadJSON("rapidcontext/procedure/" + name, params, options);
441     d.addCallback(function (res) {
442         if (res.trace != null) {
443             RapidContext.Log.log("Server trace " + name, res.trace);
444         }
445         if (res.error != null) {
446             RapidContext.Log.warn("Call error " + name, res.error);
447             throw new Error(res.error);
448         } else {
449             RapidContext.Log.log("Call response " + name, res.data);
450         }
451         return res.data;
452     });
453     if (name.indexOf("System.") == 0) {
454         d.addCallback(function (res) {
455             if (res) {
456                 RapidContext.App._Cache.update(name, res);
457             }
458             return res;
459         });
460     }
461     return d;
462 };

464 /**
465  * Performs an asynchronous login. This function returns a deferred object
466  * that will produce either a `callback` or an `errback` depending on the
467  * success of the login attempt. If the current session is already bound to a
468  * user, that session will be terminated and a new one will be created. If an
469  * authentication token is specified, the login and password fields are not
470  * used (can be null).
471  *
472  * @param {String} login the user login name or email address
473  * @param {String} password the password to autheticate the user
474  * @param {String} [token] the authentication token to indentify user/password
475  *
476  * @return {Deferred} a `MochiKit.Async.Deferred` object that will
477  *         callback with the response data on success
478  */
479 RapidContext.App.login = function (login, password, token) {
480     var d = MochiKit.Async.wait(0, false);
481     var user = RapidContext.App.user();
482     if (user && user.id) {
483         d.addCallback(RapidContext.App.logout);
484     }
485     if (token) {
486         d.addCallback(function () {
487             var args = [token];
488             return RapidContext.App.callProc("System.Session.AuthenticateToken", args);
489         });
490     } else {
491         if (/@/.test(login)) {
492             d.addCallback(function () {
493                 return RapidContext.App.callProc("System.User.Search", [login]);
494             });
495             d.addCallback(function (user) {
496                 if (user && user.id) {
497                     login = user.id;
498                     return login;
499                 } else {
500                     throw new Error("no user with that email address");
501                 }
502             });
503         }
504         d.addCallback(function () {
505             return RapidContext.App.callProc("System.Session.Current");
506         });
507         d.addCallback(function (session) {
508             var realm = RapidContext.App.status().realm;
509             var hash = CryptoJS.MD5(login + ":" + realm + ":" + password);
510             hash = CryptoJS.MD5(hash.toString() + ":" + session.nonce).toString();
511             var args = [login, session.nonce, hash];
512             return RapidContext.App.callProc("System.Session.Authenticate", args);
513         });
514     }
515     return d;
516 };

519  * Performs an asyncronous logout. This function terminates the current
520  * session and either reloads the browser window or returns a deferred object
521  * that will produce either a `callback` or an `errback` response.
522  *
523  * @param {Boolean} [reload] the reload browser flag, defaults to true

533         });

537 /**
538  * Performs an asynchronous HTTP request for a JSON data document and returns a
539  * deferred response. If no request method has been specified, the `POST` or
540  * `GET` methods are chosen depending on whether or not the params argument is
541  * `null`. The request parameters are specified as an object that will be
542  * encoded by the `MochiKit.Base.queryString` function. In addition to the
543  * default options in `MochiKit.Async.doXHR`, this function also accepts a
544  * timeout option for automatic request cancellation.
545  *
546  * Note that this function is unsuitable for loading JavaScript source code,
547  * since using `eval()` will confuse some browser error messages and debuggers
548  * regarding the actual source location.
549  *
550  * @param {String} url the URL to request
551  * @param {Object} [params] the request parameters, or `null`
552  * @param {Object} [options] the request options, or `null`
553  * @config {String} [method] the HTTP method, "GET" or "POST"
554  * @config {Number} [timeout] the timeout in seconds, default is no timeout
555  * @config {Object} [headers] the specific HTTP headers to use
556  * @config {String} [mimeType] the override MIME type, default is
557  *             none
558  *
559  * @return {Deferred} a `MochiKit.Async.Deferred` object that will
560  *         callback with the response text on success
561  */
562 RapidContext.App.loadJSON = function (url, params, options) {
563     var d = RapidContext.App.loadXHR(url, params, options);
564     d.addCallback(function (res) {
565         return JSON.parse(res.responseText);
566     });
567     return d;
568 };

601 /**
602  * Performs an asynchronous HTTP request for an XML document and returns a
603  * deferred response. If no request method has been specified, the `POST` or
604  * `GET` methods are chosen depending on whether or not the params argument is
605  * `null`. The request parameters are specified as an object that will be
606  * encoded by the `MochiKit.Base.queryString` function. In addition to the
607  * default options in `MochiKit.Async.doXHR`, this function also accepts a
608  * timeout option for automatic request cancellation.
609  *
610  * @param {String} url the URL to request
611  * @param {Object} [params] the request parameters, or `null`
612  * @param {Object} [options] the request options, or `null`
613  * @config {String} [method] the HTTP method, "GET" or "POST"
614  * @config {Number} [timeout] the timeout in seconds, default is no timeout
615  * @config {Object} [headers] the specific HTTP headers to use
616  * @config {String} [mimeType] the override MIME type, default is
617  *             none
618  *
619  * @return {Deferred} a `MochiKit.Async.Deferred` object that will
620  *         callback with the parsed response XML document on success
621  */
622 RapidContext.App.loadXML = function (url, params, options) {
623     var d = RapidContext.App.loadXHR(url, params, options);
624     d.addCallback(function (res) { return res.responseXML; });
625     return d;
626 };

628 /**
629  * Performs an asynchronous HTTP request and returns a deferred response. If no
630  * request method has been specified, the `POST` or `GET` methods are chosen
631  * depending on whether or not the `params` argument is `null`. The request
632  * parameters are specified as an object that will be encoded by the
633  * `MochiKit.Base.queryString` function. In addition to the default options in
634  * `MochiKit.Async.doXHR`, this function also accepts a timeout option for
635  * automatic request cancellation.
636  *
637  * Note that this function is unsuitable for loading JavaScript source code,
638  * since using `eval()` will confuse some browser error messages and debuggers
639  * regarding the actual source location.
640  *
641  * @param {String} url the URL to request
642  * @param {Object} [params] the request parameters, or `null`
643  * @param {Object} [options] the request options, or `null`
644  * @config {String} [method] the HTTP method, "GET" or "POST"
645  * @config {Number} [timeout] the timeout in seconds, default is no timeout
646  * @config {Object} [headers] the specific HTTP headers to use
647  * @config {String} [mimeType] the override MIME type, default is
648  *             none
649  *
650  * @return {Deferred} a `MochiKit.Async.Deferred` object that will
651  *         callback with the XMLHttpRequest instance on success
652  */
653 RapidContext.App.loadXHR = function (url, params, options) {
654     options = options || {};
655     if (options.method == "GET" && params != null) {
656         url += "?" + MochiKit.Base.queryString(params);
657     } else if (params != null) {
658         options.method = options.method || "POST";
659         options.headers = options.headers || {};
660         options.headers["Content-Type"] = "application/x-www-form-urlencoded";
661         options.sendContent = MochiKit.Base.queryString(params);
662     } else {
663         options.method = options.method || "GET";
664     }
665     var nonCachedUrl = RapidContext.App._nonCachedUrl(url);
666     RapidContext.Log.log("Starting XHR loading", nonCachedUrl, options);
667     var d = MochiKit.Async.doXHR(nonCachedUrl, options);
668     if (options.timeout) {
669         var canceller = function () {
670             // TODO: Supply error to cancel() instead of this hack, when
671             // supported in MochiKit (#323). This work-around is necessary
672             // due to MochiKit internally using wait() in doXHR().
673             d.results[0].canceller();
674             d.results[0].errback("Timeout on request to " + url);
675         };
676         var timer = MochiKit.Async.callLater(options.timeout, canceller);
677         d.addCallback(function (res) {
678             timer.cancel();
679             return res;
680         });
681     }
682     d.addBoth(function (res) {
683         if (res instanceof MochiKit.Async.CancelledError) {
684             RapidContext.Log.log("Cancelled XHR loading", nonCachedUrl);
685         } else if (res instanceof Error) {
686             RapidContext.Log.warn("Failed XHR loading", nonCachedUrl, res);
687         } else {
688             RapidContext.Log.log("Completed XHR loading", nonCachedUrl);
689         }
690         return res;
691     });
692     RapidContext.App._addErrbackLogger(d, "RapidContext.App.loadXHR(" + nonCachedUrl + ")");
693     return d;
694 };

696 /**
697  * Loads a JavaScript to the the current page asynchronously and returns a
698  * deferred response. This function is only suitable for loading JavaScript
699  * source code, and not JSON data, since it loads the script by inserting a
700  * `<script>` tag in the document `<head>` tag. All function definitions and
701  * values must therefore be stored to global variables by the script to be
702  * accessible after loading. The deferred callback function will therefore not
703  * provide any data even on successful callback.
704  *
705  * This method of script loading has the advantage that JavaScript debuggers
706  * (such as Firebug) will be able to handle the code properly (error messages,
707  * breakpoints, etc). If the script fails to load due to errors however, the
708  * returned deferred object may fail to `errback` in some cases.
709  *
710  * @param {String} url the URL to the script
711  *
712  * @return {Deferred} a `MochiKit.Async.Deferred` object that will
713  *         callback when the script has been loaded
714  */
715 RapidContext.App.loadScript = function (url) {
716     var absoluteUrl = RapidContext.Util.resolveURI(url);
717     var selector1 = 'script[src^="' + url + '"]';
718     var selector2 = 'script[src^="' + absoluteUrl + '"]';
719     var elems = MochiKit.Selector.findDocElements(selector1, selector2);
720     if (elems.length > 0) {
721         RapidContext.Log.log("Script already loaded, skipping", url);
722         return MochiKit.Async.wait(0);
723     }
724     RapidContext.Log.log("Starting script loading", url);
725     var d = MochiKit.Async.loadScript(RapidContext.App._nonCachedUrl(url));
726     d.addCallback(function () {
727         RapidContext.Log.log("Completed loading script", url);
728     });
729     d.addErrback(function (e) {
730         RapidContext.Log.warn("Failed loading script", url + ": " + e.message);
731         return e;
732     });
733     RapidContext.App._addErrbackLogger(d, "RapidContext.App.loadScript(" + url + ")");
734     return d;
735 };

737 /**
738  * Loads a CSS stylesheet to the the current page asynchronously and returns a
739  * deferred response. The stylesheet is loaded by inserting a `<link>` tag in
740  * the document head tag, which means that the deferred callback function will
741  * not be provided with any data.
742  *
743  * @param {String} url the URL to the stylesheet
744  *
745  * @return {Deferred} a `MochiKit.Async.Deferred` object that will
746  *         callback when the stylesheet has been loaded
747  */
748 RapidContext.App.loadStyles = function (url) {
749     function findStylesheet(url) {
750         var styles = document.styleSheets;
751         for (var i = 0; i < styles.length; i++) {
752             if (MochiKit.Text.startsWith(url, styles[i].href)) {
753                 return styles[i];
754             }
755         }
756         return null;
757     }
758     function isStylesheetLoaded(url, absoluteUrl) {
759         var styles = findStylesheet(url) || findStylesheet(absoluteUrl);
760         var rules = styles && (styles.cssRules ||styles.rules);
761         return !!rules && rules.length > 0;
762     }
763     var absoluteUrl = RapidContext.Util.resolveURI(url);
764     if (findStylesheet(url) || findStylesheet(absoluteUrl)) {
765         RapidContext.Log.log("Stylesheet already loaded, skipping", url);
766         return MochiKit.Async.wait(0);
767     }
768     RapidContext.Log.log("Starting stylesheet loading", url);
769     var loadUrl = RapidContext.App._nonCachedUrl(url);
770     var link = MochiKit.DOM.LINK({ rel: "stylesheet", type: "text/css", href: loadUrl });
771     document.getElementsByTagName("head")[0].appendChild(link);
772     var d = new MochiKit.Async.Deferred();
773     var img = MochiKit.DOM.IMG();
774     img.onerror = d.callback.bind(d, 'URL loading finished');
775     img.src = loadUrl;
776     d.addBoth(function () {
777         if (!isStylesheetLoaded(url, absoluteUrl)) {
778             // Add 250 ms delay after URL loading to allow CSS parsing for
779             // browsers needing it (WebKit, Safari, Chrome)
780             return MochiKit.Async.wait(0.25);
781         }
782     });
783     d.addBoth(function () {
784         if (isStylesheetLoaded(url, absoluteUrl)) {
785             RapidContext.Log.log("Completed loading stylesheet", url);
786         } else {
787             RapidContext.Log.warn("Failed loading stylesheet", url);
788             throw new URIError("Failed loading stylesheet " + url, absoluteUrl);
789         }
790     });
791     RapidContext.App._addErrbackLogger(d, "RapidContext.App.loadStyles()");
792     return d;
793 }

795 /**
796  * Downloads a file to the user desktop. This works by creating a new
797  * window or an inner frame which downloads the file from the server.
798  * Due to `Content-Disposition` headers being set on the server, the
799  * web browser will popup a dialog for the user to save the file.
800  * This function can also be used for saving a file that doesn't
801  * exist by first posting the file (text) content to the server.
802  *
803  * @param {String} url the URL or filename to download
804  * @param {String} [data] the optional file data (if not available
805  *            on the server-side)
806  */
807 RapidContext.App.downloadFile = function (url, data) {
808     if (data == null) {
809         url = url + ((url.indexOf("?") < 0) ? "?" : "&") + "download";
810         var attrs = { src: url, border: "0", frameborder: "0",
811                       height: "0", width: "0" };
812         var iframe = MochiKit.DOM.createDOM("iframe", attrs);
813         document.body.appendChild(iframe);
814     } else {
815         var name = MochiKit.DOM.INPUT({ name: "fileName", value: url });
816         var file = MochiKit.DOM.INPUT({ name: "fileData", value: data });
817         var flag = MochiKit.DOM.INPUT({ name: "download", value: "1" });
818         var attrs = { action: "rapidcontext/download", method: "POST",
819                       target: "_blank", style: { display: "none" } };
820         var form = MochiKit.DOM.FORM(attrs, name, file, flag);
821         document.body.appendChild(form);
822         form.submit();
823     }
824 };

829  * @param {String} url the URL to modify
830  *

845 /**
846  * Returns a non-cacheable version of the specified URL. This
847  * function will add a request query parameter consisting of the
848  * current time in milliseconds. Most web servers will ignore this
849  * additional parameter, but due to the URL containing a unique
850  * value, the web browser will not use its cache for the content.
851  *
852  * @param {String} url the URL to modify
853  *
854  * @return {String} the URL including the current timestamp
855  */
856 RapidContext.App._nonCachedUrl = function (url) {
857     var timestamp = new Date().getTime() % 100000;
858     return url + ((url.indexOf("?") < 0) ? "?" : "&") + timestamp;
859 };

861 /**
862  * Adds an error logger to a `MochiKit.Async.Deferred` object. The
863  * logger will actually be added in the next JavaScript event loop,
864  * ensuring that any other callbacks or handlers have already been
865  * added to the deferred object. This is useful for catching
866  * programmer errors and similar that cause exceptions inside
867  * callback functions.
868  *
869  * @param {Deferred} d the `MochiKit.Async.Deferred` object to modify
870  * @param {String} logCtx the log context to identify the source
871  */
872 RapidContext.App._addErrbackLogger = function (d, logCtx) {
873     function logger(err) {
874         if (!d.chained && !(err instanceof MochiKit.Async.CancelledError)) {
875             RapidContext.Log.warn(logCtx + " deferred: unhandled error", err);
876         }
877         return err;
878     }
879     function adder() {
880         var lastCb = d.chain && d.chain[d.chain.length - 1];
881         var errback = lastCb && lastCb[1];
882         if (!d.chained && !errback) {
883             RapidContext.Log.warn(logCtx + " deferred: no error handler");
884             d.addErrback(logger);
885         }
886     }
887     MochiKit.Async.callLater(0, adder);
888 };

890 /**
891  * The application data cache. Contains the most recently retrieved
892  * data for some commonly used objects in the execution environment.
893  */
894 RapidContext.App._Cache = {
895     status: null,
896     user: null,
897     apps: [],
898     // Compares two object on the 'id' property
899     compareId: function (a, b) {
900         // TODO: replace with MochiKit.Base.keyComparator once #331 is fixed
901         if (a == null || b == null) {
902             return MochiKit.Base.compare(a, b);
903         } else {
904             return this._cmpId(a, b);
905         }
906     },
907     // Object comparator for 'id' property
908     _cmpId: MochiKit.Base.keyComparator("id"),
909     // Builds an icon DOM node from a resource
910     _buildIcon: function (res) {
911         if (res.url) {
912             var url = RapidContext.App._rebaseUrl(res.url);
913             return $("<img/>").attr({ src: url })[0];
914         } else if (res.html) {
915             var node = $("<span/>").html(res.html)[0];
916             return node.childNodes.length === 1 ? node.childNodes[0] : node;
917         } else if (res["class"]) {
918             return $("<i/>").addClass(res["class"])[0];
919         } else {
920             return null;
921         }
922     },
923     // Updates the cache data with the results from a procedure.
924     update: function (proc, data) {
925         switch (proc) {
926         case "System.Status":
927             // TODO: use deep clone
928             data = MochiKit.Base.update(null, data);
929             this.status = data;
930             RapidContext.Log.log("Updated cached status", this.status);
931             break;
932         case "System.Session.Current":
933             if (this.compareId(this.user, data.user) != 0) {
934                 // TODO: use deep clone
935                 data = MochiKit.Base.update(null, data.user);
936                 if (data.name != "") {
937                     data.longName = data.name + " (" + data.id + ")";
938                 } else {
939                     data.longName = data.id;
940                 }
941                 this.user = data;
942                 RapidContext.Log.log("Updated cached user", this.user);
943             }
944             break;
945         case "System.App.List":
946             var idxs = {};
947             for (var i = 0; i < this.apps.length; i++) {
948                 idxs[this.apps[i].id] = i;
949             }
950             for (var i = 0; i < data.length; i++) {
951                 // TODO: use deep clone
952                 var launcher = MochiKit.Base.update(null, data[i]);
953                 launcher.launch = launcher.launch || "manual";
954                 if (!(launcher.resources instanceof Array)) {
955                     launcher.resources = [];
956                 }
957                 for (var j = 0; j < launcher.resources.length; j++) {
958                     if (launcher.resources[j].type == "icon") {
959                         launcher.icon = this._buildIcon(launcher.resources[j]);
960                     }
961                 }
962                 if (!launcher.icon) {
963                     var cssClass = "fa fa-4x fa-question-circle unimportant";
964                     launcher.icon = this._buildIcon({ "class": cssClass });
965                 }
966                 if (launcher.className == null) {
967                     RapidContext.Log.error("App missing 'className' property", launcher);
968                 }
969                 var pos = idxs[launcher.id];
970                 if (pos >= 0) {
971                     delete idxs[launcher.id];
972                     MochiKit.Base.update(this.apps[pos], launcher);
973                 } else {
974                     this.apps.push(launcher);
975                 }
976             }
977             // TODO: use separate list of app instances...
978             var arr = MochiKit.Base.values(idxs).reverse();
979             for (var i = 0; i < arr.length; i++) {
980                 var pos = arr[i];
981                 if (this.apps[pos].instances == null) {
982                     this.apps.splice(pos, 1);
983                 }
984             }
985             RapidContext.Log.log("Updated cached apps", this.apps);
986             break;
987         }
988     }
989 };
