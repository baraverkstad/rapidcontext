 32 /**
 33  * Initializes the platform, API:s and RapidContext UI. If an app
 34  * identifier is provided, the default platform UI will not be
 35  * created. Instead the app will be launched with the root document
 36  * as its UI container.
 37  *
 38  * @param {String/Object} [app] the app id or class name to start
 39  *
 40  * @return {Promise} a promise that will resolve when initialization has
 41  *         either completed or failed
 42  */
 43 RapidContext.App.init = function (app) {
 44     // Setup libraries
 45     RapidContext.Log.context("RapidContext.App.init()");
 46     RapidContext.Log.info("Initializing RapidContext");
 47     RapidContext.Util.registerFunctionNames(RapidContext, "RapidContext");
 48     if (!RapidContext.Browser.isSupported()) {
 49         MochiKit.DOM.removeElementClass("unsupported-browser", "hidden");
 50         return Promise.reject(new Error("Unsupported browser"));
 51     }
 52
 53     // Setup UI
 54     RapidContext.Util.registerSizeConstraints(document.body, "100%-20", "100%-20");
 55     var resizer = MochiKit.Base.partial(RapidContext.Util.resizeElements, document.body);
 56     MochiKit.Signal.connect(window, "onresize", resizer);
 57     RapidContext.Util.resizeElements(document.body);
 58     var overlay = new RapidContext.Widget.Overlay({ message: "Loading..." });
 59     MochiKit.DOM.replaceChildNodes(document.body, overlay);
 60
 61     // Load platform data (into cache)
 62     var cachedData = [
 63         RapidContext.App.callProc("System.Status"),
 64         RapidContext.App.callProc("System.Session.Current"),
 65         RapidContext.App.callProc("System.App.List")
 66     ];
 67
 68     // Launch app
 69     return Promise.all(cachedData)
 70         .then(function () {
 71             if (app && app !== "start") {
 72                 return RapidContext.App.startApp(app || "start").catch(function (err) {
 73                     RapidContext.UI.showError(err);
 74                     return RapidContext.App.startApp("start");
 75                 });
 76             } else {
 77                 return RapidContext.App.startApp("start");
 78             }
 79         })
 80         .catch(function (err) {
 81             RapidContext.UI.showError(err);
 82             return Promise.reject(err);
 83         })
 84         .finally(function () {
 85             RapidContext.Log.context(null);
 86         });
 87 };

124 /**
125  * Returns an array with running app instances.
126  *
127  * @return {Array} the array of app instances
128  */
129 RapidContext.App._instances = function () {
130     var res = [];
131     var apps = RapidContext.App.apps();
132     for (var i = 0; i < apps.length; i++) {
133         res = res.concat(apps[i].instances || []);
134     }
135     return res;

138 /**
139  * Finds the app launcher from an app instance, class name or
140  * launcher. In the last case, the matching cached launcher will be
141  * returned.
142  *
143  * @param {String/Object} app the app id, instance, class name or
144  *        launcher
145  *
146  * @return {Object} the read-only app launcher, or
147  *         `null` if not found
148  */
149 RapidContext.App.findApp = function (app) {
150     if (app == null) {
151         return null;
152     }
153     var apps = RapidContext.App.apps();
154     for (var i = 0; i < apps.length; i++) {
155         var l = apps[i];
156         if (l.className == null) {
157             RapidContext.Log.error("Launcher does not have 'className' property", l);
158         } else if (l.id == app || l.className == app || l.id == app.id) {
159             return l;
160         }
161     }
162     return null;
163 };

165 /**
166  * Creates and starts an app instance. Normally apps are started in the default
167  * app tab container or in a provided widget DOM node. By specifying a newly
168  * created window object as the parent container, apps can also be launched
169  * into separate windows.
170  *
171  * Note that when a new window is used, the returned deferred will callback
172  * immediately with a `null` app instance (normal app communication is not
173  * possible cross-window).
174  *
175  * @param {String/Object} app the app id, class name or launcher
176  * @param {Widget/Window} [container] the app container widget or
177  *            window, defaults to create a new pane in the app tab
178  *            container
179  *
180  * @return {Promise} a `RapidContext.Async` promise that will
181  *         resolve when the app has launched
182  *
183  * @example
184  * // Starts the help app in a new tab
185  * RapidContext.App.startApp('help');
186  *
187  * // Starts the help app in a new window
188  * RapidContext.App.startApp('help', window.open());
189  */
190 RapidContext.App.startApp = function (app, container) {
191     function loadResource(launcher, res) {
192         if (res.type == "code") {
193             return RapidContext.App.loadScript(res.url);
194         } else if (res.type == "style") {
195             return RapidContext.App.loadStyles(res.url);
196         } else if (res.type == "ui") {
197             return RapidContext.App.loadXML(res.url).then(function (node) {
198                 launcher.ui = node;
199             });
200         } else if (res.type == "json" && res.id != null) {
201             return RapidContext.App.loadJSON(res.url).then(function (data) {
202                 launcher.resource[res.id] = data;
203             });
204         } else {
205             if (res.id != null) {
206                 launcher.resource[res.id] = res.url;
207             }
208             return Promise.resolve(res);
209         }
210     }
211     function load(launcher) {
212         RapidContext.Log.info("Loading app/" + launcher.id + " resources", launcher);
213         launcher.resource = {};
214         var promises = launcher.resources.map(loadResource.bind(null, launcher));
215         return Promise.all(promises).then(function () {
216             launcher.creator = window[launcher.className];
217             if (launcher.creator == null) {
218                 RapidContext.Log.error("App constructor " + launcher.className + " not defined", launcher);
219                 throw new Error("App constructor " + launcher.className + " not defined");
220             }
221             RapidContext.Util.registerFunctionNames(launcher.creator, launcher.className);
222         });
223     }
224     function buildUI(parent, ids, ui) {
225         var root = ui.documentElement;
226         for (var i = 0; i < root.attributes.length; i++) {
227             var attr = root.attributes[i];
228             if (attr.specified && typeof(parent.setAttrs) === "function") {
229                 var o = {};
230                 o[attr.name] = attr.value;
231                 parent.setAttrs(o);
232             } else if (attr.specified && attr.name === "class") {
233                 MochiKit.DOM.addElementClass(parent, attr.value);
234             } else if (attr.specified) {
235                 parent.setAttribute(attr.name, attr.value);
236             }
237         }
238         MochiKit.DOM.appendChildNodes(parent, RapidContext.UI.buildUI(ui, ids));
239         RapidContext.Util.resizeElements(parent);
240     }
241     function launch(launcher, ui) {
242         RapidContext.Log.context("RapidContext.App.startApp(" + launcher.id + ")");
243         return RapidContext.Async.wait(0)
244             .then((launcher.creator == null) ? load.bind(null, launcher) : function () {})
245             .then(function () {
246                 RapidContext.Log.info("Starting app/" + launcher.id, launcher);
247                 /* eslint new-cap: "off" */
248                 var instance = new launcher.creator();
249                 launcher.instances.push(instance);
250                 var props = MochiKit.Base.setdefault({ ui: ui }, launcher);
251                 delete props.creator;
252                 delete props.instances;
253                 MochiKit.Base.setdefault(instance, props);
254                 MochiKit.Signal.disconnectAll(ui.root, "onclose");
255                 var halt = MochiKit.Base.partial(RapidContext.App.stopApp, instance);
256                 MochiKit.Signal.connect(ui.root, "onclose", halt);
257                 if (launcher.ui != null) {
258                     buildUI(ui.root, ui, launcher.ui);
259                 }
260                 ui.overlay.hide();
261                 ui.overlay.setAttrs({ message: "Working..." });
262                 return RapidContext.App.callApp(instance, "start");
263             })
264             .catch(function (err) {
265                 RapidContext.Log.error("Failed to start app", err);
266                 MochiKit.Signal.disconnectAll(ui.root, "onclose");
267                 var label = MochiKit.DOM.STRONG(null, "Error: ");
268                 MochiKit.DOM.appendChildNodes(ui.root, label, err.message);
269                 ui.overlay.hide();
270                 return Promise.reject(err);
271             })
272             .finally(function () {
273                 RapidContext.Log.context(null);
274             });
275     }
276     function moveAppToStart(instance, elems) {
277         var start = RapidContext.App.findApp("start").instances[0];
278         var opts = { title: instance.name, closeable: false, background: true };
279         var ui = start.initAppPane(null, opts);
280         ui.root.removeAll();
281         ui.root.addAll(elems);
282         RapidContext.Util.resizeElements(ui.root);
283     }
284     return new RapidContext.Async(function (resolve, reject) {
285         var launcher = RapidContext.App.findApp(app);
286         if (launcher == null) {
287             var msg = "No matching app launcher found";
288             RapidContext.Log.error(msg, app);
289             throw new Error([msg, ": ", app].join(""));
290         }
291         var instances = RapidContext.App._instances();
292         var start = RapidContext.App.findApp("start");
293         if ($.isWindow(container)) {
294             // Launch app into separate window/tab
295             var href = "rapidcontext/app/" + launcher.id;
296             container.location.href = RapidContext.Util.resolveURI(href);
297             resolve();
298         } else if (start && start.instances && start.instances.length > 0) {
299             // Launch app into start app tab
300             var paneOpts = { title: launcher.name, closeable: (launcher.launch != "once") };
301             var paneUi = start.instances[0].initAppPane(container, paneOpts);
302             resolve(launch(launcher, paneUi));
303         } else if (instances.length > 0) {
304             // Switch from single-app to multi-app mode
305             var elems = Array.prototype.slice.call(document.body.childNodes);
306             var moveApp = moveAppToStart.bind(null, instances[0], elems);
307             var overlay = new RapidContext.Widget.Overlay({ message: "Loading..." });
308             document.body.insertBefore(overlay, document.body.childNodes[0]);
309             var promise = launch(start, { root: document.body, overlay: overlay }).then(moveApp);
310             // FIXME: double calls to this ?
311             var recall = RapidContext.App.startApp.bind(null, launcher.id, container);
312             resolve((launcher.id === "start") ? promise : promise.then(recall));
313         } else {
314             // Launch single-app mode
315             var ui = { root: document.body, overlay: document.body.childNodes[0] };
316             resolve(launch(launcher, ui));
317         }
318     });
319 };

321 /**
322  * Stops an app instance. If only the class name or launcher is specified, the
323  * most recently created instance will be stopped.
324  *
325  * @param {String/Object} app the app id, instance, class name or launcher
326  *
327  * @return {Promise} a `RapidContext.Async` promise that will
328  *         resolve when the app has been stopped
329  */
330 RapidContext.App.stopApp = function (app) {
331     return new RapidContext.Async(function (resolve, reject) {
332         var launcher = RapidContext.App.findApp(app);
333         if (!launcher || launcher.instances.length <= 0) {
334             var msg = "No running app instance found";
335             RapidContext.Log.error(msg, app);
336             throw new Error([msg, ": ", app].join(""));
337         }
338         RapidContext.Log.info("Stopping app " + launcher.name);
339         var pos = MochiKit.Base.findIdentical(launcher.instances, app);
340         if (pos < 0) {
341             app = launcher.instances.pop();
342         } else {
343             launcher.instances.splice(pos, 1);
344         }
345         app.stop();
346         if (app.ui.root != null) {
347             RapidContext.Widget.destroyWidget(app.ui.root);
348         }
349         for (var k in app.ui) {
350             delete app.ui[k];
351         }
352         for (var n in app) {
353             delete app[n];
354         }
355         MochiKit.Signal.disconnectAllTo(app);
356         resolve();
357     });
358 };

360 /**
361  * Performs an asynchronous call to a method in an app. If only the class name
362  * or launcher is specified, the most recently created instance will be used.
363  * If no instance is running, one will be started. Also, before calling the app
364  * method, the app UI will be focused.
365  *
366  * @param {String/Object} app the app id, instance, class name or launcher
367  * @param {String} method the app method name
368  * @param {Mixed} [args] additional parameters sent to method
369  *
370  * @return {Promise} a `RapidContext.Async` promise that will
371  *         resolve with the result of the call on success
372  */
373 RapidContext.App.callApp = function (app, method) {
374     return new RapidContext.Async(function (resolve, reject) {
375         var args = Array.prototype.slice.call(arguments, 2);
376         var launcher = RapidContext.App.findApp(app);
377         if (launcher == null) {
378             var msg = "No matching app launcher found";
379             RapidContext.Log.error(msg, app);
380             throw new Error([msg, ": ", app].join(""));
381         }
382         var exists = launcher.instances.length > 0;
383         var promise = exists ? RapidContext.Async.wait(0) : RapidContext.App.startApp(app);
384         return promise.then(function () {
385             RapidContext.Log.context("RapidContext.App.callApp(" + launcher.id + "," + method + ")");
386             var pos = MochiKit.Base.findIdentical(launcher.instances, app);
387             var instance = (pos >= 0) ? app : launcher.instances[launcher.instances.length - 1];
388             var child = instance.ui.root;
389             var parent = MochiKit.DOM.getFirstParentByTagAndClassName(child, null, "widget");
390             if (parent != null && typeof(parent.selectChild) == "function") {
391                 parent.selectChild(child);
392             }
393             var methodName = launcher.className + "." + method;
394             if (instance[method] == null) {
395                 var msg = "No app method " + methodName + " found";
396                 RapidContext.Log.error(msg);
397                 throw new Error(msg);
398             }
399             RapidContext.Log.log("Calling app method " + methodName, args);
400             try {
401                 return instance[method].apply(instance, args);
402             } catch (e) {
403                 var reason = "Caught error in " + methodName;
404                 RapidContext.Log.error(reason, e);
405                 throw new Error(reason + ": " + e.toString(), { cause: e });
406             }
407         }).finally(function () {
408             RapidContext.Log.context(null);
409         });
410     });
411 };

413 /**
414  * Performs an asynchronous procedure call.
415  *
416  * @param {String} name the procedure name
417  * @param {Array} [args] the array of arguments, or `null`
418  *
419  * @return {Promise} a `RapidContext.Async` promise that will
420  *         resolve with the response data on success
421  */
422 RapidContext.App.callProc = function (name, args) {
423     // TODO: remove this legacy name conversion
424     if (name.indexOf("RapidContext.") == 0) {
425         name = "System" + name.substring(8);
426     }
427     RapidContext.Log.log("Call request " + name, args);
428     var params = {};
429     for (var i = 0; args != null && i < args.length; i++) {
430         if (args[i] == null) {
431             params["arg" + i] = "null";
432         } else {
433             params["arg" + i] = JSON.stringify(args[i]);
434         }
435     }
436     var logLevel = RapidContext.Log.level();
437     if (logLevel == "log" || logLevel == "all") {
438         params["system:trace"] = 1;
439     }
440     var url = "rapidcontext/procedure/" + name;
441     var options = { method: "POST", timeout: 60000 };
442     return RapidContext.App.loadJSON(url, params, options).then(function (res) {
443         if (res.trace != null) {
444             RapidContext.Log.log("Server trace " + name, res.trace);
445         }
446         if (res.error != null) {
447             RapidContext.Log.warn("Call error " + name, res.error);
448             throw new Error(res.error);
449         } else {
450             RapidContext.Log.log("Call response " + name, res.data);
451         }
452         if (name.indexOf("System.") == 0 && !res.error && res.data) {
453             RapidContext.App._Cache.update(name, res.data);
454         }
455         return res.data;
456     });
457 };

459 /**
460  * Performs an asynchronous login. If the current session is already bound to a
461  * user, that session will be terminated and a new one will be created. If an
462  * authentication token is specified, the login and password fields are not
463  * used (can be null).
464  *
465  * @param {String} login the user login name or email address
466  * @param {String} password the password to authenticate the user
467  * @param {String} [token] the authentication token to identify user/password
468  *
469  * @return {Promise} a `RapidContext.Async` promise that resolves when
470  *         the authentication has either succeeded or failed
471  */
472 RapidContext.App.login = function (login, password, token) {
473     function searchLogin() {
474         var proc = "System.User.Search";
475         return RapidContext.App.callProc(proc, [login]).then(function (user) {
476             if (user && user.id) {
477                 return login = user.id;
478             } else {
479                 throw new Error("no user with that email address");
480             }
481         });
482     }
483     function getNonce() {
484         var proc = "System.Session.Current";
485         return RapidContext.App.callProc(proc).then(function (session) {
486             return session.nonce;
487         });
488     }
489     function passwordAuth(nonce) {
490         var realm = RapidContext.App.status().realm;
491         var hash = CryptoJS.MD5(login + ":" + realm + ":" + password);
492         hash = CryptoJS.MD5(hash.toString() + ":" + nonce).toString();
493         var args = [login, nonce, hash];
494         return RapidContext.App.callProc("System.Session.Authenticate", args);
495     }
496     function tokenAuth() {
497         return RapidContext.App.callProc("System.Session.AuthenticateToken", [token]);
498     }
499     function verifyAuth(res) {
500         if (!res.success || res.error) {
501             console.info("login failed", login, res.error);
502             throw new Error(res.error || "authentication failed");
503         }
504     }
505     var promise = RapidContext.Async.wait(0);
506     var user = RapidContext.App.user();
507     if (user && user.id) {
508         promise = promise.then(RapidContext.App.logout);
509     }
510     if (token) {
511         return promise.then(tokenAuth).then(verifyAuth);
512     } else {
513         if (/@/.test(login)) {
514             promise = promise.then(searchLogin);
515         }
516         return promise.then(getNonce).then(passwordAuth).then(verifyAuth);
517     }
518 };

520 /**
521  * Performs an asyncronous logout. This function terminates the current
522  * session and either reloads the browser window or returns a deferred object
523  * that will produce either a `callback` or an `errback` response.
524  *
525  * @param {Boolean} [reload] the reload browser flag, defaults to true
526  *
527  * @return {Promise} a `RapidContext.Async` promise that will
528  *         resolve when user is logged out
529  */
530 RapidContext.App.logout = function (reload) {
531     var promise = RapidContext.App.callProc("System.Session.Terminate", [null]);
532     if (reload !== false) {
533         promise.then(function () {
534             window.location.reload();
535         });
536     }
537     return promise;
538 };

541  * Performs an asynchronous HTTP request and parses the JSON response. The
542  * request parameters are automatically encoded to query string or JSON format,
543  * depending on the `Content-Type` header. The parameters will be sent either
544  * in the URL or as the request payload (depending on the HTTP `method`).

546  * @param {String} url the URL to request
547  * @param {Object} [params] the request parameters, or `null`
548  * @param {Object} [options] the request options, or `null`

550  * @config {Number} [timeout] the timeout in milliseconds, default is 30s
551  * @config {Object} [headers] the specific HTTP headers to use
552  *
553  * @return {Promise} a `RapidContext.Async` promise that will
554  *         resolve with the parsed response JSON on success
555  */
556 RapidContext.App.loadJSON = function (url, params, options) {
557     var opts = MochiKit.Base.update({ responseType: "json" }, options);
558     return RapidContext.App.loadXHR(url, params, opts).then(function (xhr) {
559         return xhr.response;
560     });
561 };

611  * automatically encoded to query string or JSON format, depending on the
612  * `Content-Type` header. The parameters will be sent either in the URL or as
613  * the request payload (depending on the HTTP `method`).
614  *
615  * @param {String} url the URL to request
616  * @param {Object} [params] the request parameters, or `null`
617  * @param {Object} [options] the request options, or `null`
618  * @config {String} [method] the HTTP method, default is `GET`
619  * @config {Number} [timeout] the timeout in milliseconds, default is 30s
620  * @config {Object} [headers] the specific HTTP headers to use
621  *
622  * @return {Promise} a `RapidContext.Async` promise that will
623  *         resolve with the XMLHttpRequest instance on success
624  */
625 RapidContext.App.loadXHR = function (url, params, options) {
626     var opts = MochiKit.Base.update({ method: "GET", headers: {}, timeout: 30000 }, options);
627     opts.timeout = (opts.timeout < 1000) ? opts.timeout * 1000 : opts.timeout;
628     var hasBody = params && ["PATCH", "POST", "PUT"].indexOf(opts.method) >= 0;
629     url += (params && !hasBody) ? "?" + MochiKit.Base.queryString(params) : "";
630     if (params && hasBody && opts.headers["Content-Type"] === "application/json") {
631         opts.body = JSON.stringify(params);
632     } else if (params && hasBody) {
633         opts.headers["Content-Type"] = "application/x-www-form-urlencoded";
634         opts.body = MochiKit.Base.queryString(params);
635     }
636     var nonCachedUrl = RapidContext.App._nonCachedUrl(url);
637     RapidContext.Log.log("Starting XHR loading", nonCachedUrl, opts);
638     return RapidContext.Async.xhr(nonCachedUrl, opts).then(
639         function (res) {
640             RapidContext.Log.log("Completed XHR loading", nonCachedUrl);
641             return res;
642         },
643         function (err) {
644             RapidContext.Log.warn("Failed XHR loading", nonCachedUrl, err);
645             return Promise.reject(err);
646         }
647     );
648 };

650 /**
651  * Loads a JavaScript to the the current page. The script is loaded by
652  * inserting a `<script>` tag in the document `<head>`. Function definitions
653  * and values must therefore be stored to global variables by the script to
654  * become accessible after loading. If the script is already loaded, the
655  * promise will resolve immediately.
656  *
657  * @param {String} url the URL to the script
658  *
659  * @return {Promise} a `RapidContext.Async` promise that will
660  *         resolve when the script has loaded
661  *
662  * @see Use dynamic `import()` instead, if supported by the environment.
663  */
664 RapidContext.App.loadScript = function (url) {
665     var selector = ["script[src*='", url, "']"].join("");
666     if (document.querySelectorAll(selector).length > 0) {
667         RapidContext.Log.log("script already loaded, skipping", url);
668         return RapidContext.Async.wait(0);
669     } else {
670         RapidContext.Log.log("loading script", url);
671         return RapidContext.Async.script(RapidContext.App._nonCachedUrl(url));
672     }
673 };

696 /**
697  * Downloads a file to the user desktop. This works by creating a new
698  * window or an inner frame which downloads the file from the server.
699  * Due to `Content-Disposition` headers being set on the server, the
700  * web browser will popup a dialog for the user to save the file.
701  * This function can also be used for saving a file that doesn't
702  * exist by first posting the file (text) content to the server.
703  *
704  * @param {String} url the URL or filename to download
705  * @param {String} [data] the optional file data (if not available
706  *            on the server-side)
707  */
708 RapidContext.App.downloadFile = function (url, data) {
709     if (data == null) {
710         url = url + ((url.indexOf("?") < 0) ? "?" : "&") + "download";
711         var attrs = {
712             src: url,
713             border: "0",
714             frameborder: "0",
715             height: "0",
716             width: "0"
717         };
718         var iframe = MochiKit.DOM.createDOM("iframe", attrs);
719         document.body.appendChild(iframe);
720     } else {
721         var name = MochiKit.DOM.INPUT({ name: "fileName", value: url });
722         var file = MochiKit.DOM.INPUT({ name: "fileData", value: data });
723         var flag = MochiKit.DOM.INPUT({ name: "download", value: "1" });
724         var formAttrs = {
725             action: "rapidcontext/download",
726             method: "POST",
727             target: "_blank",
728             style: { display: "none" }
729         };
730         var form = MochiKit.DOM.FORM(formAttrs, name, file, flag);
731         document.body.appendChild(form);
732         form.submit();
733     }
734 };

737  * Uploads a file to the server. The upload is handled by an
738  * asynchronous HTTP request.
739  *
740  * @param {String} id the upload identifier to use

757 /**
758  * Returns a new relative URL adapted for a non-standard base path.
759  *
760  * @param {String} url the URL to modify
761  *
762  * @return {String} the rebased URL
763  */
764 RapidContext.App._rebaseUrl = function (url) {
765     url = url || "";
766     if (RapidContext._basePath) {
767         if (url.indexOf(RapidContext._basePath) == 0) {
768             url = url.substring(RapidContext._basePath.length);
769         } else if (url.indexOf(":") < 0) {
770             url = "rapidcontext/files/" + url;
771         }
772     }
773     return url;
774 };

776 /**
777  * Returns a non-cacheable version of the specified URL. This
778  * function will add a request query parameter consisting of the
779  * current time in milliseconds. Most web servers will ignore this
780  * additional parameter, but due to the URL containing a unique
781  * value, the web browser will not use its cache for the content.
782  *
783  * @param {String} url the URL to modify
784  *
785  * @return {String} the URL including the current timestamp
786  */
787 RapidContext.App._nonCachedUrl = function (url) {
788     var timestamp = new Date().getTime() % 100000;
789     return url + ((url.indexOf("?") < 0) ? "?" : "&") + timestamp;
790 };

792 /**
793  * The application data cache. Contains the most recently retrieved
794  * data for some commonly used objects in the execution environment.
795  */
796 RapidContext.App._Cache = {
797     status: null,
798     user: null,
799     apps: {},
800
801     // Compares two object on the 'id' property
802     compareId: function (a, b) {
803         // TODO: replace with MochiKit.Base.keyComparator once #331 is fixed
804         if (a == null || b == null) {
805             return MochiKit.Base.compare(a, b);
806         } else {
807             return this._cmpId(a, b);
808         }
809     },
810
811     // Object comparator for 'id' property
812     _cmpId: MochiKit.Base.keyComparator("id"),
813
814     // Normalizes an app manifest and its resources
815     _normalizeApp: function (app) {
816         function toType(type, url) {
817             var isJs = !type && /\.js$/i.test(url);
818             var isCss = !type && /\.css$/i.test(url);
819             if (type == "code" || type == "js" || type == "javascript" || isJs) {
820                 return "code";
821             } else if (type == "style" || type == "css" || isCss) {
822                 return "style";
823             } else if (type == "json" || (!type && /\.json$/i.test(url))) {
824                 return "json";
825             } else if (type == "ui" || (!type && /ui\.xml$/i.test(url))) {
826                 return "ui";
827             } else {
828                 return type;
829             }
830         }
831         function toIcon(res) {
832             if (res.url) {
833                 return $("<img/>").attr({ src: res.url }).get(0);
834             } else if (res.html) {
835                 var node = $("<span/>").html(res.html).get(0);
836                 return node.childNodes.length === 1 ? node.childNodes[0] : node;
837             } else if (res["class"]) {
838                 return $("<i/>").addClass(res["class"]).get(0);
839             } else {
840                 return null;
841             }
842         }
843         function toResource(res) {
844             res = (typeof(res) === "string") ? { url: res } : res;
845             res.type = toType(res.type, res.url);
846             if (res.url) {
847                 res.url = RapidContext.App._rebaseUrl(res.url);
848             }
849             if (!app.icon && res.type === "icon") {
850                 app.icon = toIcon(res);
851             }
852             return res;
853         }
854         app = MochiKit.Base.update(null, app);
855         app.launch = app.launch || "manual";
856         app.resources = [].concat(app.resources).filter(Boolean).map(toResource);
857         if (!app.icon) {
858             var cssClass = "fa fa-4x fa-question-circle unimportant";
859             app.icon = toIcon({ "class": cssClass });
860         }
861         return app;
862     },
863
864     // Updates the cache data with the results from a procedure.
865     update: function (proc, data) {
866         switch (proc) {
867         case "System.Status":
868             // TODO: use deep clone
869             data = MochiKit.Base.update(null, data);
870             this.status = data;
871             RapidContext.Log.log("Updated cached status", this.status);
872             break;
873         case "System.Session.Current":
874             if (this.compareId(this.user, data.user) != 0) {
875                 // TODO: use deep clone
876                 data = MochiKit.Base.update(null, data.user);
877                 if (data.name != "") {
878                     data.longName = data.name + " (" + data.id + ")";
879                 } else {
880                     data.longName = data.id;
881                 }
882                 this.user = data;
883                 RapidContext.Log.log("Updated cached user", this.user);
884             }
885             break;
886         case "System.App.List":
887             for (var i = 0; i < data.length; i++) {
888                 var launcher = this._normalizeApp(data[i]);
889                 if (launcher.className == null) {
890                     RapidContext.Log.error("App missing 'className' property", launcher);
891                 }
892                 launcher.instances = (this.apps[launcher.id] || {}).instances || [];
893                 this.apps[launcher.id] = MochiKit.Base.update(this.apps[launcher.id], launcher);
894             }
895             RapidContext.Log.log("Updated cached apps", this.apps);
896             break;
897         }
898     }
899 };
