 21 /**
 22  * Creates a new text area (or text box) widget.
 23  *
 24  * @constructor
 25  * @param {Object} attrs the widget and node attributes
 26  * @param {String} [attrs.name] the form field name
 27  * @param {String} [attrs.value] the field value, defaults to ""
 28  * @param {String} [attrs.helpText] the help text when empty (deprecated)
 29  * @param {Boolean} [attrs.disabled] the disabled widget flag, defaults to
 30  *            false
 31  * @param {Boolean} [attrs.hidden] the hidden widget flag, defaults to false
 32  * @param {Object} [...] the initial text content
 33  *
 34  * @return {Widget} the widget DOM node
 35  *
 36  * @class The text area widget class. Used to provide a text input field
 37  *     spanning multiple rows, using the `<textarea>` HTML element.
 38  * @property {Boolean} disabled The read-only widget disabled flag.
 39  * @property {String} defaultValue The value to use on form reset.
 40  * @extends RapidContext.Widget
 41  *
 42  * @example {JavaScript}
 43  * var attrs = { name="description", placeholder: "Description Text" };
 44  * var field = RapidContext.Widget.TextArea(attrs);
 45  *
 46  * @example {User Interface XML}
 47  * <TextArea name="description" placeholder="Description Text" />
 48  */
 49 RapidContext.Widget.TextArea = function (attrs/*, ...*/) {
 50     function scrape(val) {
 51         return String(val && val.textContent || val || "");
 52     }
 53     var text = (attrs && attrs.value) || "";
 54     text += Array.prototype.slice.call(arguments, 1).map(scrape).join("");
 55     var o = MochiKit.DOM.TEXTAREA({ value: text });
 56     RapidContext.Widget._widgetMixin(o, RapidContext.Widget.TextArea);
 57     o.addClass("widgetTextArea");
 58     o.setAttrs(MochiKit.Base.update({}, attrs, { value: text }));
 59     o.addEventListener("input", o._handleChange);
 60     return o;
 61 };

 63 // Register widget class
 64 RapidContext.Widget.Classes.TextArea = RapidContext.Widget.TextArea;

 66 /**
 67  * Emitted when the text is modified. This event is triggered by either
 68  * user events (keypress, paste, cut, blur) or by setting the value via
 69  * setAttrs(). The DOM standard onchange event has no 'event.detail'
 70  * data and is triggered on blur. The synthetic onchange events all
 71  * contain an 'event.detail' object with 'before', 'after' and 'cause'
 72  * properties.
 73  *
 74  * @name RapidContext.Widget.TextArea#onchange
 75  * @event
 76  */
 77
 78 /**
 79  * Destroys this widget.
 80  */
 81 RapidContext.Widget.TextArea.prototype.destroy = function () {
 82     // FIXME: Use AbortSignal instead to disconnect
 83     this.removeEventListener("input", this._handleChange);
 84 };
 85
 86 /**
 87  * Updates the widget or HTML DOM node attributes.
 88  *
 89  * @param {Object} attrs the widget and node attributes to set
 90  * @param {String} [attrs.name] the form field name
 91  * @param {String} [attrs.value] the field value
 92  * @param {String} [attrs.helpText] the help text when empty (deprecated)
 93  * @param {Boolean} [attrs.disabled] the disabled widget flag
 94  * @param {Boolean} [attrs.hidden] the hidden widget flag
 95  */
 96 RapidContext.Widget.TextArea.prototype.setAttrs = function (attrs) {
 97     attrs = MochiKit.Base.update({}, attrs);
 98     var locals = RapidContext.Util.mask(attrs, ["helpText", "value"]);
 99     if ("helpText" in locals) {
100         attrs.placeholder = attrs.placeholder || locals.helpText;
101     }
102     if ("value" in locals) {
103         this.value = locals.value || "";
104         this._handleChange(null);
105     }
106     this.__setAttrs(attrs);
107 };
108
109 /**
110  * Resets the text area form value to the initial value.
111  */
112 RapidContext.Widget.TextArea.prototype.reset = function () {
113     this.setAttrs({ value: this.defaultValue });
114 };
115
116 /**
117  * Returns the text area value. This function is slightly different from using
118  * the `value` property directly, since it will attempt to normalize newlines
119  * in the value.
120  *
121  * @return {String} the field value
122  *
123  * @example
124  * var str = field.getValue();
125  * var lines = str.split("\n").map((s) => s.trim());
126  * field.setAttrs({ "value": lines.join("\n") });
127  */
128 RapidContext.Widget.TextArea.prototype.getValue = function () {
129     var str = this.value;
130     // This is a hack to remove multiple newlines caused by
131     // platforms inserting or failing to normalize newlines
132     // within the HTML textarea control.
133     str = str.replace(/\r\n\n/g, "\n");
134     if (this.value != str) {
135         this.value = str;
136     }
137     return str;
138 };
139
140 /**
141  * Handles input events for this this widget.
142  *
143  * @param {Event} [evt] the DOM Event object or null for manual
144  */
145 RapidContext.Widget.TextArea.prototype._handleChange = function (evt) {
146     var cause = (evt && evt.inputType) || "set";
147     var detail = { before: this.storedValue || "", after: this.value, cause: cause };
148     this._dispatch("change", { detail: detail, bubbles: true });
149     this.storedValue = this.value;
150 };
