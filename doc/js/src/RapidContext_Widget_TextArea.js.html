 16 if (typeof(RapidContext) == "undefined") {
 17     RapidContext = {};
 18 }
 19 RapidContext.Widget = RapidContext.Widget || { Classes: {} };

 21 /**
 22  * Creates a new text area (or text box) widget.
 23  *
 24  * @constructor
 25  * @param {Object} attrs the widget and node attributes
 26  * @param {String} [attrs.name] the form field name
 27  * @param {String} [attrs.value] the field value, defaults to ""
 28  * @param {String} [attrs.helpText] the help text shown on empty input,
 29  *            defaults to ""
 30  * @param {Boolean} [attrs.disabled] the disabled widget flag, defaults to
 31  *            false
 32  * @param {Boolean} [attrs.hidden] the hidden widget flag, defaults to false
 33  * @param {Object} [...] the initial text content
 34  *
 35  * @return {Widget} the widget DOM node
 36  *
 37  * @class The text area widget class. Used to provide a text input field
 38  *     spanning multiple rows, using the `<textarea>` HTML element.
 39  * @property {Boolean} disabled The read-only widget disabled flag.
 40  * @property {Boolean} focused The read-only widget focused flag.
 41  * @property {String} defaultValue The value to use on form reset.
 42  * @extends RapidContext.Widget
 43  *
 44  * @example {JavaScript}
 45  * var attrs = { name="description", helpText: "Description Text" };
 46  * var field = RapidContext.Widget.TextArea(attrs);
 47  *
 48  * @example {User Interface XML}
 49  * <TextArea name="description" helpText="Description Text" />
 50  */
 51 RapidContext.Widget.TextArea = function (attrs/*, ...*/) {
 52     var text = "";
 53     if (attrs != null && attrs.value != null) {
 54         text = attrs.value;
 55     }
 56     for (var i = 1; i < arguments.length; i++) {
 57         var arg = arguments[i];
 58         if (RapidContext.Util.isDOM(arg)) {
 59             text += MochiKit.DOM.scrapeText(arg);
 60         } else if (arg != null) {
 61             text += String(arg);
 62         }
 63     }
 64     var o = MochiKit.DOM.TEXTAREA({ value: text });
 65     RapidContext.Widget._widgetMixin(o, RapidContext.Widget.TextArea);
 66     o.addClass("widgetTextArea");
 67     o.focused = false;
 68     o.setAttrs(MochiKit.Base.update({ helpText: "", value: text }, attrs));
 69     var changeHandler = RapidContext.Widget._eventHandler(null, "_handleChange");
 70     o.onkeyup = changeHandler;
 71     o.oncut = changeHandler;
 72     o.onpaste = changeHandler;
 73     var focusHandler = RapidContext.Widget._eventHandler(null, "_handleFocus");
 74     o.onfocus = focusHandler;
 75     o.onblur = focusHandler;
 76     return o;
 77 };

 79 // Register widget class
 80 RapidContext.Widget.Classes.TextArea = RapidContext.Widget.TextArea;

 82 /**
 83  * Emitted when the text is modified. This event is triggered by both
 84  * keypress and paste events if the text content is modified. The DOM
 85  * standard onchange event may also be trigged (on blur).
 86  *
 87  * @name RapidContext.Widget.TextArea#onchange
 88  * @event
 89  */
 90
 91 /**
 92  * Updates the widget or HTML DOM node attributes.
 93  *
 94  * @param {Object} attrs the widget and node attributes to set
 95  * @param {String} [attrs.name] the form field name
 96  * @param {String} [attrs.value] the field value
 97  * @param {String} [attrs.helpText] the help text shown on empty input
 98  * @param {Boolean} [attrs.disabled] the disabled widget flag
 99  * @param {Boolean} [attrs.hidden] the hidden widget flag
100  */
101 RapidContext.Widget.TextArea.prototype.setAttrs = function (attrs) {
102     attrs = MochiKit.Base.update({}, attrs);
103     var locals = RapidContext.Util.mask(attrs, ["helpText", "value"]);
104     if (typeof(locals.helpText) != "undefined") {
105         var str = MochiKit.Format.strip(locals.helpText);
106         if ("placeholder" in this) {
107             attrs.placeholder = str;
108         } else {
109             this.helpText = str;
110         }
111     }
112     if (typeof(locals.value) != "undefined") {
113         this.value = locals.value;
114         this._handleChange();
115     }
116     this.__setAttrs(attrs);
117     this._render();
118 };
119
120 /**
121  * Resets the text area form value to the initial value.
122  */
123 RapidContext.Widget.TextArea.prototype.reset = function () {
124     this.setAttrs({ value: this.defaultValue });
125 };
126
127 /**
128  * Returns the text area value. This function is slightly different from using
129  * the `value` property directly, since it will always return the actual value
130  * instead of the temporary help text displayed when the text area is empty and
131  * unfocused.
132  *
133  * @return {String} the field value
134  *
135  * @example
136  * var value = field.getValue();
137  * var lines = value.split("\n");
138  * lines = MochiKit.Base.map(MochiKit.Format.strip, lines);
139  * value = lines.join("\n");
140  * field.setAttrs({ "value": value });
141  */
142 RapidContext.Widget.TextArea.prototype.getValue = function () {
143     var str = (this.focused) ? this.value : this.storedValue;
144     // This is a hack to remove multiple newlines caused by
145     // platforms inserting or failing to normalize newlines
146     // within the HTML textarea control.
147     if (/\r\n\n/.test(str)) {
148         str = str.replace(/\r\n\n/g, "\n");
149     }
150     if (this.focused && this.value != str) {
151         this.value = str;
152     }
153     return str;
154 };
155
156 /**
157  * Handles keypress and paste events for this this widget.
158  *
159  * @param evt the MochiKit.Signal.Event object
160  */
161 RapidContext.Widget.TextArea.prototype._handleChange = function (evt) {
162     if (evt) {
163         setTimeout(MochiKit.Base.bind("_handleChange", this));
164     } else if (this.storedValue != this.value) {
165         this.storedValue = this.value;
166         RapidContext.Widget.emitSignal(this, "onchange", this.value);
167     }
168 };
169
170 /**
171  * Handles focus and blur events for this widget.
172  *
173  * @param evt the MochiKit.Signal.Event object
174  */
175 RapidContext.Widget.TextArea.prototype._handleFocus = function (evt) {
176     var value = this.getValue();
177     if (evt.type() == "focus") {
178         this.focused = true;
179         if (this.value != value) {
180             this.value = value;
181         }
182     } else if (evt.type() == "blur") {
183         this.focused = false;
184         this.storedValue = value;
185     }
186     this._render();
187 };
188
189 /**
190  * Updates the display of the widget content.
191  */
192 RapidContext.Widget.TextArea.prototype._render = function () {
193     var value = this.getValue();
194     var str = MochiKit.Format.strip(value);
195     if (!this.focused && str == "" && this.helpText) {
196         this.value = this.helpText;
197         this.addClass("widgetTextAreaHelp");
198     } else {
199         if (this.value != value) {
200             this.value = value;
201         }
202         this.removeClass("widgetTextAreaHelp");
203     }
204 };
