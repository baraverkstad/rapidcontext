 21 /**
 22  * Creates a new popup widget.
 23  *
 24  * @constructor
 25  * @param {Object} attrs the widget and node attributes
 26  * @param {Number} [attrs.delay] the widget auto-hide delay in
 27  *            milliseconds, defaults to `5000`
 28  * @param {Object} [attrs.showAnim] the optional animation options
 29               when showing the popup, defaults to none
 30  * @param {Object} [attrs.hideAnim] the optional animation options
 31  *            when hiding the popup, defaults to none
 32  * @param {Boolean} [attrs.hidden] the hidden widget flag, defaults to `true`
 33  * @param {Widget} [...] the child widgets or DOM nodes
 34  *
 35  * @return {Widget} the widget DOM node
 36  *
 37  * @class The popup widget class. Used to provide a popup menu or information
 38  *     area, using a `<div>` HTML element. The `Popup` widget will automatically
 39  *     disappear after a configurable amount of time, unless the user performs
 40  *     keyboard or mouse actions related to the popup.
 41  * @extends RapidContext.Widget
 42  *
 43  * @example {JavaScript}
 44  * var attrs1 = { value: 1, "class": "widgetPopupItem" };
 45  * var attrs2 = { value: 2, "class": "widgetPopupItem" };
 46  * var attrs3 = { value: 3, "class": "widgetPopupItem widgetPopupDisabled" };
 47  * var item1 = MochiKit.DOM.DIV(attrs1, "First Item");
 48  * var item2 = MochiKit.DOM.DIV(attrs2, "Second Item");
 49  * var item3 = MochiKit.DOM.DIV(attrs3, "Third Item");
 50  * var hr = MochiKit.DOM.HR();
 51  * var examplePopup = RapidContext.Widget.Popup({}, item1, item2, hr, item3);
 52  *
 53  * @example {User Interface XML}
 54  * <Popup id="examplePopup">
 55  *   <div value="1" class="widgetPopupItem">» First Item</div>
 56  *   <div value="2" class="widgetPopupItem">» Second Item</div>
 57  *   <hr />
 58  *   <div value="3" class="widgetPopupItem widgetPopupDisabled">» Third Item</div>
 59  * </Popup>
 60  */
 61 RapidContext.Widget.Popup = function (attrs/*, ...*/) {
 62     var o = MochiKit.DOM.DIV();
 63     RapidContext.Widget._widgetMixin(o, arguments.callee);
 64     o.addClass("widgetPopup");
 65     o._setHidden(true);
 66     o.selectedIndex = -1;
 67     o._delayTimer = null;
 68     o.setAttrs(MochiKit.Base.update({ delay: 5000 }, attrs));
 69     o.addAll(MochiKit.Base.extend(null, arguments, 1));
 70     MochiKit.Signal.connect(o, "onmousemove", o, "_handleMouseMove");
 71     MochiKit.Signal.connect(o, "onclick", o, "_handleMouseClick");
 72     return o;
 73 };

 94 /**
 95  * Updates the widget or HTML DOM node attributes.
 96  *
 97  * @param {Object} attrs the widget and node attributes to set
 98  * @param {Number} [attrs.delay] the widget auto-hide delay in
 99  *            milliseconds, defaults to 5000
100  * @param {Object} [attrs.showAnim] the optional animation options
101               when showing the popup, defaults to none
102  * @param {Object} [attrs.hideAnim] the optional animation options
103  *            when hiding the popup, defaults to none
104  * @param {Boolean} [attrs.hidden] the hidden widget flag
105  */
106 RapidContext.Widget.Popup.prototype.setAttrs = function (attrs) {
107     attrs = MochiKit.Base.update({}, attrs);
108     var locals = RapidContext.Util.mask(attrs, ["delay", "showAnim", "hideAnim", "hidden"]);
109     if (typeof(locals.delay) != "undefined") {
110         this.delay = parseInt(locals.delay);
111         this.resetDelay();
112     }
113     if (typeof(locals.showAnim) != "undefined") {
114         this.showAnim = locals.showAnim;
115     }
116     if (typeof(locals.hideAnim) != "undefined") {
117         this.hideAnim = locals.hideAnim;
118     }
119     if (typeof(locals.hidden) != "undefined") {
120         this._setHiddenPopup(locals.hidden);
121     }
122     this.__setAttrs(attrs);
123 };
124
125 /**
126  * Performs the changes corresponding to setting the `hidden`
127  * widget attribute for the `Popup` widget.
128  *
129  * @param {Boolean} value the new attribute value
130  */
131 RapidContext.Widget.Popup.prototype._setHiddenPopup = function (value) {
132     if (value && !this.isHidden()) {
133         this._setHidden(true);
134         if (this.hideAnim) {
135             this.animate(this.hideAnim);
136         }
137         RapidContext.Widget.emitSignal(this, "onhide");
138     } else if (!value && this.isHidden()) {
139         this.selectChild(-1);
140         this._setHidden(false);
141         if (this.showAnim) {
142             this.animate(this.showAnim);
143         }
144         this.scrollTop = 0;
145         RapidContext.Widget.emitSignal(this, "onshow");
146     }
147     this.resetDelay();
148 };
149
150 /**
151  * Resets the popup auto-hide timer. Might be called manually when
152  * receiving events on other widgets related to this one.
153  */
154 RapidContext.Widget.Popup.prototype.resetDelay = function () {
155     if (this._delayTimer) {
156         clearTimeout(this._delayTimer);
157         this._delayTimer = null;
158     }
159     if (!this.isHidden() && this.delay > 0) {
160         this._delayTimer = setTimeout(MochiKit.Base.bind("hide", this), this.delay);
161     }
162 };

164 /**
165  * Returns the currently selected child node.
166  *
167  * @return {Node} the currently selected child node, or
168  *         null if no node is selected
169  */
170 RapidContext.Widget.Popup.prototype.selectedChild = function () {
171     return RapidContext.Util.childNode(this, this.selectedIndex);
172 };
173
174 /**
175  * Marks a popup child as selected. The currently selected child will
176  * automatically be unselected by this method.
177  *
178  * @param {Number/Node} indexOrNode the child node index or DOM node,
179  *            use a negative value to unselect
180  *
181  * @return the index of the newly selected child, or
182  *         -1 if none was selected
183  */
184 RapidContext.Widget.Popup.prototype.selectChild = function (indexOrNode) {
185     var node = this.selectedChild();
186     if (node != null) {
187         MochiKit.DOM.removeElementClass(node, "widgetPopupSelected");
188     }
189     var node = RapidContext.Util.childNode(this, indexOrNode);
190     if (typeof(indexOrNode) == "number") {
191         var index = indexOrNode;
192     } else {
193         var index = MochiKit.Base.findIdentical(this.childNodes, node);
194     }
195     if (index >= 0 && node != null) {
196         this.selectedIndex = index;
197         MochiKit.DOM.addElementClass(node, "widgetPopupSelected");
198         var top = node.offsetTop;
199         var bottom = top + node.offsetHeight + 5;
200         if (this.scrollTop + this.clientHeight < bottom) {
201             this.scrollTop = bottom - this.clientHeight;
202         }
203         if (this.scrollTop > top) {
204             this.scrollTop = top;
205         }
206     } else {
207         this.selectedIndex = -1;
208     }
209     return this.selectedIndex;
210 };
211
212 /**
213  * Moves the current selection by a numeric offset.
214  *
215  * @param {Number} offset the selection offset (a positive or
216  *            negative number)
217  *
218  * @return the index of the newly selected child, or
219  *         -1 if none was selected
220  */
221 RapidContext.Widget.Popup.prototype.selectMove = function (offset) {
222     var index = this.selectedIndex + offset;
223     if (index >= this.childNodes.length) {
224         index = 0;
225     }
226     if (index < 0) {
227         index = this.childNodes.length - 1;
228     }
229     return this.selectChild(index);
230 };

232 /**
233  * Handles mouse move events over the popup.
234  *
235  * @param {Event} evt the MochiKit.Signal.Event object
236  */
237 RapidContext.Widget.Popup.prototype._handleMouseMove = function (evt) {
238     this.show();
239     var node = RapidContext.Util.childNode(this, evt.target());
240     if (node != null && MochiKit.DOM.hasElementClass(node, "widgetPopupItem")) {
241         this.selectChild(node);
242     } else {
243         this.selectChild(-1);
244     }
245 };

247 /**
248  * Handles mouse click events on the popup.
249  *
250  * @param {Event} evt the `MochiKit.Signal.Event` object
251  */
252 RapidContext.Widget.Popup.prototype._handleMouseClick = function (evt) {
253     var node = RapidContext.Util.childNode(this, evt.target());
254     if (node != null && MochiKit.DOM.hasElementClass(node, "widgetPopupItem")) {
255         this.selectChild(node);
256     } else {
257         this.selectChild(-1);
258     }
