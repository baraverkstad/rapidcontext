148     /**
149      * Checks if the browser supports all required APIs.
150      *
151      * @return {Boolean} true if the browser is supported, or false otherwise
152      *
153      * @memberof RapidContext.Browser
154      */
155     function isSupported() {
156         var hasRequired = supports(REQUIRED);
157         var hasOptional = supports(OPTIONAL);
158         if (!hasRequired) {
159             console.error("browser: required features are missing", info());
160         } else if (!hasOptional) {
161             console.warn("browser: some recommended features are missing", info());
162         }
163         return hasRequired;
164     }

166     /**
167      * Checks for browser support for one or more specified APIs. Supports
168      * checking JavaScript APIs, JavaScript syntax and CSS support.
169      *
170      * @param {String/Object/Array} feature one or more features to check
171      * @param {String/Object} [...] the additional features to check
172      * @return {Boolean} true if supported, or false otherwise
173      *
174      * @example
175      * RapidContext.Browser.supports("Array.isArray") ==> true;
176      * RapidContext.Browser.supports({ test: "let a = 2; a === 2", name: "Let statements" }) ==> true;
177      * RapidContext.Browser.supports("display: flex") ==> false;
178      *
179      * @memberof RapidContext.Browser
180      */
181     function supports(feature/*, ...*/) {
182         function checkCSS(code) {
183             try {
184                 return CSS.supports(code);
185             } catch (ignore) {
186                 return false;
187             }
188         }
189         function checkPath(base, path) {
190             while (base && path.length > 0) {
191                 base = base[path.shift()];
192             }
193             return typeof(base) === "function";
194         }
195         function checkEval(code) {
196             try {
197                 var val = eval(code);
198                 return val !== undefined && val !== null;
199             } catch (ignore) {
200                 return false;
201             }
202         }
203         function check(test) {
204             var isCSS = /^[a-z-]+:/i.test(test);
205             var isPath = /^[a-z]+(\.[a-z]+)*$/i.test(test);
206             var isValid = (
207                 (isCSS && checkCSS(test)) ||
208                 (isPath && checkPath(window, test.split("."))) ||
209                 (!isCSS && !isPath && checkEval(test))
210             );
211             return isValid;
212         }
213         var res = true;
214         var features = Array.isArray(feature) ? feature : Array.prototype.slice.call(arguments);
215         for (var i = 0; i < features.length; i++) {
216             var def = features[i].test ? features[i] : { test: features[i] };
217             if (!check(def.test)) {
218                 var explain = [def.name, def.test].filter(Boolean).join(": ");
219                 console.warn("browser: missing support for " + explain);
220                 res = false;
221             }
222         }
223         return res;
224     }

226     /**
227      * Returns browser information version and platform information.
228      *
229      * @param {String} [userAgent] the agent string, or undefined for this browser
230      * @return {Object} a browser meta-data object
231      *
232      * @memberof RapidContext.Browser
233      */
234     function info(userAgent) {
235         function firstMatch(patterns, text) {
236             for (var k in patterns) {
237                 var m = patterns[k].exec(text);
238                 if (m) {
239                     var extra = m[1] ? " " + m[1] : "";
240                     if (/^[ \d_.]+$/.test(extra)) {
241                         extra = extra.replace(/_/g, ".");
242                     }
243                     return k + extra;
244                 }
245             }
246             return null;
247         }
248         function clone(obj, keys) {
249             var res = {};
250             for (var i = 0; i < keys.length; i++) {
251                 var k = keys[i];
252                 res[k] = obj[k];
253             }
254             return res;
255         }
256         var BROWSERS = {
257             "Edge": /Edg(?:e|A|iOS|)\/([^\s;]+)/,
258             "Chrome": /Chrome\/([^\s;]+)/,
259             "Chrome iOS": /CriOS\/([^\s;]+)/,
260             "Firefox": /Firefox\/([^\s;]+)/,
261             "Firefox iOS": /FxiOS\/([^\s;]+)/,
262             "Safari": /Version\/(\S+).* Safari\/[^\s;]+/,
263             "MSIE": /(?:MSIE |Trident\/.*rv:)([^\s;)]+)/
264         };
265         var PLATFORMS = {
266             "Android": /Android ([^;)]+)/,
267             "Chrome OS": /CrOS [^\s]+ ([^;)]+)/,
268             "iOS": /(?:iPhone|CPU) OS ([\d_]+)/,
269             "macOS": /Mac OS X ([^;)]+)/,
270             "Linux": /Linux ([^;)]+)/,
271             "Windows": /Windows ([^;)]+)/
272         };
273         var DEVICES = {
274             "iPad": /iPad/,
275             "iPhone": /iPhone/,
276             "Tablet": /Tablet/,
277             "Mobile": /Mobile|Android/
278         };
279         var ua = userAgent || window.navigator.userAgent;
280         var browser = firstMatch(BROWSERS, ua);
281         var platform = firstMatch(PLATFORMS, ua);
282         var device = firstMatch(DEVICES, ua);
283         var res = {};
284         if (browser && platform) {
285             res["browser"] = browser;
286             res["platform"] = platform;
287             res["device"] = device || "Desktop/Other";
288         } else {
289             res["userAgent"] = String(ua);
290         }
291         if (!userAgent) {
292             res["language"] = window.navigator.language;
293             res["screen"] = clone(window.screen, ["width", "height", "colorDepth"]);
294             res["window"] = clone(window, ["innerWidth", "innerHeight", "devicePixelRatio"]);
295             res["cookies"] = _cookies();
296         }
297         return res;
298     }
299
300     function _cookies() {
301         var cookies = {};
302         var pairs = window.document.cookie.split(/\s*;\s*/g);
303         pairs.forEach(function (pair) {
304             var name = pair.split("=")[0];
305             var value = pair.substr(name.length + 1);
306             cookies[decodeURIComponent(name)] = decodeURIComponent(value);
307         });
308         return cookies;
309     }

311     /**
312      * Gets, sets or removes browser cookies.
313      *
314      * @param {String} [name] the cookie name to get/set
315      * @param {String} [value] the cookie value to set, or null to remove
316      * @return {Object/String} all cookie values or a single value
317      *
318       * @memberof RapidContext.Browser
319      */
320     function cookie(name, value) {
321         if (name === undefined || name === null) {
322             return _cookies();
323         } else if (value === undefined) {
324             return _cookies()[name];
325         } else if (value === null) {
326             var prefix = encodeURIComponent(name) + "=";
327             var suffix = ";path=/;expires=" + new Date(0).toUTCString();
328             var domain = window.location.hostname.split(".");
329             while (domain.length > 1) {
330                 var domainPart = ";domain=" + domain.join(".");
331                 window.document.cookie = prefix + domainPart + suffix;
332                 domain.shift();
333             }
334             window.document.cookie = prefix + suffix;
335             return null;
336         } else {
337             window.document.cookie = encodeURIComponent(name) + "=" + encodeURIComponent(value) + ";path=/";
338             return value;
339         }
340     }
