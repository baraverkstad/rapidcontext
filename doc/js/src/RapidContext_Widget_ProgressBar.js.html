 21 /**
 22  * Creates a new progress bar widget.
 23  *
 24  * @constructor
 25  * @param {Object} attrs the widget and node attributes
 26  * @param {Number} [attrs.min] the minimum range value, defaults to 0
 27  * @param {Number} [attrs.max] the maximum range value, defaults to 100
 28  * @param {Number} [attrs.value] the progress value, a number between `min`
 29  *            and `max`, defaults to 0
 30  * @param {Number} [attrs.ratio] the progress ratio, a floating-point number
 31  *            between 0.0 and 1.0, defaults to 0.0
 32  * @param {Number} [attrs.text] the additional information text, defaults to
 33  *            blank
 34  * @param {Boolean} [attrs.noratio] the hide ratio (in percent) display flag,
 35  *            defaults to `false`
 36  * @param {Boolean} [attrs.novalue] the hide value display flag, defaults to
 37  *            `false`
 38  * @param {Boolean} [attrs.notime] the hide remaining time display flag,
 39  *            defaults to `false`
 40  * @param {Boolean} [attrs.hidden] the hidden widget flag, defaults to false
 41  *
 42  * @return {Widget} the widget DOM node
 43  *
 44  * @class The progress bar widget class. Used to provide a dynamic progress
 45  *     meter, using a `<div>` HTML elements. The progress bar also provides a
 46  *     completion time estimation that is displayed in the bar. Whenever the
 47  *     range is modified, the time estimation is reset.
 48  * @extends RapidContext.Widget
 49  *
 50  * @example {JavaScript}
 51  * var attrs = { text: "Working", noratio: true, notime: true };
 52  * var w = RapidContext.Widget.ProgressBar(attrs);
 53  *
 54  * @example {User Interface XML}
 55  * <ProgressBar text="Working" noratio="true" notime="true" />
 56  */
 57 RapidContext.Widget.ProgressBar = function (attrs) {
 58     var meter = MochiKit.DOM.DIV({ "class": "widgetProgressBarMeter" });
 59     var text = MochiKit.DOM.DIV({ "class": "widgetProgressBarText" });
 60     var o = MochiKit.DOM.DIV({}, meter, text);
 61     RapidContext.Widget._widgetMixin(o, RapidContext.Widget.ProgressBar);
 62     o.addClass("widgetProgressBar");
 63     o.setAttrs(Object.assign({ min: 0, max: 100, value: 0 }, attrs));
 64     return o;
 65 };

 68 RapidContext.Widget.Classes.ProgressBar = RapidContext.Widget.ProgressBar;

 79 /**
 80  * Updates the widget or HTML DOM node attributes. Note that updating the
 81  * value will automatically also update the ratio. All calls to this method
 82  * may trigger a new remaining time estimation.
 83  *
 84  * @param {Object} attrs the widget and node attributes to set
 85  * @param {Number} [attrs.min] the minimum range value, if modified the current
 86  *            value and ratio are reset
 87  * @param {Number} [attrs.max] the maximum range value, if modified the current
 88  *            value and ratio are reset
 89  * @param {Number} [attrs.value] the progress value, a number between `min`
 90  *            and `max`
 91  * @param {Number} [attrs.ratio] the progress ratio, a floating-point number
 92  *            between 0.0 and 1.0
 93  * @param {Number} [attrs.text] the additional information text
 94  * @param {Boolean} [attrs.noratio] the hide ratio (in percent) display flag
 95  * @param {Boolean} [attrs.novalue] the hide value display flag
 96  * @param {Boolean} [attrs.notime] the hide remaining time display flag
 97  * @param {Boolean} [attrs.hidden] the hidden widget flag
 98  */
 99 RapidContext.Widget.ProgressBar.prototype.setAttrs = function (attrs) {
100     attrs = Object.assign({}, attrs);
101     var nowTime = new Date().getTime();
102     if (typeof(attrs.min) != "undefined" || typeof(attrs.max) != "undefined") {
103         this.min = attrs.min = Math.max(parseInt(attrs.min) || this.min || 0, 0);
104         this.max = attrs.max = Math.max(parseInt(attrs.max) || this.max || 100, this.min);
105         attrs.value = attrs.value || null;
106         attrs.ratio = attrs.ratio || 0.0;
107         this.startTime = nowTime;
108         this.updateTime = nowTime;
109         this.timeRemaining = null;
110     }
111     if (typeof(attrs.value) != "undefined") {
112         var value = Math.min(Math.max(parseFloat(attrs.value), this.min), this.max);
113         if (!isNaN(value)) {
114             attrs.value = value;
115             var calc = (value - this.min) / (this.max - this.min);
116             attrs.ratio = attrs.ratio || calc;
117         } else {
118             delete attrs.value;
119         }
120     }
121     if (typeof(attrs.ratio) != "undefined") {
122         var ratio = Math.min(Math.max(parseFloat(attrs.ratio), 0.0), 1.0);
123         if (!isNaN(ratio)) {
124             attrs.ratio = ratio;
125         } else {
126             delete attrs.ratio;
127         }
128     }
129     if (typeof(attrs.noratio) != "undefined") {
130         attrs.noratio = MochiKit.Base.bool(attrs.noratio) || null;
131     }
132     if (typeof(attrs.novalue) != "undefined") {
133         attrs.novalue = MochiKit.Base.bool(attrs.novalue) || null;
134     }
135     if (typeof(attrs.notime) != "undefined") {
136         attrs.notime = MochiKit.Base.bool(attrs.notime) || null;
137         this.timeRemaining = null;
138     }
139     this.__setAttrs(attrs);
140     if (!this.notime && nowTime - this.updateTime > 1000) {
141         this.updateTime = nowTime;
142         var time = this._remainingTime();
143         this.timeRemaining = (time && time.text) ? time.text : null;
144     }
145     this._render();
146 };
147
148 /**
149  * Returns the remaining time.
150  *
151  * @return {Object} the remaining time object, or
152  *         `null` if not possible to estimate
153  */
154 RapidContext.Widget.ProgressBar.prototype._remainingTime = function () {
155     var duration = new Date().getTime() - this.startTime;
156     duration = Math.max(Math.round(duration / this.ratio - duration), 0);
157     if (isFinite(duration) && !isNaN(duration)) {
158         var res = {
159             total: duration,
160             days: Math.floor(duration / 86400000),
161             hours: Math.floor(duration / 3600000) % 24,
162             minutes: Math.floor(duration / 60000) % 60,
163             seconds: Math.floor(duration / 1000) % 60,
164             millis: duration % 1000
165         };
166         var pad = MochiKit.Text.padLeft;
167         if (res.days >= 10) {
168             res.text = res.days + " days";
169         } else if (res.days >= 1) {
170             res.text = res.days + " days " + res.hours + " hours";
171         } else if (res.hours >= 1) {
172             res.text = res.hours + ":" + pad("" + res.minutes, 2, "0") + " hours";
173         } else if (res.minutes >= 1) {
174             res.text = res.minutes + ":" + pad("" + res.seconds, 2, "0") + " min";
175         } else {
176             res.text = res.seconds + " sec";
177         }
178         return res;
179     }
180     return null;
181 };
182
183 /**
184  * Redraws the progress bar meter and text.
185  */
186 RapidContext.Widget.ProgressBar.prototype._render = function () {
187     var percent = Math.round(this.ratio * 1000) / 10;
188     MochiKit.Style.setElementDimensions(this.firstChild, { w: percent }, "%");
189     this.firstChild.classList.toggle("animated", percent < 100);
190     var info = [];
191     if (!this.noratio) {
192         info.push(Math.round(percent) + "%");
193     }
194     if (!this.novalue && typeof(this.value) == "number") {
195         var pos = this.value - this.min;
196         var total = this.max - this.min;
197         info.push(pos + " of " + total);
198     }
199     if (this.text) {
200         info.push(this.text);
201     }
202     if (this.timeRemaining && percent < 100) {
203         info.push(this.timeRemaining + " remaining");
204     }
205     MochiKit.DOM.replaceChildNodes(this.lastChild, info.join(" \u2022 "));
206 };
