 29 /**
 30  * Displays an error message for the user. This operation may or may
 31  * not block the user interface, while the message is being
 32  * displayed (depending on implementation). This function can be
 33  * used as an errback function in deferred calls. All arguments to
 34  * this function will be concatenated and displayed.
 35  *
 36  * @param {String/Error} [args] the messages or errors to display
 37  */
 38 RapidContext.UI.showError = function () {
 39     if (arguments.length == 1 &&
 40         arguments[0] instanceof MochiKit.Async.CancelledError) {
 41         // Ignore async cancellation errors
 42         return;
 43     }
 44     var msg = "";
 45     for (var i = 0; i < arguments.length; i++) {
 46         if (arguments[i] == null) {
 47             msg += "null";
 48         } else if (arguments[i] instanceof Error) {
 49             msg += arguments[i].message;
 50         } else {
 51             msg += arguments[i].toString();
 52         }
 53     }
 54     alert("Error: " + msg);
 55     return arguments[arguments.length - 1];
 56 };

 58 /**
 59  * Creates a tree of widgets from a parsed XML document. This
 60  * function will call `createWidget()` for any XML element node found,
 61  * performing some basic adjustments on the element attributes
 62  * before sending them as attributes to the widget constructor. Text
 63  * nodes with non-whitespace content will be mapped to HTML DOM text
 64  * nodes.
 65  *
 66  * @param {Object} node the XML document or node
 67  * @param {Object} [ids] the optional node id mappings
 68  *
 69  * @return {Array/Object} an array or an object with the root
 70  *         widget(s) created
 71  */
 72 RapidContext.UI.buildUI = function (node, ids) {
 73     if (node.documentElement) {
 74         return RapidContext.UI.buildUI(node.documentElement.childNodes, ids);
 75     } else if (typeof(node.item) != "undefined" && typeof(node.length) == "number") {
 76         var res = [];
 77         var elems = MochiKit.Base.extend(null, node);
 78         for (var i = 0; i < elems.length; i++) {
 79             var elem = RapidContext.UI.buildUI(elems[i], ids);
 80             if (elem) {
 81                 res.push(elem);
 82             }
 83         }
 84         return res;
 85     } else if (node.nodeType === 1) { // Node.ELEMENT_NODE
 86         try {
 87             return RapidContext.UI._buildUIElem(node, ids);
 88         } catch (e) {
 89             RapidContext.Log.error("Failed to build UI element", node, e);
 90         }
 91     } else if (node.nodeType === 3) { // Node.TEXT_NODE
 92         var str = node.nodeValue;
 93         if (str != null && MochiKit.Format.strip(str) != "") {
 94             return RapidContext.Util.createTextNode(str.replace(/\s+/g, " "));
 95         }
 96     } else if (node.nodeType === 4) { // Node.CDATA_SECTION_NODE
 97         var str = node.nodeValue;
 98         if (str != null && str != "") {
 99             return RapidContext.Util.createTextNode(str);
100         }
101     }
102     return null;
103 };

105 /**
106  * Creates a widget from a parsed XML element. This function will
107  * call `createWidget()`, performing some basic adjustments on the
108  * element attributes before sending them as attributes to the widget
109  * constructor.
110  *
111  * @param {Object} node the XML document or node
112  * @param {Object} [ids] the optional node id mappings
113  *
114  * @return {Object} an object with the widget created
115  */
116 RapidContext.UI._buildUIElem = function (node, ids) {
117     var name = node.nodeName;
118     if (name == "style") {
119         RapidContext.UI._buildUIStylesheet(MochiKit.DOM.scrapeText(node));
120         node.parentNode.removeChild(node);
121         return null;
122     }
123     var attrs = RapidContext.Util.dict(RapidContext.Util.attributeArray(node));
124     var locals = RapidContext.Util.mask(attrs, ["id", "w", "h", "a"]);
125     var children = RapidContext.UI.buildUI(node.childNodes, ids);
126     if (RapidContext.Widget.Classes[name]) {
127         if (name == "Table" && attrs.multiple) {
128             // TODO: remove deprecated code, eventually...
129             RapidContext.Log.warn("Table 'multiple' attribute is deprecated, use 'select'", node);
130             attrs.select = MochiKit.Base.bool(attrs.multiple) ? "multiple" : "one";
131             delete attrs.multiple;
132         }
133         var widget = RapidContext.Widget.createWidget(name, attrs, children);
134     } else {
135         var widget = MochiKit.DOM.createDOM(name, attrs, children);
136     }
137     if (locals.id) {
138         if (ids) {
139             ids[locals.id] = widget;
140         } else {
141             widget.id = locals.id;
142         }
143     }
144     if (locals.w || locals.h || locals.a) {
145         RapidContext.Util.registerSizeConstraints(widget, locals.w, locals.h, locals.a);
146     }
147     return widget;
148 };

150 /**
151  * Creates and injects a stylesheet element from a set of CSS rules.
152  *
153  * @param {String} css the CSS rules to inject
154  */
155 RapidContext.UI._buildUIStylesheet = function (css) {
156     var style = document.createElement('style');
157     style.setAttribute('type', 'text/css');
158     document.getElementsByTagName('head')[0].appendChild(style);
159     try {
160         style.innerHTML = css;
161     } catch (e) {
162         var parts = css.split(/\s*[{}]\s*/);
163         for (var i = 0; i < parts.length; i += 2) {
164             var rules = parts[i].split(/\s*,\s*/);
165             var styles = parts[i+1];
166             for (var j = 0; j < rules.length; j++) {
167                 var rule = MochiKit.Format.strip(rules[j].replace(/\s+/, " "));
168                 style.styleSheet.addRule(rule, styles);
169             }
170         }
171     }
172 }

174 /**
175  * Connects the default UI signals for a procedure. This includes a default
176  * error handler, a loading icon with cancellation handler and a reload icon
177  * with the appropriate click handler.
178  *
179  * @param {Procedure} proc the `RapidContext.Procedure` instance
180  * @param {Icon} [loadingIcon] the loading icon, or `null`
181  * @param {Icon} [reloadIcon] the reload icon, or `null`
182  *
183  * @see RapidContext.Procedure
184  */
185 RapidContext.UI.connectProc = function (proc, loadingIcon, reloadIcon) {
186     // TODO: error signal not automatically cleaned up on stop()...
187     MochiKit.Signal.connect(proc, "onerror", RapidContext.UI, "showError");
188     if (loadingIcon) {
189         MochiKit.Signal.connect(proc, "oncall", loadingIcon, "show");
190         MochiKit.Signal.connect(proc, "onresponse", loadingIcon, "hide");
191         MochiKit.Signal.connect(loadingIcon, "onclick", proc, "cancel");
192     }
193     if (reloadIcon) {
194         MochiKit.Signal.connect(proc, "oncall", reloadIcon, "hide");
195         MochiKit.Signal.connect(proc, "onresponse", reloadIcon, "show");
196         MochiKit.Signal.connect(reloadIcon, "onclick", proc, "recall");
197     }
198 };
