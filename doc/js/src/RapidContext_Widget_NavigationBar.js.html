 15 // Namespace initialization
 16 if (typeof(RapidContext) == "undefined") {
 17     RapidContext = {};
 18 }
 19 RapidContext.Widget = RapidContext.Widget || { Classes: {} };
 20
 21 /**
 22  * Creates a new navigation bar widget.
 23  *
 24  * @constructor
 25  * @param {Object} attrs the widget and node attributes
 26  * @param {Boolean} [attrs.hidden] the hidden widget flag, defaults to false
 27  *
 28  * @return {Widget} the widget DOM node
 29  *
 30  * @class The navigation bar widget class. Used to show step-based progress or
 31  *     location in a tree structure. In both cases, it allows the user to go
 32  *     back to a previous step. The widget uses a `<table>` HTML element.
 33  * @extends RapidContext.Widget
 34  * @property {Number} position The current position in the path (from 0)
 35  * @property {Number} maxPosition The maximum position reached when
 36  *     navigating.
 37  * @property {Array} path The path being navigated.
 38  *
 39  * @example {JavaScript}
 40  * var navBar = RapidContext.NavigationBar();
 41  * navBar.moveTo(0, ["Step 1", "Step 2", "Step 3"]);
 42  *
 43  * @example {User Interface XML}
 44  * <NavigationBar id="navBar" />
 45  * <!-- No support for menu items in XML yet -->
 46  */
 47 RapidContext.Widget.NavigationBar = function (attrs) {
 48     var tr = MochiKit.DOM.TR();
 49     var tbody = MochiKit.DOM.TBODY({}, tr);
 50     var o = MochiKit.DOM.TABLE({}, tbody);
 51     RapidContext.Widget._widgetMixin(o, RapidContext.Widget.NavigationBar);
 52     o.addClass("widgetNavigationBar");
 53     o.setAttrs(attrs);
 54     o.moveTo(0, ["Start"]);
 55     o.onclick = RapidContext.Widget._eventHandler(null, "_handleClick");
 56     return o;
 57 };
 58
 59 // Register widget class
 60 RapidContext.Widget.Classes.NavigationBar = RapidContext.Widget.NavigationBar;
 61
 62 /**
 63  * Returns the widget container DOM node.
 64  *
 65  * @return {Node} returns null, since child nodes are not supported
 66  */
 67 RapidContext.Widget.NavigationBar.prototype._containerNode = function () {
 68     return null;
 69 };
 70
 71 /**
 72  * Emitted when the navigation position changes. This is triggered
 73  * either by the user moving forward or backward, or by a call to the
 74  * `moveTo()` method. This event signal carries a reference to the
 75  * widget itself.
 76  *
 77  * @name RapidContext.Widget.NavigationBar#onchange
 78  * @event
 79  */
 80
 81 /**
 82  * Resets the position and maximum position counters and moves to
 83  * the first step in the path.
 84  */
 85 RapidContext.Widget.NavigationBar.prototype.reset = function () {
 86     this.position = 0;
 87     this.maxPosition = 0;
 88     this.moveTo(0);
 89 };
 90
 91 /**
 92  * Moves the navigation bar to the specified position in the path.
 93  * Optionally a new path may also be provided.
 94  *
 95  * @param {Number} pos the new position to move to (from 0)
 96  * @param {Array} [path] the new path to navigate
 97  */
 98 RapidContext.Widget.NavigationBar.prototype.moveTo = function (pos, path) {
 99     this.position = pos;
100     this.maxPosition = Math.max(this.maxPosition, pos);
101     if (path != null) {
102         this.path = path.slice(0);
103     }
104     var tr = this.firstChild.firstChild;
105     MochiKit.DOM.replaceChildNodes(tr);
106     for (var i = 0; i < this.path.length; i++) {
107         var className = this._posClass(i);
108         tr.appendChild(MochiKit.DOM.TD({ "class": className }, this.path[i]));
109         className = className + "-" + this._posClass(i + 1);
110         tr.appendChild(MochiKit.DOM.TD({ "class": className }, "\xa0"));
111     }
112     RapidContext.Widget.emitSignal(this, "onchange", this);
113 };
114
115 /**
116  * Returns the class to use for a specified path position.
117  *
118  * @param {Number} pos the position to render
119  *
120  * @return {String} the CSS class corresponding to the position
121  */
122 RapidContext.Widget.NavigationBar.prototype._posClass = function (pos) {
123     if (pos >= this.path.length) {
124         return "end";
125     } else if (pos == this.position) {
126         return "active";
127     } else if (pos < this.maxPosition) {
128         return "prev";
129     } else {
130         return "prev";
131     }
132 };
133
134 /**
135  * Handles click events on the navigation bar.
136  *
137  * @param {Event} evt the MochiKit.Signal.Event object
138  */
139 RapidContext.Widget.NavigationBar.prototype._handleClick = function (evt) {
140     var node = evt.target();
141     if (node.className === "prev") {
142         var step = 0;
143         while ((node = node.previousSibling) != null) {
144             step++;
145         }
146         this.moveTo(step / 2);
147     }
148 };
