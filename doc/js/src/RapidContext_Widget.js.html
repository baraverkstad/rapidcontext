 44 /**
 45  * Checks if the specified object is a widget. Any non-null object
 46  * that looks like a DOM node and has the element class "widget"
 47  * will cause this function to return `true`. Otherwise, `false` will
 48  * be returned. As an option, this function can also check if the
 49  * widget has a certain class by checking for an additional CSS
 50  * class "widget<className>" (which is a standard followed by all
 51  * widgets).
 52  *
 53  * @param {Object} obj the object to check
 54  * @param {String} [className] the optional widget class name
 55  *
 56  * @return {Boolean} `true` if the object looks like a widget, or
 57  *         `false` otherwise
 58  *
 59  * @static
 60  */
 61 RapidContext.Widget.isWidget = function (obj, className) {
 62     if (className != null) {
 63         return RapidContext.Util.isHTML(obj) &&
 64                MochiKit.DOM.hasElementClass(obj, "widget") &&
 65                MochiKit.DOM.hasElementClass(obj, "widget" + className);
 66     } else {
 67         return RapidContext.Util.isHTML(obj) &&
 68                MochiKit.DOM.hasElementClass(obj, "widget");
 69     }
 70 };

 72 /**
 73  * Checks if the specified object is a form field. Any non-null object that
 74  * looks like a DOM node and is either an standard HTML form field (`<input>`,
 75  * `<textarea>` or `<select>`) or one with a "value" property will cause this
 76  * function to return `true`. Otherwise, `false` will be returned.
 77  *
 78  * @param {Object} obj the object to check
 79  *
 80  * @return {Boolean} `true` if the object looks like a form field, or
 81  *         `false` otherwise
 82  *
 83  * @static
 84  */
 85 RapidContext.Widget.isFormField = function (obj) {
 86     if (!RapidContext.Util.isHTML(obj) || typeof(obj.tagName) !== "string") {
 87         return false;
 88     }
 89     var tagName = obj.tagName.toUpperCase();
 90     return tagName == "INPUT" ||
 91            tagName == "TEXTAREA" ||
 92            tagName == "SELECT" ||
 93            RapidContext.Widget.isWidget(obj, "Field");
 94 };

 96 /**
 97  * Adds all functions from a widget class to a DOM node. This will also convert
 98  * the DOM node into a widget by adding the "widget" CSS class and all the
 99  * default widget functions from `Widget.prototype`.
100  *
101  * The default widget functions are added non-destructively, using the prefix
102  * "__" if also defined in the widget class.
103  *
104  * @param {Node} node the DOM node to modify
105  * @param {Object/Function} [...] the widget class or constructor
106  *
107  * @return {Widget} the widget DOM node
108  */
109 RapidContext.Widget._widgetMixin = function (node/*, objOrClass, ...*/) {
110     MochiKit.DOM.addElementClass(node, "widget");
111     var protos = MochiKit.Base.extend([], arguments, 1);
112     protos.push(RapidContext.Widget);
113     while (protos.length > 0) {
114         var obj = protos.pop();
115         if (typeof(obj) === "function") {
116             obj = obj.prototype;
117         }
118         for (var key in obj) {
119             var prevKey = "__" + key;
120             if (key in node) {
121                 node[prevKey] = node[key];
122             }
123             try {
124                 node[key] = obj[key];
125             } catch (e) {
126                 var msg = "failed to overwrite '" + key + "' in DOM node";
127                 RapidContext.Log.warn(msg, e, node);
128             }
129         }
130     }
131     return node;
132 };

134 /**
135  * Creates a new widget with the specified name, attributes and
136  * child widgets or DOM nodes. The widget class name must have been
137  * registered in the `RapidContext.Widget.Classes` lookup table, or an
138  * exception will be thrown. This function is identical to calling
139  * the constructor function directly.
140  *
141  * @param {String} name the widget class name
142  * @param {Object} attrs the widget and node attributes
143  * @param {Object} [...] the child widgets or DOM nodes
144  *
145  * @return {Widget} the widget DOM node
146  *
147  * @throws {ReferenceError} if the widget class name couldn't be
148  *             found in `RapidContext.Widget.Classes`
149  *
150  * @static
151  */
152 RapidContext.Widget.createWidget = function (name, attrs/*, ...*/) {
153     var cls = RapidContext.Widget.Classes[name];
154     if (cls == null) {
155         throw new ReferenceError("failed to find widget '" + name +
156                                  "' in RapidContext.Widget.Classes");
157     }
158     return cls.apply(this, MochiKit.Base.extend([], arguments, 1));
159 };

161 /**
162  * Destroys a widget or a DOM node. This function will remove the DOM
163  * node from the tree, disconnect all signals and call all widget
164  * destructor functions. The same procedure will also be applied
165  * recursively to all child nodes. Once destroyed, all references to
166  * the widget object should be cleared in order for the browser to
167  * be able to reclaim the memory used.
168  *
169  * @param {Widget/Node/Array} node the (widget) DOM node or list
170  *
171  * @static
172  */
173 RapidContext.Widget.destroyWidget = function (node) {
174     if (node.nodeType != null) {
175         if (typeof(node.destroy) == "function") {
176             node.destroy();
177         }
178         if (node.parentNode != null) {
179             MochiKit.DOM.removeElement(node);
180         }
181         MochiKit.Signal.disconnectAll(node);
182         MochiKit.Signal.disconnectAllTo(node);
183         while (node.firstChild != null) {
184             RapidContext.Widget.destroyWidget(node.firstChild);
185         }
186     } else if (MochiKit.Base.isArrayLike(node)) {
187         for (var i = node.length - 1; i >= 0; i--) {
188             RapidContext.Widget.destroyWidget(node[i]);
189         }
190     }
191 };

193 /**
194  * Creates an event handler function that will forward any calls to
195  * another function. The other function must exist as a property in
196  * a parent widget of the specified class.
197  *
198  * @param {String} className the parent widget class name, or null
199  *                     to use the same node
200  * @param {String} methodName the name of the method to call
201  * @param {Object} [...] the additional method arguments
202  *
203  * @return {Function} a function that forwards calls as specified
204  */
205 RapidContext.Widget._eventHandler = function (className, methodName/*, ...*/) {
206     var baseArgs = MochiKit.Base.extend([], arguments, 2);
207     return function (evt) {
208         var node = this;
209         while (!RapidContext.Widget.isWidget(node, className)) {
210             node = node.parentNode;
211         }
212         var e = new MochiKit.Signal.Event(this, evt);
213         return node[methodName].apply(node, baseArgs.concat([e]));
214     };
215 };

217 /**
218  * Emits an asynchronous signal to any listeners connected with
219  * MochiKit.Signal.
220  *
221  * Note that this function is an internal helper function for the
222  * widgets and shouldn't be called by external code.
223  *
224  * @param {Widget} node the widget DOM node
225  * @param {String} sig the signal name ("onclick" or similar)
226  * @param {Object} [...] the optional signal arguments
227  *
228  * @deprecated Use _fireEvent() instead to emit proper Event object.
229  */
230 RapidContext.Widget.emitSignal = function (node, sig/*, ...*/) {
231     var args = $.makeArray(arguments);
232     function later() {
233         try {
234             MochiKit.Signal.signal.apply(MochiKit.Signal, args);
235         } catch (e) {
236             var msg = "exception in signal '" + sig + "' handler";
237             RapidContext.Log.error(msg, node, e);
238         }
239     }
240     setTimeout(later);
241 };

243 /**
244  * Creates a DOM event object.
245  *
246  * @param {String} evtType the event type name ("click" or similar)
247  * @param {Object} [detail] the optional event "detail" value
248  *
249  * @return {Event} a new Event object of the specified type
250  */
251 RapidContext.Widget._createEvent = function (evtType, detail) {
252     var evt;
253     try {
254         evt = new Event(evtType);
255     } catch (msieFailsOfCourse) {
256         if (document.createEvent) {
257             evt = document.createEvent("Event");
258             evt.initEvent(evtType, true, true);
259         } else {
260             evt = document.createEventObject();
261             evt.type = evtType;
262         }
263     }
264     if (detail) {
265         evt.detail = detail;
266     }
267     return evt;
268 };

270 /**
271  * Fires an event (asynchronously) to all listeners. This function uses
272  * native DOM APIs, so all type of event listeners should be reached
273  * by the event.
274  *
275  * @param {Widget/Node} node the (widget) DOM node emitting the event
276  * @param {Event/String} evt the event object or type
277  * @param {Object} [detail] the optional event "detail" value
278  */
279 RapidContext.Widget._fireEvent = function (node, evt, detail) {
280     if (typeof(evt) == "string") {
281         evt = RapidContext.Widget._createEvent(evt, detail);
282     }
283     function later() {
284         try {
285             if (node.dispatchEvent) {
286                 node.dispatchEvent(evt);
287             } else {
288                 var msg = "cannot fire event, no dispatchEvent method";
289                 RapidContext.Log.warn(msg, evt, node);
290             }
291         } catch (e) {
292             RapidContext.Log.warn("exception when firing event", evt, node, e);
293         }
294     }
295     setTimeout(later);
296 };

344 /**
345  * Updates the widget or HTML DOM node attributes. This method is
346  * sometimes overridden by individual widgets to allow modification
347  * of additional widget attributes.
348  *
349  * @param {Object} attrs the widget and node attributes to set
350  * @param {Boolean} [attrs.disabled] the disabled widget flag
351  * @param {Boolean} [attrs.hidden] the hidden widget flag
352  * @param {String} [attrs.class] the CSS class names
353  */
354 RapidContext.Widget.prototype.setAttrs = function (attrs) {
355     /* eslint max-depth: "off" */
356     for (var name in attrs) {
357         var value = attrs[name];
358         if (name == "disabled") {
359             this._setDisabled(value);
360         } else if (name == "hidden") {
361             this._setHidden(value);
362         } else if (name == "class") {
363             var elem = this._styleNode();
364             this.removeClass.apply(this, elem.className.split(/\s+/));
365             this.addClass.apply(this, value.split(/\s+/));
366         } else if (name == "style") {
367             if (typeof(value) == "string") {
368                 var styles = {};
369                 var parts = value.split(";");
370                 for (var i = 0; i < parts.length; i++) {
371                     var a = parts[i].split(":");
372                     var k = MochiKit.Format.strip(a[0]);
373                     if (k != "" && a.length > 1) {
374                         styles[k] = MochiKit.Format.strip(a[1]);
375                     }
376                 }
377                 value = styles;
378             }
379             this.setStyle(value);
380         } else if (value != null) {
381             MochiKit.DOM.setNodeAttribute(this, name, value);
382             if (typeof(value) != "object") {
383                 this[name] = value;
384             }
385         } else {
386             this.removeAttribute(name);
387             delete this[name];
388         }
389     }
390 };

392 /**
393  * Updates the CSS styles of this HTML DOM node. This method is
394  * identical to `MochiKit.Style.setStyle`, but uses "this" as the
395  * first argument.
396  *
397  * @param {Object} styles an object with the styles to set
398  *
399  * @example
400  * widget.setStyle({ "font-size": "bold", "color": "red" });
401  */
402 RapidContext.Widget.prototype.setStyle = function (styles) {
403     styles = MochiKit.Base.update({}, styles);
404     var posDimNames = [
405         "width", "height", "zIndex", "z-index",
406         "position", "top", "bottom", "left", "right"
407     ];
408     var posDimStyles = RapidContext.Util.mask(styles, posDimNames);
409     MochiKit.Style.setStyle(this, posDimStyles);
410     MochiKit.Style.setStyle(this._styleNode(), styles);
411 };

413 /**
414  * Checks if this HTML DOM node has the specified CSS class names.
415  * Note that more than one CSS class name may be checked, in which
416  * case all must be present.
417  *
418  * @param {String} [...] the CSS class names to check
419  *
420  * @return {Boolean} `true` if all CSS classes were present, or
421  *         `false` otherwise
422  */
423 RapidContext.Widget.prototype.hasClass = function (/* ... */) {
424     var elem = this._styleNode();
425     for (var i = 0; i < arguments.length; i++) {
426         if (!MochiKit.DOM.hasElementClass(elem, arguments[i])) {
427             return false;
428         }
429     }
430     return true;
431 };

433 /**
434  * Adds the specified CSS class names to this HTML DOM node.
435  *
436  * @param {String} [...] the CSS class names to add
437  */
438 RapidContext.Widget.prototype.addClass = function (/* ... */) {
439     var elem = this._styleNode();
440     for (var i = 0; i < arguments.length; i++) {
441         MochiKit.DOM.addElementClass(elem, arguments[i]);
442     }
443 };
444
445 /**
446  * Removes the specified CSS class names from this HTML DOM node.
447  * Note that this method will not remove any class starting with
448  * "widget".
449  *
450  * @param {String} [...] the CSS class names to remove
451  */
452 RapidContext.Widget.prototype.removeClass = function (/* ... */) {
453     var elem = this._styleNode();
454     for (var i = 0; i < arguments.length; i++) {
455         var name = "" + arguments[i];
456         if (name.indexOf("widget") != 0) {
457             MochiKit.DOM.removeElementClass(elem, name);
458         }
459     }
460 };

462 /**
463  * Toggles adding and removing the specified CSS class names to and
464  * from this HTML DOM node. If all the CSS classes are already set,
465  * they will be removed. Otherwise they will be added.
466  *
467  * @param {String} [...] the CSS class names to remove
468  *
469  * @return {Boolean} `true` if the CSS classes were added, or
470  *         `false` otherwise
471  */
472 RapidContext.Widget.prototype.toggleClass = function (/* ... */) {
473     if (this.hasClass.apply(this, arguments)) {
474         this.removeClass.apply(this, arguments);
475         return false;
476     } else {
477         this.addClass.apply(this, arguments);
478         return true;
479     }
480 };

482 /**
483  * Checks if this widget is disabled. This method checks both the
484  * "widgetDisabled" CSS class and the `disabled` property. Changes
485  * to the disabled status can be made with `enable()`, `disable()` or

487  *
488  * @return {Boolean} `true` if the widget is disabled, or

493            MochiKit.DOM.hasElementClass(this, "widgetDisabled");

496 /**
497  * Performs the changes corresponding to setting the `disabled`
498  * widget attribute.
499  *
500  * @param {Boolean} value the new attribute value
501  */
502 RapidContext.Widget.prototype._setDisabled = function (value) {
503     value = MochiKit.Base.bool(value);
504     if (value) {
505         MochiKit.DOM.addElementClass(this, "widgetDisabled");
506     } else {
507         MochiKit.DOM.removeElementClass(this, "widgetDisabled");
508     }
509     MochiKit.DOM.setNodeAttribute(this, "disabled", value);
510     this.disabled = value;
511 };

579 /**
580  * Performs a visual effect animation on this widget. This is
581  * implemented using the `MochiKit.Visual` effect package. All options
582  * sent to this function will be passed on to the appropriate
583  * `MochiKit.Visual` function.
584  *
585  * @param {Object} opts the visual effect options
586  * @param {String} opts.effect the MochiKit.Visual effect name
587  * @param {String} opts.queue the MochiKit.Visual queue handling,
588  *            defaults to "replace" and a unique scope for each widget
589  *            (see `MochiKit.Visual` for full options)
590  *
591  * @example
592  * widget.animate({ effect: "fade", duration: 0.5 });
593  * widget.animate({ effect: "Move", transition: "spring", y: 300 });
594  */
595 RapidContext.Widget.prototype.animate = function (opts) {
596     var queue = { scope: this.uid(), position: "replace" };
597     opts = MochiKit.Base.updatetree({ queue: queue }, opts);
598     if (typeof(opts.queue) == "string") {
599         queue.position = opts.queue;
600         opts.queue = queue;
601     }
602     var func = MochiKit.Visual[opts.effect];
603     if (typeof(func) == "function") {
604         func.call(null, this, opts);
605     }
606 };

663 /**
664  * Adds one or more children to this widget. This method will flatten any
665  * arrays among the arguments and ignores any `null` or `undefined` arguments.
666  * Any DOM nodes or widgets will be added to the end, and other objects will be
667  * converted to a text node first. Subclasses should normally override the
668  * `addChildNode()` method instead of this one, since that is the basis for
669  * DOM node insertion.
670  *
671  * @param {Object} [...] the children to add
672  */
673 RapidContext.Widget.prototype.addAll = function (/* ... */) {
674     var args = MochiKit.Base.flattenArray(arguments);
675     for (var i = 0; i < args.length; i++) {
676         if (args[i] == null) {
677             // Ignore null values
678         } else if (RapidContext.Util.isDOM(args[i])) {
679             this.addChildNode(args[i]);
680             // TODO: remove this call for performance
681             RapidContext.Util.resizeElements(args[i]);
682         } else {
683             this.addChildNode(RapidContext.Util.createTextNode(args[i]));
684         }
685     }
686 };

688 /**
689  * Removes all children to this widget. This method will also destroy and child
690  * widgets and disconnect all signal listeners. This method uses the
691  * `getChildNodes()` and `removeChildNode()` methods to find and remove the
692  * individual child nodes.
693  */
694 RapidContext.Widget.prototype.removeAll = function () {
695     var children = this.getChildNodes();
696     for (var i = children.length - 1; i >= 0; i--) {
697         this.removeChildNode(children[i]);
698         RapidContext.Widget.destroyWidget(children[i]);
699     }
700 };
