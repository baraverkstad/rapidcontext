 20 /**
 21  * @namespace The base class for the HTML user interface widgets.
 22  *     The Widget class shouldn't be instantiated directly, instead
 23  *     one of the subclasses should be instantiated.

 29 /**
 30  * The global widget registry. This is a widget lookup table where
 31  * all widgets should have an entry. The entries should be added as
 32  * the JavaScript file are loaded. Each widget is indexed by the
 33  * widget name (class name) and point to the constructor function.
 34  */
 35 RapidContext.Widget.Classes = {};

 37 /**
 38  * Function to return unique identifiers.
 39  *
 40  * @return {Number} the next number in the sequence
 41  */
 42 RapidContext.Widget._nextId = MochiKit.Base.counter();

 44 /**
 45  * Checks if the specified object is a widget. Any non-null object
 46  * that looks like a DOM node and has the element class "widget"
 47  * will cause this function to return `true`. Otherwise, `false` will
 48  * be returned. As an option, this function can also check if the
 49  * widget has a certain class by checking for an additional CSS
 50  * class "widget<className>" (which is a standard followed by all
 51  * widgets).
 52  *
 53  * @param {Object} obj the object to check
 54  * @param {String} [className] the optional widget class name
 55  *
 56  * @return {Boolean} `true` if the object looks like a widget, or
 57  *         `false` otherwise
 58  *
 59  * @static
 60  */
 61 RapidContext.Widget.isWidget = function (obj, className) {
 62     return RapidContext.Util.isHTML(obj) &&
 63            MochiKit.DOM.hasElementClass(obj, "widget") &&
 64            (!className || MochiKit.DOM.hasElementClass(obj, "widget" + className));
 65 };

 67 /**
 68  * Adds all functions from a widget class to a DOM node. This will also convert
 69  * the DOM node into a widget by adding the "widget" CSS class and all the
 70  * default widget functions from `Widget.prototype`.
 71  *
 72  * The default widget functions are added non-destructively, using the prefix
 73  * "__" if also defined in the widget class.
 74  *
 75  * @param {Node} node the DOM node to modify
 76  * @param {Object/Function} [...] the widget class or constructor
 77  *
 78  * @return {Widget} the widget DOM node
 79  */
 80 RapidContext.Widget._widgetMixin = function (node/*, objOrClass, ...*/) {
 81     MochiKit.DOM.addElementClass(node, "widget");
 82     var protos = Array.prototype.slice.call(arguments, 1);
 83     protos.push(RapidContext.Widget);
 84     while (protos.length > 0) {
 85         var obj = protos.pop();
 86         if (typeof(obj) === "function") {
 87             obj = obj.prototype;
 88         }
 89         for (var key in obj) {
 90             var prevKey = "__" + key;
 91             if (key in node) {
 92                 node[prevKey] = node[key];
 93             }
 94             try {
 95                 node[key] = obj[key];
 96             } catch (e) {
 97                 var msg = "failed to overwrite '" + key + "' in DOM node";
 98                 RapidContext.Log.warn(msg, e, node);
 99             }
100         }
101     }
102     return node;
103 };

105 /**
106  * Creates a new widget with the specified name, attributes and
107  * child widgets or DOM nodes. The widget class name must have been
108  * registered in the `RapidContext.Widget.Classes` lookup table, or an
109  * exception will be thrown. This function is identical to calling
110  * the constructor function directly.
111  *
112  * @param {String} name the widget class name
113  * @param {Object} attrs the widget and node attributes
114  * @param {Object} [...] the child widgets or DOM nodes
115  *
116  * @return {Widget} the widget DOM node
117  *
118  * @throws {ReferenceError} if the widget class name couldn't be
119  *             found in `RapidContext.Widget.Classes`
120  *
121  * @static
122  */
123 RapidContext.Widget.createWidget = function (name, attrs/*, ...*/) {
124     var cls = RapidContext.Widget.Classes[name];
125     if (cls == null) {
126         throw new ReferenceError("failed to find widget '" + name +
127                                  "' in RapidContext.Widget.Classes");
128     }
129     var args = Array.prototype.slice.call(arguments, 1);
130     return cls.apply(this, MochiKit.Base.flattenArray(args));
131 };

133 /**
134  * Destroys a widget or a DOM node. This function will remove the DOM
135  * node from its parent, disconnect any signals and call destructor
136  * functions. It is also applied recursively to to all child nodes.
137  * Once destroyed, all references to the widget object should be
138  * cleared to reclaim browser memory.
139  *
140  * @param {Widget/Node/NodeList/Array} node the DOM node or list
141  *
142  * @static
143  */
144 RapidContext.Widget.destroyWidget = function (node) {
145     if (node && node.nodeType === 1) {
146         if (typeof(node.destroy) == "function") {
147             node.destroy();
148         }
149         if (node.parentNode != null) {
150             MochiKit.DOM.removeElement(node);
151         }
152         MochiKit.Signal.disconnectAll(node);
153         MochiKit.Signal.disconnectAllTo(node);
154         RapidContext.Widget.destroyWidget(node.childNodes);
155     } else if (node && typeof(node.length) === "number") {
156         Array.prototype.slice.call(node).forEach(RapidContext.Widget.destroyWidget);
157     }
158 };

160 /**
161  * Returns the unique identifier for this DOM node. If a node id has
162  * already been set, that id will be returned. Otherwise a new id
163  * will be generated and assigned to the widget DOM node.
164  *
165  * @return {String} the the unique DOM node identifier
166  */
167 RapidContext.Widget.prototype.uid = function () {
168     if (!this.id) {
169         this.id = "widget" + RapidContext.Widget._nextId();
170     }
171     return this.id;
172 };

206 /**
207  * Dispatches a custom event from this DOM node. The event will be
208  * created and emitted asynchronously (via setTimeout).
209  *
210  * @param {String} type the event type (e.g. "validate")
211  * @param {Object} [opts] the event options (e.g. "{ bubbles: true }")
212  */
213 RapidContext.Widget.prototype._dispatch = function (type, opts) {
214     var self = this;
215     function later() {
216         self.dispatchEvent(new CustomEvent(type, opts));
217     }
218     setTimeout(later);
219 };

221 /**
222  * Updates the widget or HTML DOM node attributes. This method is
223  * sometimes overridden by individual widgets to allow modification
224  * of additional widget attributes.
225  *
226  * @param {Object} attrs the widget and node attributes to set
227  * @param {Boolean} [attrs.disabled] the disabled widget flag
228  * @param {Boolean} [attrs.hidden] the hidden widget flag
229  * @param {String} [attrs.class] the CSS class names
230  */
231 RapidContext.Widget.prototype.setAttrs = function (attrs) {
232     /* eslint max-depth: "off" */
233     for (var name in attrs) {
234         var value = attrs[name];
235         if (name == "disabled") {
236             this._setDisabled(value);
237         } else if (name == "hidden") {
238             this._setHidden(value);
239         } else if (name == "class") {
240             var elem = this._styleNode();
241             this.removeClass.apply(this, elem.className.split(/\s+/));
242             this.addClass.apply(this, value.split(/\s+/));
243         } else if (name == "style") {
244             if (typeof(value) == "string") {
245                 var styles = {};
246                 var parts = value.split(";");
247                 for (var i = 0; i < parts.length; i++) {
248                     var a = parts[i].split(":");
249                     var k = a[0].trim();
250                     if (k != "" && a.length > 1) {
251                         styles[k] = a[1].trim();
252                     }
253                 }
254                 value = styles;
255             }
256             this.setStyle(value);
257         } else if (value != null) {
258             MochiKit.DOM.setNodeAttribute(this, name, value);
259             if (typeof(value) != "object") {
260                 this[name] = value;
261             }
262         } else {
263             this.removeAttribute(name);
264             delete this[name];
265         }
266     }
267 };

269 /**
270  * Updates the CSS styles of this HTML DOM node. This method is
271  * identical to `MochiKit.Style.setStyle`, but uses "this" as the
272  * first argument.
273  *
274  * @param {Object} styles an object with the styles to set
275  *
276  * @example
277  * widget.setStyle({ "font-size": "bold", "color": "red" });
278  */
279 RapidContext.Widget.prototype.setStyle = function (styles) {
280     styles = MochiKit.Base.update({}, styles);
281     var posDimNames = [
282         "width", "height", "zIndex", "z-index",
283         "position", "top", "bottom", "left", "right"
284     ];
285     var posDimStyles = RapidContext.Util.mask(styles, posDimNames);
286     MochiKit.Style.setStyle(this, posDimStyles);
287     MochiKit.Style.setStyle(this._styleNode(), styles);
288 };

290 /**
291  * Checks if this HTML DOM node has the specified CSS class names.
292  * Note that more than one CSS class name may be checked, in which
293  * case all must be present.
294  *
295  * @param {String} [...] the CSS class names to check
296  *
297  * @return {Boolean} `true` if all CSS classes were present, or
298  *         `false` otherwise
299  */
300 RapidContext.Widget.prototype.hasClass = function (/* ... */) {
301     var elem = this._styleNode();
302     for (var i = 0; i < arguments.length; i++) {
303         if (!MochiKit.DOM.hasElementClass(elem, arguments[i])) {
304             return false;
305         }
306     }
307     return true;
308 };
309
310 /**
311  * Adds the specified CSS class names to this HTML DOM node.
312  *
313  * @param {String} [...] the CSS class names to add
314  */
315 RapidContext.Widget.prototype.addClass = function (/* ... */) {
316     var elem = this._styleNode();
317     for (var i = 0; i < arguments.length; i++) {
318         MochiKit.DOM.addElementClass(elem, arguments[i]);
319     }
320 };
321
322 /**
323  * Removes the specified CSS class names from this HTML DOM node.
324  * Note that this method will not remove any class starting with
325  * "widget".
326  *
327  * @param {String} [...] the CSS class names to remove
328  */
329 RapidContext.Widget.prototype.removeClass = function (/* ... */) {
330     var elem = this._styleNode();
331     for (var i = 0; i < arguments.length; i++) {
332         var name = "" + arguments[i];
333         if (name.indexOf("widget") != 0) {
334             MochiKit.DOM.removeElementClass(elem, name);
335         }
336     }
337 };
338
339 /**
340  * Toggles adding and removing the specified CSS class names to and
341  * from this HTML DOM node. If all the CSS classes are already set,
342  * they will be removed. Otherwise they will be added.
343  *
344  * @param {String} [...] the CSS class names to remove
345  *
346  * @return {Boolean} `true` if the CSS classes were added, or
347  *         `false` otherwise
348  */
349 RapidContext.Widget.prototype.toggleClass = function (/* ... */) {
350     if (this.hasClass.apply(this, arguments)) {
351         this.removeClass.apply(this, arguments);
352         return false;
353     } else {
354         this.addClass.apply(this, arguments);
355         return true;
356     }
357 };

359 /**
360  * Checks if this widget is disabled. This method checks both the
361  * "widgetDisabled" CSS class and the `disabled` property. Changes
362  * to the disabled status can be made with `enable()`, `disable()` or
363  * `setAttrs()`.
364  *
365  * @return {Boolean} `true` if the widget is disabled, or
366  *         `false` otherwise
367  */
368 RapidContext.Widget.prototype.isDisabled = function () {
369     return this.disabled === true &&
370            MochiKit.DOM.hasElementClass(this, "widgetDisabled");
371 };
372
373 /**
374  * Performs the changes corresponding to setting the `disabled`
375  * widget attribute.
376  *
377  * @param {Boolean} value the new attribute value
378  */
379 RapidContext.Widget.prototype._setDisabled = function (value) {
380     value = MochiKit.Base.bool(value);
381     if (value) {
382         MochiKit.DOM.addElementClass(this, "widgetDisabled");
383     } else {
384         MochiKit.DOM.removeElementClass(this, "widgetDisabled");
385     }
386     MochiKit.DOM.setNodeAttribute(this, "disabled", value);
387     this.disabled = value;
388 };
389
390 /**
391  * Enables this widget if it was previously disabled. This is
392  * equivalent to calling `setAttrs({ disabled: false })`.
393  */
394 RapidContext.Widget.prototype.enable = function () {
395     this.setAttrs({ disabled: false });
396 };

398 /**
399  * Disables this widget if it was previously enabled. This method is
400  * equivalent to calling `setAttrs({ disabled: true })`.
401  */
402 RapidContext.Widget.prototype.disable = function () {
403     this.setAttrs({ disabled: true });

406 /**
407  * Checks if this widget node is hidden. This method checks for the
408  * existence of the `widgetHidden` CSS class. It does NOT check the
409  * actual widget visibility (the `display` style property set by
410  * animations for example).
411  *
412  * @return {Boolean} `true` if the widget is hidden, or
413  *         `false` otherwise
414  */
415 RapidContext.Widget.prototype.isHidden = function () {
416     return MochiKit.DOM.hasElementClass(this, "widgetHidden");
417 };

419 /**
420  * Performs the changes corresponding to setting the `hidden`
421  * widget attribute.
422  *
423  * @param {Boolean} value the new attribute value
424  */
425 RapidContext.Widget.prototype._setHidden = function (value) {
426     value = MochiKit.Base.bool(value);
427     if (value) {
428         MochiKit.DOM.addElementClass(this, "widgetHidden");
429     } else {
430         MochiKit.DOM.removeElementClass(this, "widgetHidden");
431     }
432     MochiKit.DOM.setNodeAttribute(this, "hidden", value);
433     this.hidden = value;
434 };

436 /**
437  * Shows this widget node if it was previously hidden. This method is
438  * equivalent to calling `setAttrs({ hidden: false })`. It is safe
439  * for all types of widgets, since it only removes the `widgetHidden`
440  * CSS class instead of setting the `display` style property.
441  */
442 RapidContext.Widget.prototype.show = function () {
443     this.setAttrs({ hidden: false });

450  * CSS class instead of setting the `display` style property.

456 /**
457  * Performs a visual effect animation on this widget. This is
458  * implemented using the `MochiKit.Visual` effect package. All options
459  * sent to this function will be passed on to the appropriate
460  * `MochiKit.Visual` function.
461  *
462  * @param {Object} opts the visual effect options
463  * @param {String} opts.effect the MochiKit.Visual effect name
464  * @param {String} opts.queue the MochiKit.Visual queue handling,
465  *            defaults to "replace" and a unique scope for each widget
466  *            (see `MochiKit.Visual` for full options)
467  *
468  * @example
469  * widget.animate({ effect: "fade", duration: 0.5 });
470  * widget.animate({ effect: "Move", transition: "spring", y: 300 });
471  */
472 RapidContext.Widget.prototype.animate = function (opts) {
473     var queue = { scope: this.uid(), position: "replace" };
474     opts = MochiKit.Base.updatetree({ queue: queue }, opts);
475     if (typeof(opts.queue) == "string") {
476         queue.position = opts.queue;
477         opts.queue = queue;
478     }
479     var func = MochiKit.Visual[opts.effect];
480     if (typeof(func) == "function") {
481         func.call(null, this, opts);
482     }
483 };

494 /**
495  * Returns an array with all child DOM nodes. Note that the array is
496  * a real JavaScript array, not a dynamic `NodeList`. This method is
497  * sometimes overridden by child widgets in order to hide
498  * intermediate DOM nodes required by the widget.
499  *
500  * @return {Array} the array of child DOM nodes
501  */
502 RapidContext.Widget.prototype.getChildNodes = function () {
503     var elem = this._containerNode();
504     return elem ? Array.prototype.slice.call(elem.childNodes) : [];
505 };
506
507 /**
508  * Adds a single child DOM node to this widget. This method is
509  * sometimes overridden by child widgets in order to hide or control
510  * intermediate DOM nodes required by the widget.
511  *
512  * @param {Widget/Node} child the DOM node to add
513  */
514 RapidContext.Widget.prototype.addChildNode = function (child) {
515     var elem = this._containerNode();
516     if (elem) {
517         elem.appendChild(child);
518     } else {
519         throw new Error("cannot add child node, widget is not a container");
520     }
521 };
522
523 /**
524  * Removes a single child DOM node from this widget. This method is
525  * sometimes overridden by child widgets in order to hide or control
526  * intermediate DOM nodes required by the widget.
527  *
528  * Note that this method will NOT destroy the removed child widget,
529  * so care must be taken to ensure proper child widget destruction.
530  *
531  * @param {Widget/Node} child the DOM node to remove
532  */
533 RapidContext.Widget.prototype.removeChildNode = function (child) {
534     var elem = this._containerNode();
535     if (elem) {
536         elem.removeChild(child);
537     }
538 };

540 /**
541  * Adds one or more children to this widget. This method will flatten any
542  * arrays among the arguments and ignores any `null` or `undefined` arguments.
543  * Any DOM nodes or widgets will be added to the end, and other objects will be
544  * converted to a text node first. Subclasses should normally override the
545  * `addChildNode()` method instead of this one, since that is the basis for
546  * DOM node insertion.
547  *
548  * @param {Object} [...] the children to add
549  */
550 RapidContext.Widget.prototype.addAll = function (/* ... */) {
551     var args = MochiKit.Base.flattenArray(arguments);
552     for (var i = 0; i < args.length; i++) {
553         if (args[i] == null) {
554             // Ignore null values
555         } else if (RapidContext.Util.isDOM(args[i])) {
556             this.addChildNode(args[i]);
557             // TODO: remove this call for performance
558             RapidContext.Util.resizeElements(args[i]);
559         } else {
560             this.addChildNode(RapidContext.Util.createTextNode(args[i]));
561         }
562     }
563 };
564
565 /**
566  * Removes all children to this widget. This method will also destroy and child
567  * widgets and disconnect all signal listeners. This method uses the
568  * `getChildNodes()` and `removeChildNode()` methods to find and remove the
569  * individual child nodes.
570  */
571 RapidContext.Widget.prototype.removeAll = function () {
572     var children = this.getChildNodes();
573     for (var i = children.length - 1; i >= 0; i--) {
574         this.removeChildNode(children[i]);
575         RapidContext.Widget.destroyWidget(children[i]);
576     }
577 };
