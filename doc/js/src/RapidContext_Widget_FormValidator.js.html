 21 /**
 22  * Creates a new form validator widget.
 23  *
 24  * @constructor
 25  * @param {Object} attrs the widget and node attributes
 26  * @param {String} attrs.name the form field name to validate
 27  * @param {Boolean} [attrs.mandatory] the mandatory field flag,
 28  *            defaults to `true`
 29  * @param {String/RegExp} [attrs.regex] the regular expression to
 30  *            match the field value against, defaults to `null`
 31  * @param {String} [attrs.display] the validator display setting
 32  *            (either "none", "icon", "text" or "both"), defaults
 33  *            to "both"
 34  * @param {String} [attrs.message] the message to display, defaults
 35  *            to the validator function error message
 36  * @param {Function} [attrs.validator] the validator function
 37  * @param {Boolean} [attrs.hidden] the hidden widget flag, defaults to `false`
 38  *
 39  * @return {Widget} the widget DOM node
 40  *
 41  * @class The form validator widget class. Provides visual feedback on form
 42  *     validation failures, using a `<span>` HTML element. It is normally
 43  *     hidden by default and may be configured to only modify its related form
 44  *     field.
 45  * @property {String} name The form field name to validate.
 46  * @property {String} message The default validation message.
 47  * @property {Function} validator The validator function in use.
 48  * @extends RapidContext.Widget
 49  *
 50  * @example {JavaScript}
 51  * var field = RapidContext.Widget.TextField({ name: "name", placeholder: "Your Name Here" });
 52  * var attrs = { name: "name", message: "Please enter your name to proceed." };
 53  * var valid = RapidContext.Widget.FormValidator(attrs);
 54  * var exampleForm = RapidContext.Widget.Form({}, field, valid);
 55  *
 56  * @example {User Interface XML}
 57  * <Form id="exampleForm">
 58  *   <TextField name="name" placeholder="Your Name Here" />
 59  *   <FormValidator name="name" message="Please enter your name to proceed." />
 60  * </Form>
 61  */
 62 RapidContext.Widget.FormValidator = function (attrs) {
 63     var o = MochiKit.DOM.SPAN();
 64     RapidContext.Widget._widgetMixin(o, RapidContext.Widget.FormValidator);
 65     o.addClass("widgetFormValidator");
 66     var defaults = { name: "", mandatory: true, display: "both", message: null, validator: null };
 67     o.setAttrs(Object.assign(defaults, attrs));
 68     o.fields = [];
 69     o.hide();
 70     return o;
 71 };

 73 // Register widget class
 74 RapidContext.Widget.Classes.FormValidator = RapidContext.Widget.FormValidator;

 76 /**
 77  * Updates the widget or HTML DOM node attributes.
 78  *
 79  * @param {Object} attrs the widget and node attributes to set
 80  * @param {String} [attrs.name] the form field name to validate
 81  * @param {Boolean} [attrs.mandatory] the mandatory field flag
 82  * @param {String/RegExp} [attrs.regex] the regular expression to
 83  *            match the field value against
 84  * @param {String} [attrs.display] the validator display setting
 85  *            (either "none", "icon", "text" or "both")
 86  * @param {String} [attrs.message] the message to display
 87  * @param {Function} [attrs.validator] the validator function
 88  * @param {Boolean} [attrs.hidden] the hidden widget flag
 89  */
 90 RapidContext.Widget.FormValidator.prototype.setAttrs = function (attrs) {
 91     attrs = Object.assign({}, attrs);
 92     var locals = RapidContext.Util.mask(attrs, ["name", "mandatory", "regex", "display", "message", "validator"]);
 93     if (typeof(locals.name) != "undefined") {
 94         this.name = locals.name;
 95     }
 96     if (typeof(locals.mandatory) != "undefined") {
 97         this.mandatory = MochiKit.Base.bool(locals.mandatory);
 98     }
 99     if (typeof(locals.regex) != "undefined") {
100         if (locals.regex instanceof RegExp) {
101             this.regex = locals.regex;
102         } else {
103             if (!locals.regex.startsWith("^")) {
104                 locals.regex = "^" + locals.regex;
105             }
106             if (!locals.regex.endsWith("$")) {
107                 locals.regex += "$";
108             }
109             this.regex = new RegExp(locals.regex);
110         }
111     }
112     if (typeof(locals.display) != "undefined") {
113         this.display = locals.display;
114     }
115     if (typeof(locals.message) != "undefined") {
116         this.message = locals.message;
117     }
118     if (typeof(locals.validator) != "undefined") {
119         this.validator = locals.validator;
120     }
121     this.__setAttrs(attrs);
122 };

124 /**
125  * Resets this form validator. This will hide any error messages and mark all
126  * invalidated fields as valid.
127  *
128  * Note that this method is normally not called directly, instead the
129  * validation is reset by the `RapidContext.Widget.Form` widget.
130  *
131  * @see RapidContext.Widget.Form#validateReset
132  */
133 RapidContext.Widget.FormValidator.prototype.reset = function () {
134     this.fields.forEach(function (field) {
135         field.classList.remove("invalid");
136     });
137     this.fields = [];
138     this.hide();
139     this.removeAll();
140 };
141
142 /**
143  * Verifies a form field with this validator. If the form field value doesn't
144  * match this validator, the field will be invalidated until this validator is
145  * reset.
146  *
147  * Note that this method is normally not called directly, instead the
148  * validation is performed by the `RapidContext.Widget.Form` widget.
149  *
150  * @param {Widget/Node} field the form field DOM node
151  * @param {String} [value] the form field value to check
152  *
153  * @return {Boolean} `true` if the form validated successfully, or
154  *         `false` if the validation failed
155  *
156  * @see RapidContext.Widget.Form#validate
157  */
158 RapidContext.Widget.FormValidator.prototype.verify = function (field, value) {
159     if (!field.disabled) {
160         if (arguments.length == 1 && typeof(field.getValue) == "function") {
161             value = field.getValue();
162         } else if (arguments.length == 1) {
163             value = field.value;
164         }
165         var str = String(value).trim();
166         if (field.validationMessage) {
167             this.addError(field, field.validationMessage);
168             return false;
169         } else if (this.mandatory && str == "") {
170             this.addError(field, "This field is required");
171             return false;
172         } else if (this.regex && str && !this.regex.test(str)) {
173             this.addError(field, "The field format is incorrect");
174             return false;
175         } else if (typeof(this.validator) == "function") {
176             var res = this.validator(value);
177             if (res !== true) {
178                 this.addError(field, res || "Field validation failed");
179                 return false;
180             }
181         }
182     }
183     return true;
184 };

186 /**
187  * Adds a validation error message for the specified field. If the field is
188  * already invalid, this method will not do anything.
189  *
190  * Note that this method is normally not called directly, instead the
191  * validation is performed by the `RapidContext.Widget.Form` widget.
192  *
193  * @param {Widget/Node} field the field DOM node
194  * @param {String} message the validation error message
195  *
196  * @see RapidContext.Widget.Form#validate
197  */
198 RapidContext.Widget.FormValidator.prototype.addError = function (field, message) {
199     if (!field.classList.contains("invalid")) {
200         this.fields.push(field);
201         field.classList.add("invalid");
202         if (this.display !== "none") {
203             message = this.message || message;
204             var span = null;
205             var icon = null;
206             if (!this.display || this.display === "both") {
207                 this.addClass("block");
208             }
209             if (this.display !== "icon") {
210                 span = MochiKit.DOM.SPAN({}, message);
211             }
212             if (this.display !== "text") {
213                 icon = RapidContext.Widget.Icon({ ref: "ERROR", tooltip: message });
214             }
215             if (!this.childNodes.length) {
216                 this.addAll(icon, span);
217             }
218             this.show();
219         }
220     }
221 };
