 17     function isFunction(value) {
 18         return typeof(value) === "function";
 19     }
 20
 21     /**
 22      * Checks if an object conforms to the `Promise` API.
 23      *
 24      * @memberOf RapidContext.Async
 25      * @param {Object} value the object to check
 26      *
 27      * @return {Boolean} `true` if the object is a promise, or `false` otherwise
 28      */
 29     function isPromise(value) {
 30         return !!value && isFunction(value.then) && isFunction(value.catch);
 31     }
 32
 33     function isDeferred(value) {
 34         return !isPromise(value) && value instanceof MochiKit.Async.Deferred;
 35     }
 36
 37     function isCancellable(value) {
 38         return isPromise(value) && isFunction(value.cancel);
 39     }
 40
 41     /**
 42      * Creates a new cancellable promise. The cancellation callback must be a
 43      * no-op if the action is already performed.
 44      *
 45      * @constructor
 46      * @param {Promise/Deferred/Function/Error/Object} promise the promise to wrap
 47      * @param {Function} [onCancelled] the cancellation callback function
 48      *
 49      * @name RapidContext.Async
 50      * @class A cancellable Promise that is backwards-compatible with
 51      * `MochiKit.Async.Deferred`. These promises can be used either as a deferred
 52      * or as a normal promise (recommended).
 53      *
 54      * Instances of this class are `instanceof MochiKit.Async.Deferred` (and
 55      * NOT `instanceof Promise`) due to backwards compatibility. Use
 56      * `RapidContext.Async.isPromise` to check for a promise-compatible API.
 57      */
 58     function Async(promise, onCancelled) {
 59         if (isPromise(promise)) {
 60             this._promise = promise;
 61         } else if (isDeferred(promise)) {
 62             this._promise = new Promise(function (resolve, reject) {
 63                 promise.addBoth(function (res) {
 64                     var cb = (res instanceof Error) ? reject : resolve;
 65                     cb(res);
 66                 });
 67             });
 68         } else if (isFunction(promise)) {
 69             this._promise = new Promise(promise);
 70         } else if (promise instanceof Error) {
 71             this._promise = Promise.reject(promise);
 72         } else {
 73             this._promise = Promise.resolve(promise);
 74         }
 75         this._cancel = onCancelled;
 76     }

 79     // FIXME: Use Object.create() here...
 80     // FIXME: Change to Promise.prototype once MochiKit is removed
 81     Async.prototype = MochiKit.Base.clone(MochiKit.Async.Deferred.prototype);

 83     /**#@+ @methodOf RapidContext.Async.prototype */
 84     // FIXME: Use Object.assign() here...
 85     MochiKit.Base.update(Async.prototype, {
 86         constructor: Async,
 87
 88         /**
 89          * Registers one or two callback functions to this promise.
 90          *
 91          * @name then
 92          * @param {Function} [onFulfilled] a callback if promise fulfilled
 93          * @param {Function} [onRejected] a callback if promise rejected
 94          * @returns {Async} a new promise that resolves to whatever value the
 95          *     callback functions return
 96          */
 97         then: function (onFulfilled, onRejected) {
 98             var promise = wrapPromise(this, onFulfilled, onRejected);
 99             var onCancel = this.cancel.bind(this);
100             return new Async(promise, onCancel);
101         },
102
103         /**
104          * Registers a reject callback function to this promise.
105          *
106          * @name catch
107          * @param {Function} onRejected a callback if promise rejected
108          * @returns {Async} a new promise that resolves to whatever value the
109          *     callback function returned
110          */
111         catch: function (onRejected) {
112             return this.then(undefined, onRejected);
113         },
114
115         /**
116          * Registers a finalizer callback function to this promise. Note that
117          * the finalizer MAY NOT BE CALLED if the promise is cancelled.
118          *
119          * @name finally
120          * @param {Function} onFinally a callback for promise resolved
121          * @returns {Async} a new promise
122          */
123         finally: function (onFinally) {
124             var promise = this._promise.finally(onFinally);
125             var onCancel = this.cancel.bind(this);
126             return new Async(promise, onCancel);
127         },
128
129         /**
130          * Cancels this promise and calls the registered cancellation handler.
131          * No other callbacks will be triggered after the promise has been
132          * cancelled (except finalizer).
133          *
134          * @name cancel
135          */
136         cancel: function () {
137             this._cancelled = true;
138             isFunction(this._cancel) && this._cancel();
139             isCancellable(this._result) && this._result.cancel();
140         },
141
142         /**
143          * Registers one or two callback functions to this promise.
144          *
145          * @name addCallbacks
146          * @param {Function} [onFulfilled] a callback if promise fulfilled
147          * @param {Function} [onRejected] a callback if promise rejected
148          * @returns {Async} this same promise
149          * @deprecated Provided for `MochiKit.Async.Deferred` compatibility
150          */
151         addCallbacks: function (callback, errback) {
152             this._promise = wrapPromise(this, callback, errback);
153             return this;
154         },
155
156         /**
157          * Registers a fulfilled callback function to this promise.
158          *
159          * @name addCallback
160          * @param {Function} [callback] a callback if promise fulfilled
161          * @returns {Async} this same promise
162          * @deprecated Provided for `MochiKit.Async.Deferred` compatibility
163          */
164         addCallback: function (callback) {
165             if (arguments.length > 1) {
166                 var args = Array.prototype.slice.call(arguments);
167                 args.splice(1, 0, undefined);
168                 callback = callback.bind.apply(callback, args);
169             }
170             this._promise = wrapPromise(this, callback, undefined);
171             return this;
172         },
173
174         /**
175          * Registers a reject callback function to this promise.
176          *
177          * @name addErrback
178          * @param {Function} errback a callback if promise rejected
179          * @returns {Async} this same promise
180          * @deprecated Provided for `MochiKit.Async.Deferred` compatibility
181          */
182         addErrback: function (errback) {
183             if (arguments.length > 1) {
184                 var args = Array.prototype.slice.call(arguments);
185                 args.splice(1, 0, undefined);
186                 errback = errback.bind.apply(errback, args);
187             }
188             this._promise = wrapPromise(this, undefined, errback);
189             return this;
190         },
191
192         /**
193          * Registers a callback function to this promise.
194          *
195          * @name addBoth
196          * @param {Function} [callback] a callback if promise either fulfilled
197          *        or rejected
198          * @returns {Async} this same promise
199          * @deprecated Provided for `MochiKit.Async.Deferred` compatibility
200          */
201         addBoth: function (callback) {
202             if (arguments.length > 1) {
203                 var args = Array.prototype.slice.call(arguments);
204                 args.splice(1, 0, undefined);
205                 callback = callback.bind.apply(callback, args);
206             }
207             this._promise = wrapPromise(this, callback, callback);
208             return this;
209         }
210     });
211     /**#@- */
212
213     function wrapPromise(self, callback, errback) {
214         var onSuccess = isFunction(callback) ? wrapCallback(self, callback) : callback;
215         var onError = isFunction(errback) ? wrapCallback(self, errback) : errback;
216         return self._promise.then(onSuccess, onError);
217     }
218
219     function wrapCallback(self, callback) {
220         return function (val) {
221             var res = self._cancelled ? undefined : callback(val);
222             return self._result = (isDeferred(res) ? new Async(res) : res);
223         };
224     }

226     /**
227      * Returns a delayed value.
228      *
229      * @memberOf RapidContext.Async
230      * @param {Number} millis the number of milliseconds to wait
231      * @param {Object} [value] the value to resolve with
232      * @return {Async} a new promise that resolves with the value
233      */
234     function wait(millis, value) {
235         var timer = null;
236         function callLater(resolve) {
237             timer = setTimeout(resolve.bind(null, value), millis);
238         }
239         return new Async(callLater, clearTimeout.bind(null, timer));
240     }

265     /**
266      * Injects a JavaScript to the current page.
267      *

276     function create(tag, attrs, parent) {
277         return new Async(function (resolve, reject) {
278             var el = document.createElement(tag);
279             el.onload = function () {
280                 el = el.onload = el.onerror = null;
281                 resolve(el);
282             };
283             el.onerror = function (err) {
284                 var url = el.src || el.href;
285                 el = el.onload = el.onerror = null;
286                 reject(new URIError("failed to load: " + url, url));
287             };
288             // FIXME: Use Object.assign() here...
289             MochiKit.Base.update(el, attrs);
290             parent && parent.appendChild(el);
291         });
292     }

294     /**
295      * Performs an XmlHttpRequest to a URL.
296      *
297      * @memberOf RapidContext.Async
298      * @param {String} url the URL to request
299      * @param {Object} [options] the request options
300      * @config {String} [method] the HTTP method (e.g. `GET`, `POST`...)
301      * @config {Object} [headers] the HTTP headers to send
302      * @config {Number} [timeout] the timeout in milliseconds (default is 30s)
303      * @config {String} [log] the logging prefix, or `null` (defaults to `request`)
304      * @config {String} [responseType] the expected HTTP response (e.g. `json`)
305      * @config {String} [body] the HTTP request body to send
306      *
307      * @return {Async} a new promise that resolves with either the XHR object,
308      *     or an error if the request failed
309      */
310     function xhr(url, options) {
311         var opts = { method: "GET", headers: {}, timeout: 30000, log: "request" };
312         MochiKit.Base.update(opts, options);
313         if (opts.responseType === "json" && !opts.headers["Accept"]) {
314             opts.headers["Accept"] = "application/json";
315         }
316         var xhr = new XMLHttpRequest();
317         var promise = new Promise(function (resolve, reject) {
318             xhr.open(opts.method, url, true);
319             for (var key in opts.headers) {
320                 xhr.setRequestHeader(key, opts.headers[key]);
321             }
322             xhr.responseType = opts.responseType || "text";
323             xhr.timeout = opts.timeout;
324             if (xhr.upload && isFunction(opts.progress)) {
325                 xhr.upload.addEventListener("progress", opts.progress);
326             }
327             xhr.onreadystatechange = function () {
328                 var err;
329                 if (xhr && xhr.readyState === 4) {
330                     if (xhr.status >= 200 && xhr.status <= 299 && xhr.response != null) {
331                         resolve(xhr);
332                     } else if (xhr.status === 0) {
333                         err = new Error(["communication error or timeout on", opts.method, url].join(" "));
334                         err.xhr = xhr;
335                         opts.log && console.warn([opts.log, err.message].join(": "));
336                         reject(err);
337                     } else {
338                         err = new Error(["unexpected HTTP", xhr.status, "on", opts.method, url].join(" "));
339                         err.xhr = xhr;
340                         opts.log && console.warn([opts.log, err.message].join(": "), xhr.response);
341                         reject(err);
342                     }
343                     xhr = null; // Stop duplicate events
344                 }
345             };
346             xhr.send(opts.body);
347         });
348         var cancel = function () {
349             xhr && setTimeout(xhr.abort.bind(xhr));
350             xhr = null;
351         };
352         return new Async(promise, cancel);
353     }
