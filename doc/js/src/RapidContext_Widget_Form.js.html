 21 /**
 22  * Creates a new form widget.
 23  *
 24  * @constructor
 25  * @param {Object} attrs the widget and node attributes
 26  * @param {Boolean} [attrs.hidden] the hidden widget flag, defaults to false
 27  * @param {Object} [...] the child widgets or DOM nodes
 28  *
 29  * @return {Widget} the widget DOM node
 30  *
 31  * @class The form widget class. Provides a grouping for form fields, using the
 32  *     `<form>` HTML element. The form widget supports form reset, validation
 33  *     and data retrieval.
 34  * @extends RapidContext.Widget
 35  *
 36  * @example {JavaScript}
 37  * var field = RapidContext.Widget.TextField({ name: "name", helpText: "Your Name Here" });
 38  * var attrs = { name: "name", message: "Please enter your name to proceed." };
 39  * var valid = RapidContext.Widget.FormValidator(attrs);
 40  * var exampleForm = RapidContext.Widget.Form({}, field, valid);
 41  *
 42  * @example {User Interface XML}
 43  * <Form id="exampleForm">
 44  *   <TextField name="name" helpText="Your Name Here" />
 45  *   <FormValidator name="name" message="Please enter your name to proceed." />
 46  * </Form>
 47  */
 48 RapidContext.Widget.Form = function (attrs/*, ...*/) {
 49     var o = MochiKit.DOM.FORM(attrs);
 50     RapidContext.Widget._widgetMixin(o, arguments.callee);
 51     o.addClass("widgetForm");
 52     o.setAttrs(attrs);
 53     o.addAll(MochiKit.Base.extend(null, arguments, 1));
 54     o.onsubmit = RapidContext.Widget._eventHandler(null, "_handleSubmit");
 55     return o;
 56 };

 58 // Register widget class
 59 RapidContext.Widget.Classes.Form = RapidContext.Widget.Form;

 61 /**
 62  * Returns an array with all child DOM nodes containing form fields.
 63  * The child nodes will be returned based on the results of the
 64  * `RapidContext.Widget.isFormField()` function.
 65  *
 66  * @return {Array} the array of form field elements
 67  */
 68 RapidContext.Widget.Form.prototype.fields = function () {
 69     var fields = [];
 70     MochiKit.Base.nodeWalk(this, function (elem) {
 71         if (elem.nodeType !== 1) { // !Node.ELEMENT_NODE
 72             return null;
 73         }
 74         if (RapidContext.Widget.isFormField(elem)) {
 75             fields.push(elem);
 76             return null;
 77         } else {
 78             return elem.childNodes;
 79         }
 80     });
 81     return fields;
 82 };

 84 /**
 85  * Returns a map with all child DOM nodes containing form fields with
 86  * a name attribute. If multiple fields have the same name, the
 87  * returned map will contain an array with all matching fields.
 88  *
 89  * @return {Object} the map of form field elements
 90  */
 91 RapidContext.Widget.Form.prototype.fieldMap = function () {
 92     var fields = this.fields();
 93     var map = {};
 94     for (var i = 0; i < fields.length; i++) {
 95         var name = fields[i].name;
 96         if (typeof(name) == "string" && name != "*") {
 97             if (map[name] instanceof Array) {
 98                 map[name].push(fields[i]);
 99             } else if (map[name] != null) {
100                 map[name] = [map[name], fields[i]];
101             } else {
102                 map[name] = fields[i];
103             }
104         }
105     }
106     return map;
107 };

109 /**
110  * Resets all fields in the form to their default values.
111  */
112 // TODO: Consider renaming this method, since it collides with the reset()
113 //       method on the DOM element...
114 RapidContext.Widget.Form.prototype.reset = function () {
115     this.validateReset();
116     var fields = this.fields();
117     for (var i = 0; i < fields.length; i++) {
118         var elem = fields[i];
119         // TODO: generic form field value setting
120         if (typeof(elem.reset) == "function") {
121             elem.reset();
122         } else if (elem.type == "radio" && typeof(elem.defaultChecked) == "boolean") {
123             elem.checked = elem.defaultChecked;
124         } else if (elem.type == "checkbox" && typeof(elem.defaultChecked) == "boolean") {
125             elem.checked = elem.defaultChecked;
126         } else if (typeof(elem.defaultValue) == "string") {
127             if (typeof(elem.setAttrs) == "function") {
128                 elem.setAttrs({ value: elem.defaultValue });
129             } else {
130                 elem.value = elem.defaultValue;
131             }
132         } else if (elem.options != null) {
133             for (var j = 0; j < elem.options.length; j++) {
134                 var opt = elem.options[j];
135                 opt.selected = opt.defaultSelected;
136             }
137         }
138     }
139 };

141 /**
142  * Returns a map with all form field values. If multiple fields have
143  * the same name, the value will be set to an array of all values.
144  * Any unchecked checkbox or radiobutton will be also be ignored.
145  *
146  * @return {Object} the map of form field values
147  */
148 RapidContext.Widget.Form.prototype.valueMap = function () {
149     var fields = this.fields();
150     var map = {};
151     for (var i = 0; i < fields.length; i++) {
152         var name = fields[i].name;
153         // TODO: use generic field value retrieval
154         var value = "";
155         if (typeof(fields[i].getValue) == "function") {
156             value = fields[i].getValue();
157         } else {
158             value = fields[i].value;
159         }
160         if (fields[i].type === "radio" || fields[i].type === "checkbox") {
161             if (fields[i].checked) {
162                 value = value || true;
163             } else {
164                 value = null;
165             }
166         }
167         if (typeof(name) == "string" && name != "*" && value != null) {
168             if (map[name] instanceof Array) {
169                 map[name].push(value);
170             } else if (map[name] != null) {
171                 map[name] = [map[name], value];
172             } else {
173                 map[name] = value;
174             }
175         }
176     }
177     return map;
178 };

180 /**
181  * Updates the fields in this form with a specified map of values.
182  * If multiple fields have the same name, the value will be set to
183  * all of them.
184  *
185  * @param {Object} values the map of form field values
186  */
187 RapidContext.Widget.Form.prototype.update = function (values) {
188     var fields = this.fields();
189     for (var i = 0; i < fields.length; i++) {
190         var elem = fields[i];
191         if (elem.name == "*") {
192             if (typeof(elem.setAttrs) == "function") {
193                 elem.setAttrs({ value: values });
194             }
195         } else if (elem.name in values) {
196             var value = values[elem.name];
197             // TODO: generic form field value setting
198             if (elem.type === "radio" || elem.type === "checkbox") {
199                 if (value == null) {
200                     elem.checked = false;
201                 } else if (MochiKit.Base.isArrayLike(value)) {
202                     elem.checked = (MochiKit.Base.findValue(value, elem.value) >= 0);
203                 } else {
204                     elem.checked = (elem.value === value || value === true);
205                 }
206             } else {
207                 if (typeof(elem.setAttrs) == "function") {
208                     elem.setAttrs({ value: value });
209                 } else {
210                     if (MochiKit.Base.isArrayLike(value)) {
211                         value = value.join(", ");
212                     }
213                     elem.value = value;
214                 }
215             }
216         }
217     }
218 };

220 /**
221  * Returns an array with all child DOM nodes containing form validator widgets.
222  *
223  * @return {Array} the array of form validator widgets
224  */
225 RapidContext.Widget.Form.prototype.validators = function () {
226     var res = [];
227     var elems = this.getElementsByTagName("SPAN");
228     for (var i = 0; i < elems.length; i++) {
229         if (RapidContext.Widget.isWidget(elems[i], "FormValidator")) {
230             res.push(elems[i]);
231         }
232     }
233     return res;
234 };

236 /**
237  * Validates this form using the form validators found.
238  *
239  * @return {Boolean/MochiKit.Async.Deferred} `true` if the form
240  *         validated successfully, `false` if the validation failed,
241  *         or a `MochiKit.Async.Deferred` instance if the validation
242  *         was deferred
243  */
244 RapidContext.Widget.Form.prototype.validate = function () {
245     var validators = this.validators();
246     var fields = this.fields();
247     var values = this.valueMap();
248     var success = true;
249     var defers = [];
250     for (var i = 0; i < validators.length; i++) {
251         validators[i].reset();
252     }
253     for (var i = 0; i < validators.length; i++) {
254         for (var j = 0; j < fields.length; j++) {
255             if (validators[i].name == fields[j].name) {
256                 var name = fields[j].name;
257                 var res = validators[i].verify(fields[j], values[name] || "");
258                 if (res instanceof MochiKit.Async.Deferred) {
259                     defers.push(res);
260                 } else if (res === false) {
261                     success = false;
262                 }
263             }
264         }
265     }
266     if (!success) {
267         return false;
268     } else if (defers.length > 0) {
269         return MochiKit.Async.gatherResults(defers);
270     } else {
271         return true;
272     }
273 };

275 /**
276  * Resets all form validators. This metod is automatically called upon form
277  * reset.
278  *
279  * @see #reset
280  */
281 RapidContext.Widget.Form.prototype.validateReset = function () {
282     var validators = this.validators();
283     for (var i = 0; i < validators.length; i++) {
284         validators[i].reset();
285     }
286 };

291  * @param {Event} evt the MochiKit.Signal.Event object
292  */
293 RapidContext.Widget.Form.prototype._handleSubmit = function (evt) {
294     evt.stop();
