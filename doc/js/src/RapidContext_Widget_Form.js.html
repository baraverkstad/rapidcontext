 15 // Namespace initialization
 16 if (typeof(RapidContext) == "undefined") {
 17     RapidContext = {};
 18 }
 19 RapidContext.Widget = RapidContext.Widget || { Classes: {} };

 21 /**
 22  * Creates a new form widget.
 23  *
 24  * @constructor
 25  * @param {Object} attrs the widget and node attributes
 26  * @param {Boolean} [attrs.hidden] the hidden widget flag, defaults to false
 27  * @param {Object} [...] the child widgets or DOM nodes
 28  *
 29  * @return {Widget} the widget DOM node
 30  *
 31  * @class The form widget class. Provides a grouping for form fields, using the
 32  *     `<form>` HTML element. The form widget supports form reset, validation
 33  *     and data retrieval.
 34  * @extends RapidContext.Widget
 35  *
 36  * @example {JavaScript}
 37  * var field = RapidContext.Widget.TextField({ name: "name", placeholder: "Your Name Here" });
 38  * var attrs = { name: "name", message: "Please enter your name to proceed." };
 39  * var valid = RapidContext.Widget.FormValidator(attrs);
 40  * var exampleForm = RapidContext.Widget.Form({}, field, valid);
 41  *
 42  * @example {User Interface XML}
 43  * <Form id="exampleForm">
 44  *   <TextField name="name" placeholder="Your Name Here" />
 45  *   <FormValidator name="name" message="Please enter your name to proceed." />
 46  * </Form>
 47  */
 48 RapidContext.Widget.Form = function (attrs/*, ...*/) {
 49     var o = MochiKit.DOM.FORM(attrs);
 50     o._validators = {};
 51     o._originalReset = o.reset;
 52     RapidContext.Widget._widgetMixin(o, RapidContext.Widget.Form);
 53     o.addClass("widgetForm");
 54     o.setAttrs(attrs);
 55     o.addAll(Array.prototype.slice.call(arguments, 1));
 56     o.addEventListener("input", o._handleInput);
 57     o.addEventListener("invalid", o._handleInvalid, { capture: true });
 58     o.addEventListener("submit", o._handleSubmit);
 59     return o;
 60 };
 61
 62 // Register widget class
 63 RapidContext.Widget.Classes.Form = RapidContext.Widget.Form;

 65 /**
 66  * Destroys this widget.
 67  */
 68 RapidContext.Widget.Form.prototype.destroy = function () {
 69     // FIXME: Use AbortSignal instead to disconnect
 70     this.addEventListener("input", this._handleInput);
 71     this.addEventListener("invalid", this._handleInvalid, { capture: true });
 72     this.addEventListener("submit", this._handleSubmit);
 73 };
 74
 75 /**
 76  * Applies custom validators on field input.
 77  *
 78  * @param {Event} evt the DOM event object
 79  */
 80 RapidContext.Widget.Form.prototype._handleInput = function (evt) {
 81     this._callValidators(evt.target);
 82 };
 83
 84 /**
 85  * Debounces the input invalid events and validates the form.
 86  */
 87 RapidContext.Widget.Form.prototype._handleInvalid = function () {
 88     if (this._validationTimer !== false) {
 89         this._validationTimer && clearTimeout(this._validationTimer);
 90         this._validationTimer = setTimeout(this.validate.bind(this), 10);
 91     }
 92 };
 93
 94 /**
 95  * Prevents the default submit action and validates the form.
 96  *
 97  * @param {Event} evt the DOM event object
 98  */
 99 RapidContext.Widget.Form.prototype._handleSubmit = function (evt) {
100     evt.preventDefault();
101     if (!this.validate()) {
102         evt.stopImmediatePropagation();
103     }
104 };
105
106 RapidContext.Widget.Form.prototype._fieldValue = function (field) {
107     if (field.disabled) {
108         return null;
109     } else if (field.type === "radio" || field.type === "checkbox") {
110         return field.checked ? (field.value || true) : null;
111     } else if (typeof(field.getValue) == "function") {
112         return field.getValue();
113     } else {
114         return field.value;
115     }
116 };

118 /**
119  * Returns an array with all child DOM nodes containing form fields.
120  *
121  * @return {Array} the array of form field elements
122  *
123  * @see https://developer.mozilla.org/en-US/docs/Web/API/HTMLFormElement/elements
124  */
125 RapidContext.Widget.Form.prototype.fields = function () {
126     var basics = Array.prototype.slice.call(this.elements);
127     var extras = Array.prototype.slice.call(this.querySelectorAll(".widgetField"));
128     return basics.concat(extras);
129 };
130
131 /**
132  * Returns a map with all child DOM nodes containing form fields with
133  * a name attribute. If multiple fields have the same name, the
134  * returned map will contain an array with all matching fields.
135  *
136  * @return {Object} the map of form field elements
137  */
138 RapidContext.Widget.Form.prototype.fieldMap = function () {
139     function update(o, field) {
140         var k = field.name;
141         if (k && k != "*") {
142             o[k] = (k in o) ? [].concat(o[k], field) : field;
143         }
144         return o;
145     }
146     return this.fields().reduce(update, {});
147 };
148
149 /**
150  * Resets all fields and validations to their original state.
151  */
152 RapidContext.Widget.Form.prototype.reset = function () {
153     this._originalReset();
154     var extras = Array.prototype.slice.call(this.querySelectorAll(".widgetField"));
155     extras.forEach(function (field) {
156         field.reset();
157     });
158     this.validateReset();
159 };
160
161 /**
162  * Returns a map with all form field values. If multiple fields have
163  * the same name, the value will be set to an array of all values.
164  * Disabled fields and unchecked checkboxes or radiobuttons will be
165  * ignored.
166  *
167  * @return {Object} the map of form field values
168  */
169 RapidContext.Widget.Form.prototype.valueMap = function () {
170     function update(o, field) {
171         var k = field.name;
172         var v = getValue(field);
173         if (k && k != "*" && v != null) {
174             o[k] = (k in o) ? [].concat(o[k], v) : v;
175         }
176         return o;
177     }
178     var getValue = this._fieldValue;
179     return this.fields().reduce(update, {});
180 };
181
182 /**
183  * Updates the fields in this form with a specified map of values.
184  * If multiple fields have the same name, the value will be set to
185  * all of them.
186  *
187  * @param {Object} values the map of form field values
188  */
189 RapidContext.Widget.Form.prototype.update = function (values) {
190     function setValue(field) {
191         var v = values[field.name];
192         if (field.name == "*" && typeof(field.setAttrs) == "function") {
193             field.setAttrs({ value: values });
194         } else if (!(field.name in values)) {
195             // Don't change omitted fields
196         } else if (field.type === "radio" || field.type === "checkbox") {
197             var found = MochiKit.Base.isArrayLike(v) && MochiKit.Base.findValue(v, field.value) >= 0;
198             field.checked = found || v === field.value || v === true;
199         } else if (typeof(field.setAttrs) == "function") {
200             field.setAttrs({ value: v });
201         } else {
202             field.value = MochiKit.Base.isArrayLike(v) ? v.join(", ") : v;
203         }
204     }
205     this.fields().forEach(setValue);
206 };

208 /**
209  * Adds a custom form validator for a named form field. The function will be
210  * called as `[field].validator([value], [field], [form])` and should return
211  * `true`, `false` or a validation error message. The validator will be called
212  * on each `input` event and before form submission for enabled fields.
213  *
214  * Note: Checkbox validators will be called once for each `<input>` element,
215  * regardless of checked state. Radio validators will only be called with
216  * either the first or the checked `<input>` element.
217  *
218  * @param {String/Element} field the form field or name
219  * @param {Function} validator the validator function
220  */
221 RapidContext.Widget.Form.prototype.addValidator = function (field, validator) {
222     var name = String(field.name || field);
223     var arr = [].concat(this._validators[name], validator).filter(Boolean);
224     this._validators[name] = arr;
225 };
226
227 /**
228  * Removes all custom form validators for a named form field.
229  *
230  * @param {String/Element} [field] the form field, name, or null for all
231  */
232 RapidContext.Widget.Form.prototype.removeValidators = function (field) {
233     if (field) {
234         var name = String(field.name || field);
235         delete this._validators[name];
236     } else {
237         this._validators = {};
238     }
239 };
240
241 /**
242  * Calls all custom validators for a form field. The validation result will
243  * update the `setCustomValidity()` on the field.
244  *
245  * @param {Element} field the form field
246  */
247 RapidContext.Widget.Form.prototype._callValidators = function (field) {
248     var validators = this._validators[field.name];
249     if (!field.disabled && validators) {
250         var self = this;
251         var res = true;
252         validators.forEach(function (validator) {
253             if (res === true) {
254                 res = validator.call(field, self._fieldValue(field), field, self);
255             }
256         });
257         field.setCustomValidity((res === true) ? "" : (res || "Validation failed"));
258     }
259 };

261 /**
262  * Returns an array with all child DOM nodes containing form validator widgets.
263  *
264  * @return {Array} the array of form validator widgets
265  */
266 RapidContext.Widget.Form.prototype.validators = function () {
267     var nodes = this.querySelectorAll(".widgetFormValidator");
268     return Array.prototype.slice.call(nodes);
269 };
270
271 /**
272  * Validates this form using the form validators found.
273  *
274  * @return {Boolean} `true` if the form validated successfully, or
275  *         `false` if the validation failed
276  */
277 RapidContext.Widget.Form.prototype.validate = function () {
278     this._validationTimer && clearTimeout(this._validationTimer);
279     this._validationTimer = false;
280     var self = this;
281     var fields = this.fieldMap();
282     var values = this.valueMap();
283     var success = true;
284     this.validateReset();
285     Object.keys(this._validators).forEach(function (name) {
286         [].concat(fields[name]).filter(Boolean).forEach(function (f) {
287             if (f.type !== "radio" || f.checked) {
288                 self._callValidators(f);
289             }
290         });
291     });
292     this.validators().forEach(function (validator) {
293         [].concat(fields[validator.name]).filter(Boolean).forEach(function (f) {
294             success = validator.verify(f, values[f.name] || "") && success;
295         });
296     });
297     success = this.checkValidity() && success;
298     success || this.addClass("invalid");
299     delete this._validationTimer;
300     this._dispatch("validate", { detail: success, bubbles: true });
301     return success;
302 };
303
304 /**
305  * Resets all form validations. This method is automatically called when form
306  * is reset.
307  *
308  * @see #reset
309  */
310 RapidContext.Widget.Form.prototype.validateReset = function () {
311     $(this).find(".invalid").addBack().removeClass("invalid");
312     this.validators().forEach(function (validator) {
313         validator.reset();
314     });
315 };
