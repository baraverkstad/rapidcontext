 21 /**
 22  * Creates a new wizard widget.
 23  *
 24  * @constructor
 25  * @param {Object} attrs the widget and node attributes
 26  * @param {Boolean} [attrs.hidden] the hidden widget flag, defaults to false
 27  * @param {Widget} [...] the child widgets or DOM nodes (should be
 28  *            Pane widgets)
 29  *
 30  * @return {Widget} the widget DOM node
 31  *
 32  * @class The wizard widget class. Used to provide a sequence of pages, where
 33  *     the user can step forward and backward with buttons. Internally it uses
 34  *     a `<div>` HTML element containing Pane widgets that are hidden and shown
 35  *     according to the page transitions.
 36  * @extends RapidContext.Widget
 37  *
 38  * @example {JavaScript}
 39  * var page1 = RapidContext.Widget.Pane({ pageTitle: "The first step" });
 40  * ...
 41  * var page2 = RapidContext.Widget.Pane({ pageTitle: "The second step" });
 42  * ...
 43  * var attrs = { style: { width: "100%", height: "100%" } };
 44  * var exampleWizard = RapidContext.Widget.Wizard(attrs, page1, page2);
 45  * var exampleDialog = RapidContext.Widget.Dialog({ title: "Example Dialog" }, exampleWizard);
 46  * RapidContext.Util.registerSizeConstraints(exampleDialog, "80%", "50%");
 47  *
 48  * @example {User Interface XML}
 49  * <Dialog id="exampleDialog" title="Example Dialog" w="80%" h="50%">
 50  *   <Wizard id="exampleWizard" style="width: 100%; height: 100%;">
 51  *     <Pane pageTitle="The first step">
 52  *       ...
 53  *     </Pane>
 54  *     <Pane pageTitle="The second step">
 55  *       ...
 56  *     </Pane>
 57  *   </Wizard>
 58  * </Dialog>
 59  */
 60 RapidContext.Widget.Wizard = function (attrs/*, ... */) {
 61     var o = MochiKit.DOM.DIV(attrs);
 62     RapidContext.Widget._widgetMixin(o, RapidContext.Widget.Wizard);
 63     o.addClass("widgetWizard");
 64     o.resizeContent = o._resizeContent;
 65     o._selectedIndex = -1;
 66     o.appendChild(MochiKit.DOM.H3({ "class": "widgetWizardTitle" }));
 67     var bCancel = RapidContext.Widget.Button(
 68         { icon: "fa fa-lg fa-times", "class": "mr-2" },
 69         "Cancel"
 70     );
 71     var bPrev = RapidContext.Widget.Button(
 72         { icon: "fa fa-lg fa-caret-left", "class": "mr-2" },
 73         "Previous"
 74     );
 75     var bNext = RapidContext.Widget.Button(
 76         {},
 77         "Next",
 78         RapidContext.Widget.Icon({ class: "fa fa-lg fa-caret-right" })
 79     );
 80     var bDone = RapidContext.Widget.Button(
 81         { icon: "fa fa-lg fa-check", highlight: true },
 82         "Finish"
 83     );
 84     bCancel.hide();
 85     var divAttrs = { "class": "widgetWizardButtons" };
 86     o.appendChild(MochiKit.DOM.DIV(divAttrs, bCancel, bPrev, bNext, bDone));
 87     MochiKit.Signal.connect(bCancel, "onclick", o, "cancel");
 88     MochiKit.Signal.connect(bPrev, "onclick", o, "previous");
 89     MochiKit.Signal.connect(bNext, "onclick", o, "next");
 90     MochiKit.Signal.connect(bDone, "onclick", o, "done");
 91     o._updateStatus();
 92     o.setAttrs(attrs);
 93     o.addAll(Array.from(arguments).slice(1));
 94     return o;
 95 };

121 /**
122  * Returns an array with all child pane widgets. Note that the array
123  * is a real JavaScript array, not a dynamic NodeList.
124  *
125  * @return {Array} the array of child wizard page widgets
126  */
127 RapidContext.Widget.Wizard.prototype.getChildNodes = function () {
128     return Array.from(this.childNodes).slice(2);
129 };
130
131 /**
132  * Adds a single child page widget to this widget. The child widget should be a
133  * `RapidContext.Widget.Pane` widget, or a new one will be created where it
134  * will be added.
135  *
136  * @param {Widget} child the page widget to add
137  */
138 RapidContext.Widget.Wizard.prototype.addChildNode = function (child) {
139     if (!RapidContext.Widget.isWidget(child, "Pane")) {
140         child = RapidContext.Widget.Pane(null, child);
141     }
142     RapidContext.Util.registerSizeConstraints(child, "100%", "100%-65");
143     child.hide();
144     this.appendChild(child);
145     child.style.position = "absolute";
146     // TODO: remove hard-coded size here...
147     MochiKit.Style.setElementPosition(child, { x: 0, y: 24 });
148     if (this.getChildNodes().length == 1) {
149         this.activatePage(0);
150     } else {
151         this._updateStatus();
152     }
153 };

157 /**
158  * Updates the wizard status indicators, such as the title and the
159  * current buttons.
160  */
161 RapidContext.Widget.Wizard.prototype._updateStatus = function () {
162     var h3 = this.childNodes[0];
163     var bCancel = this.childNodes[1].childNodes[0];
164     var bPrev = this.childNodes[1].childNodes[1];
165     var bNext = this.childNodes[1].childNodes[2];
166     var bDone = this.childNodes[1].childNodes[3];
167     var page = this.activePage();
168     var status = RapidContext.Widget.Pane.FORWARD;
169     var title = null;
170     var info = "(No pages available)";
171     var icon = null;
172     if (page != null) {
173         status = page.pageStatus || RapidContext.Widget.Pane.ANY;
174         title = page.pageTitle;
175         info = " (Step " + (this._selectedIndex + 1) + " of " +
176                this.getChildNodes().length + ")";
177     }
178     if (status === RapidContext.Widget.Pane.WORKING) {
179         bCancel.show();
180         bPrev.hide();
181         icon = { ref: "LOADING", "class": "widgetWizardWait" };
182         icon = RapidContext.Widget.Icon(icon);
183     } else {
184         bCancel.hide();
185         bPrev.show();
186     }
187     if (this._selectedIndex >= this.getChildNodes().length - 1) {
188         bNext.hide();
189         bDone.show();
190     } else {
191         bNext.show();
192         bDone.hide();
193     }
194     bPrev.setAttrs({ disabled: (this._selectedIndex <= 0) || !status.previous });
195     bNext.setAttrs({ disabled: !status.next });
196     bDone.setAttrs({ disabled: !status.next });
197     info = MochiKit.DOM.SPAN({ "class": "widgetWizardInfo" }, info);
198     MochiKit.DOM.replaceChildNodes(h3, icon, title, info);
199 };

201 /**
202  * Returns the active page.
203  *
204  * @return {Widget} the active page, or
205  *         null if no pages have been added
206  */
207 RapidContext.Widget.Wizard.prototype.activePage = function () {
208     if (this._selectedIndex >= 0) {
209         return this.childNodes[this._selectedIndex + 2];
210     } else {
211         return null;
212     }
213 };
214
215 /**
216  * Returns the active page index.
217  *
218  * @return the active page index, or
219  *         -1 if no page is active
220  */
221 RapidContext.Widget.Wizard.prototype.activePageIndex = function () {
222     return this._selectedIndex;
223 };

225 /**
226  * Activates a new page and sends the `onchange` signal. If the page is moved
227  * forward, the old page must pass a form validation check, or nothing will
228  * happen.
229  *
230  * @param {Number/Widget} indexOrPage the page index or page DOM node
231  *
232  * @see #next
233  * @see #previous
234  */
235 RapidContext.Widget.Wizard.prototype.activatePage = function (indexOrPage) {
236     var index, page;
237     if (typeof(indexOrPage) == "number") {
238         index = indexOrPage;
239         page = this.childNodes[index + 2];
240     } else {
241         page = indexOrPage;
242         index = MochiKit.Base.findIdentical(this.childNodes, page, 2) - 2;
243     }
244     if (index < 0 || index >= this.getChildNodes().length) {
245         throw new RangeError("Page index out of bounds: " + index);
246     }
247     var oldIndex = this._selectedIndex;
248     var oldPage = this.activePage();
249     if (oldPage != null && oldPage !== page) {
250         if (!oldPage._handleExit({ hide: false, validate: this._selectedIndex < index })) {
251             // Old page blocked page transition
252             return;
253         }
254     }
255     this._selectedIndex = index;
256     this._updateStatus();
257     if (oldPage != null && oldPage !== page) {
258         var dim = MochiKit.Style.getElementDimensions(this);
259         var offset = (oldIndex < index) ? dim.w : -dim.w;
260         MochiKit.Style.setElementPosition(page, { x: offset });
261         page._handleEnter({ validateReset: true });
262         var cleanup = function () {
263             oldPage.hide();
264             MochiKit.Style.setElementPosition(oldPage, { x: 0 });
265         };
266         var opts = { duration: 0.5, x: -offset, afterFinish: cleanup };
267         MochiKit.Visual.Move(oldPage, opts);
268         MochiKit.Visual.Move(page, opts);
269     } else {
270         page._handleEnter({ validateReset: true });
271     }
272     var detail = { index: index, page: page };
273     this._dispatch("change", { detail: detail });
274 };

276 /**
277  * Cancels the active page operation. This method will also reset the page
278  * status of the currently active page to `ANY`. This method is triggered when
279  * the user presses the "Cancel" button.
280  *
281  * @see RapidContext.Widget.Pane.ANY
282  */
283 RapidContext.Widget.Wizard.prototype.cancel = function () {
284     var page = this.activePage();
285     page.setAttrs({ pageStatus: RapidContext.Widget.Pane.ANY });
286     this._dispatch("cancel");
287 };
288
289 /**
290  * Moves the wizard backward to the previous page and sends the `onchange`
291  * signal. This method is triggered when the user presses the "Previous"
292  * button.
293  */
294 RapidContext.Widget.Wizard.prototype.previous = function () {
295     if (this._selectedIndex > 0) {
296         this.activatePage(this._selectedIndex - 1);
297     }
298 };
299
300 /**
301  * Moves the wizard forward to the next page and sends the `onchange` signal.
302  * The page will not be changed if the active page fails a validation check.
303  * This method is triggered when the user presses the "Next" button.
304  */
305 RapidContext.Widget.Wizard.prototype.next = function () {
306     if (this._selectedIndex < this.getChildNodes().length - 1) {
307         this.activatePage(this._selectedIndex + 1);
308     }
309 };
310
311 /**
312  * Sends the wizard `onclose` signal. This method is triggered when the user
313  * presses the "Finish" button.
314  */
315 RapidContext.Widget.Wizard.prototype.done = function () {
316     var page = this.activePage();
317     if (page != null) {
318         if (!page._handleExit({ validate: true })) {
319             // Page blocked wizard completion
320             return;
321         }
322     }
323     this._dispatch("close");
324 };

326 /**
327  * Resizes the current wizard page. This method need not be called
328  * directly, but is automatically called whenever a parent node is
329  * resized. It optimizes the resize chain by only resizing those DOM
330  * child nodes that are visible.
331  */
332 RapidContext.Widget.Wizard.prototype._resizeContent = function () {
333     var page = this.activePage();
334     if (page != null) {
335         RapidContext.Util.resizeElements(page);
336     }
337 };
