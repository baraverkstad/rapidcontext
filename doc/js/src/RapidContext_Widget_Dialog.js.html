 21 /**
 22  * Creates a new dialog widget.
 23  *
 24  * @constructor
 25  * @param {Object} attrs the widget and node attributes
 26  * @param {String} [attrs.title] the dialog title, defaults to "Dialog"
 27  * @param {Boolean} [attrs.modal] the modal dialog flag, defaults to `false`
 28  * @param {Boolean} [attrs.system] the system dialog flag, implies modal,
 29  *            defaults to `false`
 30  * @param {Boolean} [attrs.center] the center dialog flag, defaults to `true`
 31  * @param {Boolean} [attrs.closeable] the closeable dialog flag, defaults to
 32  *            `true`
 33  * @param {Boolean} [attrs.resizeable] the resize dialog flag, defaults to
 34  *            `true`
 35  * @param {Boolean} [attrs.hidden] the hidden widget flag, defaults to `true`
 36  * @param {Object} [...] the child widgets or DOM nodes
 37  *
 38  * @return {Widget} the widget DOM node
 39  *
 40  * @class The dialog widget class. Used to provide a resizeable and
 41  *     moveable window within the current page. Internally it uses a
 42  *     number of `<div>` HTML elements.
 43  * @extends RapidContext.Widget
 44  *
 45  * @example {JavaScript}
 46  * var h1 = MochiKit.DOM.H1({}, "Hello, world!");
 47  * var attrs = { title: "Hello", modal: true };
 48  * var helloDialog = RapidContext.Widget.Dialog(attrs, h1);
 49  * RapidContext.Util.registerSizeConstraints(helloDialog, "200", "75");
 50  *
 51  * @example {User Interface XML}
 52  * <Dialog id="helloDialog" title="Hello" modal="true" w="200" h="75">
 53  *   <h1>Hello, world!</h1>
 54  * </Dialog>
 55  */
 56 RapidContext.Widget.Dialog = function (attrs/*, ... */) {
 57     var title = MochiKit.DOM.DIV({ "class": "widgetDialogTitle" }, "Dialog");
 58     var close = RapidContext.Widget.Icon({ "class": "widgetDialogClose fa fa-close", tooltip: "Close",  });
 59     var resize = RapidContext.Widget.Icon({ ref: "RESIZE", "class": "widgetDialogResize" });
 60     var content = MochiKit.DOM.DIV({ "class": "widgetDialogContent" });
 61     var o = MochiKit.DOM.DIV({}, title, close, resize, content);
 62     RapidContext.Widget._widgetMixin(o, arguments.callee);
 63     MochiKit.DOM.addElementClass(o, "widgetDialog");
 64     o.resizeContent = o._resizeContent;
 65     o._setHidden(true);
 66     o.setAttrs(MochiKit.Base.update({ modal: false, system: false, center: true }, attrs));
 67     o.addAll(MochiKit.Base.extend(null, arguments, 1));
 68     title.onmousedown = RapidContext.Widget._eventHandler("Dialog", "_handleMoveStart");
 69     close.onclick = RapidContext.Widget._eventHandler("Dialog", "hide");
 70     resize.onmousedown = RapidContext.Widget._eventHandler("Dialog", "_handleResizeStart");
 71     return o;
 72 };

129 /**
130  * Updates the dialog or HTML DOM node attributes.
131  *
132  * @param {Object} attrs the widget and node attributes to set
133  * @param {String} [attrs.title] the dialog title
134  * @param {Boolean} [attrs.modal] the modal dialog flag
135  * @param {Boolean} [attrs.system] the system dialog flag, implies modal
136  * @param {Boolean} [attrs.center] the center dialog flag
137  * @param {Boolean} [attrs.closeable] the closeable dialog flag
138  * @param {Boolean} [attrs.resizeable] the resize dialog flag
139  * @param {Boolean} [attrs.hidden] the hidden widget flag
140  */
141 RapidContext.Widget.Dialog.prototype.setAttrs = function (attrs) {
142     attrs = MochiKit.Base.update({}, attrs);
143     var locals = RapidContext.Util.mask(attrs, ["title", "modal", "system", "center", "resizeable", "closeable", "hidden"]);
144     if (typeof(locals.title) != "undefined") {
145         MochiKit.DOM.replaceChildNodes(this.firstChild, locals.title);
146     }
147     if (typeof(locals.modal) != "undefined") {
148         this.modal = MochiKit.Base.bool(locals.modal);
149     }
150     if (typeof(locals.system) != "undefined") {
151         this.system = MochiKit.Base.bool(locals.system);
152     }
153     if (typeof(locals.center) != "undefined") {
154         this.center = MochiKit.Base.bool(locals.center);
155     }
156     if (typeof(locals.resizeable) != "undefined") {
157         var resize = this.childNodes[2];
158         resize.setAttrs({ hidden: !MochiKit.Base.bool(locals.resizeable) });
159     }
160     if (typeof(locals.closeable) != "undefined") {
161         var close = this.childNodes[1];
162         close.setAttrs({ hidden: !MochiKit.Base.bool(locals.closeable) });
163     }
164     if (typeof(locals.hidden) != "undefined") {
165         this._setHiddenDialog(locals.hidden);
166     }
167     this.__setAttrs(attrs);
168 };
169
170 /**
171  * Performs the changes corresponding to setting the `hidden`
172  * widget attribute for the Dialog widget.
173  *
174  * @param {Boolean} value the new attribute value
175  */
176 RapidContext.Widget.Dialog.prototype._setHiddenDialog = function (value) {
177     if (value) {
178         if (this._modalNode != null) {
179             RapidContext.Widget.destroyWidget(this._modalNode);
180             this._modalNode = null;
181         }
182         this.blurAll();
183         this._setHidden(true);
184         RapidContext.Widget.emitSignal(this, "onhide");
185     } else {
186         if (this.parentNode == null) {
187             throw new Error("Cannot show Dialog widget without setting a parent DOM node");
188         }
189         if (this.modal || this.system) {
190             var attrs = { loading: false, message: "", style: { "z-index": "99" } };
191             if (this.system) {
192                 attrs.dark = true;
193             }
194             this._modalNode = RapidContext.Widget.Overlay(attrs);
195             this.parentNode.appendChild(this._modalNode);
196         }
197         this._setHidden(false);
198         this.moveTo(0, 0);
199         var dim = MochiKit.Style.getElementDimensions(this);
200         this.resizeTo(dim.w, dim.h);
201         if (this.center) {
202             this.moveToCenter();
203         }
204         this.resetScroll();
205         RapidContext.Widget.emitSignal(this, "onshow");
206     }
207 };

209 /**
210  * Moves the dialog to the specified position (relative to the
211  * parent DOM node). The position will be restrained by the parent
212  * DOM node size.
213  *
214  * @param {Number} x the horizontal position (in pixels)
215  * @param {Number} y the vertical position (in pixels)
216  */
217 RapidContext.Widget.Dialog.prototype.moveTo = function (x, y) {
218     var parentDim = MochiKit.Style.getElementDimensions(this.parentNode);
219     var dim = MochiKit.Style.getElementDimensions(this);
220     var pos = { x: Math.max(0, Math.min(x, parentDim.w - dim.w - 2)),
221                 y: Math.max(0, Math.min(y, parentDim.h - dim.h - 2)) };
222     MochiKit.Style.setElementPosition(this, pos);
223     RapidContext.Widget.emitSignal(this, "onmove", pos);
224 };
225
226 /**
227  * Moves the dialog to the apparent center (relative to the parent DOM
228  * node). The vertical position actually uses the golden ratio instead
229  * of the geometric center for improved visual alignment.
230  */
231 RapidContext.Widget.Dialog.prototype.moveToCenter = function () {
232     var parentDim = MochiKit.Style.getElementDimensions(this.parentNode);
233     var dim = MochiKit.Style.getElementDimensions(this);
234     var pos = { x: Math.round(Math.max(0, (parentDim.w - dim.w) / 2)),
235                 y: Math.round(Math.max(0, (parentDim.h - dim.h) / 2.618)) };
236     MochiKit.Style.setElementPosition(this, pos);
237     RapidContext.Widget.emitSignal(this, "onmove", pos);
238 };

240 /**
241  * Resizes the dialog to the specified size (in pixels). The size
242  * will be restrained by the parent DOM node size.
243  *
244  * @param {Number} width the width (in pixels)
245  * @param {Number} height the height (in pixels)
246  *
247  * @return {Dimensions} an object with "w" and "h" properties for the
248  *         actual size used
249  */
250 RapidContext.Widget.Dialog.prototype.resizeTo = function (width, height) {
251     var parentDim = MochiKit.Style.getElementDimensions(this.parentNode);
252     var pos = MochiKit.Style.getElementPosition(this, this.parentNode);
253     var dim = { w: Math.max(150, Math.min(width, parentDim.w - pos.x - 2)),
254                 h: Math.max(100, Math.min(height, parentDim.h - pos.y - 2)) };
255     MochiKit.Style.setElementDimensions(this, dim);
256     RapidContext.Util.registerSizeConstraints(this, null, null);
257     MochiKit.Base.update(this, dim);
258     this._resizeContent();
259     RapidContext.Widget.emitSignal(this, "onresize", dim);
260     return dim;
261 };
262
263 /**
264  * Resizes the dialog to the optimal size for the current content.
265  * Note that the size reported by the content may vary depending on
266  * if it has already been displayed, is absolutely positioned, etc.
267  * The size will be restrained by the parent DOM node size.
268  *
269  * @return {Dimensions} an object with "w" and "h" properties for the
270  *         actual size used
271  */
272 RapidContext.Widget.Dialog.prototype.resizeToContent = function () {
273     var content = this.lastChild;
274     MochiKit.Style.setStyle(content, { width: "auto", height: "auto", overflow: "hidden" });
275     var x = Math.max(content.scrollWidth, content.offsetWidth) + 4;
276     var y = Math.max(content.scrollHeight, content.offsetHeight) + content.offsetTop + 2;
277     MochiKit.Style.setStyle(content, { overflow: "auto" });
278     return this.resizeTo(Math.round(x), Math.round(y));
279 }
280
281 /**
282  * Called when dialog content should be resized.
283  */
284 RapidContext.Widget.Dialog.prototype._resizeContent = function () {
285     // TODO: Allow content node to have different padding
286     var content = this.lastChild;
287     var dim = { w: Math.max(0, this.w - 20) || undefined,
288                 h: Math.max(0, this.h - 18 - content.offsetTop) || undefined };
289     MochiKit.Style.setElementDimensions(content, dim);
290     MochiKit.Base.update(content, dim);
291     RapidContext.Util.resizeElements(content);
292 }

294 /**
295  * Resets the scroll offsets for all child elements in the dialog.
296  */
297 RapidContext.Widget.Dialog.prototype.resetScroll = function () {
298     function visitor(node) {
299         if (node.nodeType == 1) {
300             node.scrollTop = 0;
301             node.scrollLeft = 0;
302             return node.childNodes;
303         }
304     }
305     MochiKit.Base.nodeWalk(this, visitor);
306 }

308 /**
309  * Initiates a dialog move drag operation. This will install a mouse
310  * event handler on the parent document.
311  *
312  * @param {Event} evt the MochiKit.Signal.Event object
313  */
314 RapidContext.Widget.Dialog.prototype._handleMoveStart = function (evt) {
315     var pos = MochiKit.Style.getElementPosition(this.parentNode);
316     this._offsetPos = MochiKit.Style.getElementPosition(this, pos);
317     this._startPos = evt.mouse().page;
318     evt.stop();
319     MochiKit.Signal.connect(document, "onmousemove", this, "_handleMove");
320     MochiKit.Signal.connect(document, "onmouseup", this, "_stopDrag");
321 };
322
323 /**
324  * Handles a dialog move drag operation.
325  *
326  * @param {Event} evt the MochiKit.Signal.Event object
327  */
328 RapidContext.Widget.Dialog.prototype._handleMove = function (evt) {
329     var pos = evt.mouse().page;
330     this.moveTo(this._offsetPos.x + pos.x - this._startPos.x,
331                 this._offsetPos.y + pos.y - this._startPos.y);
332 };
333
334 /**
335  * Initiates a dialog resize drag operation. This will install a
336  * mouse event handler on the parent document.
337  *
338  * @param {Event} evt the MochiKit.Signal.Event object
339  */
340 RapidContext.Widget.Dialog.prototype._handleResizeStart = function (evt) {
341     this._offsetDim = MochiKit.Style.getElementDimensions(this);
342     this._startPos = evt.mouse().page;
343     evt.stop();
344     // TODO: correct handling of drag event, since IE seems to get
345     //       problems when mouse enters other HTML elements
346     MochiKit.Signal.connect(document, "onmousemove", this, "_handleResize");
347     MochiKit.Signal.connect(document, "onmousedown", function (evt) { evt.stop(); });
348     MochiKit.Signal.connect(document, "onmouseup", this, "_stopDrag");
349 };
350
351 /**
352  * Handles a dialog resize drag operation.
353  *
354  * @param {Event} evt the MochiKit.Signal.Event object
355  */
356 RapidContext.Widget.Dialog.prototype._handleResize = function (evt) {
357     var pos = evt.mouse().page;
358     this.resizeTo(this._offsetDim.w + pos.x - this._startPos.x,
359                   this._offsetDim.h + pos.y - this._startPos.y);
360 };
361
362 /**
363  * Stops a dialog resize or move drag operation.
364  *
365  * @param {Event} evt the MochiKit.Signal.Event object
366  */
367 RapidContext.Widget.Dialog.prototype._stopDrag = function (evt) {
368     MochiKit.Signal.disconnectAll(document, "onmousemove");
369     MochiKit.Signal.disconnectAll(document, "onmousedown");
370     MochiKit.Signal.disconnectAll(document, "onmouseup");
371 };
