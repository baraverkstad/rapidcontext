 21 /**
 22  * Creates a new dialog widget.
 23  *
 24  * @constructor
 25  * @param {Object} attrs the widget and node attributes
 26  * @param {String} [attrs.title] the dialog title, defaults to "Dialog"
 27  * @param {Boolean} [attrs.modal] the modal dialog flag, defaults to `false`
 28  * @param {Boolean} [attrs.system] the system dialog flag, implies modal,
 29  *            defaults to `false`
 30  * @param {Boolean} [attrs.center] the center dialog flag, defaults to `true`
 31  * @param {Boolean} [attrs.closeable] the closeable dialog flag, defaults to
 32  *            `true`
 33  * @param {Boolean} [attrs.resizeable] the resize dialog flag, defaults to
 34  *            `true`
 35  * @param {Boolean} [attrs.hidden] the hidden widget flag, defaults to `true`
 36  * @param {Object} [...] the child widgets or DOM nodes
 37  *
 38  * @return {Widget} the widget DOM node
 39  *
 40  * @class The dialog widget class. Used to provide a resizeable and
 41  *     moveable window within the current page. Internally it uses a
 42  *     number of `<div>` HTML elements.
 43  * @extends RapidContext.Widget
 44  *
 45  * @example {JavaScript}
 46  * var h1 = MochiKit.DOM.H1({}, "Hello, world!");
 47  * var attrs = { title: "Hello", modal: true };
 48  * var helloDialog = RapidContext.Widget.Dialog(attrs, h1);
 49  * RapidContext.Util.registerSizeConstraints(helloDialog, "200", "75");
 50  *
 51  * @example {User Interface XML}
 52  * <Dialog id="helloDialog" title="Hello" modal="true" w="200" h="75">
 53  *   <h1>Hello, world!</h1>
 54  * </Dialog>
 55  */
 56 RapidContext.Widget.Dialog = function (attrs/*, ... */) {
 57     var DIV = MochiKit.DOM.DIV;
 58     var title = DIV({ "class": "widgetDialogTitle", "data-dialog": "move" }, "Dialog");
 59     var close = RapidContext.Widget.Icon({
 60         "class": "widgetDialogClose fa fa-times",
 61         "title": "Close",
 62         "data-dialog": "close"
 63     });
 64     var resize = DIV({ "class": "widgetDialogResize", "data-dialog": "resize" });
 65     var content = DIV({ "class": "widgetDialogContent" });
 66     var o = DIV({}, title, close, resize, content);
 67     RapidContext.Widget._widgetMixin(o, RapidContext.Widget.Dialog);
 68     o.classList.add("widgetDialog");
 69     o.resizeContent = o._resizeContent;
 70     o._setHidden(true);
 71     o.setAttrs(Object.assign({ modal: false, system: false, center: true }, attrs));
 72     o.addAll(Array.from(arguments).slice(1));
 73     o.addEventListener("click", o._handleClick);
 74     o.addEventListener("mousedown", o._handleMouseDown);
 75     return o;
 76 };

138 /**
139  * Handles click events in the dialog. Will close the dialog if an element
140  * with `data-dialog="close"` attribute was clicked.
141  *
142  * @param {Event} evt the DOM Event object
143  */
144 RapidContext.Widget.Dialog.prototype._handleClick = function (evt) {
145     var action = $(evt.target).closest("[data-dialog]").data("dialog");
146     if (action == "close") {
147         this.hide();
148     }
149 };
150
151 /**
152  * Handles mouse down events in the dialog. Will start move or resize actions
153  * if an element with a `data-dialog="move"` or `data-dialog="resize"`
154  * attribute was clicked.
155  *
156  * @param {Event} evt the DOM Event object
157  */
158 RapidContext.Widget.Dialog.prototype._handleMouseDown = function (evt) {
159     var action = $(evt.target).closest("[data-dialog]").data("dialog");
160     if (action == "move" || action == "resize") {
161         var isDim = action == "resize";
162         var x = (isDim ? this.offsetWidth : this.offsetLeft) - evt.pageX;
163         var y = (isDim ? this.offsetHeight : this.offsetTop) - evt.pageY;
164         document._drag = { target: this, action: action, x: x, y: y };
165         evt.preventDefault();
166         document.addEventListener("mouseup", this._handleMouseUp);
167         document.addEventListener("mousemove", this._handleMouseMove);
168     }
169 };
170
171 /**
172  * Stops a dialog resize or move drag operation and removes event listeners.
173  * Note that this event handler is attached to the root `document`.
174  *
175  * @param {Event} evt the DOM Event object
176  */
177 RapidContext.Widget.Dialog.prototype._handleMouseUp = function (evt) {
178     var o = document._drag;
179     if (o && o.target) {
180         // FIXME: Use AbortSignal instead to disconnect
181         document.removeEventListener("mouseup", o.target._handleMouseUp);
182         document.removeEventListener("mousemove", o.target._handleMouseMove);
183     }
184     delete document._drag;
185 };
186
187 /**
188  * Handles a dialog move drag operation.
189  *
190  * @param {Event} evt the DOM Event object
191  */
192 RapidContext.Widget.Dialog.prototype._handleMouseMove = function (evt) {
193     var o = document._drag;
194     if (o && o.action == "move") {
195         o.target.moveTo(o.x + evt.pageX, o.y + evt.pageY);
196     } else if (o && o.action == "resize") {
197         o.target.resizeTo(o.x + evt.pageX, o.y + evt.pageY);
198     }
199 };

201 /**
202  * Updates the dialog or HTML DOM node attributes.
203  *
204  * @param {Object} attrs the widget and node attributes to set
205  * @param {String} [attrs.title] the dialog title
206  * @param {Boolean} [attrs.modal] the modal dialog flag
207  * @param {Boolean} [attrs.system] the system dialog flag, implies modal
208  * @param {Boolean} [attrs.center] the center dialog flag
209  * @param {Boolean} [attrs.closeable] the closeable dialog flag
210  * @param {Boolean} [attrs.resizeable] the resize dialog flag
211  * @param {Boolean} [attrs.hidden] the hidden widget flag
212  */
213 RapidContext.Widget.Dialog.prototype.setAttrs = function (attrs) {
214     attrs = Object.assign({}, attrs);
215     var locals = RapidContext.Util.mask(attrs, [
216         "title", "modal", "system", "center", "resizeable", "closeable", "hidden"
217     ]);
218     if (typeof(locals.title) != "undefined") {
219         MochiKit.DOM.replaceChildNodes(this.firstChild, locals.title);
220     }
221     if (typeof(locals.modal) != "undefined") {
222         this.modal = MochiKit.Base.bool(locals.modal);
223     }
224     if (typeof(locals.system) != "undefined") {
225         this.system = MochiKit.Base.bool(locals.system);
226     }
227     if (typeof(locals.center) != "undefined") {
228         this.center = MochiKit.Base.bool(locals.center);
229     }
230     if (typeof(locals.resizeable) != "undefined") {
231         var resize = this.childNodes[2];
232         $(resize).toggleClass("hidden", !MochiKit.Base.bool(locals.resizeable));
233     }
234     if (typeof(locals.closeable) != "undefined") {
235         var close = this.childNodes[1];
236         close.setAttrs({ hidden: !MochiKit.Base.bool(locals.closeable) });
237     }
238     if (typeof(locals.hidden) != "undefined") {
239         this._setHiddenDialog(locals.hidden);
240     }
241     this.__setAttrs(attrs);
242 };
243
244 /**
245  * Performs the changes corresponding to setting the `hidden`
246  * widget attribute for the Dialog widget.
247  *
248  * @param {Boolean} value the new attribute value
249  */
250 RapidContext.Widget.Dialog.prototype._setHiddenDialog = function (value) {
251     if (value) {
252         if (this._modalNode != null) {
253             RapidContext.Widget.destroyWidget(this._modalNode);
254             this._modalNode = null;
255         }
256         this.blurAll();
257         this._setHidden(true);
258         this._dispatch("hide");
259     } else {
260         if (this.parentNode == null) {
261             throw new Error("Cannot show Dialog widget without setting a parent DOM node");
262         }
263         if (this.modal || this.system) {
264             var attrs = { loading: false, message: "", style: { "z-index": "99" } };
265             if (this.system) {
266                 attrs.dark = true;
267             }
268             this._modalNode = RapidContext.Widget.Overlay(attrs);
269             this.parentNode.appendChild(this._modalNode);
270         }
271         this._setHidden(false);
272         this.resetScroll();
273         this._resizeContent();
274         this._dispatch("show");
275     }
276 };

278 /**
279  * Moves the dialog to the specified position (relative to the
280  * parent DOM node). The position will be restrained by the parent
281  * DOM node size.
282  *
283  * @param {Number} x the horizontal position (in pixels)
284  * @param {Number} y the vertical position (in pixels)
285  */
286 RapidContext.Widget.Dialog.prototype.moveTo = function (x, y) {
287     var max = {
288         x: this.parentNode.offsetWidth - this.offsetWidth - 2,
289         y: this.parentNode.offsetHeight - this.offsetHeight - 2
290     };
291     var pos = {
292         x: Math.round(Math.max(0, Math.min(x, max.x))),
293         y: Math.round(Math.max(0, Math.min(y, max.y)))
294     };
295     this.style.left = pos.x + "px";
296     this.style.top = pos.y + "px";
297     var el = this.lastChild;
298     el.style.maxWidth = (this.parentNode.offsetWidth - pos.x - el.offsetLeft - 2) + "px";
299     el.style.maxHeight = (this.parentNode.offsetHeight - pos.y - el.offsetTop - 2) + "px";
300     this.center = false;
301     this._dispatch("move", { detail: pos });
302 };
303
304 /**
305  * Moves the dialog to the apparent center (relative to the parent DOM
306  * node). The vertical position actually uses the golden ratio instead
307  * of the geometric center for improved visual alignment.
308  */
309 RapidContext.Widget.Dialog.prototype.moveToCenter = function () {
310     this.style.left = "0px";
311     this.style.top = "0px";
312     this.lastChild.style.maxWidth = "";
313     this.lastChild.style.maxHeight = "";
314     var x = (this.parentNode.offsetWidth - this.offsetWidth) / 2;
315     var y = (this.parentNode.offsetHeight - this.offsetHeight) / 2.618;
316     this.moveTo(x, y);
317     this.center = true;
318 };

320 /**
321  * Resizes the dialog to the specified size (in pixels). The size
322  * will be restrained by the parent DOM node size.
323  *
324  * @param {Number} w the width (in pixels)
325  * @param {Number} h the height (in pixels)
326  *
327  * @return {Dimensions} an object with "w" and "h" properties for the
328  *         actual size used
329  */
330 RapidContext.Widget.Dialog.prototype.resizeTo = function (w, h) {
331     var max = {
332         w: this.parentNode.offsetWidth - this.offsetLeft - 2,
333         h: this.parentNode.offsetHeight - this.offsetTop - 2
334     };
335     var dim = {
336         w: Math.round(Math.max(150, Math.min(w, max.w))),
337         h: Math.round(Math.max(100, Math.min(h, max.h)))
338     };
339     this.style.width = dim.w + "px";
340     this.style.height = dim.h + "px";
341     delete this.sizeConstraints; // FIXME: Remove with RapidContext.Util.registerSizeConstraints
342     this.center = false;
343     this._resizeContent();
344     this._dispatch("resize", { detail: dim });
345     return dim;
346 };
347
348 /**
349  * Resizes the dialog to the optimal size for the current content.
350  * Note that the size reported by the content may vary depending on
351  * if it has already been displayed, is absolutely positioned, etc.
352  * The size will be restrained by the parent DOM node size.
353  *
354  * @return {Dimensions} an object with "w" and "h" properties for the
355  *         actual size used
356  */
357 RapidContext.Widget.Dialog.prototype.resizeToContent = function () {
358     var el = this.lastChild;
359     var w = Math.max(el.scrollWidth, el.offsetWidth) + 2;
360     var h = Math.max(el.scrollHeight, el.offsetHeight) + el.offsetTop + 2;
361     return this.resizeTo(w, h);
362 };
363
364 /**
365  * Called when dialog content should be resized.
366  */
367 RapidContext.Widget.Dialog.prototype._resizeContent = function () {
368     if (!this.isHidden()) {
369         RapidContext.Util.resizeElements(this.lastChild);
370         if (this.center) {
371             this.moveToCenter();
372         }
373     }
374 };
375
376 /**
377  * Resets the scroll offsets for all child elements in the dialog.
378  */
379 RapidContext.Widget.Dialog.prototype.resetScroll = function () {
380     function scrollReset(el) {
381         el.scrollTop = 0;
382         el.scrollLeft = 0;
383     }
384     Array.from(this.querySelectorAll("*")).forEach(scrollReset);
385 };
