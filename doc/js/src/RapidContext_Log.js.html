 15 /**
 16  * @name RapidContext.Log
 17  * @namespace Provides a logging service for debugging apps and server calls.
 18  *
 19  * All log messages are filtered by log level and either discarded or
 20  * stored to an internal array. Log messages on the error or warning levels
 21  * are also sent to the server for remote logging.
 22  *
 23  * This module replaces the built-in `console.error()`, `console.warn()`,
 24  * `console.info()`, `console.log()` and `console.debug()` functions with its
 25  * own versions, passing through the log messages if not filtered.
 26  */
 27 (function (window) {
 28
 29     // The original console logger functions
 30     var backup = {};
 31
 32     // The current log state
 33     var state = {
 34         count: 0,
 35         level: 3,
 36         context: null,
 37         history: [],
 38         publish: {
 39             last: 0,
 40             timer: null
 41         }
 42     };
 43
 44     /**
 45      * Modifies the `console` object for logging. This replaces the default
 46      * `console.error`, `console.warn`, `console.info`, `console.log` and
 47      * `console.debug` functions.
 48      */
 49     function _init() {
 50         function overwrite(obj, key, fn) {
 51             if (obj[key] !== fn) {
 52                 backup[key] = obj[key] || function () {};
 53                 obj[key] = fn;
 54             }
 55         }
 56         if (typeof(window.console) !== "object") {
 57             window.console = {};
 58         }
 59         overwrite(window.console, "error", error);
 60         overwrite(window.console, "warn", warn);
 61         overwrite(window.console, "info", info);
 62         overwrite(window.console, "log", debug);
 63         overwrite(window.console, "debug", debug);
 64         overwrite(window, "onerror", _onerror);
 65     }
 66
 67     /**
 68      * Clears the log console and the array of stored messages.
 69      *
 70      * @see RapidContext.Log.history
 71      *
 72      * @memberof RapidContext.Log
 73      */
 74     function clear() {
 75         state.history = [];
 76         if (window.console && typeof(window.console.clear) === "function") {
 77             window.console.clear();
 78         }
 79     }
 80
 81     /**
 82      * Returns the history of filtered log entries. Each log entry is a plain
 83      * object with properties -- `id`, `time`, `level`, `context`, `message`
 84      * and `data`.
 85      *
 86      * @return {Array} the array of log entries
 87      *
 88      * @memberof RapidContext.Log
 89      */
 90     function history() {
 91         return state.history.slice(0);
 92     }
 93
 94     /**
 95      * Returns and optionally sets the current log level. The supported log
 96      * level values are -- "none", "error", "warn", "info", "log" and "all".
 97      *
 98      * @param {String} [value] the new log level
 99      * @return {String} the current log level
100      *
101      * @memberof RapidContext.Log
102      */
103     function level(value) {
104         if (typeof(value) !== "undefined") {
105             if (value === 0 || /^none/i.test(value)) {
106                 state.level = 0;
107             } else if (value === 1 || /^err/i.test(value)) {
108                 state.level = 1;
109             } else if (value === 2 || /^warn/i.test(value)) {
110                 state.level = 2;
111             } else if (value === 3 || /^info/i.test(value)) {
112                 state.level = 3;
113             } else if (value === 4 || /^(log|debug|trace)/i.test(value)) {
114                 state.level = 4;
115             } else {
116                 state.level = 5;
117             }
118         }
119         if (state.level <= 0) {
120             return "none";
121         } else if (state.level <= 1) {
122             return "error";
123         } else if (state.level <= 2) {
124             return "warn";
125         } else if (state.level <= 3) {
126             return "info";
127         } else if (state.level <= 4) {
128             return "log";
129         } else {
130             return "all";
131         }
132     }
133
134     /**
135      * Returns and optionally sets the current log context. The log context is
136      * used to tag all subsequent log messages until the context is removed or
137      * modified.
138      *
139      * @param {String} [value] the new log context, or null to clear
140      *
141      * @return {String} the current log context, or null for none
142      *
143      * @example
144      * RapidContext.Log.context('mybutton.onclick');
145      * ...
146      * console.warn('unsupported xyz value:', value);
147      * ...
148      * RapidContext.Log.context(null);
149      *
150      * @memberof RapidContext.Log
151      */
152     function context(value) {
153         if (typeof(value) !== "undefined") {
154             state.context = value;
155             if (!value) {
156                 // Clear group immediately, but create new group on first log
157                 _group();
158             }
159         }
160         return state.context;
161     }
162
163     /**
164      * Logs an error message with optional data. Also available as the global
165      * `console.error()` function.
166      *
167      * @param {String} msg the log message
168      * @param {Object} [...] the additional log data or messages
169      *
170      * @example
171      * console.error('failed to initialize module');
172      *
173      * @memberof RapidContext.Log
174      */
175     function error(msg/**, ...*/) {
176         if (state.level >= 1) {
177             var args = Array.prototype.slice.apply(arguments);
178             _log("error", args);
179             _store("error", args.map(stringify));
180         }
181     }
182
183     /**
184      * Logs a warning message with optional data. Also available as the global
185      * `console.warn()` function.
186      *
187      * @param {String} msg the log message
188      * @param {Object} [...] the additional log data or messages
189      *
190      * @example
191      * console.warn('missing "data" attribute on document root:', document.body);
192      *
193      * @memberof RapidContext.Log
194      */
195     function warn(msg/**, ...*/) {
196         if (state.level >= 2) {
197             var args = Array.prototype.slice.apply(arguments);
198             _log("warn", args);
199             _store("warn", args.map(stringify));
200         }
201     }
202
203     /**
204      * Logs an information message with optional data. Also available as the
205      * global `console.info()` function.
206      *
207      * @param {String} msg the log message
208      * @param {Object} [...] the additional log data or messages
209      *
210      * @example
211      * console.info('authorization failed, user not logged in');
212      *
213      * @memberof RapidContext.Log
214      */
215     function info(msg/**, ...*/) {
216         if (state.level >= 3) {
217             var args = Array.prototype.slice.apply(arguments);
218             _log("info", args);
219             _store("info", args.map(stringify));
220         }
221     }
222
223     /**
224      * Logs a debug message with optional data. Also available as the global
225      * `console.log()` and `console.debug()` functions.
226      *
227      * @param {String} msg the log message
228      * @param {Object} [...] the additional log data or messages
229      *
230      * @example
231      * console.log('init AJAX call to URL:', url);
232      * ...
233      * console.log('done AJAX call to URL:', url, responseCode);
234      *
235      * @memberof RapidContext.Log
236      */
237     function debug(msg/**, ...*/) {
238         if (state.level >= 4) {
239             var args = Array.prototype.slice.apply(arguments);
240             _log("log", args);
241             _store("log", args.map(stringify));
242         }
243     }
244
245     /**
246      * Handles window.onerror events (global uncaught errors).
247      */
248     function _onerror(msg, url, line, col, err) {
249         if (err instanceof Error && err.stack) {
250             error(msg || "Uncaught error", err);
251         } else {
252             var location = [url, line, col].filter(Boolean).join(":");
253             error(msg || "Uncaught error", location, err);
254         }
255         return true;
256     }
257
258     /**
259      * Logs a message to one of the console loggers.
260      *
261      * @param {String} level the log level (i.e. function 'error', 'warn'...)
262      * @param {Array} args the log message & data (as raw objects)
263      */
264     function _log(level, args) {
265         var logger = backup[level];
266         if (typeof(logger) === "function") {
267             _group(state.context);
268             logger.apply(window.console, args);
269         }
270     }
271
272     /**
273      * Calls the `console.group` and `console.groupEnd` functions (if
274      * they exist) to change the group label (if needed).
275      *
276      * @param {String} label the log context label
277      */
278     function _group(label) {
279         var console = window.console;
280         if (console._group !== label) {
281             if (console._group) {
282                 delete console._group;
283                 if (typeof(console.groupEnd) === "function") {
284                     console.groupEnd();
285                 }
286             }
287             if (label) {
288                 console._group = label;
289                 if (typeof(console.group) === "function") {
290                     console.group(label + ":");
291                 }
292             }
293         }
294     }
295
296     /**
297      * Stores a log message to the history.
298      *
299      * @param {String} level the message log level
300      * @param {Array} args the log message arguments (as strings)
301      */
302     function _store(level, args) {
303         state.history.push({
304             id: ++state.count,
305             time: new Date(),
306             level: level,
307             context: state.context,
308             message: args[0],
309             data: (args.length > 1) ? args.slice(1).join("\n") : null
310         });
311         state.history.splice(0, Math.max(0, state.history.length - 100));
312         if (!state.publish.timer) {
313             state.publish.timer = setTimeout(_publish, 100);
314         }
315     }
316
317     /**
318      * Publishes stored events (above a threshold level) to the server.
319      */
320     function _publish() {
321         var events = [];
322         var lastId = 0;
323         for (var i = 0; i < state.history.length; i++) {
324             var evt = state.history[i];
325             lastId = evt.id;
326             if (evt.id > state.publish.last && /error|warn/.test(evt.level)) {
327                 events.push(evt);
328             }
329         }
330         if (events.length <= 0) {
331             state.publish.last = lastId;
332             state.publish.timer = null;
333             return;
334         }
335         var opts = {
336             method: "POST",
337             contentType: "application/json",
338             data: JSON.stringify(events)
339         };
340         var d = $.ajax("rapidcontext/log", opts);
341         d.then(function (res) {
342             state.publish.last = lastId;
343             state.publish.timer = setTimeout(_publish, 10000);
344         });
345         d.fail(function (err) {
346             info("failed to upload errorlog", err);
347             state.publish.timer = setTimeout(_publish, 10000);
348         });
349     }

351     /**
352      * Creates a string representation (suitable for logging) for any value or
353      * object. The returned string is similar to a JSON representation of the
354      * value, but may be simplified for increased readability.
355      *
356      * @param {Object} val the value or object to convert
357      *
358      * @return {String} the string representation of the value
359      *
360      * @memberof RapidContext.Log
361      */
362     function stringify(val) {
363         var isObject = Object.prototype.toString.call(val) === "[object Object]";
364         var isArray = val instanceof Array;
365         var isSerializable = val && typeof(val.toJSON) === "function";
366         if (val && (isObject || isArray || isSerializable)) {
367             try {
368                 return JSON.stringify(val, null, 2);
369             } catch (e) {
370                 return String(val);
371             }
372         } else if (val instanceof Error) {
373             var parts = [val.toString()];
374             if (val.stack) {
375                 var stack = String(val.stack).trim()
376                     .replace(val.toString() + "\n", "")
377                     .replace(/https?:.*\//g, "")
378                     .replace(/\?[^:\s]+:/g, ":")
379                     .replace(/@(.*)/mg, " ($1)")
380                     .replace(/^(\s+at\s+)?/mg, "    at ");
381                 parts.push("\n", stack);
382             }
383             return parts.join("");
384         } else if (val && (val.nodeType === 1 || val.nodeType === 9)) {
385             var el = val.documentElement || val.document || val;
386             var xml = el.outerHTML || el.outerXML || el.xml || String(el);
387             var children = el.childNodes && el.childNodes.length;
388             return xml.replace(/>[^<]*/, children ? ">..." : "/>");
389         } else {
390             return String(val);
391         }
392     }

395     var RapidContext = window.RapidContext || (window.RapidContext = {});
396     var module = RapidContext.Log || (RapidContext.Log = {});

398     // Export namespace symbols
399     module.history = history;
400     module.clear = clear;
401     module.level = level;
402     module.context = context;
403     module.error = error;
404     module.warn = warn;
405     module.info = info;
406     module.log = module.debug = module.trace = debug;
407     module.stringify = stringify;
408
409     // Install console loggers and global error handler
410     _init();

412 })(this);
