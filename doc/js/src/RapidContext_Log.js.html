 15 /**
 16  * @name RapidContext.Log
 17  * @namespace Provides a logging service for debugging apps and server calls.
 18  *
 19  * All log messages are filtered by log level and either discarded or
 20  * stored to an internal array. This makes it possible to either show these
 21  * in a developer console or to send them to a server for remote storage.
 22  *
 23  * This module also replaces the built-in `console.error()`, `console.warn()`,
 24  * `console.info()` and `console.log()` functions with its own versions,
 25  * passing through the filtered log messages to the original functions. For
 26  * browsers without a `console` object, a minimal version is provided.
 27  */
 28 (function (window, undefined) {
 29
 30     // Function-level strict mode
 31     "use strict";
 32
 33     // Detect MSIE browser to force simplified console object logging
 34     var isMSIE = /MSIE/.test(window.navigator.userAgent);
 35
 36     // The original console log functions
 37     var consoleBackup = {};
 38
 39     // The current log level
 40     var logLevel = 3;
 41
 42     // The current log message count
 43     var logCount = 0;
 44
 45     // The current log history
 46     var logHistory = [];
 47
 48     // The current log context
 49     var logContext = null;
 50
 51     // The last published log event
 52     var publishedId = -1;
 53
 54     // The log levels filterer out for publishing
 55     var publishLevels = /error|warn/;
 56
 57     // The currently running publisher timer
 58     var publisherTimer = null;
 59
 60     /**
 61      * Clears the log console and the array of stored messages.
 62      *
 63      * @see RapidContext.Log.history
 64      *
 65      * @memberof RapidContext.Log
 66      */
 67     function clear() {
 68         logHistory = [];
 69         if (window.console && window.console.clear) {
 70             window.console.clear();
 71         }
 72     }

 74     /**
 75      * Returns the history of filtered log entries. Each log entry is a plain
 76      * object with properties -- `id`, `time`, `level`, `context`, `message`
 77      * and `data`.
 78      *
 79      * @return {Array} the array of log entries
 80      *
 81      * @memberof RapidContext.Log
 82      */
 83     function history() {
 84         return logHistory.slice(0);
 85     }
 86
 87     /**
 88      * Returns and optionally sets the current log level. The supported log
 89      * level values are -- "none", "error", "warn", "info", "log" and "all".
 90      *
 91      * @param {String} [value] the new log level
 92      * @return {String} the current log level
 93      *
 94      * @memberof RapidContext.Log
 95      */
 96     function level(value) {
 97         if (typeof(value) !== "undefined") {
 98             if (value === 0 || /^none/i.test(value)) {
 99                 logLevel = 0;
100             } else if (value === 1 || /^err/i.test(value)) {
101                 logLevel = 1;
102             } else if (value === 2 || /^warn/i.test(value)) {
103                 logLevel = 2;
104             } else if (value === 3 || /^info/i.test(value)) {
105                 logLevel = 3;
106             } else if (value === 4 || /^(log|debug|trace)/i.test(value)) {
107                 logLevel = 4;
108             } else {
109                 logLevel = 5;
110             }
111         }
112         if (logLevel <= 0) {
113             return "none";
114         } else if (logLevel <= 1) {
115             return "error";
116         } else if (logLevel <= 2) {
117             return "warn";
118         } else if (logLevel <= 3) {
119             return "info";
120         } else if (logLevel <= 4) {
121             return "log";
122         } else {
123             return "all";
124         }
125     }

127     /**
128      * Returns and optionally sets the current log context. The log context is
129      * used to tag all subsequent log messages until the context is removed or
130      * modified.
131      *
132      * @param {String} [value] the new log context, or null to clear
133      *
134      * @return {String} the current log context, or null for none
135      *
136      * @example
137      * RapidContext.Log.context('mybutton.onclick');
138      * ...
139      * console.warn('unsupported xyz value:', value);
140      * ...
141      * RapidContext.Log.context(null);
142      *
143      * @memberof RapidContext.Log
144      */
145     function context(value) {
146         if (typeof(value) !== "undefined") {
147             logContext = value;
148             if (value == null) {
149                 _consoleContext();
150             }
151         }
152         return logContext;
153     }

155     /**
156      * Logs an error message with optional data. Also available as the global
157      * `console.error()` function.
158      *
159      * @param {String} msg the log message
160      * @param {Object} [...] the additional log data or messages
161      *
162      * @example
163      * console.error('failed to initialize module');
164      *
165      * @name error
166      * @function
167      * @memberof RapidContext.Log
168      */
169     function logError(msg/**, ...*/) {
170         if (logLevel >= 1) {
171             var args = Array.prototype.slice.apply(arguments);
172             var strs = args.map(stringify);
173             _consoleLog("error", args, strs);
174             _store("error", strs);
175         }
176     }

178     /**
179      * Logs a warning message with optional data. Also available as the global
180      * `console.warn()` function.
181      *
182      * @param {String} msg the log message
183      * @param {Object} [...] the additional log data or messages
184      *
185      * @example
186      * console.warn('missing "data" attribute on document root:', document.body);
187      *
188      * @name warn
189      * @function
190      * @memberof RapidContext.Log
191      */
192     function logWarn(msg/**, ...*/) {
193         if (logLevel >= 2) {
194             var args = Array.prototype.slice.apply(arguments);
195             var strs = args.map(stringify);
196             _consoleLog("warn", args, strs);
197             _store("warn", strs);
198         }
199     }

201     /**
202      * Logs an information message with optional data. Also available as the
203      * global `console.info()` function.
204      *
205      * @param {String} msg the log message
206      * @param {Object} [...] the additional log data or messages
207      *
208      * @example
209      * console.info('authorization failed, user not logged in');
210      *
211      * @name info
212      * @function
213      * @memberof RapidContext.Log
214      */
215     function logInfo(msg/**, ...*/) {
216         if (logLevel >= 3) {
217             var args = Array.prototype.slice.apply(arguments);
218             var strs = args.map(stringify);
219             _consoleLog("info", args, strs);
220             _store("info", strs);
221         }
222     }

224     /**
225      * Logs a debug message with optional data. Also available as the global
226      * `console.log()` and `console.debug()` functions.
227      *
228      * @param {String} msg the log message
229      * @param {Object} [...] the additional log data or messages
230      *
231      * @example
232      * console.log('init AJAX call to URL:', url);
233      * ...
234      * console.log('done AJAX call to URL:', url, responseCode);
235      *
236      * @name debug
237      * @function
238      * @memberof RapidContext.Log
239      */
240     function logDebug(msg/**, ...*/) {
241         if (logLevel >= 4) {
242             var args = Array.prototype.slice.apply(arguments);
243             var strs = args.map(stringify);
244             _consoleLog("log", args, strs);
245             _store("log", strs);
246         }
247     }

249     /**
250      * Logs a number of browser meta-data parameters (INFO level).
251      *
252      * @memberof RapidContext.Log
253      */
254     function logBrowserInfo() {
255         function copy(obj, keys) {
256             var res = {};
257             for (var i = 0; i < keys.length; i++) {
258                 var k = keys[i];
259                 res[k] = obj[k];
260             }
261             return res;
262         }
263         context('Browser')
264         logInfo("location.href", location.href);
265         logInfo("navigator.userAgent", navigator.userAgent);
266         logInfo("navigator.language", navigator.language);
267         logInfo("screen", copy(screen, ["width", "height", "colorDepth"]));
268         logInfo("window", copy(window, ["innerWidth", "innerHeight", "devicePixelRatio"]));
269         logInfo("document.cookie", document.cookie);
270         context(null);
271     }

273     /**
274      * Handles window.onerror events (global uncaught errors).
275      */
276     function _windowErrorHandler(msg, src, line, col, error) {
277         var location = "unknown source";
278         if (error && error.stack) {
279             location = error.stack;
280         } else if (src && src !== "undefined") {
281             location = [src, line || 0, col || 0].join(':');
282         }
283         logError(msg, location);
284         return true;
285     }

287     /**
288      * Modifies the `console` object for logging. This replaces the default
289      * `console.error`, `console.warn`, `console.info`, `console.log` and
290      * `console.debug` functions (if not previously modified).
291      */
292     function _consoleSetup() {
293         function overwrite(obj, key, value, backup) {
294             if (obj[key] !== value) {
295                 backup[key] = obj[key];
296                 obj[key] = value;
297             }
298         }
299         window.console || (window.console = {});
300         overwrite(window.console, 'error', logError, consoleBackup);
301         overwrite(window.console, 'warn', logWarn, consoleBackup);
302         overwrite(window.console, 'info', logInfo, consoleBackup);
303         overwrite(window.console, 'log', logDebug, consoleBackup);
304         overwrite(window.console, 'debug', logDebug, consoleBackup);
305     }
306
307     /**
308      * Calls the `console.group` and `console.groupEnd` functions (if
309      * they exist) to adjust for the current log context.
310      */
311     function _consoleContext() {
312         if (window.console._group && window.console._group !== logContext) {
313             window.console._group = null;
314             window.console.groupEnd && window.console.groupEnd();
315         }
316         if (logContext && window.console._group !== logContext) {
317             window.console._group = logContext;
318             window.console.group && window.console.group(logContext + ":");
319         }
320     }

322     /**
323      * Logs a message to one of the console loggers.
324      *
325      * @param {String} loggerName the console logger name ('error', 'warn'...)
326      * @param {Array} args the log message & data (as raw objects)
327      * @param {Array} strs the log message & data (as strings)
328      */
329     function _consoleLog(loggerName, args, strs) {
330         _consoleSetup();
331         _consoleContext();
332         var logger = consoleBackup[loggerName];
333         if (typeof(logger) === 'function' && typeof(logger.apply) === 'function') {
334             logger.apply(window.console, args);
335         } else if (logger && isMSIE) {
336             var ctx = (logContext ? logContext + ": " : "");
337             logger(ctx + strs.join(", "));
338         }
339     }

341     /**
342      * Stores a log message to the history.
343      *
344      * @param {String} level the message log level
345      * @param {Array} args the log message arguments (as strings)
346      *
347      * @private
348      */
349     function _store(level, args) {
350         var data = (args.length > 1) ? args.slice(1).join("\n") : null;
351         logHistory.push({ id: ++logCount, time: new Date(), level: level,
352                           context: logContext, message: args[0], data: data });
353         while (logHistory.length > 100) {
354             logHistory.shift();
355         }
356         if (publisherTimer == null) {
357             publisherTimer = setTimeout(_publishEvents, 100);
358         }
359     }
360
361     /**
362      * Publishes stored events (above a threshold level) to the server.
363      *
364      * @private
365      */
366     function _publishEvents() {
367         var events = [];
368         var lastId = 0;
369         for (var i = 0; i < logHistory.length; i++) {
370             var evt = logHistory[i];
371             lastId = evt.id;
372             if (evt.id > publishedId && publishLevels.test(evt.level)) {
373                 events.push(evt);
374             }
375         }
376         if (events.length <= 0) {
377             publishedId = lastId;
378             publisherTimer = null;
379             return;
380         }
381         var opts = {
382             method: "POST",
383             contentType: "application/json",
384             data: JSON.stringify(events)
385         };
386         var d = $.ajax("rapidcontext/log", opts);
387         d.then(function (res) {
388             publishedId = lastId;
389             publisherTimer = setTimeout(_publishEvents, 10000);
390         });
391         d.fail(function (err) {
392             logInfo("failed to upload errorlog", err);
393             publisherTimer = setTimeout(_publishEvents, 10000);
394         });
395     }

397     /**
398      * Creates a string representation (suitable for logging) for any value or
399      * object. The returned string is similar to a JSON representation of the
400      * value, but may be simplified for increased readability.
401      *
402      * @param {Object} o the value or object to convert
403      *
404      * @return {String} the string representation of the value
405      *
406      * @memberof RapidContext.Log
407      */
408     function stringify(o) {
409         return _stringify(o, 0);
410     }
411
412     /**
413      * Internal stringify implementation that tracks of object graph
414      * depth.
415      *
416      * @param {Object} o the value or object to convert
417      * @param {Number} depth the current object depth (max is 4)
418      *
419      * @return {Array} the array of string representations
420      *
421      * @private
422      */
423     function _stringify(o, depth) {
424         var type = typeof(o);
425         if (o == null || type == "boolean" || type == "number") {
426             return "" + o;
427         } else if (type == "string") {
428             return (depth > 0) ? JSON.stringify(o) : o;
429         } else if (type == "function") {
430             var name = o.name || o.displayName || "";
431             var signature = o.toString().replace(/\)[\s\S]*/, ")");
432             if (/function\s+\(/.test(signature) && name) {
433                 signature = signature.replace(/function\s+/, "function " + name);
434             } else if (!(/function/.test(signature))) {
435                 signature = "function ()";
436             }
437             return signature + " {...}";
438         } else if (o instanceof Error) {
439             var arr = [o.toString()];
440             if (typeof(o.number) === "number" && o.number != 0) {
441                 arr.push(" (code: ", o.number & 0xFFFF);
442                 arr.push(", facility: ", o.number >> 16 & 0x1FFF, ")");
443             }
444             if (o.stack) {
445                 var stack = ("" + o.stack).trim();
446                 stack = stack.replace(o.toString() + "\n", "");
447                 stack = stack.replace(/https?:.*\//g, "");
448                 stack = stack.replace(/\?[^:\s]+:/g, ":");
449                 stack = stack.replace(/@(.*)/mg, " ($1)");
450                 stack = stack.replace(/^(\s+at\s+)?/mg, "    at ");
451                 arr.push("\n", stack);
452             }
453             return arr.join("");
454         } else if (o.nodeType === 1 || o.nodeType === 9) {
455             o = o.documentElement || o.document || o;
456             var xml = o.outerHTML || o.outerXML || o.xml || "" + o;
457             if (o.childNodes && o.childNodes.length) {
458                 return xml.replace(/>[\s\S]*</, ">...<");
459             } else {
460                 return xml.replace(/>[\s\S]*/, " />");
461             }
462         } else if (depth > 3) {
463             return "...";
464         } else if (typeof(o.length) == "number") {
465             var arr = [];
466             for (var i = 0; i < o.length; i++) {
467                 if (i > 20) {
468                     arr.push("...");
469                     break;
470                 }
471                 arr.push(_stringify(o[i], depth + 1));
472             }
473             return "[" + arr.join(", ") + "]";
474         } else {
475             var arr = [];
476             for (var k in o) {
477                 if (arr.length > 20) {
478                     arr.push("...");
479                     break;
480                 }
481                 arr.push(k + ": " + _stringify(o[k], depth + 1));
482             }
483             return "{" + arr.join(", ") + "}";
484         }
485     }
