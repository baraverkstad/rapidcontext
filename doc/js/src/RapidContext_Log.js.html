 15 /**
 16  * @name RapidContext.Log
 17  * @namespace Provides a logging service for debugging apps and server calls.
 18  *
 19  * All log messages are filtered by log level and either discarded or
 20  * stored to an internal array. Log messages on the error or warning levels
 21  * are also sent to the server for remote logging.
 22  *
 23  * This module replaces the built-in `console.error()`, `console.warn()`,
 24  * `console.info()`, `console.log()` and `console.debug()` functions with its
 25  * own versions, passing through the log messages if not filtered.
 26  */
 27 (function (window) {
 28
 29     // The original console logger functions
 30     var backup = {};
 31
 32     // The current log state
 33     var state = {
 34         count: 0,
 35         level: 3,
 36         context: null,
 37         history: [],
 38         publish: {
 39             last: 0,
 40             timer: null
 41         }
 42     };
 43
 44     // The configuration settings
 45     var config = {
 46         interval: 10000,
 47         url: "rapidcontext/log",
 48         filter: null,
 49         publisher: null
 50     };
 51
 52     /**
 53      * Initializes and configures the logging module. Will modify the `console`
 54      * object for logging. This replaces the default `console.error`,
 55      * `console.warn`, `console.info`, `console.log` and `console.debug`
 56      * functions. Safe to call multiple times to change or update config.
 57      *
 58      * @param {Object} [opts] the log configuration options, or null
 59      * @config {Number} [interval] the publish delay in millis, default is 10s
 60      * @config {String} [url] the publish URL endpoint
 61      * @config {Function} [filter] the event filter, returns a boolean
 62      * @config {Function} [publisher] the event publisher, returns a Promise
 63      *
 64      * @memberof RapidContext.Log
 65      */
 66     function init(opts) {
 67         function overwrite(obj, key, fn) {
 68             if (obj[key] !== fn) {
 69                 backup[key] = obj[key] || function () {};
 70                 obj[key] = fn;
 71             }
 72         }
 73         if (typeof(window.console) !== "object") {
 74             window.console = {};
 75         }
 76         overwrite(window.console, "error", error);
 77         overwrite(window.console, "warn", warn);
 78         overwrite(window.console, "info", info);
 79         overwrite(window.console, "log", debug);
 80         overwrite(window.console, "debug", debug);
 81         overwrite(window, "onerror", _onerror);
 82         opts = opts || {};
 83         config.interval = opts.interval || config.interval;
 84         config.url = opts.url || config.url;
 85         config.filter = opts.filter || _isErrorOrWarning;
 86         config.publisher = opts.publisher || _publish;
 87     }
 88
 89     /**
 90      * Clears the log console and the array of stored messages.
 91      *
 92      * @see RapidContext.Log.history
 93      *
 94      * @memberof RapidContext.Log
 95      */
 96     function clear() {
 97         state.history = [];
 98         if (window.console && typeof(window.console.clear) === "function") {
 99             window.console.clear();
100         }
101     }
102
103     /**
104      * Returns the history of filtered log entries. Each log entry is a plain
105      * object with properties -- `id`, `time`, `level`, `context`, `message`
106      * and `data`.
107      *
108      * @return {Array} the array of log entries
109      *
110      * @memberof RapidContext.Log
111      */
112     function history() {
113         return state.history.slice(0);
114     }
115
116     /**
117      * Returns and optionally sets the current log level. The supported log
118      * level values are -- "none", "error", "warn", "info", "log" and "all".
119      *
120      * @param {String} [value] the new log level
121      * @return {String} the current log level
122      *
123      * @memberof RapidContext.Log
124      */
125     function level(value) {
126         if (typeof(value) !== "undefined") {
127             if (value === 0 || /^none/i.test(value)) {
128                 state.level = 0;
129             } else if (value === 1 || /^err/i.test(value)) {
130                 state.level = 1;
131             } else if (value === 2 || /^warn/i.test(value)) {
132                 state.level = 2;
133             } else if (value === 3 || /^info/i.test(value)) {
134                 state.level = 3;
135             } else if (value === 4 || /^(log|debug|trace)/i.test(value)) {
136                 state.level = 4;
137             } else {
138                 state.level = 5;
139             }
140         }
141         if (state.level <= 0) {
142             return "none";
143         } else if (state.level <= 1) {
144             return "error";
145         } else if (state.level <= 2) {
146             return "warn";
147         } else if (state.level <= 3) {
148             return "info";
149         } else if (state.level <= 4) {
150             return "log";
151         } else {
152             return "all";
153         }
154     }
155
156     /**
157      * Returns and optionally sets the current log context. The log context is
158      * used to tag all subsequent log messages until the context is removed or
159      * modified.
160      *
161      * @param {String} [value] the new log context, or null to clear
162      *
163      * @return {String} the current log context, or null for none
164      *
165      * @example
166      * RapidContext.Log.context('mybutton.onclick');
167      * ...
168      * console.warn('unsupported xyz value:', value);
169      * ...
170      * RapidContext.Log.context(null);
171      *
172      * @memberof RapidContext.Log
173      */
174     function context(value) {
175         if (typeof(value) !== "undefined") {
176             state.context = value;
177             if (!value) {
178                 // Clear group immediately, but create new group on first log
179                 _group();
180             }
181         }
182         return state.context;
183     }
184
185     /**
186      * Logs an error message with optional data. Also available as the global
187      * `console.error()` function.
188      *
189      * @param {String} msg the log message
190      * @param {Object} [...] the additional log data or messages
191      *
192      * @example
193      * console.error('failed to initialize module');
194      *
195      * @memberof RapidContext.Log
196      */
197     function error(msg/**, ...*/) {
198         if (state.level >= 1) {
199             var args = Array.prototype.slice.call(arguments);
200             _log("error", args);
201             _store("error", args.map(stringify));
202         }
203     }
204
205     /**
206      * Logs a warning message with optional data. Also available as the global
207      * `console.warn()` function.
208      *
209      * @param {String} msg the log message
210      * @param {Object} [...] the additional log data or messages
211      *
212      * @example
213      * console.warn('missing "data" attribute on document root:', document.body);
214      *
215      * @memberof RapidContext.Log
216      */
217     function warn(msg/**, ...*/) {
218         if (state.level >= 2) {
219             var args = Array.prototype.slice.call(arguments);
220             _log("warn", args);
221             _store("warn", args.map(stringify));
222         }
223     }
224
225     /**
226      * Logs an information message with optional data. Also available as the
227      * global `console.info()` function.
228      *
229      * @param {String} msg the log message
230      * @param {Object} [...] the additional log data or messages
231      *
232      * @example
233      * console.info('authorization failed, user not logged in');
234      *
235      * @memberof RapidContext.Log
236      */
237     function info(msg/**, ...*/) {
238         if (state.level >= 3) {
239             var args = Array.prototype.slice.call(arguments);
240             _log("info", args);
241             _store("info", args.map(stringify));
242         }
243     }
244
245     /**
246      * Logs a debug message with optional data. Also available as the global
247      * `console.log()` and `console.debug()` functions.
248      *
249      * @param {String} msg the log message
250      * @param {Object} [...] the additional log data or messages
251      *
252      * @example
253      * console.log('init AJAX call to URL:', url);
254      * ...
255      * console.log('done AJAX call to URL:', url, responseCode);
256      *
257      * @memberof RapidContext.Log
258      */
259     function debug(msg/**, ...*/) {
260         if (state.level >= 4) {
261             var args = Array.prototype.slice.call(arguments);
262             _log("log", args);
263             _store("log", args.map(stringify));
264         }
265     }
266
267     /**
268      * Handles window.onerror events (global uncaught errors).
269      */
270     function _onerror(msg, url, line, col, err) {
271         if (err instanceof Error && err.stack) {
272             error(msg || "Uncaught error", err);
273         } else {
274             var location = [url, line, col].filter(Boolean).join(":");
275             error(msg || "Uncaught error", location, err);
276         }
277         return true;
278     }
279
280     /**
281      * Logs a message to one of the console loggers.
282      *
283      * @param {String} level the log level (i.e. function 'error', 'warn'...)
284      * @param {Array} args the log message & data (as raw objects)
285      */
286     function _log(level, args) {
287         var logger = backup[level];
288         if (typeof(logger) === "function") {
289             _group(state.context);
290             logger.apply(window.console, args);
291         }
292     }
293
294     /**
295      * Calls the `console.group` and `console.groupEnd` functions (if
296      * they exist) to change the group label (if needed).
297      *
298      * @param {String} label the log context label
299      */
300     function _group(label) {
301         var console = window.console;
302         if (console._group !== label) {
303             if (console._group) {
304                 delete console._group;
305                 if (typeof(console.groupEnd) === "function") {
306                     console.groupEnd();
307                 }
308             }
309             if (label) {
310                 console._group = label;
311                 if (typeof(console.group) === "function") {
312                     console.group(label + ":");
313                 }
314             }
315         }
316     }
317
318     /**
319      * Stores a log message to the history.
320      *
321      * @param {String} level the message log level
322      * @param {Array} args the log message arguments (as strings)
323      */
324     function _store(level, args) {
325         var m = 1;
326         for (; m < args.length; m++) {
327             if (args[m] && args[m].indexOf("\n") >= 0) {
328                 break;
329             }
330         }
331         state.history.push({
332             id: ++state.count,
333             time: new Date(),
334             level: level,
335             context: state.context,
336             message: args.slice(0, m).join(" "),
337             data: (m < args.length) ? args.slice(m).join("\n") : null
338         });
339         state.history.splice(0, Math.max(0, state.history.length - 100));
340         if (!state.publish.timer) {
341             state.publish.timer = setTimeout(_publishLoop, 100);
342         }
343     }
344
345     /**
346      * Handles stored events publishing in a timer loop.
347      */
348     function _publishLoop() {
349         function isValid(evt) {
350             lastId = evt.id;
351             return evt.id > state.publish.last && config.filter(evt);
352         }
353         function onSuccess() {
354             state.publish.last = lastId;
355             state.publish.timer = setTimeout(_publishLoop, config.interval);
356         }
357         function onError(err) {
358             info("error publishing log events", err);
359             state.publish.timer = setTimeout(_publishLoop, config.interval);
360         }
361         var lastId = 0;
362         var events = state.history.filter(isValid);
363         if (events.length <= 0) {
364             state.publish.last = lastId;
365             state.publish.timer = null;
366         } else {
367             try {
368                 config.publisher(events).then(onSuccess, onError);
369             } catch (e) {
370                 onError(e);
371             }
372         }
373     }
374
375     /**
376      * Publishes events to the server.
377      */
378     function _publish(events) {
379         return $.ajax(config.url, {
380             method: "POST",
381             contentType: "application/json",
382             data: JSON.stringify(events)
383         });
384     }
385
386     /**
387      * Checks if an event is an error or a warning.
388      */
389     function _isErrorOrWarning(evt) {
390         return !!evt && /error|warn/.test(evt.level);
391     }

393     /**
394      * Creates a string representation (suitable for logging) for any value or
395      * object. The returned string is similar to a JSON representation of the
396      * value, but may be simplified for increased readability.
397      *
398      * @param {Object} val the value or object to convert
399      *
400      * @return {String} the string representation of the value
401      *
402      * @memberof RapidContext.Log
403      */
404     function stringify(val) {
405         var isObject = Object.prototype.toString.call(val) === "[object Object]";
406         var isArray = val instanceof Array;
407         var isSerializable = val && typeof(val.toJSON) === "function";
408         if (val && (isObject || isArray || isSerializable)) {
409             try {
410                 return JSON.stringify(val, null, 2);
411             } catch (e) {
412                 return String(val);
413             }
414         } else if (val instanceof Error) {
415             var parts = [val.toString()];
416             if (val.stack) {
417                 var stack = String(val.stack).trim()
418                     .replace(val.toString() + "\n", "")
419                     .replace(/https?:.*\//g, "")
420                     .replace(/\?[^:\s]+:/g, ":")
421                     .replace(/@(.*)/mg, " ($1)")
422                     .replace(/^(\s+at\s+)?/mg, "    at ");
423                 parts.push("\n", stack);
424             }
425             return parts.join("");
426         } else if (val && (val.nodeType === 1 || val.nodeType === 9)) {
427             var el = val.documentElement || val.document || val;
428             var xml = el.outerHTML || el.outerXML || el.xml || String(el);
429             var children = el.childNodes && el.childNodes.length;
430             return xml.replace(/>[^<]*/, children ? ">..." : "/>");
431         } else {
432             return String(val);
433         }
434     }
