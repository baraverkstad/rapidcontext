 15 /**
 16  * @name RapidContext.Log
 17  * @namespace Provides a logging service for debugging apps and server calls.
 18  *
 19  * All log messages are filtered by log level and either discarded or
 20  * stored to an internal array. This makes it possible to either show these
 21  * in a developer console or to send them to a server for remote storage.
 22  *
 23  * This module also replaces the built-in `console.error()`, `console.warn()`,
 24  * `console.info()` and `console.log()` functions with its own versions,
 25  * passing through the filtered log messages to the original functions. For
 26  * browsers without a `console` object, a minimal version is provided.
 27  */
 28 (function (window, undefined) {
 29
 30     // Function-level strict mode
 31     "use strict";
 32
 33     // Detect MSIE browser to force simplified console object logging
 34     var isMSIE = /MSIE/.test(window.navigator.userAgent);
 35
 36     // The original console log functions
 37     var consoleBackup = {};
 38
 39     // The current log level
 40     var logLevel = 3;
 41
 42     // The current log message count
 43     var logCount = 0;
 44
 45     // The current log history
 46     var logHistory = [];
 47
 48     // The current log context
 49     var logContext = null;
 50
 51     /**
 52      * Clears the log console and the array of stored messages.
 53      *
 54      * @see RapidContext.Log.history
 55      *
 56      * @memberof RapidContext.Log
 57      */
 58     function clear() {
 59         logHistory = [];
 60         if (window.console && window.console.clear) {
 61             window.console.clear();
 62         }
 63     }
 64
 65     /**
 66      * Returns the history of filtered log entries. Each log entry is a plain
 67      * object with properties -- `id`, `time`, `level`, `context`, `message`
 68      * and `data`.
 69      *
 70      * @return {Array} the array of log entries
 71      *
 72      * @memberof RapidContext.Log
 73      */
 74     function history() {
 75         return logHistory.slice(0);
 76     }
 77
 78     /**
 79      * Returns and optionally sets the current log level. The supported log
 80      * level values are -- "none", "error", "warn", "info", "log" and "all".
 81      *
 82      * @param {String} [value] the new log level
 83      * @return {String} the current log level
 84      *
 85      * @memberof RapidContext.Log
 86      */
 87     function level(value) {
 88         if (typeof(value) !== "undefined") {
 89             if (value === 0 || /^none/i.test(value)) {
 90                 logLevel = 0;
 91             } else if (value === 1 || /^err/i.test(value)) {
 92                 logLevel = 1;
 93             } else if (value === 2 || /^warn/i.test(value)) {
 94                 logLevel = 2;
 95             } else if (value === 3 || /^info/i.test(value)) {
 96                 logLevel = 3;
 97             } else if (value === 4 || /^(log|debug|trace)/i.test(value)) {
 98                 logLevel = 4;
 99             } else {
100                 logLevel = 5;
101             }
102         }
103         if (logLevel <= 0) {
104             return "none";
105         } else if (logLevel <= 1) {
106             return "error";
107         } else if (logLevel <= 2) {
108             return "warn";
109         } else if (logLevel <= 3) {
110             return "info";
111         } else if (logLevel <= 4) {
112             return "log";
113         } else {
114             return "all";
115         }
116     }
117
118     /**
119      * Returns and optionally sets the current log context. The log context is
120      * used to tag all subsequent log messages until the context is removed or
121      * modified.
122      *
123      * @param {String} [value] the new log context, or null to clear
124      *
125      * @return {String} the current log context, or null for none
126      *
127      * @example
128      * RapidContext.Log.context('mybutton.onclick');
129      * ...
130      * console.warn('unsupported xyz value:', value);
131      * ...
132      * RapidContext.Log.context(null);
133      *
134      * @memberof RapidContext.Log
135      */
136     function context(value) {
137         if (typeof(value) !== "undefined") {
138             logContext = value;
139             if (value == null) {
140                 _consoleContext();
141             }
142         }
143         return logContext;
144     }

146     /**
147      * Logs an error message with optional data. Also available as the global
148      * `console.error()` function.
149      *
150      * @param {String} msg the log message
151      * @param {Object} [...] the additional log data or messages
152      *
153      * @example
154      * console.error('failed to initialize module');
155      *
156      * @name error
157      * @function
158      * @memberof RapidContext.Log
159      */
160     function logError(msg/**, ...*/) {
161         if (logLevel >= 1) {
162             var args = Array.prototype.slice.apply(arguments);
163             var strs = args.map(stringify);
164             _consoleLog("error", args, strs);
165             _store("error", strs);
166         }
167     }

169     /**
170      * Logs a warning message with optional data. Also available as the global
171      * `console.warn()` function.
172      *
173      * @param {String} msg the log message
174      * @param {Object} [...] the additional log data or messages
175      *
176      * @example
177      * console.warn('missing "data" attribute on document root:', document.body);
178      *
179      * @name warn
180      * @function
181      * @memberof RapidContext.Log
182      */
183     function logWarn(msg/**, ...*/) {
184         if (logLevel >= 2) {
185             var args = Array.prototype.slice.apply(arguments);
186             var strs = args.map(stringify);
187             _consoleLog("warn", args, strs);
188             _store("warn", strs);
189         }
190     }

192     /**
193      * Logs an information message with optional data. Also available as the
194      * global `console.info()` function.
195      *
196      * @param {String} msg the log message
197      * @param {Object} [...] the additional log data or messages
198      *
199      * @example
200      * console.info('authorization failed, user not logged in');
201      *
202      * @name info
203      * @function
204      * @memberof RapidContext.Log
205      */
206     function logInfo(msg/**, ...*/) {
207         if (logLevel >= 3) {
208             var args = Array.prototype.slice.apply(arguments);
209             var strs = args.map(stringify);
210             _consoleLog("info", args, strs);
211             _store("info", strs);
212         }
213     }

215     /**
216      * Logs a debug message with optional data. Also available as the global
217      * `console.log()` and `console.debug()` functions.
218      *
219      * @param {String} msg the log message
220      * @param {Object} [...] the additional log data or messages
221      *
222      * @example
223      * console.log('init AJAX call to URL:', url);
224      * ...
225      * console.log('done AJAX call to URL:', url, responseCode);
226      *
227      * @name debug
228      * @function
229      * @memberof RapidContext.Log
230      */
231     function logDebug(msg/**, ...*/) {
232         if (logLevel >= 4) {
233             var args = Array.prototype.slice.apply(arguments);
234             var strs = args.map(stringify);
235             _consoleLog("log", args, strs);
236             _store("log", strs);
237         }
238     }

240     /**
241      * Logs a number of browser meta-data parameters (INFO level).
242      *
243      * @memberof RapidContext.Log
244      */
245     function logBrowserInfo() {
246         function copy(obj, keys) {
247             var res = {};
248             for (var i = 0; i < keys.length; i++) {
249                 var k = keys[i];
250                 res[k] = obj[k];
251             }
252             return res;
253         }
254         context('Browser')
255         logInfo("location.href", location.href);
256         logInfo("navigator.userAgent", navigator.userAgent);
257         logInfo("navigator.language", navigator.language);
258         logInfo("screen", copy(screen, ["width", "height", "colorDepth"]));
259         logInfo("window", copy(window, ["innerWidth", "innerHeight", "devicePixelRatio"]));
260         logInfo("document.cookie", document.cookie);
261         context(null);
262     }

264     /**
265      * Handles window.onerror events (global uncaught errors).
266      */
267     function _windowErrorHandler(msg, src, line, col, error) {
268         var location = "unknown source";
269         if (error && error.stack) {
270             location = error.stack;
271         } else if (src && src !== "undefined") {
272             location = [src, line || 0, col || 0].join(':');
273         }
274         logError(msg, location);
275         return true;
276     }

278     /**
279      * Modifies the `console` object for logging. This replaces the default
280      * `console.error`, `console.warn`, `console.info`, `console.log` and
281      * `console.debug` functions (if not previously modified).
282      */
283     function _consoleSetup() {
284         function overwrite(obj, key, value, backup) {
285             if (obj[key] !== value) {
286                 backup[key] = obj[key];
287                 obj[key] = value;
288             }
289         }
290         window.console || (window.console = {});
291         overwrite(window.console, 'error', logError, consoleBackup);
292         overwrite(window.console, 'warn', logWarn, consoleBackup);
293         overwrite(window.console, 'info', logInfo, consoleBackup);
294         overwrite(window.console, 'log', logDebug, consoleBackup);
295         overwrite(window.console, 'debug', logDebug, consoleBackup);
296     }
297
298     /**
299      * Calls the `console.group` and `console.groupEnd` functions (if
300      * they exist) to adjust for the current log context.
301      */
302     function _consoleContext() {
303         if (window.console._group && window.console._group !== logContext) {
304             window.console._group = null;
305             window.console.groupEnd && window.console.groupEnd();
306         }
307         if (logContext && window.console._group !== logContext) {
308             window.console._group = logContext;
309             window.console.group && window.console.group(logContext + ":");
310         }
311     }

313     /**
314      * Logs a message to one of the console loggers.
315      *
316      * @param {String} loggerName the console logger name ('error', 'warn'...)
317      * @param {Array} args the log message & data (as raw objects)
318      * @param {Array} strs the log message & data (as strings)
319      */
320     function _consoleLog(loggerName, args, strs) {
321         _consoleSetup();
322         _consoleContext();
323         var logger = consoleBackup[loggerName];
324         if (typeof(logger) === 'function' && typeof(logger.apply) === 'function') {
325             logger.apply(window.console, args);
326         } else if (logger && isMSIE) {
327             var ctx = (logContext ? logContext + ": " : "");
328             logger(ctx + strs.join(", "));
329         }
330     }

332     /**
333      * Stores a log message to the history.
334      *
335      * @param {String} level the message log level
336      * @param {Array} args the log message arguments (as strings)
337      *
338      * @private
339      */
340     function _store(level, args) {
341         var data = (args.length > 1) ? args.slice(1).join(", ") : null;
342         logHistory.push({ id: ++logCount, time: new Date(), level: level,
343                           context: logContext, message: args[0], data: data });
344         while (logHistory.length > 100) {
345             logHistory.shift();
346         }
347     }

349     /**
350      * Creates a string representation (suitable for logging) for any value or
351      * object. The returned string is similar to a JSON representation of the
352      * value, but may be simplified for increased readability.
353      *
354      * @param {Object} o the value or object to convert
355      *
356      * @return {String} the string representation of the value
357      *
358      * @memberof RapidContext.Log
359      */
360     function stringify(o) {
361         return _stringify(o, 0);
362     }
363
364     /**
365      * Internal stringify implementation that tracks of object graph
366      * depth.
367      *
368      * @param {Object} o the value or object to convert
369      * @param {Number} depth the current object depth (max is 4)
370      *
371      * @return {Array} the array of string representations
372      *
373      * @private
374      */
375     function _stringify(o, depth) {
376         var type = typeof(o);
377         if (o == null || type == "boolean" || type == "number") {
378             return "" + o;
379         } else if (type == "string" && depth > 0) {
380             return '"' + o.replace(/"/g, '\\"').replace(/\n/g, "\\n") + '"';
381         } else if (type == "string") {
382             return o;
383         } else if (type == "function") {
384             var name = o.name || o.displayName || "";
385             var signature = o.toString().replace(/\)[\s\S]*/, "") + ")";
386             if (/function\s+\(/.test(signature) && name) {
387                 signature = signature.replace(/function\s+/, "function " + name);
388             } else if (!(/function/.test(signature))) {
389                 signature = "function ()";
390             }
391             return signature + " {...}";
392         } else if (o.nodeType === 1 || o.nodeType === 9) {
393             o = o.documentElement || o.document || o;
394             var xml = o.outerHTML || o.outerXML || o.xml || "" + o;
395             if (o.childNodes && o.childNodes.length) {
396                 return xml.replace(/>[\s\S]*</, ">...<");
397             } else {
398                 return xml.replace(/>[\s\S]*/, " />");
399             }
400         } else if (depth > 3) {
401             return "...";
402         } else if (typeof(o.length) == "number") {
403             var arr = [];
404             for (var i = 0; i < o.length; i++) {
405                 if (i > 20) {
406                     arr.push("...");
407                     break;
408                 }
409                 arr.push(_stringify(o[i], depth + 1));
410             }
411             return "[" + arr.join(", ") + "]";
412         } else {
413             var arr = [];
414             for (var k in o) {
415                 if (arr.length > 20) {
416                     arr.push("...");
417                     break;
418                 }
419                 arr.push(k + ": " + _stringify(o[k], depth + 1));
420             }
421             return "{" + arr.join(", ") + "}";
422         }
423     }
