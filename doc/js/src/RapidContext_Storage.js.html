 15 /**
 16  * @name RapidContext.Storage
 17  * @namespace Provides functions for accessing the server storage (BETA).
 18  */
 19 (function (window) {
 20
 21     /**
 22      * Returns a storage URL for a resource. If the resource is an
 23      * object, its 'type' and 'id' properties will be used to form
 24      * the path.
 25      *
 26      * @param {String/Object} pathOrObj the path or object
 27      *
 28      * @return {String} the URL to the resource
 29      *
 30      * @throws {Error} if the object didn't have both 'type' and 'id'
 31      *     properties
 32      *
 33      * @private
 34      */
 35     function storageUrl(pathOrObj) {
 36         var ident = (typeof(pathOrObj) === "string") ? pathOrObj : path(pathOrObj);
 37         if (!ident) {
 38             throw new Error("Invalid object or storage path");
 39         }
 40         return "rapidcontext/storage/" + ident.replace(/^\//, "");
 41     }
 42
 43     /**
 44      * Returns the storage path for an object. The object must have
 45      * both the 'type' and 'id' properties set.
 46      *
 47      * @param {Object} obj the object to store
 48      *
 49      * @return {String} the storage path, or null if not available
 50      *
 51      * @memberof RapidContext.Storage
 52      */
 53     function path(obj) {
 54         return (obj && obj.type && obj.id) ? obj.type + "/" + obj.id : null;
 55     }
 56
 57     /**
 58      * Reads an object on a storage path. Note that this will return
 59      * a JSON representation of the object, regardless of the actual
 60      * object type.
 61      *
 62      * @param {String/Object} pathOrObj the path or object to read
 63      *
 64      * @return {Promise} a `RapidContext.Async` promise that will
 65      *         resolve with the object data
 66      *
 67      * @memberof RapidContext.Storage
 68      */
 69     function read(pathOrObj) {
 70         var url = storageUrl(pathOrObj);
 71         url += MochiKit.Text.endsWith("/", url) ? "index.json" : ".json";
 72         return RapidContext.App.loadJSON(url, null, null);
 73     }
 74
 75     /**
 76      * Writes an object to a storage path. Any previous object on the
 77      * specified path will be removed. If a path is specified without
 78      * data, only the removal is performed.
 79      *
 80      * @param {String/Object} pathOrObj the path or object to write
 81      * @param {Object} [data] the object to write (if path was string)
 82      *
 83      * @return {Promise} a `RapidContext.Async` promise that will
 84      *         resolve when the object has been written
 85      *
 86      * @memberof RapidContext.Storage
 87      */
 88     function write(pathOrObj, data) {
 89         var url = storageUrl(pathOrObj);
 90         if (typeof(pathOrObj) == "string" && data == null) {
 91             return RapidContext.App.loadXHR(url, null, { method: "DELETE" });
 92         } else {
 93             var headers = { "Content-Type": "application/json" };
 94             var opts = { method: "POST", headers: headers };
 95             return RapidContext.App.loadXHR(url, data || pathOrObj, opts);
 96         }
 97     }
 98
 99     /**
100      * Updates an object with properties from a partial object. The
101      * properties in the partial object will overwrite any previous
102      * properties with the same name in the destination object. No
103      * merging of property values will be performed.
104      *
105      * @param {String/Object} pathOrObj the path or object to write
106      * @param {Object} data the partial object properties
107      *
108      * @return {Promise} a `RapidContext.Async` promise that will
109      *         resolve with the updated data object on success
110      *
111      * @memberof RapidContext.Storage
112      */
113     function update(pathOrObj, data) {
114         var url = storageUrl(pathOrObj);
115         var headers = { "Content-Type": "application/json" };
116         var opts = { method: "PATCH", headers: headers };
117         return RapidContext.App.loadJSON(url, data, opts);
118     }
119
120     // Create namespaces
121     var RapidContext = window.RapidContext || (window.RapidContext = {});
122     var Storage = RapidContext.Storage || (RapidContext.Storage = {});
123
124     // Export symbols
125     Storage.path = path;
126     Storage.read = read;
127     Storage.write = write;
128     Storage.update = update;
129
130 })(this);
