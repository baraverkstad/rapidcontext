 15 /**
 16  * @name RapidContext.Storage
 17  * @namespace Provides functions for accessing the server storage (BETA).
 18  */
 19 (function (window, undefined) {
 20
 21     /**
 22      * Returns a storage URL for a resource. If the resource is an
 23      * object, its 'type' and 'id' properties will be used to form
 24      * the path.
 25      *
 26      * @param {String/Object} pathOrObj the path or object
 27      *
 28      * @return {String} the URL to the resource
 29      *
 30      * @throws {Error} if the object didn't have both 'type' and 'id'
 31      *     properties
 32      *
 33      * @private
 34      */
 35     function storageUrl(pathOrObj) {
 36         if (typeof(pathOrObj) != "string") {
 37             pathOrObj = path(pathOrObj);
 38         }
 39         if (!pathOrObj) {
 40             throw new Error("Invalid object or storage path");
 41         }
 42         return "rapidcontext/storage/" + pathOrObj.replace(/^\//, "");
 43     }
 44
 45     /**
 46      * Returns the storage path for an object. The object must have
 47      * both the 'type' and 'id' properties set.
 48      *
 49      * @param {Object} obj the object to store
 50      *
 51      * @return {String} the storage path, or null if not available
 52      *
 53      * @memberof RapidContext.Storage
 54      */
 55     function path(obj) {
 56         return (obj && obj.type && obj.id) ? obj.type + "/" + obj.id : null;
 57     }
 58
 59     /**
 60      * Reads an object on a storage path. Note that this will return
 61      * a JSON representation of the object, regardless of the actual
 62      * object type.
 63      *
 64      * @param {String/Object} pathOrObj the path or object to read
 65      *
 66      * @return {Deferred} a `MochiKit.Async.Deferred` object that will
 67      *         callback with the JSON data object
 68      *
 69      * @memberof RapidContext.Storage
 70      */
 71     function read(pathOrObj) {
 72         var url = storageUrl(pathOrObj) + ".json";
 73         var d = RapidContext.App.loadJSON(url, null, null);
 74         d.addCallback(MochiKit.Base.itemgetter("data"));
 75         return d;
 76     }
 77
 78     /**
 79      * Writes an object to a storage path. Any previous object on the
 80      * specified path will be removed. If a path is specified without
 81      * data, only the removal is performed.
 82      *
 83      * @param {String/Object} pathOrObj the path or object to write
 84      * @param {Object} [data] the object to write (if path was string)
 85      *
 86      * @return {Deferred} a `MochiKit.Async.Deferred` object that will
 87      *         callback with the XMLHttpRequest instance on success
 88      *
 89      * @memberof RapidContext.Storage
 90      */
 91     function write(pathOrObj, data) {
 92         if (typeof(pathOrObj) == "string" && data == null) {
 93             var opts = { method: "DELETE" };
 94             return RapidContext.App.loadXHR(storageUrl(pathOrObj), null, opts);
 95         } else {
 96             var json = JSON.stringify(data || pathOrObj);
 97             var headers = { "Content-Type": "application/json" };
 98             var opts = { method: "POST", sendContent: json, headers: headers };
 99             return RapidContext.App.loadXHR(storageUrl(pathOrObj), null, opts);
100         }
101     }
102
103     /**
104      * Updates an object with properties from a partial object. The
105      * properties in the partial object will overwrite any previous
106      * properties with the same name in the destination object. No
107      * merging of property values will be performed.
108      *
109      * @param {String/Object} pathOrObj the path or object to write
110      * @param {Object} data the partial object properties
111      *
112      * @return {Deferred} a `MochiKit.Async.Deferred` object that will
113      *         callback with the updated data object on success
114      *
115      * @memberof RapidContext.Storage
116      */
117     function update(pathOrObj, data) {
118         var json = JSON.stringify(data);
119         var headers = { "Content-Type": "application/json" };
120         var opts = { method: "PATCH", sendContent: json, headers: headers };
121         return RapidContext.App.loadJSON(storageUrl(pathOrObj), null, opts);
122     }
123
124     // Create namespaces
125     var RapidContext = window.RapidContext || (window.RapidContext = {});
126     var Storage = RapidContext.Storage || (RapidContext.Storage = {});
127
128     // Export symbols
129     Storage.path = path;
130     Storage.read = read;
131     Storage.write = write;
132     Storage.update = update;
133
134 })(this);
