 21 /**
 22  * Creates a new icon widget.
 23  *
 24  * @constructor
 25  * @param {Object} attrs the widget and node attributes to set
 26  * @param {String} [attrs.ref] the predefined icon name (reference)
 27  * @param {String} [attrs.class] the icon CSS class names
 28  * @param {String} [attrs.url] the icon image URL
 29  * @param {String} [attrs.position] the icon image position
 30  * @param {Number} [attrs.width] the icon width (in pixels)
 31  * @param {Number} [attrs.height] the icon height (in pixels)
 32  * @param {String} [attrs.tooltip] the icon tooltip text
 33  * @param {Boolean} [attrs.disabled] the disabled widget flag, defaults to
 34  *            false
 35  * @param {Boolean} [attrs.hidden] the hidden widget flag, defaults to false
 36  *
 37  * @return {Widget} the widget DOM node
 38  *
 39  * @class The icon widget class. Used to provide a small clickable image, using
 40  *     the `<i>` HTML element. The icons can be either image- or font-based,
 41  *     causing slightly different behaviors with respect to size. A number of
 42  *     predefined icons suitable for different actions are available (using the
 43  *     Font-Awesome icon font).
 44  * @extends RapidContext.Widget
 45  *
 46  * @example {JavaScript}
 47  * var style = { "margin-left": "3px" };
 48  * var dataReload = RapidContext.Widget.Icon({ ref: "RELOAD", style: style });
 49  * var dataLoading = RapidContext.Widget.Icon({ ref: "LOADING", hidden: true, style: style });
 50  * var h3 = MochiKit.DOM.H3({}, "Data List:", dataReload, dataLoading);
 51  *
 52  * @example {User Interface XML}
 53  * <h3>
 54  *   Data List:
 55  *   <Icon id="dataReload" ref="RELOAD" style="margin-left: 3px;" />
 56  *   <Icon id="dataLoading" ref="LOADING" hidden="true" style="margin-left: 3px;" />
 57  * </h3>
 58  */
 59 RapidContext.Widget.Icon = function (attrs) {
 60     var o = MochiKit.DOM.createDOM('i');
 61     RapidContext.Widget._widgetMixin(o, arguments.callee);
 62     o.setAttrs(attrs);
 63     o.addClass("widgetIcon");
 64     return o;
 65 };

 87 /**
 88  * Updates the icon or HTML DOM node attributes.
 89  *
 90  * @param {Object} attrs the widget and node attributes to set
 91  * @param {String} [attrs.ref] the predefined icon name (reference)
 92  * @param {String} [attrs.class] the icon CSS class names
 93  * @param {String} [attrs.url] the icon image URL
 94  * @param {String} [attrs.position] the icon image position
 95  * @param {Number} [attrs.width] the icon width (in pixels)
 96  * @param {Number} [attrs.height] the icon height (in pixels)
 97  * @param {String} [attrs.tooltip] the icon tooltip text
 98  * @param {Boolean} [attrs.disabled] the disabled widget flag
 99  * @param {Boolean} [attrs.hidden] the hidden widget flag
100  */
101 RapidContext.Widget.Icon.prototype.setAttrs = function (attrs) {
102     var locals = MochiKit.Base.update({}, attrs);
103     while (locals.ref) {
104         var ref = RapidContext.Widget.Icon[locals.ref] || {};
105         delete locals.ref;
106         MochiKit.Base.setdefault(locals, ref);
107     }
108     var styles = {};
109     if (locals.url) {
110         this.addClass("widgetIconSprite");
111         styles.backgroundImage = 'url("' + locals.url + '")';
112         delete locals.url;
113     }
114     if (locals.position) {
115         styles.backgroundPosition = locals.position;
116         delete locals.position;
117     }
118     if (locals.width) {
119         styles.width = locals.width + "px";
120         delete locals.width;
121     }
122     if (locals.height) {
123         styles.height = locals.height + "px";
124         delete locals.height;
125     }
126     this.setStyle(styles);
127     if (locals.tooltip && !locals.title) {
128         locals.title = locals.tooltip;
129         delete locals.tooltip;
130     }
131     this.__setAttrs(locals);
132 };
