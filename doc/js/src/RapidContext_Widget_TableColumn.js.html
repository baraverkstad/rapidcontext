 21 /**
 22  * Creates a new data table column widget.
 23  *
 24  * @constructor
 25  * @param {Object} attrs the widget and node attributes
 26  * @param {String} attrs.title the column title
 27  * @param {String} attrs.field the data property name
 28  * @param {String} [attrs.type] the data property type, one of
 29  *            "string", "number", "date", "time", "datetime",
 30  *            "boolean" or "object" (defaults to "string")
 31  * @param {String} [attrs.sort] the sort direction, one of "asc",
 32  *            "desc", "none" (disabled) or null (unsorted)
 33  * @param {Number} [attrs.maxLength] the maximum data length,
 34  *            overflow will be displayed as a tooltip, only used by
 35  *            the default renderer
 36  * @param {Boolean} [attrs.key] the unique key value flag, only to be
 37  *            set for a single column per table
 38  * @param {String} [attrs.tooltip] the tooltip text to display on the
 39  *            column header
 40  * @param {Function} [attrs.renderer] the function that renders the converted
 41  *            data value into a table cell, called as
 42  *            `renderer(<td>, value, data)` with the DOM node, field value and
 43  *            data object as arguments
 44  *
 45  * @return {Widget} the widget DOM node
 46  *
 47  * @class The table column widget class. Used to provide a sortable data table
 48  *     column, using a `<th>` HTML element for the header (and rendering data
 49  *     to `<td>` HTML elements).
 50  * @extends RapidContext.Widget
 51  *
 52  * @example {JavaScript}
 53  * var attrs1 = { field: "id", title: "Identifier", key: true, type: "number" };
 54  * var attrs2 = { field: "name", title: "Name", maxLength: 50, sort: "asc" };
 55  * var attrs3 = { field: "modified", title: "Last Modified", type: "datetime" };
 56  * var col1 = RapidContext.Widget.TableColumn(attrs1);
 57  * var col2 = RapidContext.Widget.TableColumn(attrs2);
 58  * var col3 = RapidContext.Widget.TableColumn(attrs3);
 59  * var exampleTable = RapidContext.Widget.Table({}, col1, col2, col3);
 60  * RapidContext.Util.registerSizeConstraints(exampleTable, "50%", "100%");
 61  *
 62  * @example {User Interface XML}
 63  * <Table id="exampleTable" w="50%" h="100%">
 64  *   <TableColumn field="id" title="Identifier" key="true" type="number" />
 65  *   <TableColumn field="name" title="Name" maxLength="50" sort="asc" />
 66  *   <TableColumn field="modified" title="Last Modified" type="datetime" />
 67  * </Table>
 68  */
 69 RapidContext.Widget.TableColumn = function (attrs) {
 70     if (attrs.field == null) {
 71         throw new Error("The 'field' attribute cannot be null for a TableColumn");
 72     }
 73     var o = MochiKit.DOM.TH();
 74     RapidContext.Widget._widgetMixin(o, RapidContext.Widget.TableColumn);
 75     o.addClass("widgetTableColumn");
 76     o.setAttrs(Object.assign({ title: attrs.field, type: "string", key: false }, attrs));
 77     o.addEventListener("click", o._handleClick);
 78     return o;
 79 };

101 /**
102  * Updates the widget or HTML DOM node attributes. Note that some
103  * updates will not take effect until the parent table is cleared
104  * or data is reloaded.
105  *
106  * @param {Object} attrs the widget and node attributes to set
107  * @param {String} [attrs.title] the column title
108  * @param {String} [attrs.field] the data property name
109  * @param {String} [attrs.type] the data property type, one of
110  *            "string", "number", "date", "time", "datetime",
111  *            "boolean" or "object"
112  * @param {String} [attrs.sort] the sort direction, one of "asc",
113  *            "desc", "none" (disabled) or null (unsorted)
114  * @param {Number} [attrs.maxLength] the maximum data length,
115  *            overflow will be displayed as a tooltip, only used by
116  *            the default renderer
117  * @param {Boolean} [attrs.key] the unique key value flag, only to be
118  *            set for a single column per table
119  * @param {String} [attrs.tooltip] the tooltip text to display on the
120  *            column header
121  * @param {Function} [attrs.renderer] the function that renders the converted
122  *            data value into a table cell, called as
123  *            `renderer(<td>, value, data)` with the DOM node, field value and
124  *            data object as arguments
125  */
126 RapidContext.Widget.TableColumn.prototype.setAttrs = function (attrs) {
127     attrs = Object.assign({}, attrs);
128     var locals = RapidContext.Util.mask(attrs, [
129         "title", "field", "type", "sort", "maxLength", "key", "tooltip", "renderer"
130     ]);
131     if (typeof(locals.title) !== "undefined") {
132         MochiKit.DOM.replaceChildNodes(this, locals.title);
133     }
134     if (typeof(locals.field) !== "undefined") {
135         this.field = locals.field;
136     }
137     if (typeof(locals.type) !== "undefined") {
138         this.type = locals.type;
139     }
140     if (typeof(locals.sort) !== "undefined") {
141         this.sort = locals.sort;
142         $(this).toggleClass("sortNone", locals.sort === "none");
143         $(this).toggleClass("sortDesc", locals.sort === "desc");
144         $(this).toggleClass("sortAsc", locals.sort === "asc");
145     }
146     if (typeof(locals.maxLength) !== "undefined") {
147         this.maxLength = parseInt(locals.maxLength);
148     }
149     if (typeof(locals.key) !== "undefined") {
150         this.key = MochiKit.Base.bool(locals.key);
151     }
152     if (typeof(locals.tooltip) !== "undefined") {
153         this.title = locals.tooltip;
154     }
155     if (typeof(locals.renderer) === "function") {
156         this.renderer = locals.renderer;
157     }
158     this.__setAttrs(attrs);
159 };

161 /**
162  * Maps the column field from one object onto another. This method
163  * will also convert the data depending on the column data type.
164  *
165  * @param src                the source object (containing the field)
166  * @param dst                the destination object
167  */
168 RapidContext.Widget.TableColumn.prototype._map = function (src, dst) {
169     var value = src[this.field];
170     if (value != null) {
171         if (this.key) {
172             dst.$id = value;
173         }
174         switch (this.type) {
175         case "number":
176             if (value instanceof Number) {
177                 value = value.valueOf();
178             } else if (typeof(value) != "number") {
179                 value = parseFloat(value);
180             }
181             break;
182         case "date":
183             if (/^@\d+$/.test(value)) {
184                 value = new Date(+value.substr(1));
185             }
186             if (value instanceof Date) {
187                 value = MochiKit.DateTime.toISODate(value);
188             } else {
189                 value = MochiKit.Text.truncate(value, 10);
190             }
191             break;
192         case "datetime":
193             if (/^@\d+$/.test(value)) {
194                 value = new Date(+value.substr(1));
195             }
196             if (value instanceof Date) {
197                 value = MochiKit.DateTime.toISOTimestamp(value);
198             } else {
199                 value = MochiKit.Text.truncate(value, 19);
200             }
201             break;
202         case "time":
203             if (/^@\d+$/.test(value)) {
204                 value = new Date(+value.substr(1));
205             }
206             if (value instanceof Date) {
207                 value = MochiKit.DateTime.toISOTime(value);
208             } else {
209                 if (typeof(value) !== "string") {
210                     value = value.toString();
211                 }
212                 if (value.length > 8) {
213                     value = value.substring(value.length - 8);
214                 }
215             }
216             break;
217         case "boolean":
218             if (typeof(value) !== "boolean") {
219                 value = MochiKit.Base.bool(value);
220             }
221             break;
222         case "string":
223             if (typeof(value) !== "string") {
224                 value = value.toString();
225             }
226             break;
227         }
228     }
229     dst[this.field] = value;
230 };

232 /**
233  * Renders the column field value into a table cell.
234  *
235  * @param obj                the data object (containing the field)
236  *
237  * @return the table cell DOM node
238  */
239 RapidContext.Widget.TableColumn.prototype._render = function (obj) {
240     var td = MochiKit.DOM.TD();
241     var value = obj[this.field];
242     if (typeof(this.renderer) === "function") {
243         try {
244             this.renderer(td, value, obj.$data);
245         } catch (e) {
246             td.appendChild(RapidContext.Util.createTextNode(e));
247         }
248     } else if (typeof(value) == "boolean") {
249         var css = value ? "fa fa-check-square-o" : "fa fa-square-o";
250         td.appendChild(RapidContext.Widget.Icon(css));
251     } else {
252         if (value == null || (typeof(value) == "number" && isNaN(value))) {
253             value = "";
254         } else if (typeof(value) != "string") {
255             value = value.toString();
256         }
257         if (this.maxLength && this.maxLength < value.length) {
258             td.title = value;
259             value = MochiKit.Text.truncate(value, this.maxLength, "...");
260         }
261         td.appendChild(RapidContext.Util.createTextNode(value));
262     }
263     return td;
264 };

266 /**
267  * Handles click events on the column header.
268  */
269 RapidContext.Widget.TableColumn.prototype._handleClick = function () {
270     if (this.parentNode != null) {
271         var dir = (this.sort == "asc") ? "desc" : "asc";
272         var tr = this.parentNode;
273         var thead = tr.parentNode;
274         var table = thead.parentNode;
275         table.parentNode.sortData(this.field, dir);
276     }
277 };
