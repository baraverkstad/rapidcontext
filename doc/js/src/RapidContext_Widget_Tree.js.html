 21 /**
 22  * Creates a new tree widget.
 23  *
 24  * @constructor
 25  * @param {Object} attrs the widget and node attributes
 26  * @param {Boolean} [attrs.hidden] the hidden widget flag, defaults to false
 27  * @param {Widget} [...] the child tree node widgets
 28  *
 29  * @return {Widget} the widget DOM node
 30  *
 31  * @class The tree widget class. Used to provide a dynamic tree with expandable
 32  *     tree nodes, using a number of `<div>` HTML elements.
 33  * @extends RapidContext.Widget
 34  *
 35  * @example {JavaScript}
 36  * var tree = RapidContext.Widget.Tree({ style: { width: "200px", height: "400px" } });
 37  * var root = RapidContext.Widget.TreeNode({ folder: true, name: "Root" });
 38  * var child = RapidContext.Widget.TreeNode({ name: "Child" });
 39  * root.addAll(child);
 40  * tree.addAll(root);
 41  *
 42  * @example {User Interface XML}
 43  * <Tree style="width: 200px; height: 400px;">
 44  *   <TreeNode name="Root">
 45  *     <TreeNode name="Child" />
 46  *   </TreeNode>
 47  * </Tree>
 48  */
 49 RapidContext.Widget.Tree = function (attrs/*, ...*/) {
 50     var o = MochiKit.DOM.DIV(attrs);
 51     RapidContext.Widget._widgetMixin(o, RapidContext.Widget.Tree);
 52     o.classList.add("widgetTree");
 53     o.resizeContent = o._resizeContent;
 54     o.selectedPath = null;
 55     o.setAttrs(attrs);
 56     o.addAll(Array.from(arguments).slice(1));
 57     return o;
 58 };

 93 /**
 94  * Adds a single child tree node widget to this widget.
 95  *
 96  * @param {Widget} child the tree node widget to add
 97  */
 98 RapidContext.Widget.Tree.prototype.addChildNode = function (child) {
 99     if (!RapidContext.Widget.isWidget(child, "TreeNode")) {
100         throw new Error("Tree widget can only have TreeNode children");
101     }
102     this.appendChild(child);
103 };
104
105 /**
106  * Removes all marked tree nodes. When adding or updating tree nodes, any
107  * node modified is automatically unmarked (e.g. by calling `setAttrs` on the
108  * tree nodes or `addPath` on the tree). This makes it easy to prune a tree
109  * after an update, by initially marking all tree nodes with `markAll()`,
110  * inserting or touching all nodes to keep, and finally calling this method to
111  * remove the remaining nodes.
112  *
113  * @example
114  * tree.markAll();
115  * tree.addPath(["Root", "Child"]);
116  * ...
117  * tree.removeAllMarked();
118  */
119 RapidContext.Widget.Tree.prototype.removeAllMarked = function () {
120     var children = this.getChildNodes();
121     for (var i = 0; i < children.length; i++) {
122         if (children[i].marked === true) {
123             this.removeChildNode(children[i]);
124         } else {
125             children[i].removeAllMarked();
126         }
127     }
128 };
129
130 /**
131  * Marks this tree node and all child nodes recursively. When adding or
132  * updating tree nodes, any node modified is automatically unmarked (e.g. by
133  * calling `setAttrs` on the tree nodes or `addPath` on the tree). This makes
134  * it easy to prune a tree after an update, by initially marking all tree
135  * nodes, inserting or touching all nodes to keep, and finally calling
136  * `removeAllMarked()` to remove the remaining nodes.
137  *
138  * @example
139  * tree.markAll();
140  * tree.addPath(["Root", "Child"]);
141  * ...
142  * tree.removeAllMarked();
143  */
144 RapidContext.Widget.Tree.prototype.markAll = function () {
145     var children = this.getChildNodes();
146     for (var i = 0; i < children.length; i++) {
147         children[i].markAll();
148     }
149 };

151 /**
152  * Finds a root tree node with the specified name.
153  *
154  * @param {String} name the root tree node name
155  *
156  * @return {Widget} the root tree node found, or
157  *         null if not found
158  */
159 RapidContext.Widget.Tree.prototype.findRoot = function (name) {
160     var children = this.getChildNodes();
161     for (var i = 0; i < children.length; i++) {
162         if (children[i].name == name) {
163             return children[i];
164         }
165     }
166     return null;
167 };
168
169 /**
170  * Searches for a tree node from the specified path.
171  *
172  * @param {Array} path the tree node path (array of names)
173  *
174  * @return {Widget} the descendant tree node found, or
175  *         null if not found
176  */
177 RapidContext.Widget.Tree.prototype.findByPath = function (path) {
178     if (path == null || path.length < 1) {
179         return null;
180     }
181     var root = this.findRoot(path[0]);
182     if (root != null) {
183         return root.findByPath(path.slice(1));
184     } else {
185         return null;
186     }
187 };
188
189 /**
190  * Returns the currently selected tree node.
191  *
192  * @return {Widget} the currently selected tree node, or
193  *         null if no node is selected
194  */
195 RapidContext.Widget.Tree.prototype.selectedChild = function () {
196     if (this.selectedPath == null) {
197         return null;
198     } else {
199         return this.findByPath(this.selectedPath);
200     }
201 };
202
203 /**
204  * Sets the currently selected node in the tree. This method is only
205  * called from the tree node `select()` and `unselect()` methods.
206  *
207  * @param {Widget} node the new selected tree node, or null for none
208  */
209 RapidContext.Widget.Tree.prototype._handleSelect = function (node) {
210     var prev = this.selectedChild();
211     if (node == null) {
212         this.selectedPath = null;
213         this._dispatch("unselect");
214     } else {
215         if (prev != null && prev !== node) {
216             prev.unselect();
217         }
218         this.selectedPath = node.path();
219         this._dispatch("select");
220     }
221 };

223 /**
224  * Recursively expands all nodes. If a depth is specified,
225  * expansions will not continue below that depth.
226  *
227  * @param {Number} [depth] the optional maximum depth
228  */
229 RapidContext.Widget.Tree.prototype.expandAll = function (depth) {
230     if (typeof(depth) !== "number") {
231         depth = 10;
232     }
233     var children = this.getChildNodes();
234     for (var i = 0; depth > 0 && i < children.length; i++) {
235         children[i].expandAll(depth - 1);
236     }
237 };
238
239 /**
240  * Recursively collapses all nodes. If a depth is specified, only
241  * nodes below that depth will be collapsed.
242  *
243  * @param {Number} [depth] the optional minimum depth
244  */
245 RapidContext.Widget.Tree.prototype.collapseAll = function (depth) {
246     if (typeof(depth) !== "number") {
247         depth = 0;
248     }
249     var children = this.getChildNodes();
250     for (var i = 0; i < children.length; i++) {
251         children[i].collapseAll(depth - 1);
252     }
253 };

255 /**
256  * Adds a path to the tree as a recursive list of child nodes. If
257  * nodes in the specified path already exists, they will be used
258  * instead of creating new nodes.
259  *
260  * @param {Array} path the tree node path (array of names)
261  *
262  * @return {Widget} the last node in the path
263  */
264 RapidContext.Widget.Tree.prototype.addPath = function (path) {
265     if (path == null || path.length < 1) {
266         return null;
267     }
268     var node = this.findRoot(path[0]);
269     if (node == null) {
270         node = RapidContext.Widget.TreeNode({ name: path[0], folder: (path.length > 1) });
271         this.addChildNode(node);
272     }
273     node.marked = false;
274     for (var i = 1; i < path.length; i++) {
275         var child = node.findChild(path[i]);
276         if (child == null) {
277             var attrs = { name: path[i], folder: (path.length > i + 1) };
278             child = RapidContext.Widget.TreeNode(attrs);
279             node.addChildNode(child);
280         }
281         child.marked = false;
282         node = child;
283     }
284     return node;
285 };

287 /**
288  * Called when tree content should be resized. This method is also called when
289  * the widget is made visible in a container after being hidden.
290  */
291 RapidContext.Widget.Tree.prototype._resizeContent = function () {
292     // Work-around to restore scrollTop for WebKit browsers
293     if (this.scrollTop == 0 && this.selectedPath) {
294         var node = this.selectedChild();
295         var h = this.clientHeight;
296         var y = node.offsetTop + node.offsetHeight;
297         this.scrollTop = Math.round(y - h / 2);
298     }
299 };
