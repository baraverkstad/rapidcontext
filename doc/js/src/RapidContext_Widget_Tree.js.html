 21 /**
 22  * Creates a new tree widget.
 23  *
 24  * @constructor
 25  * @param {Object} attrs the widget and node attributes
 26  * @param {Boolean} [attrs.hidden] the hidden widget flag, defaults to false
 27  * @param {Widget} [...] the child tree node widgets
 28  *
 29  * @return {Widget} the widget DOM node
 30  *
 31  * @class The tree widget class. Used to provide a dynamic tree with expandable
 32  *     tree nodes, using a number of `<div>` HTML elements.
 33  * @extends RapidContext.Widget
 34  *
 35  * @example {JavaScript}
 36  * var tree = RapidContext.Widget.Tree({ style: { width: "200px", height: "400px" } });
 37  * var root = RapidContext.Widget.TreeNode({ folder: true, name: "Root" });
 38  * var child = RapidContext.Widget.TreeNode({ name: "Child" });
 39  * root.addAll(child);
 40  * tree.addAll(root);
 41  *
 42  * @example {User Interface XML}
 43  * <Tree style="width: 200px; height: 400px;">
 44  *   <TreeNode name="Root">
 45  *     <TreeNode name="Child" />
 46  *   </TreeNode>
 47  * </Tree>
 48  */
 49 RapidContext.Widget.Tree = function (attrs/*, ...*/) {
 50     var o = MochiKit.DOM.DIV(attrs);
 51     RapidContext.Widget._widgetMixin(o, RapidContext.Widget.Tree);
 52     MochiKit.DOM.addElementClass(o, "widgetTree");
 53     o.resizeContent = o._resizeContent;
 54     o.selectedPath = null;
 55     o.setAttrs(attrs);
 56     o.addAll(MochiKit.Base.extend(null, arguments, 1));
 57     return o;
 58 };

 97 /**
 98  * Adds a single child tree node widget to this widget.
 99  *
100  * @param {Widget} child the tree node widget to add
101  */
102 RapidContext.Widget.Tree.prototype.addChildNode = function (child) {
103     if (!RapidContext.Widget.isWidget(child, "TreeNode")) {
104         throw new Error("Tree widget can only have TreeNode children");
105     }
106     this.appendChild(child);
107 };
108
109 /**
110  * Removes all marked tree nodes. When adding or updating tree nodes, any
111  * node modified is automatically unmarked (e.g. by calling `setAttrs` on the
112  * tree nodes or `addPath` on the tree). This makes it easy to prune a tree
113  * after an update, by initially marking all tree nodes with `markAll()`,
114  * inserting or touching all nodes to keep, and finally calling this method to
115  * remove the remaining nodes.
116  *
117  * @example
118  * tree.markAll();
119  * tree.addPath(["Root", "Child"]);
120  * ...
121  * tree.removeAllMarked();
122  */
123 RapidContext.Widget.Tree.prototype.removeAllMarked = function () {
124     var children = this.getChildNodes();
125     for (var i = 0; i < children.length; i++) {
126         if (children[i].marked === true) {
127             this.removeChildNode(children[i]);
128         } else {
129             children[i].removeAllMarked();
130         }
131     }
132 };
133
134 /**
135  * Marks this tree node and all child nodes recursively. When adding or
136  * updating tree nodes, any node modified is automatically unmarked (e.g. by
137  * calling `setAttrs` on the tree nodes or `addPath` on the tree). This makes
138  * it easy to prune a tree after an update, by initially marking all tree
139  * nodes, inserting or touching all nodes to keep, and finally calling
140  * `removeAllMarked()` to remove the remaining nodes.
141  *
142  * @example
143  * tree.markAll();
144  * tree.addPath(["Root", "Child"]);
145  * ...
146  * tree.removeAllMarked();
147  */
148 RapidContext.Widget.Tree.prototype.markAll = function () {
149     var children = this.getChildNodes();
150     for (var i = 0; i < children.length; i++) {
151         children[i].markAll();
152     }
153 };

155 /**
156  * Finds a root tree node with the specified name.
157  *
158  * @param {String} name the root tree node name
159  *
160  * @return {Widget} the root tree node found, or
161  *         null if not found
162  */
163 RapidContext.Widget.Tree.prototype.findRoot = function (name) {
164     var children = this.getChildNodes();
165     for (var i = 0; i < children.length; i++) {
166         if (children[i].name == name) {
167             return children[i];
168         }
169     }
170     return null;
171 };
172
173 /**
174  * Searches for a tree node from the specified path.
175  *
176  * @param {Array} path the tree node path (array of names)
177  *
178  * @return {Widget} the descendant tree node found, or
179  *         null if not found
180  */
181 RapidContext.Widget.Tree.prototype.findByPath = function (path) {
182     if (path == null || path.length < 1) {
183         return null;
184     }
185     var root = this.findRoot(path[0]);
186     if (root != null) {
187         return root.findByPath(path.slice(1));
188     } else {
189         return null;
190     }
191 };
192
193 /**
194  * Returns the currently selected tree node.
195  *
196  * @return {Widget} the currently selected tree node, or
197  *         null if no node is selected
198  */
199 RapidContext.Widget.Tree.prototype.selectedChild = function () {
200     if (this.selectedPath == null) {
201         return null;
202     } else {
203         return this.findByPath(this.selectedPath);
204     }
205 };
206
207 /**
208  * Sets the currently selected node in the tree. This method is only
209  * called from the tree node `select()` and `unselect()` methods.
210  *
211  * @param {Widget} node the new selected tree node, or null for none
212  */
213 RapidContext.Widget.Tree.prototype._handleSelect = function (node) {
214     var prev = this.selectedChild();
215     if (node == null) {
216         this.selectedPath = null;
217         RapidContext.Widget.emitSignal(this, "onunselect", prev);
218     } else {
219         if (prev != null && prev !== node) {
220             prev.unselect();
221         }
222         this.selectedPath = node.path();
223         RapidContext.Widget.emitSignal(this, "onselect", node);
224     }
225 };

245 /**
246  * Recursively expands all nodes. If a depth is specified,
247  * expansions will not continue below that depth.
248  *
249  * @param {Number} [depth] the optional maximum depth
250  */
251 RapidContext.Widget.Tree.prototype.expandAll = function (depth) {
252     if (typeof(depth) !== "number") {
253         depth = 10;
254     }
255     var children = this.getChildNodes();
256     for (var i = 0; depth > 0 && i < children.length; i++) {
257         children[i].expandAll(depth - 1);
258     }
259 };
260
261 /**
262  * Recursively collapses all nodes. If a depth is specified, only
263  * nodes below that depth will be collapsed.
264  *
265  * @param {Number} [depth] the optional minimum depth
266  */
267 RapidContext.Widget.Tree.prototype.collapseAll = function (depth) {
268     if (typeof(depth) !== "number") {
269         depth = 0;
270     }
271     var children = this.getChildNodes();
272     for (var i = 0; i < children.length; i++) {
273         children[i].collapseAll(depth - 1);
274     }
275 };

277 /**
278  * Adds a path to the tree as a recursive list of child nodes. If
279  * nodes in the specified path already exists, they will be used
280  * instead of creating new nodes.
281  *
282  * @param {Array} path the tree node path (array of names)
283  *
284  * @return {Widget} the last node in the path
285  */
286 RapidContext.Widget.Tree.prototype.addPath = function (path) {
287     if (path == null || path.length < 1) {
288         return null;
289     }
290     var node = this.findRoot(path[0]);
291     if (node == null) {
292         node = RapidContext.Widget.TreeNode({ name: path[0], folder: (path.length > 1) });
293         this.addChildNode(node);
294     }
295     node.marked = false;
296     for (var i = 1; i < path.length; i++) {
297         var child = node.findChild(path[i]);
298         if (child == null) {
299             var attrs = { name: path[i], folder: (path.length > i + 1) };
300             child = RapidContext.Widget.TreeNode(attrs);
301             node.addChildNode(child);
302         }
303         child.marked = false;
304         node = child;
305     }
306     return node;
307 };

309 /**
310  * Called when tree content should be resized. This method is also called when
311  * the widget is made visible in a container after being hidden.
312  */
313 RapidContext.Widget.Tree.prototype._resizeContent = function () {
314     // Work-around to restore scrollTop for WebKit browsers
315     if (this.scrollTop == 0 && this.selectedPath) {
316         var node = this.selectedChild();
317         var h = this.clientHeight;
318         var y = node.offsetTop + node.offsetHeight;
319         this.scrollTop = Math.round(y - h / 2);
320     }
321 };
