 20 /**
 21  * Creates a new procedure caller function. This function can be called either
 22  * as a constructor or as a plain function. In both cases it returns a new
 23  * JavaScript function with additional methods.
 24  *
 25  * @constructor
 26  * @param {String} procedure the procedure name
 27  * @property {String} procedure The procedure name.
 28  * @property {Array} args The arguments used in the last call.
 29  *
 30  * @class The procedure wrapper function. Used to provide a simplified way of
 31  *   calling a procedure and connecting results through signals (instead of
 32  *   using deferred callbacks).
 33  *
 34  *   The actual calls are performed with normal function calls, but the results
 35  *   are asynchronous. When called, the procedure function returns a
 36  *   `MochiKit.Async.Deferred` object (as the normal API call), but the results
 37  *   will also be signalled through the `onsuccess` signal.
 38  *
 39  *   Differing from normal functions, a procedure function will also ensure
 40  *   that only a single call is in progress at any time, automatically
 41  *   cancelling any previous call if needed.
 42  */
 43 RapidContext.Procedure = function (procedure) {
 44     var proc = function () {
 45         var self = arguments.callee;
 46         self.args = MochiKit.Base.extend(null, arguments);
 47         return self.recall();
 48     }
 49     MochiKit.Base.setdefault(proc, RapidContext.Procedure.prototype);
 50     proc.procedure = procedure;
 51     proc.args = null;
 52     proc._deferred = null;
 53     return proc;
 54 };

 56 /**
 57  * Creates a new procedure caller for each key-value-pair in the specified
 58  * object.
 59  *
 60  * @param {Object} obj an object mapping keys to procedure names
 61  *
 62  * @return {Object} an object mapping keys to procedure instances
 63  */
 64 RapidContext.Procedure.mapAll = function (obj) {
 65     var res = {};
 66     for (var k in obj) {
 67         res[k] = new RapidContext.Procedure(obj[k]);
 68     }
 69     return res;
 70 };

 87  * @name RapidContext.Procedure#onupdate

 92  * Emitted when the procedure response has been received. Each call corresponds

 94  * cancelled). The call response or error object will be sent as the event
 95  * data.
 96  *
 97  * @name RapidContext.Procedure#onresponse
 98  * @event

102  * Emitted when a procedure call returned a result. This event is emitted after

104  * Use the `onerror` or `oncancel` signals for other result statuses. The call
105  * response object will be sent as the event data.

107  * @name RapidContext.Procedure#onsuccess
108  * @event

112  * Emitted when a procedure call failed. This event is emitted after the
113  * `onresponse` event, but only if the procedure call returned an error. Use
114  * the `onsuccess` or `oncancel` for other result statuses. The call error
115  * object will be sent as the event data.
116  *
117  * @name RapidContext.Procedure#onerror
118  * @event
119  */
120
121 /**
122  * Emitted when a procedure call was cancelled. This event is emitted after the
123  * `onresponse` event, but only if the procedure call was cancelled. Use the
124  * `onsuccess` or `onerror` for other result statuses. No event data will be
125  * sent.
126  *
127  * @name RapidContext.Procedure#oncancel
128  * @event
129  */

131 /**
132  * Calls the procedure with the same arguments as used in the last call. The
133  * call is asynchronous, so results will not be returned by this method.
134  * Instead the results will be available through the `onsuccess` signal, for
135  * example.
136  *
137  * Note that any previously running call will automatically be cancelled, since
138  * only a single call can be processed at any time.
139  *
140  * @return {Deferred} a `MochiKit.Async.Deferred` object that will
141  *         callback with the response data or error
142  */
143 RapidContext.Procedure.prototype.recall = function () {
144     if (this.args === null) {
145         throw new Error("No arguments supplied for procedure call to " + this.procedure);
146     }
147     this.cancel();
148     MochiKit.Signal.signal(this, "oncall");
149     this._deferred = RapidContext.App.callProc(this.procedure, this.args);
150     this._deferred.addBoth(MochiKit.Base.bind("_callback", this));
151     return this._deferred;
152 };
153
154 /**
155  * The procedure deferred callback handler. Dispatches the appropriate signals
156  * depending on the result.
157  *
158  * @param {Object/Error} res the procedure result object
159  */
160 RapidContext.Procedure.prototype._callback = function (res) {
161     this._deferred = null;
162     MochiKit.Signal.signal(this, "onresponse", res);
163     if (res instanceof MochiKit.Async.CancelledError) {
164         MochiKit.Signal.signal(this, "oncancel");
165     } else if (res instanceof Error) {
166         MochiKit.Signal.signal(this, "onerror", res);
167     } else {
168         MochiKit.Signal.signal(this, "onsuccess", res);
169     }
170     return res;
171 };

173 /**
174  * Calls the procedure multiple times (in sequence) with different arguments
175  * (supplied as an array of argument arrays). The calls are asynchronous, so
176  * results will not be returned by this method. Instead an array with the
177  * results will be available through the `onupdate` and `onsuccess` signals,
178  * for example.
179  *
180  * Note that any previously running call will automatically be cancelled, since
181  * only a single call can be processed at any time. A result `transform`
182  * function can be supplied to transform each individual result. If the
183  * `transform` function throws an error, that result will be omitted.
184  *
185  * @param {Array} args the array of argument arrays
186  * @param {Function} [transform] the optional result transform function
187  *
188  * @return {Deferred} a `MochiKit.Async.Deferred` object that will
189  *         callback with the response data or error
190  */
191 RapidContext.Procedure.prototype.multicall = function (args, transform) {
192     this.cancel();
193     this._mapArgs = args;
194     this._mapPos = 0;
195     this._mapRes = [];
196     this._mapTransform = transform
197     this._nextCall();
198 }
199
200 /**
201  * The multicall deferred callback handler. Dispatches the appropriate signals
202  * depending on the result.
203  *
204  * @param {Object/Error} res the procedure result object
205  */
206 RapidContext.Procedure.prototype._nextCall = function (res) {
207     this._deferred = null;
208     if (typeof(res) != "undefined") {
209         MochiKit.Signal.signal(this, "onresponse", res);
210         if (res instanceof MochiKit.Async.CancelledError) {
211             MochiKit.Signal.signal(this, "oncancel");
212             return res;
213         } else if (res instanceof Error) {
214             MochiKit.Signal.signal(this, "onerror", res);
215             return res;
216         } else {
217             if (this._mapTransform == null) {
218                 this._mapRes.push(res);
219             } else {
220                 try {
221                     res = this._mapTransform(res);
222                     this._mapRes.push(res);
223                 } catch (ignore) {
224                     // Skip results with mapping errors
225                 }
226             }
227             MochiKit.Signal.signal(this, "onupdate", this._mapRes);
228         }
229     }
230     if (this._mapPos < this._mapArgs.length) {
231         this.args = this._mapArgs[this._mapPos++];
232         MochiKit.Signal.signal(this, "oncall");
233         this._deferred = RapidContext.App.callProc(this.procedure, this.args);
234         this._deferred.addBoth(MochiKit.Base.bind("_nextCall", this));
235         return this._deferred;
236     } else {
237         MochiKit.Signal.signal(this, "onsuccess", this._mapRes);
238         return this._mapRes;
239     }
240 }

242 /**
243  * Cancels any current execution of this procedure. This method does nothing if
244  * no procedure call was currently executing.
245  */
246 RapidContext.Procedure.prototype.cancel = function () {
247     if (this._deferred !== null) {
248         this._deferred.cancel();
249         return true;
250     } else {
251         return false;
252     }
253 };
254
255 /**
256  * Cancels any current execution and removes the reference to the arguments of
257  * this procedure.
258  */
259 RapidContext.Procedure.prototype.reset = function () {
260     this.cancel();
261     this.args = null;
262 };
