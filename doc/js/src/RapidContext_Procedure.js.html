 21 /**
 22  * Creates a new procedure caller function. This function can be called either
 23  * as a constructor or as a plain function. In both cases it returns a new
 24  * JavaScript function with additional methods.
 25  *
 26  * @constructor
 27  * @param {String} procedure the procedure name
 28  * @property {String} procedure The procedure name.
 29  * @property {Array} args The arguments used in the last call.
 30  *
 31  * @class The procedure wrapper function. Used to provide a simplified way of
 32  *   calling a procedure and connecting results through signals (instead of
 33  *   using deferred callbacks).
 34  *
 35  *   The actual calls are performed with normal function calls, but the results
 36  *   are asynchronous. When called, the procedure function returns a
 37  *   `MochiKit.Async.Deferred` object (as the normal API call), but the results
 38  *   will also be signalled through the `onsuccess` signal.
 39  *
 40  *   Differing from normal functions, a procedure function will also ensure
 41  *   that only a single call is in progress at any time, automatically
 42  *   cancelling any previous call if needed.
 43  */
 44 RapidContext.Procedure = function (procedure) {
 45     var proc = function () {
 46         var self = arguments.callee;
 47         self.args = MochiKit.Base.extend(null, arguments);
 48         return self.recall();
 49     }
 50     MochiKit.Base.setdefault(proc, RapidContext.Procedure.prototype);
 51     proc.procedure = procedure;
 52     proc.args = null;
 53     proc._deferred = null;
 54     return proc;
 55 };

 57 /**
 58  * Creates a new procedure caller for each key-value-pair in the specified
 59  * object.
 60  *
 61  * @param {Object} obj an object mapping keys to procedure names
 62  *
 63  * @return {Object} an object mapping keys to procedure instances
 64  */
 65 RapidContext.Procedure.mapAll = function (obj) {
 66     var res = {};
 67     for (var k in obj) {
 68         res[k] = new RapidContext.Procedure(obj[k]);
 69     }
 70     return res;
 71 };

 93  * Emitted when the procedure response has been received. Each call corresponds

 96  * data.
 97  *
 98  * @name RapidContext.Procedure#onresponse

103  * Emitted when a procedure call returned a result. This event is emitted after

108  * @name RapidContext.Procedure#onsuccess

113  * Emitted when a procedure call failed. This event is emitted after the
114  * `onresponse` event, but only if the procedure call returned an error. Use
115  * the `onsuccess` or `oncancel` for other result statuses. The call error
116  * object will be sent as the event data.
117  *
118  * @name RapidContext.Procedure#onerror
119  * @event
120  */
121
122 /**
123  * Emitted when a procedure call was cancelled. This event is emitted after the
124  * `onresponse` event, but only if the procedure call was cancelled. Use the
125  * `onsuccess` or `onerror` for other result statuses. No event data will be
126  * sent.
127  *
128  * @name RapidContext.Procedure#oncancel
129  * @event
130  */

132 /**
133  * Calls the procedure with the same arguments as used in the last call. The
134  * call is asynchronous, so results will not be returned by this method.
135  * Instead the results will be available through the `onsuccess` signal, for
136  * example.
137  *
138  * Note that any previously running call will automatically be cancelled, since
139  * only a single call can be processed at any time.
140  *
141  * @return {Deferred} a `MochiKit.Async.Deferred` object that will
142  *         callback with the response data or error
143  */
144 RapidContext.Procedure.prototype.recall = function () {
145     if (this.args === null) {
146         throw new Error("No arguments supplied for procedure call to " + this.procedure);
147     }
148     this.cancel();
149     MochiKit.Signal.signal(this, "oncall");
150     this._deferred = RapidContext.App.callProc(this.procedure, this.args);
151     this._deferred.addBoth(MochiKit.Base.bind("_callback", this));
152     return this._deferred;
153 };
154
155 /**
156  * The procedure deferred callback handler. Dispatches the appropriate signals
157  * depending on the result.
158  *
159  * @param {Object/Error} res the procedure result object
160  */
161 RapidContext.Procedure.prototype._callback = function (res) {
162     this._deferred = null;
163     MochiKit.Signal.signal(this, "onresponse", res);
164     if (res instanceof MochiKit.Async.CancelledError) {
165         MochiKit.Signal.signal(this, "oncancel");
166     } else if (res instanceof Error) {
167         MochiKit.Signal.signal(this, "onerror", res);
168     } else {
169         MochiKit.Signal.signal(this, "onsuccess", res);
170     }
171     return res;
172 };

174 /**
175  * Calls the procedure multiple times (in sequence) with different arguments
176  * (supplied as an array of argument arrays). The calls are asynchronous, so
177  * results will not be returned by this method. Instead an array with the
178  * results will be available through the `onupdate` and `onsuccess` signals,
179  * for example.
180  *
181  * Note that any previously running call will automatically be cancelled, since
182  * only a single call can be processed at any time. A result `transform`
183  * function can be supplied to transform each individual result. If the
184  * `transform` function throws an error, that result will be omitted.
185  *
186  * @param {Array} args the array of argument arrays
187  * @param {Function} [transform] the optional result transform function
188  *
189  * @return {Deferred} a `MochiKit.Async.Deferred` object that will
190  *         callback with the response data or error
191  */
192 RapidContext.Procedure.prototype.multicall = function (args, transform) {
193     this.cancel();
194     this._mapArgs = args;
195     this._mapPos = 0;
196     this._mapRes = [];
197     this._mapTransform = transform
198     this._nextCall();
199 }
200
201 /**
202  * The multicall deferred callback handler. Dispatches the appropriate signals
203  * depending on the result.
204  *
205  * @param {Object/Error} res the procedure result object
206  */
207 RapidContext.Procedure.prototype._nextCall = function (res) {
208     this._deferred = null;
209     if (typeof(res) != "undefined") {
210         MochiKit.Signal.signal(this, "onresponse", res);
211         if (res instanceof MochiKit.Async.CancelledError) {
212             MochiKit.Signal.signal(this, "oncancel");
213             return res;
214         } else if (res instanceof Error) {
215             MochiKit.Signal.signal(this, "onerror", res);
216             return res;
217         } else {
218             if (this._mapTransform == null) {
219                 this._mapRes.push(res);
220             } else {
221                 try {
222                     res = this._mapTransform(res);
223                     this._mapRes.push(res);
224                 } catch (ignore) {
225                     // Skip results with mapping errors
226                 }
227             }
228             MochiKit.Signal.signal(this, "onupdate", this._mapRes);
229         }
230     }
231     if (this._mapPos < this._mapArgs.length) {
232         this.args = this._mapArgs[this._mapPos++];
233         MochiKit.Signal.signal(this, "oncall");
234         this._deferred = RapidContext.App.callProc(this.procedure, this.args);
235         this._deferred.addBoth(MochiKit.Base.bind("_nextCall", this));
236         return this._deferred;
237     } else {
238         MochiKit.Signal.signal(this, "onsuccess", this._mapRes);
239         return this._mapRes;
240     }
241 }

243 /**
244  * Cancels any current execution of this procedure. This method does nothing if
245  * no procedure call was currently executing.
246  */
247 RapidContext.Procedure.prototype.cancel = function () {
248     if (this._deferred !== null) {
249         this._deferred.cancel();
250         return true;
251     } else {
252         return false;
253     }
254 };
255
256 /**
257  * Cancels any current execution and removes the reference to the arguments of
258  * this procedure.
259  */
260 RapidContext.Procedure.prototype.reset = function () {
261     this.cancel();
262     this.args = null;
263 };
