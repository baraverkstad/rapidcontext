172 /**
173  * Moves a node to the top of the SVG drawing order (i.e. z-index).
174  * Note that this will only have effect on other SVG DOM nodes with
175  * the same parent node. Otherwise, the node must be moved down in
176  * the SVG DOM tree by changing parent node.
177  *
178  * @param {Node/String} node the SVG DOM node or unique id
179  */
180 RapidContext.SVG.moveToTop = function (node) {
181     node = MochiKit.DOM.getElement(node);
182     if (node != null) {
183         var parent = node.parentNode;
184         if (parent && parent.lastChild !== node) {
185             parent.appendChild(node);
186         }
187     }
188 };
189
190 /**
191  * Moves a node to the bottom of the SVG drawing order (i.e. z-index).
192  * Note that this will only have effect on other SVG DOM nodes with
193  * the same parent node. Otherwise, the node must be moved up in
194  * the SVG DOM tree by changing parent node.
195  *
196  * @param {Node/String} node the SVG DOM node or unique id
197  */
198 RapidContext.SVG.moveToBottom = function (node) {
199     node = MochiKit.DOM.getElement(node);
200     if (node != null) {
201         var parent = node.parentNode;
202         if (parent && parent.firstChild !== node) {
203             parent.insertBefore(node, parent.firstChild);
204         }
205     }
206 };

208 /**
209  * Adds a rotation transform to a SVG DOM node. Any previous
210  * rotation transform will be kept intact.
211  *
212  * @param {Node/String} node the SVG DOM node or unique id
213  * @param {String/Number} angle the numeric angle
214  * @param {String/Number} [x] the x coordinate value
215  * @param {String/Number} [y] the y coordinate value
216  */
217 RapidContext.SVG.rotate = function (node, angle, x, y) {
218     var str = MochiKit.DOM.getNodeAttribute(node, "transform");
219     x = x || 0;
220     y = y || 0;
221     if (str == null || str == "") {
222         str = "";
223     } else {
224         str += " ";
225     }
226     str += "rotate(" + angle + "," + x + "," + y + ")";
227     MochiKit.DOM.setNodeAttribute(node, "transform", str);
228 };
