 21 /**
 22  * Creates a new tree node widget.
 23  *
 24  * @constructor
 25  * @param {Object} attrs the widget and node attributes
 26  * @param {String} attrs.name the tree node name
 27  * @param {Boolean} [attrs.folder] the folder flag, defaults to `false` if no
 28  *            child nodes are provided in constructor call
 29  * @param {String} [attrs.icon] the icon reference to use, defaults
 30  *            to "FOLDER" for folders and "DOCUMENT" otherwise
 31  * @param {String} [attrs.tooltip] the tooltip text when hovering
 32  * @param {Boolean} [attrs.hidden] the hidden widget flag, defaults to `false`
 33  * @param {Widget} [...] the child tree node widgets
 34  *
 35  * @return {Widget} the widget DOM node
 36  *
 37  * @class The tree node widget class. Used to provide a tree node in a tree,
 38  *     using a number of `<div>` HTML elements. Note that events should
 39  *     normally not be listened for on individual tree nodes, but rather on
 40  *     the tree as a whole.
 41  * @extends RapidContext.Widget
 42  *
 43  * @example {JavaScript}
 44  * var parent = RapidContext.Widget.TreeNode({ folder: true, name: "Parent" });
 45  * var child = RapidContext.Widget.TreeNode({ name: "Child" });
 46  * parent.addAll(child);
 47  *
 48  * @example {User Interface XML}
 49  * <TreeNode name="Parent">
 50  *   <TreeNode name="Child" />
 51  * </TreeNode>
 52  */
 53 RapidContext.Widget.TreeNode = function (attrs/*, ...*/) {
 54     var toggle = RapidContext.Widget.Icon("fa fa-fw");
 55     var icon = RapidContext.Widget.Icon("fa fa-fw fa-dot-circle-o");
 56     var label = MochiKit.DOM.SPAN({ "class": "widgetTreeNodeText" });
 57     var cls = "widgetTreeNodeLabel overflow-ellipsis text-nowrap";
 58     var div = MochiKit.DOM.DIV({ "class": cls }, toggle, icon, label);
 59     var o = MochiKit.DOM.DIV({}, div);
 60     RapidContext.Widget._widgetMixin(o, RapidContext.Widget.TreeNode);
 61     MochiKit.DOM.addElementClass(o, "widgetTreeNode");
 62     var isFolder = (arguments.length > 1);
 63     attrs = MochiKit.Base.update({ name: "Tree Node", folder: isFolder }, attrs);
 64     o.setAttrs(attrs);
 65     o.addAll(Array.prototype.slice.call(arguments, 1));
 66     // FIXME: Consider using a single click handler in parent Tree instead...
 67     div.addEventListener("click", o._handleLabelClick);
 68     return o;
 69 };

 72 RapidContext.Widget.Classes.TreeNode = RapidContext.Widget.TreeNode;

 74 /**
 75  * Destroys this widget.
 76  */
 77 RapidContext.Widget.TreeNode.prototype.destroy = function () {
 78     // FIXME: Use AbortSignal instead to disconnect
 79     this.firstChild.removeEventListener("click", this._handleLabelClick);
 80 };
 81
 82 /**
 83  * Handles label click events.
 84  *
 85  * @param {Event} evt the DOM Event object
 86  */
 87 RapidContext.Widget.TreeNode.prototype._handleLabelClick = function (evt) {
 88     var node = MochiKit.DOM.getFirstParentByTagAndClassName(this, "DIV", "widgetTreeNode");
 89     if (RapidContext.Widget.isWidget(evt.target, "Icon")) {
 90         node.toggle();
 91     } else {
 92         node.select();
 93     }
 94 };

 96 /**
 97  * Returns and optionally creates the widget container DOM node. If a
 98  * child container is created, it will be hidden by default.
 99  *
100  * @param {Boolean} [create] the create flag, defaults to `false`
101  *
102  * @return {Node} the container DOM node, or
103  *         null if this widget has no container (yet)
104  */
105 RapidContext.Widget.TreeNode.prototype._containerNode = function (create) {
106     var container = this.lastChild;
107     if (MochiKit.DOM.hasElementClass(container, "widgetTreeNodeContainer")) {
108         return container;
109     } else if (create) {
110         container = MochiKit.DOM.DIV({ "class": "widgetTreeNodeContainer widgetHidden" });
111         this.appendChild(container);
112         this.firstChild.childNodes[0].setAttrs({ "class": "fa fa-fw fa-plus-square-o" });
113         if (!this.attributes.icon) {
114             this.firstChild.childNodes[1].setAttrs({ "class": "fa fa-fw fa-folder" });
115         }
116         return container;
117     } else {
118         return null;
119     }
120 };

122 /**
123  * Updates the widget or HTML DOM node attributes.
124  *
125  * @param {Object} attrs the widget and node attributes to set
126  * @param {String} [attrs.name] the tree node name
127  * @param {Boolean} [attrs.folder] the folder flag, cannot be
128  *            reverted to `false` once set (implicitly or explicitly)
129  * @param {Icon/Object/String} [attrs.icon] icon the icon to set, or
130  *            null to remove
131  * @param {String} [attrs.tooltip] the tooltip text when hovering
132  * @param {Boolean} [attrs.hidden] the hidden widget flag
133  */
134 RapidContext.Widget.TreeNode.prototype.setAttrs = function (attrs) {
135     attrs = MochiKit.Base.update({}, attrs);
136     this.marked = false;
137     var locals = RapidContext.Util.mask(attrs, ["name", "folder", "icon", "tooltip"]);
138     if (typeof(locals.name) != "undefined") {
139         this.name = locals.name;
140         var node = this.firstChild.firstChild;
141         while (!MochiKit.DOM.hasElementClass(node, "widgetTreeNodeText")) {
142             node = node.nextSibling;
143         }
144         MochiKit.DOM.replaceChildNodes(node, locals.name);
145     }
146     if (MochiKit.Base.bool(locals.folder)) {
147         this._containerNode(true);
148     }
149     if (typeof(locals.icon) != "undefined") {
150         this.setAttribute("icon", String(locals.icon));
151         var icon = RapidContext.Widget.Icon(locals.icon);
152         MochiKit.DOM.swapDOM(this.firstChild.childNodes[1], icon);
153     }
154     if (typeof(locals.tooltip) != "undefined") {
155         this.firstChild.title = locals.tooltip;
156     }
157     this.__setAttrs(attrs);
158 };

160 /**
161  * Adds a single child tree node widget to this widget.
162  *
163  * @param {Widget} child the tree node widget to add
164  */
165 RapidContext.Widget.TreeNode.prototype.addChildNode = function (child) {
166     if (!RapidContext.Widget.isWidget(child, "TreeNode")) {
167         throw new Error("TreeNode widget can only have TreeNode children");
168     }
169     this._containerNode(true).appendChild(child);
170 };

172 /**
173  * Removes all marked tree nodes. When adding or updating tree nodes, any
174  * node modified is automatically unmarked (e.g. by calling `setAttrs`). This
175  * makes it easy to prune a tree after an update, by initially marking all
176  * tree nodes with `markAll()`, inserting or touching all nodes to keep, and
177  * finally calling this method to remove the remaining nodes.
178  *
179  * @example
180  * parent.markAll();
181  * parent.setAttrs();
182  * child.setAttrs();
183  * ...
184  * parent.removeAllMarked();
185  */
186 RapidContext.Widget.TreeNode.prototype.removeAllMarked = function () {
187     var children = this.getChildNodes();
188     for (var i = 0; i < children.length; i++) {
189         if (children[i].marked === true) {
190             this.removeChildNode(children[i]);
191         } else {
192             children[i].removeAllMarked();
193         }
194     }
195 };
196
197 /**
198  * Marks this tree node and all child nodes recursively. When adding or
199  * updating tree nodes, any node modified is automatically unmarked (e.g. by
200  * calling `setAttrs`). This makes it easy to prune a tree after an update, by
201  * initially marking all tree nodes, inserting or touching all nodes to keep,
202  * and finally calling `removeAllMarked()` to remove the remaining nodes.
203  *
204  * @example
205  * parent.markAll();
206  * parent.setAttrs();
207  * child.setAttrs();
208  * ...
209  * parent.removeAllMarked();
210  */
211 RapidContext.Widget.TreeNode.prototype.markAll = function () {
212     this.marked = true;
213     var children = this.getChildNodes();
214     for (var i = 0; i < children.length; i++) {
215         children[i].markAll();
216     }
217 };

219 /**
220  * Checks if this node is a folder.
221  *
222  * @return {Boolean} `true` if this node is a folder, or
223  *         `false` otherwise
224  */
225 RapidContext.Widget.TreeNode.prototype.isFolder = function () {
226     return this._containerNode() != null;
227 };
228
229 /**
230  * Checks if this folder node is expanded.
231  *
232  * @return {Boolean} `true` if this node is expanded, or
233  *         `false` otherwise
234  */
235 RapidContext.Widget.TreeNode.prototype.isExpanded = function () {
236     var container = this._containerNode();
237     return container != null &&
238            !MochiKit.DOM.hasElementClass(container, "widgetHidden");
239 };

241 /**
242  * Checks if this node is selected.
243  *
244  * @return {Boolean} `true` if the node is selected, or
245  *         `false` otherwise
246  */
247 RapidContext.Widget.TreeNode.prototype.isSelected = function () {
248     return MochiKit.DOM.hasElementClass(this.firstChild, "selected");
249 };

251 /**
252  * Returns the ancestor tree widget.
253  *
254  * @return {Widget} the ancestor tree widget, or
255  *         null if none was found
256  */
257 RapidContext.Widget.TreeNode.prototype.tree = function () {
258     var parent = this.parent();
259     if (parent != null) {
260         return parent.tree();
261     }
262     if (RapidContext.Widget.isWidget(this.parentNode, "Tree")) {
263         return this.parentNode;
264     } else {
265         return null;
266     }
267 };
268
269 /**
270  * Returns the parent tree node widget.
271  *
272  * @return {Widget} the parent tree node widget, or
273  *         null if this is a root node
274  */
275 RapidContext.Widget.TreeNode.prototype.parent = function () {
276     var node = this.parentNode;
277     if (MochiKit.DOM.hasElementClass(node, "widgetTreeNodeContainer")) {
278         return node.parentNode;
279     } else {
280         return null;
281     }
282 };

284 /**
285  * Returns the path to this tree node.
286  *
287  * @return {Array} the tree node path, i.e an array of node names
288  */
289 RapidContext.Widget.TreeNode.prototype.path = function () {
290     var parent = this.parent();
291     if (parent == null) {
292         return [this.name];
293     } else {
294         var path = parent.path();
295         path.push(this.name);
296         return path;
297     }
298 };

300 /**
301  * Finds a child tree node with the specified name.
302  *
303  * @param {String} name the child tree node name
304  *
305  * @return {Widget} the child tree node found, or
306  *         null if not found
307  */
308 RapidContext.Widget.TreeNode.prototype.findChild = function (name) {
309     var children = this.getChildNodes();
310     for (var i = 0; i < children.length; i++) {
311         if (children[i].name == name) {
312             return children[i];
313         }
314     }
315     return null;
316 };
317
318 /**
319  * Searches for a descendant tree node from the specified path.
320  *
321  * @param {Array} path the tree node path (array of node names)
322  *
323  * @return {Widget} the descendant tree node found, or
324  *         null if not found
325  */
326 RapidContext.Widget.TreeNode.prototype.findByPath = function (path) {
327     var node = this;
328     for (var i = 0; node != null && path != null && i < path.length; i++) {
329         node = node.findChild(path[i]);
330     }
331     return node;
332 };

334 /**
335  * Selects this tree node.
336  */
337 RapidContext.Widget.TreeNode.prototype.select = function () {
338     MochiKit.DOM.addElementClass(this.firstChild, "selected");
339     var tree = this.tree();
340     if (tree != null) {
341         tree._handleSelect(this);
342     }
343     this.expand();
344 };
345
346 /**
347  * Unselects this tree node.
348  */
349 RapidContext.Widget.TreeNode.prototype.unselect = function () {
350     if (this.isSelected()) {
351         MochiKit.DOM.removeElementClass(this.firstChild, "selected");
352         var tree = this.tree();
353         if (tree != null) {
354             tree._handleSelect(null);
355         }
356     }
357 };
358
359 /**
360  * Expands this node to display any child nodes. If the parent node
361  * is not expanded, it will be expanded as well.
362  */
363 RapidContext.Widget.TreeNode.prototype.expand = function () {
364     var parent = this.parent();
365     if (parent != null && !parent.isExpanded()) {
366         parent.expand();
367     }
368     var container = this._containerNode();
369     if (container != null && !this.isExpanded()) {
370         this.firstChild.childNodes[0].setAttrs({ "class": "fa fa-fw fa-minus-square-o" });
371         if (!this.attributes.icon) {
372             this.firstChild.childNodes[1].setAttrs({ "class": "fa fa-fw fa-folder-open" });
373         }
374         MochiKit.DOM.removeElementClass(container, "widgetHidden");
375         var tree = this.tree();
376         if (tree != null) {
377             var detail = { tree: tree, node: this };
378             tree._dispatch("expand", { detail: detail });
379         }
380     }
381 };
382
383 /**
384  * Recursively expands this node and all its children. If a depth is
385  * specified, expansions will not continue below that depth.
386  *
387  * @param {Number} [depth] the optional maximum depth
388  */
389 RapidContext.Widget.TreeNode.prototype.expandAll = function (depth) {
390     if (typeof(depth) !== "number") {
391         depth = 10;
392     }
393     this.expand();
394     var children = this.getChildNodes();
395     for (var i = 0; depth > 0 && i < children.length; i++) {
396         children[i].expandAll(depth - 1);
397     }
398 };
399
400 /**
401  * Collapses this node to hide any child nodes.
402  */
403 RapidContext.Widget.TreeNode.prototype.collapse = function () {
404     var container = this._containerNode();
405     if (container != null && this.isExpanded()) {
406         this.firstChild.childNodes[0].setAttrs({ "class": "fa fa-fw fa-plus-square-o" });
407         if (!this.attributes.icon) {
408             this.firstChild.childNodes[1].setAttrs({ "class": "fa fa-fw fa-folder" });
409         }
410         MochiKit.DOM.addElementClass(container, "widgetHidden");
411         var tree = this.tree();
412         if (tree != null) {
413             var detail = { tree: tree, node: this };
414             tree._dispatch("collapse", { detail: detail });
415         }
416     }
417 };
418
419 /**
420  * Recursively collapses this node and all its children. If a depth
421  * is specified, only children below that depth will be collapsed.
422  *
423  * @param {Number} [depth] the optional minimum depth
424  */
425 RapidContext.Widget.TreeNode.prototype.collapseAll = function (depth) {
426     if (typeof(depth) !== "number") {
427         depth = 0;
428     }
429     if (depth <= 0) {
430         this.collapse();
431     }
432     var children = this.getChildNodes();
433     for (var i = 0; i < children.length; i++) {
434         children[i].collapseAll(depth - 1);
435     }
436 };
437
438 /**
439  * Toggles expand and collapse for this node.
440  */
441 RapidContext.Widget.TreeNode.prototype.toggle = function () {
442     if (this.isExpanded()) {
443         this.collapse();
444     } else {
445         this.expand();
446     }
447 };
