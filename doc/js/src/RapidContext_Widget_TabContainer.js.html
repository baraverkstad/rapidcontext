 21 /**
 22  * Creates a new tab container widget.
 23  *
 24  * @constructor
 25  * @param {Object} attrs the widget and node attributes
 26  * @param {Boolean} [attrs.hidden] the hidden widget flag, defaults to false
 27  * @param {Widget} [...] the child widgets or DOM nodes (should be
 28  *            Pane widgets)
 29  *
 30  * @return {Widget} the widget DOM node
 31  *
 32  * @class The tab container widget class. Used to provide a set of tabbed
 33  *     pages, where the user can switch page freely. Internally it uses a
 34  *     `<div>` HTML element containing `Pane` widgets that are hidden and shown
 35  *     according to the page transitions. If a child `Pane` widget is
 36  *     `pageCloseable`, a close button will be available on the tab label.
 37  * @extends RapidContext.Widget
 38  *
 39  * @example {JavaScript}
 40  * var page1 = RapidContext.Widget.Pane({ pageTitle: "One" });
 41  * ...
 42  * var page2 = RapidContext.Widget.Pane({ pageTitle: "Two", pageCloseable: true });
 43  * ...
 44  * var attrs = { style: { width: "100%", height: "100%" } };
 45  * var tabs = RapidContext.Widget.TabContainer(attrs, page1, page2);
 46  *
 47  * @example {User Interface XML}
 48  * <TabContainer style="width: 100%; height: 100%;">
 49  *   <Pane pageTitle="One">
 50  *     ...
 51  *   </Pane>
 52  *   <Pane pageTitle="Two" pageCloseable="true">
 53  *     ...
 54  *   </Pane>
 55  * <TabContainer>
 56  */
 57 RapidContext.Widget.TabContainer = function (attrs/*, ... */) {
 58     var labels = MochiKit.DOM.DIV({ "class": "widgetTabContainerLabels" });
 59     var container = MochiKit.DOM.DIV({ "class": "widgetTabContainerContent" });
 60     var o = MochiKit.DOM.DIV(attrs, labels, container);
 61     RapidContext.Widget._widgetMixin(o, arguments.callee);
 62     MochiKit.DOM.addElementClass(o, "widgetTabContainer");
 63     RapidContext.Util.registerSizeConstraints(container, "100% - 20", "100% - 40");
 64     o.resizeContent = o._resizeContent;
 65     container.resizeContent = MochiKit.Base.noop;
 66     o._selectedIndex = -1;
 67     o.setAttrs(attrs);
 68     o.addAll(MochiKit.Base.extend(null, arguments, 1));
 69     return o;
 70 };
 71
 72 // Register widget class
 73 RapidContext.Widget.Classes.TabContainer = RapidContext.Widget.TabContainer;
 74
 75 /**
 76  * Returns an array with all child pane widgets. Note that the array
 77  * is a real JavaScript array, not a dynamic `NodeList`.
 78  *
 79  * @return {Array} the array of child DOM nodes
 80  */
 81 RapidContext.Widget.TabContainer.prototype.getChildNodes = function () {
 82     return MochiKit.Base.extend([], this.lastChild.childNodes);
 83 };
 84
 85 /**
 86  * Adds a single child page widget to this widget. The child widget
 87  * should be a `RapidContext.Widget.Pane` widget, or it will be added to a
 88  * new one.
 89  *
 90  * @param {Widget} child the page widget to add
 91  *
 92  * @see RapidContext.Widget.Pane
 93  */
 94 RapidContext.Widget.TabContainer.prototype.addChildNode = function (child) {
 95     if (!RapidContext.Widget.isWidget(child, "Pane")) {
 96         child = RapidContext.Widget.Pane(null, child);
 97     }
 98     RapidContext.Util.registerSizeConstraints(child, "100%", "100%");
 99     child.hide();
100     var text = MochiKit.DOM.SPAN(null, child.pageTitle);
101     if (child.pageCloseable) {
102         var icon = RapidContext.Widget.Icon({ "class": "fa fa-close", tooltip: "Close" });
103         // TODO: potential memory leak with stale child object references
104         icon.onclick = RapidContext.Widget._eventHandler("TabContainer", "_handleClose", child);
105     }
106     var label = MochiKit.DOM.DIV({ "class": "widgetTabContainerLabel" },
107                                  MochiKit.DOM.DIV({}, text, icon));
108     // TODO: potential memory leak with stale child object references
109     label.onclick = RapidContext.Widget._eventHandler("TabContainer", "selectChild", child);
110     this.firstChild.appendChild(label);
111     this.lastChild.appendChild(child);
112     if (this._selectedIndex < 0) {
113         this.selectChild(0);
114     }
115 };
116
117 /**
118  * Removes a single child DOM node from this widget. This method is
119  * sometimes overridden by child widgets in order to hide or control
120  * intermediate DOM nodes required by the widget.
121  *
122  * Note that this method will NOT destroy the removed child widget,
123  * so care must be taken to ensure proper child widget destruction.
124  *
125  * @param {Widget/Node} child the DOM node to remove
126  */
127 RapidContext.Widget.TabContainer.prototype.removeChildNode = function (child) {
128     var children = this.getChildNodes();
129     var index = MochiKit.Base.findIdentical(children, child);
130     if (index < 0) {
131         throw new Error("Cannot remove DOM node that is not a TabContainer child");
132     }
133     if (this._selectedIndex == index) {
134         child._handleExit();
135         this._selectedIndex = -1;
136     }
137     RapidContext.Widget.destroyWidget(this.firstChild.childNodes[index]);
138     MochiKit.DOM.removeElement(child);
139     RapidContext.Widget.emitSignal(child, "onclose");
140     if (this._selectedIndex > index) {
141         this._selectedIndex--;
142     }
143     if (this._selectedIndex < 0 && this.getChildNodes().length > 0) {
144         this.selectChild((index == 0) ? 0 : index - 1);
145     }
146 };

150 /**
151  * Returns the index of the currently selected child in the tab
152  * container.
153  *
154  * @return {Number} the index of the selected child, or
155  *         -1 if no child is selected
156  */
157 RapidContext.Widget.TabContainer.prototype.selectedIndex = function () {
158     return this._selectedIndex;
159 };
160
161 /**
162  * Returns the child widget currently selected in the tab container.
163  *
164  * @return {Node} the child widget selected, or
165  *         null if no child is selected
166  */
167 RapidContext.Widget.TabContainer.prototype.selectedChild = function () {
168     var children = this.getChildNodes();
169     return (this._selectedIndex < 0) ? null : children[this._selectedIndex];
170 };
171
172 /**
173  * Selects a specified child in the tab container. This method can be
174  * called without arguments to re-select the currently selected tab.
175  *
176  * @param {Number/Node} [indexOrChild] the child index or node
177  */
178 RapidContext.Widget.TabContainer.prototype.selectChild = function (indexOrChild) {
179     var children = this.getChildNodes();
180     if (this._selectedIndex >= 0) {
181         var label = this.firstChild.childNodes[this._selectedIndex];
182         MochiKit.DOM.removeElementClass(label, "selected");
183         children[this._selectedIndex]._handleExit();
184     }
185     var index = -1;
186     if (indexOrChild == null) {
187         index = this._selectedIndex;
188     } else if (typeof(indexOrChild) == "number") {
189         index = indexOrChild;
190     } else {
191         index = MochiKit.Base.findIdentical(children, indexOrChild);
192     }
193     this._selectedIndex = (index < 0 || index >= children.length) ? -1 : index;
194     if (this._selectedIndex >= 0) {
195         var label = this.firstChild.childNodes[this._selectedIndex];
196         MochiKit.DOM.addElementClass(label, "selected");
197         children[this._selectedIndex]._handleEnter();
198     }
199 };
200
201 /**
202  * Resizes the currently selected child. This method need not be called
203  * directly, but is automatically called whenever a parent node is
204  * resized. It optimizes the resize chain by only resizing those DOM
205  * child nodes that are visible, i.e. the currently selected tab
206  * container child.
207  */
208 RapidContext.Widget.TabContainer.prototype._resizeContent = function () {
209     RapidContext.Util.resizeElements(this.lastChild);
210     var child = this.selectedChild();
211     if (child != null) {
212         RapidContext.Util.resizeElements(child);
213     }
214 };
215
216 /**
217  * Handles the tab close event.
218  *
219  * @param {Node} child the child DOM node
220  * @param {Event} evt the MochiKit.Signal.Event object
221  */
222 RapidContext.Widget.TabContainer.prototype._handleClose = function (child, evt) {
223     evt.stop();
224     this.removeChildNode(child);
225 };
