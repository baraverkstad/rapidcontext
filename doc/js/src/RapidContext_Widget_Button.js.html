 15 // Namespace initialization
 16 if (typeof(RapidContext) == "undefined") {
 17     RapidContext = {};
 18 }
 19 RapidContext.Widget = RapidContext.Widget || { Classes: {}};

 21 /**
 22  * Creates a new button widget.
 23  *
 24  * @constructor
 25  * @param {Object} attrs the widget and node attributes
 26  * @param {Boolean} [attrs.disabled] the disabled widget flag, defaults to
 27  *            false
 28  * @param {Boolean} [attrs.hidden] the hidden widget flag, defaults to false
 29  * @param {Boolean} [attrs.highlight] the highlight option flag,
 30  *            defaults to false
 31  * @param {String} [attrs.icon] the icon reference to use, defaults
 32  *            to null (no icon)
 33  * @param {Object} [...] the child widgets or DOM nodes
 34  *
 35  * @return {Widget} the widget DOM node
 36  *
 37  * @class The button widget class. Used to provide a simple push
 38  *     button, using the `<button>` HTML element.
 39  * @extends RapidContext.Widget
 40  *
 41  * @example {JavaScript}
 42  * var closeBtn = RapidContext.Widget.Button({ icon: "OK", highlight: true }, "Close");
 43  *
 44  * @example {User Interface XML}
 45  * <Button id="closeBtn" icon="OK" highlight="true">Close</Button>
 46  */
 47 RapidContext.Widget.Button = function (attrs/*, ...*/) {
 48     var o = MochiKit.DOM.BUTTON({ type: attrs.type ||Â "button" });
 49     RapidContext.Widget._widgetMixin(o, arguments.callee);
 50     o.addClass("widgetButton");
 51     o.setAttrs(attrs);
 52     o.addAll(MochiKit.Base.extend(null, arguments, 1));
 53     return o;
 54 };

 67 /**
 68  * Updates the widget or HTML DOM node attributes.
 69  *
 70  * @param {Object} attrs the widget and node attributes to set
 71  * @param {Boolean} [attrs.disabled] the disabled widget flag
 72  * @param {Boolean} [attrs.hidden] the hidden widget flag
 73  * @param {Boolean} [attrs.highlight] the highlight option flag
 74  * @param {Icon/Object/String} [attrs.icon] the icon reference to use
 75  */
 76 RapidContext.Widget.Button.prototype.setAttrs = function (attrs) {
 77     attrs = MochiKit.Base.update({}, attrs);
 78     var locals = RapidContext.Util.mask(attrs, ["highlight", "icon"]);
 79     if (typeof(locals.highlight) != "undefined") {
 80         if (MochiKit.Base.bool(locals.highlight)) {
 81             this.addClass("widgetButtonHighlight");
 82         } else {
 83             this.removeClass("widgetButtonHighlight");
 84         }
 85     }
 86     if (typeof(locals.icon) != "undefined") {
 87         var iconNode = this.firstChild;
 88         if (!RapidContext.Widget.isWidget(iconNode, "Icon")) {
 89             iconNode = null;
 90         }
 91         if (iconNode == null && locals.icon != null) {
 92             if (typeof(locals.icon) === "string") {
 93                 locals.icon = RapidContext.Widget.Icon({ ref: locals.icon });
 94             } else if (!RapidContext.Widget.isWidget(locals.icon, "Icon")) {
 95                 locals.icon = RapidContext.Widget.Icon(locals.icon);
 96             }
 97             this.insertBefore(locals.icon, this.firstChild);
 98         } else if (iconNode != null && locals.icon != null) {
 99             if (RapidContext.Widget.isWidget(locals.icon, "Icon")) {
100                 MochiKit.DOM.swapDOM(iconNode, locals.icon);
101             } else if (typeof(locals.icon) === "string") {
102                 iconNode.setAttrs({ ref: locals.icon });
103             } else {
104                 iconNode.setAttrs(locals.icon);
105             }
106         } else if (iconNode != null && locals.icon == null) {
107             RapidContext.Widget.destroyWidget(iconNode);
108         }
109     }
110     this.__setAttrs(attrs);
111 };
