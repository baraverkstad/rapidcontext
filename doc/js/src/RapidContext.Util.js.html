 47 /**
 48  * Creates a dictionary object from a list of keys and values. Optionally a
 49  * list of key-value pairs can be provided instead. As a third option, a single
 50  * (non-array) value can be assigned to all the keys.
 51  *
 52  * If a key is specified twice, only the last value will be used. Note that
 53  * this function is the reverse of `MochiKit.Base.items()`,
 54  * `MochiKit.Base.keys()` and `MochiKit.Base.values()`.
 55  *
 56  * @param {Array} itemsOrKeys the list of keys or items
 57  * @param {Array} [values] the list of values (optional if key-value
 58  *            pairs are specified in first argument)
 59  *
 60  * @return {Object} an object with properties for each key-value pair
 61  *
 62  * @example
 63  * RapidContext.Util.dict(['a','b'], [1, 2])
 64  * ==> { a: 1, b: 2 }
 65  *
 66  * @example
 67  * RapidContext.Util.dict([['a', 1], ['b', 2]])
 68  * ==> { a: 1, b: 2 }
 69  *
 70  * @example
 71  * RapidContext.Util.dict(['a','b'], true)
 72  * ==> { a: true, b: true }
 73  */
 74 RapidContext.Util.dict = function (itemsOrKeys, values) {
 75     var o = {};
 76     if (!MochiKit.Base.isArrayLike(itemsOrKeys)) {
 77         throw new TypeError("First argument must be array-like");
 78     }
 79     if (MochiKit.Base.isArrayLike(values) && itemsOrKeys.length !== values.length) {
 80         throw new TypeError("Both arrays must be of same length");
 81     }
 82     for (var i = 0; i < itemsOrKeys.length; i++) {
 83         var k = itemsOrKeys[i];
 84         if (k === null || k === undefined) {
 85             throw new TypeError("Key at index " + i + " is null or undefined");
 86         } else if (MochiKit.Base.isArrayLike(k)) {
 87             o[k[0]] = k[1];
 88         } else if (MochiKit.Base.isArrayLike(values)) {
 89             o[k] = values[i];
 90         } else {
 91             o[k] = values;
 92         }
 93     }
 94     return o;
 95 };

 97 /**
 98  * Filters an object by removing a list of keys. A list of key names (or an
 99  * object whose property names will be used as keys) must be specified as an
100  * argument. A new object containing the source object values for the specified
101  * keys will be returned. The source object will be modified by removing all
102  * the specified keys.
103  *
104  * @param {Object} src the source object to select and modify
105  * @param {Array/Object} keys the list of keys to remove, or an
106  *            object with the keys to remove
107  *
108  * @return {Object} a new object containing the matching keys and
109  *             values found in the source object
110  *
111  * @deprecated This function will be removed in the future.
112  *
113  * @example
114  * var o = { a: 1, b: 2 };
115  * RapidContext.Util.mask(o, ['a', 'c']);
116  * ==> { a: 1 } and modifies o to { b: 2 }
117  *
118  * @example
119  * var o = { a: 1, b: 2 };
120  * RapidContext.Util.mask(o, { a: null, c: null });
121  * ==> { a: 1 } and modifies o to { b: 2 }
122  */
123 RapidContext.Util.mask = function (src, keys) {
124     var res = {};
125     if (!MochiKit.Base.isArrayLike(keys)) {
126         keys = MochiKit.Base.keys(keys);
127     }
128     for (var i = 0; i < keys.length; i++) {
129         var k = keys[i];
130         if (k in src) {
131             res[k] = src[k];
132             delete src[k];
133         }
134     }
135     return res;
136 };

138 /**
139  * Converts a string to a title-cased string. All word boundaries are replaced
140  * with a single space and the subsequent character is capitalized.
141  *
142  * All underscore ("_"), hyphen ("-") and lower-upper character pairs are
143  * recognized as word boundaries. Note that this function does not change the
144  * capitalization of other characters in the string.
145  *
146  * @param {String} str the string to convert
147  *
148  * @return {String} the converted string
149  *
150  * @example
151  * RapidContext.Util.toTitleCase("a short heading")
152  * ==> "A Short Heading"
153  *
154  * @example
155  * RapidContext.Util.toTitleCase("camelCase")
156  * ==> "Camel Case"
157  *
158  * @example
159  * RapidContext.Util.toTitleCase("bounding-box")
160  * ==> "Bounding Box"
161  *
162  * @example
163  * RapidContext.Util.toTitleCase("UPPER_CASE_VALUE")
164  * ==> "UPPER CASE VALUE"
165  */
166 RapidContext.Util.toTitleCase = function (str) {
167     str = str.replace(/[_-]+/g, " ").trim();
168     str = str.replace(/[a-z][A-Z]/g, function (match) {
169         return match.charAt(0) + " " + match.charAt(1);
170     });
171     str = str.replace(/(^|\s)[a-z]/g, function (match) {
172         return match.toUpperCase();
173     });
174     return str;
175 };

177 /**
178  * Returns the name of a function. If the function is anonymous (i.e. the
179  * `name` property is `undefined` or an empty string), the value of the
180  * `displayName` property is returned instead.
181  *
182  * @param {Function} func the function to check
183  *
184  * @return {String} the function name, or `undefined` if not available
185  *
186  * @example
187  * var o = { test: function () {} };
188  * RapidContext.Util.registerFunctionNames(o, "o");
189  * RapidContext.Util.functionName(o.test)
190  * ==> "o.test"
191  *
192  * @see RapidContext.Util.registerFunctionNames
193  */
194 RapidContext.Util.functionName = function (func) {
195     if (func == null) {
196         return null;
197     } else if (func.name != null && func.name != "") {
198         return func.name;
199     } else {
200         return func.displayName;
201     }
202 };

204 /**
205  * Registers function names for anonymous functions. This is useful when
206  * debugging code in Firebug or similar tools, as readable stack traces can be
207  * provided.
208  *
209  * This function will add a `displayName` property to all functions without a
210  * `name` property. Both the `obj` properties and `obj.prototype` properties
211  * are processed recursively, using the base name as a namespace (i.e.
212  * `[name].[property]` or `[name].prototype.[property]`).
213  *
214  * @param {Object} obj the function or object to process
215  * @param {String} [name] the function or object (class) name
216  *
217  * @example
218  * var o = { name: "MyObject", test: function () {} };
219  * RapidContext.Util.registerFunctionNames(o);
220  * o.test.displayName
221  * ==> "MyObject.test"
222  *
223  * @see RapidContext.Util.functionName
224  */
225 RapidContext.Util.registerFunctionNames = function (obj, name) {
226     function worker(o, name, stack) {
227         var isObj = (o != null && typeof(o) === "object");
228         var isFunc = (typeof(o) === "function");
229         var isAnon = isFunc && (o.name == null || o.name == "");
230         var isProto = (o === Object.prototype || o === Function.prototype);
231         var isNode = isObj && (typeof(o.nodeType) === "number");
232         var isVisited = (MochiKit.Base.findIdentical(stack, o) >= 0);
233         if (isFunc && isAnon && !o.displayName) {
234             o.displayName = name;
235         }
236         if ((isObj || isFunc) && !isProto && !isNode && !isVisited) {
237             stack.push(o);
238             for (var prop in o) {
239                 if (Object.prototype.hasOwnProperty.call(o, prop)) {
240                     worker(o[prop], name + "." + prop, stack);
241                 }
242             }
243             worker(o.prototype, name + ".prototype", stack);
244             stack.pop();
245         }
246     }
247     worker(obj, name || obj.name || obj.displayName || obj.NAME || "", []);
248 };

250 /**
251  * Resolves a relative URI to an absolute URI. This function will return
252  * absolute URI:s directly and traverse any "../" directory paths in the
253  * specified URI. The base URI provided must be absolute.
254  *
255  * @param {String} uri the relative URI to resolve
256  * @param {String} [base] the absolute base URI, defaults to the
257  *            the current document base URI
258  *
259  * @return {String} the resolved absolute URI
260  *
261  * @deprecated This function will be removed and/or renamed in the future.
262  *     Better solutions for handling URL:s is to use a URL-parsing library such
263  *     as URL.js.
264  */
265 RapidContext.Util.resolveURI = function (uri, base) {
266     var pos;
267     base = base || document.baseURI || document.getElementsByTagName("base")[0].href;
268     if (uri.indexOf(":") > 0) {
269         return uri;
270     } else if (uri.indexOf("#") == 0) {
271         pos = base.lastIndexOf("#");
272         if (pos >= 0) {
273             base = base.substring(0, pos);
274         }
275         return base + uri;
276     } else if (uri.indexOf("/") == 0) {
277         pos = base.indexOf("/", base.indexOf("://") + 3);
278         base = base.substring(0, pos);
279         return base + uri;
280     } else if (uri.indexOf("../") == 0) {
281         pos = base.lastIndexOf("/");
282         base = base.substring(0, pos);
283         uri = uri.substring(3);
284         return RapidContext.Util.resolveURI(uri, base);
285     } else {
286         pos = base.lastIndexOf("/");
287         base = base.substring(0, pos + 1);
288         return base + uri;
289     }
290 };

303 /**
304  * Returns `true` if the specified object looks like a DOM node. Otherwise,
305  * `false` will be returned. Any non-null object with a `nodeType` > 0 will be
306  * considered a DOM node by this function.
307  *
308  * @param {Object} obj the object to check
309  *
310  * @return {Boolean} `true` if the object looks like a DOM node, or
311  *         `false` otherwise
312  */
313 RapidContext.Util.isDOM = function (obj) {
314     return obj != null &&
315            typeof(obj.nodeType) === "number" &&
316            obj.nodeType > 0;
317 };

319 /**
320  * Returns `true` if the specified object looks like an HTML or XHTML DOM node.
321  * Otherwise, `false` will be returned. Any non-null object with a
322  * `nodeType` > 0 will be considered a DOM node, but only those with a matching
323  * `namespaceURI` will be considered HTML DOM nodes.
324  *
325  * @param {Object} obj the object to check
326  *
327  * @return {Boolean} `true` if the object looks like an HTML DOM node,
328  *         or `false` otherwise
329  */
330 RapidContext.Util.isHTML = function (obj) {
331     var ns = RapidContext.Util.NS.HTML;
332     return RapidContext.Util.isDOM(obj) &&
333            MochiKit.Base.findIdentical(ns, obj.namespaceURI) >= 0;
334 };

336 /**
337  * Returns an array with DOM node attribute name and value pairs. The name and
338  * value pairs are also stored in arrays with two elements.
339  *
340  * @param {Object} node the HTML DOM node
341  *
342  * @return {Array} an array containing attribute name and value
343  *             pairs (as arrays)

346     var res = [];

348     for (var i = 0; node != null && i < node.attributes.length; i++) {

355 /**
356  * Returns an immediate child node from a parent DOM node. If a numeric index
357  * is provided, the index will be range checked and any matching child DOM
358  * node will be returned. Otherwise, the DOM tree is traversed to find the
359  * immediate child that corresponds to the specified node.
360  *
361  * @param {Node} parent the parent HTML DOM node
362  * @param {Number/Node} indexOrNode the child index or a descendant node
363  *
364  * @return {Node} the child HTML DOM node, or
365  *         `null` if no matching node was found
366  *
367  * @example
368  * var child = RapidContext.Util.childNode(parent, 2);
369  * ==> parent.childNodes[2] or null
370  *
371  * @example
372  * var child = RapidContext.Util.childNode(node, evt.target());
373  * ==> child DOM node if descendant or null otherwise
374  */
375 RapidContext.Util.childNode = function (parent, indexOrNode) {
376     parent = MochiKit.DOM.getElement(parent);
377     if (typeof(indexOrNode) == "number") {
378         if (indexOrNode < 0 || indexOrNode >= parent.childNodes.length) {
379             return null;
380         } else {
381             return parent.childNodes[indexOrNode];
382         }
383     } else {
384         var node = MochiKit.DOM.getElement(indexOrNode);
385         while (node != null && node !== parent && node.parentNode !== parent) {
386             node = node.parentNode;
387         }
388         return (node == null || node === parent) ? null : node;
389     }
390 };

392 /**
393  * Creates a DOM node with a namespace.
394  *
395  * @param {String} ns the DOM namespace
396  * @param {String} tag the DOM tag name
397  * @param {Object} [attrs] the node attributes, or null for none
398  * @param {Object} [...] the nodes or text to add as children
399  *
400  * @return {Object} the DOM node created
401  *
402  * @example
403  * RapidContext.Util.createDOMExt("http://www.w3.org/2000/svg", "g");
404  * ==> an SVG <g> element
405  */
406 RapidContext.Util.createDOMExt = function (ns, tag, attrs/*, ...*/) {
407     var doc = MochiKit.DOM.currentDocument();
408     var node = (ns) ? doc.createElementNS(ns, tag) : doc.createElement(tag);
409     MochiKit.DOM.updateNodeAttributes(node, attrs);
410     var children = Array.prototype.slice.call(arguments, 3);
411     MochiKit.DOM.appendChildNodes(node, children);
412     return node;
413 };

428 /**
429  * Returns a function for creating a specific kind of DOM nodes. The returned
430  * function will optionally require a sequence of non-null arguments that will
431  * be added as attributes to the node creation. The returned function will
432  * otherwise work similar to the `createDOMExt()` function, taking attributes
433  * and child nodes.
434  *
435  * @param {String} ns the DOM namespace, or `null` for HTML
436  * @param {String} tag the DOM tag name
437  * @param {Array} [args] the array with required arguments, or `null` for no
438  *            required arguments
439  * @param {Object} [attrs] the default node attributes, or `null` for none
440  * @param {Object} [...] the default nodes or text to add as children
441  *
442  * @return {Function} the function that creates the DOM nodes
443  */
444 RapidContext.Util.createDOMFuncExt = function (ns, tag, args, attrs/*, ...*/) {
445     args = args || [];
446     attrs = attrs || {};
447     var children = Array.prototype.slice.call(arguments, 4);
448     return function (/*arg1, ..., argN, attrs, ...*/) {
449         var myArgs = Array.prototype.slice.call(arguments);
450         var myAttrs = MochiKit.Base.update({}, attrs);
451         args.forEach(function (key, idx) {
452             if (myArgs[idx] == null) {
453                 throw new Error("Argument '" + key + "' cannot be null");
454             }
455             myAttrs[key] = myArgs[idx];
456         });
457         MochiKit.Base.update(myAttrs, myArgs[args.length]);
458         var myChildren = [].concat(children, myArgs.slice(args.length + 1));
459         return RapidContext.Util.createDOMExt(ns, tag, myAttrs, myChildren);
460     };
461 };

463 /**
464  * Blurs (unfocuses) a specified DOM node and all relevant child nodes. This
465  * function will recursively blur all `<a>`, `<button>`, `<input>`,
466  * `<textarea>` and `<select>` child nodes found.
467  *
468  * @param {Object} node the HTML DOM node
469  */
470 RapidContext.Util.blurAll = function (node) {
471     node.blur();
472     var tags = ["A", "BUTTON", "INPUT", "TEXTAREA", "SELECT"];
473     for (var i = 0; i < tags.length; i++) {
474         var nodes = node.getElementsByTagName(tags[i]);
475         for (var j = 0; j < nodes.length; j++) {
476             nodes[j].blur();
477         }
478     }

481 /**
482  * Registers size constraints for the element width and/or height. The
483  * constraints may either be fixed numeric values or simple arithmetic (in a
484  * string). The formulas will be converted to CSS calc() expressions.
485  *
486  * Legacy constraint functions are still supported and must take two arguments
487  * (parent width and height) and should return a number. The returned number is
488  * set as the new element width or height (in pixels). Any returned value will
489  * also be bounded by the parent element size to avoid calculation errors.
490  *
491  * @param {Object} node the HTML DOM node
492  * @param {Number/String/Function} [width] the width constraint
493  * @param {Number/String/Function} [height] the height constraint
494  *
495  * @see RapidContext.Util.resizeElements
496  *
497  * @example
498  * RapidContext.Util.registerSizeConstraints(node, "50%-20", "100%");
499  * ==> Sets width to 50%-20 px and height to 100% of parent dimension
500  *
501  * @example
502  * RapidContext.Util.resizeElements(node, otherNode);
503  * ==> Evaluates the size constraints for both nodes
504  */
505 RapidContext.Util.registerSizeConstraints = function (node, width, height) {
506     function toCSS(val) {
507         if (/[+-]/.test(val)) {
508             val = "calc( " + val.replace(/[+-]/g, " $& ") + " )";
509         }
510         val = val.replace(/(\d)( |$)/g, "$1px$2");
511         return val;
512     }
513     node = MochiKit.DOM.getElement(node);
514     if (typeof(width) == "number" || typeof(width) == "string") {
515         node.style.width = toCSS(String(width));
516     } else if (typeof(width) == "function") {
517         console.info("registerSizeConstraints: width function support will be removed", node);
518         node.sizeConstraints = node.sizeConstraints || { w: null, h: null };
519         node.sizeConstraints.w = width;
520     }
521     if (typeof(height) == "number" || typeof(height) == "string") {
522         node.style.height = toCSS(String(height));
523     } else if (typeof(height) == "function") {
524         console.info("registerSizeConstraints: height function support will be removed", node);
525         node.sizeConstraints = node.sizeConstraints || { w: null, h: null };
526         node.sizeConstraints.h = height;
527     }
528 };

530 /**
531  * Resizes one or more DOM nodes using their registered size constraints and
532  * their parent element sizes. The resize operation will only modify those
533  * elements that have constraints, but will perform a depth-first recursion
534  * over all element child nodes as well.
535  *
536  * Partial constraints are accepted, in which case only the width or the height
537  * is modified. Aspect ratio constraints are applied after the width and height
538  * constraints. The result will always be bounded by the parent element width
539  * or height.
540  *
541  * The recursive descent of this function can be limited by adding a
542  * `resizeContent` function to a DOM node. Such a function will be called to
543  * handle all subnode resizing, making it possible to limit or omitting the
544  * DOM tree traversal.
545  *
546  * @param {Object} [...] the HTML DOM nodes to resize
547  *
548  * @see RapidContext.Util.registerSizeConstraints
549  *
550  * @example
551  * RapidContext.Util.resizeElements(node);
552  * ==> Evaluates the size constraints for a node and all child nodes
553  *
554  * @example
555  * elem.resizeContent = MochiKit.Base.noop;
556  * ==> Assigns a no-op child resize handler to elem
557  */
558 RapidContext.Util.resizeElements = function (/* ... */) {
559     Array.prototype.slice.call(arguments).forEach(function (arg) {
560         var node = MochiKit.DOM.getElement(arg);
561         if (node && node.nodeType === 1 && node.parentNode && node.sizeConstraints) {
562             var ref = { w: node.parentNode.w, h: node.parentNode.h };
563             if (ref.w == null && ref.h == null) {
564                 ref = MochiKit.Style.getElementDimensions(node.parentNode, true);
565             }
566             var dim = RapidContext.Util._evalConstraints(node.sizeConstraints, ref);
567             MochiKit.Style.setElementDimensions(node, dim);
568             node.w = dim.w;
569             node.h = dim.h;
570         }
571         if (node && typeof(node.resizeContent) == "function") {
572             try {
573                 node.resizeContent();
574             } catch (e) {
575                 console.error("Error in resizeContent()", node, e);
576             }
577         } else if (node && node.childNodes) {
578             Array.prototype.slice.call(node.childNodes).forEach(function (child) {
579                 if (child.nodeType === 1) {
580                     RapidContext.Util.resizeElements(child);
581                 }
582             });
583         }
584     });
585 };

587 /**
588  * Evaluates the size constraint functions with a refeence dimension
589  * object. This is an internal function used to encapsulate the
590  * function calls and provide logging on errors.
591  *
592  * @param {Object} sc the size constraints object
593  * @param {Object} ref the MochiKit.Style.Dimensions maximum
594  *            reference values
595  *
596  * @return {Object} the MochiKit.Style.Dimensions with evaluated size
597  *         constraint values (some may be null)
598  */
599 RapidContext.Util._evalConstraints = function (sc, ref) {
600     var w, h;
601     if (typeof(sc.w) == "function") {
602         try {
603             w = Math.max(0, Math.min(ref.w, sc.w(ref.w, ref.h)));
604         } catch (e) {
605             console.error("Error evaluating width size constraint; " +
606                           "w: " + ref.w + ", h: " + ref.h, e);
607         }
608     }
609     if (typeof(sc.h) == "function") {
610         try {
611             h = Math.max(0, Math.min(ref.h, sc.h(ref.w, ref.h)));
612         } catch (e) {
613             console.error("Error evaluating height size constraint; " +
614                           "w: " + ref.w + ", h: " + ref.h, e);
615         }
616     }
617     if (w != null) {
618         w = Math.floor(w);
619     }
620     if (h != null) {
621         h = Math.floor(h);
622     }
623     return new MochiKit.Style.Dimensions(w, h);
624 };
