 47 /**
 48  * Creates a dictionary object from a list of keys and values. Optionally a
 49  * list of key-value pairs can be provided instead. As a third option, a single
 50  * (non-array) value can be assigned to all the keys.
 51  *
 52  * If a key is specified twice, only the last value will be used. Note that
 53  * this function is the reverse of `MochiKit.Base.items()`,
 54  * `MochiKit.Base.keys()` and `MochiKit.Base.values()`.
 55  *
 56  * @param {Array} itemsOrKeys the list of keys or items
 57  * @param {Array} [values] the list of values (optional if key-value
 58  *            pairs are specified in first argument)
 59  *
 60  * @return {Object} an object with properties for each key-value pair
 61  *
 62  * @example
 63  * RapidContext.Util.dict(['a','b'], [1, 2])
 64  * ==> { a: 1, b: 2 }
 65  *
 66  * @example
 67  * RapidContext.Util.dict([['a', 1], ['b', 2]])
 68  * ==> { a: 1, b: 2 }
 69  *
 70  * @example
 71  * RapidContext.Util.dict(['a','b'], true)
 72  * ==> { a: true, b: true }
 73  */
 74 RapidContext.Util.dict = function (itemsOrKeys, values) {
 75     var o = {};
 76     if (!MochiKit.Base.isArrayLike(itemsOrKeys)) {
 77         throw new TypeError("First argument must be array-like");
 78     }
 79     if (MochiKit.Base.isArrayLike(values) && itemsOrKeys.length !== values.length) {
 80         throw new TypeError("Both arrays must be of same length");
 81     }
 82     for (var i = 0; i < itemsOrKeys.length; i++) {
 83         var k = itemsOrKeys[i];
 84         if (k === null || k === undefined) {
 85             throw new TypeError("Key at index " + i + " is null or undefined");
 86         } else if (MochiKit.Base.isArrayLike(k)) {
 87             o[k[0]] = k[1];
 88         } else if (MochiKit.Base.isArrayLike(values)) {
 89             o[k] = values[i];
 90         } else {
 91             o[k] = values;
 92         }
 93     }
 94     return o;
 95 };

 97 /**
 98  * Filters an object by removing a list of keys. A list of key names (or an
 99  * object whose property names will be used as keys) must be specified as an
100  * argument. A new object containing the source object values for the specified
101  * keys will be returned. The source object will be modified by removing all
102  * the specified keys.
103  *
104  * @param {Object} src the source object to select and modify
105  * @param {Array/Object} keys the list of keys to remove, or an
106  *            object with the keys to remove
107  *
108  * @return {Object} a new object containing the matching keys and
109  *             values found in the source object
110  *
111  * @deprecated This function will be removed in the future.
112  *
113  * @example
114  * var o = { a: 1, b: 2 };
115  * RapidContext.Util.mask(o, ['a', 'c']);
116  * ==> { a: 1 } and modifies o to { b: 2 }
117  *
118  * @example
119  * var o = { a: 1, b: 2 };
120  * RapidContext.Util.mask(o, { a: null, c: null });
121  * ==> { a: 1 } and modifies o to { b: 2 }
122  */
123 RapidContext.Util.mask = function (src, keys) {
124     var res = {};
125     if (!MochiKit.Base.isArrayLike(keys)) {
126         keys = MochiKit.Base.keys(keys);
127     }
128     for (var i = 0; i < keys.length; i++) {
129         var k = keys[i];
130         if (k in src) {
131             res[k] = src[k];
132             delete src[k];
133         }
134     }
135     return res;
136 };

138 /**
139  * Converts a string to a title-cased string. All word boundaries are replaced
140  * with a single space and the subsequent character is capitalized.
141  *
142  * All underscore ("_"), hyphen ("-") and lower-upper character pairs are
143  * recognized as word boundaries. Note that this function does not change the
144  * capitalization of other characters in the string.
145  *
146  * @param {String} str the string to convert
147  *
148  * @return {String} the converted string
149  *
150  * @example
151  * RapidContext.Util.toTitleCase("a short heading")
152  * ==> "A Short Heading"
153  *
154  * @example
155  * RapidContext.Util.toTitleCase("camelCase")
156  * ==> "Camel Case"
157  *
158  * @example
159  * RapidContext.Util.toTitleCase("bounding-box")
160  * ==> "Bounding Box"
161  *
162  * @example
163  * RapidContext.Util.toTitleCase("UPPER_CASE_VALUE")
164  * ==> "UPPER CASE VALUE"
165  */
166 RapidContext.Util.toTitleCase = function (str) {
167     str = MochiKit.Format.strip(str.replace(/[_-]+/g, " "));
168     str = str.replace(/[a-z][A-Z]/g, function (match) {
169         return match.charAt(0) + " " + match.charAt(1);
170     });
171     str = str.replace(/(^|\s)[a-z]/g, function (match) {
172         return match.toUpperCase();
173     });
174     return str;
175 };

177 /**
178  * Returns the name of a function. If the function is anonymous (i.e. the
179  * `name` property is `undefined` or an empty string), the value of the
180  * `displayName` property is returned instead.
181  *
182  * @param {Function} func the function to check
183  *
184  * @return {String} the function name, or `undefined` if not available
185  *
186  * @example
187  * var o = { test: function () {} };
188  * RapidContext.Util.registerFunctionNames(o, "o");
189  * RapidContext.Util.functionName(o.test)
190  * ==> "o.test"
191  *
192  * @see RapidContext.Util.registerFunctionNames
193  */
194 RapidContext.Util.functionName = function (func) {
195     if (func == null) {
196         return null;
197     } else if (func.name != null && func.name != "") {
198         return func.name;
199     } else {
200         return func.displayName;
201     }
202 };

204 /**
205  * Registers function names for anonymous functions. This is useful when
206  * debugging code in Firebug or similar tools, as readable stack traces can be
207  * provided.
208  *
209  * This function will add a `displayName` property to all functions without a
210  * `name` property. Both the `obj` properties and `obj.prototype` properties
211  * are processed recursively, using the base name as a namespace (i.e.
212  * `[name].[property]` or `[name].prototype.[property]`).
213  *
214  * @param {Object} obj the function or object to process
215  * @param {String} [name] the function or object (class) name
216  *
217  * @example
218  * var o = { name: "MyObject", test: function () {} };
219  * RapidContext.Util.registerFunctionNames(o);
220  * o.test.displayName
221  * ==> "MyObject.test"
222  *
223  * @see RapidContext.Util.functionName
224  */
225 RapidContext.Util.registerFunctionNames = function (obj, name) {
226     function worker(o, name, stack) {
227         var isObj = (o != null && typeof(o) === "object");
228         var isFunc = (typeof(o) === "function");
229         var isAnon = isFunc && (o.name == null || o.name == "");
230         var isProto = (o === Object.prototype || o === Function.prototype);
231         var isNode = isObj && (typeof(o.nodeType) === "number");
232         var isVisited = (MochiKit.Base.findIdentical(stack, o) >= 0);
233         if (isFunc && isAnon && !o.hasOwnProperty("displayName")) {
234             o.displayName = name;
235         }
236         if ((isObj || isFunc) && !isProto && !isNode && !isVisited) {
237             stack.push(o);
238             for (var prop in o) {
239                 if (o.hasOwnProperty(prop)) {
240                     worker(o[prop], name + "." + prop, stack);
241                 }
242             }
243             worker(o.prototype, name + ".prototype", stack);
244             stack.pop();
245         }
246     }
247     worker(obj, name || obj.name || obj.displayName || obj.NAME || "", []);
248 };

250 /**
251  * Resolves a relative URI to an absolute URI. This function will return
252  * absolute URI:s directly and traverse any "../" directory paths in the
253  * specified URI. The base URI provided must be absolute.
254  *
255  * @param {String} uri the relative URI to resolve
256  * @param {String} [base] the absolute base URI, defaults to the
257  *            the current document base URI
258  *
259  * @return {String} the resolved absolute URI
260  *
261  * @deprecated This function will be removed and/or renamed in the future.
262  *     Better solutions for handling URL:s is to use a URL-parsing library such
263  *     as URL.js.
264  */
265 RapidContext.Util.resolveURI = function (uri, base) {
266     base = base || document.baseURI || document.getElementsByTagName("base")[0].href;
267     if (uri.indexOf(":") > 0) {
268         return uri;
269     } else if (uri.indexOf("#") == 0) {
270         var pos = base.lastIndexOf("#");
271         if (pos >= 0) {
272             base = base.substring(0, pos);
273         }
274         return base + uri;
275     } else if (uri.indexOf("/") == 0) {
276         var pos = base.indexOf("/", base.indexOf("://") + 3);
277         base = base.substring(0, pos);
278         return base + uri;
279     } else if (uri.indexOf("../") == 0) {
280         var pos = base.lastIndexOf("/");
281         base = base.substring(0, pos);
282         uri = uri.substring(3);
283         return RapidContext.Util.resolveURI(uri, base);
284     } else {
285         var pos = base.lastIndexOf("/");
286         base = base.substring(0, pos + 1);
287         return base + uri;
288     }
289 };

302 /**
303  * Returns `true` if the specified object looks like a DOM node. Otherwise,
304  * `false` will be returned. Any non-null object with a `nodeType` > 0 will be
305  * considered a DOM node by this function.
306  *
307  * @param {Object} obj the object to check
308  *
309  * @return {Boolean} `true` if the object looks like a DOM node, or
310  *         `false` otherwise
311  */
312 RapidContext.Util.isDOM = function (obj) {
313     return obj != null &&
314            typeof(obj.nodeType) === "number" &&
315            obj.nodeType > 0;
316 };

318 /**
319  * Returns `true` if the specified object looks like an HTML or XHTML DOM node.
320  * Otherwise, `false` will be returned. Any non-null object with a
321  * `nodeType` > 0 will be considered a DOM node, but only those with a matching
322  * `namespaceURI` will be considered HTML DOM nodes.
323  *
324  * @param {Object} obj the object to check
325  *
326  * @return {Boolean} `true` if the object looks like an HTML DOM node,
327  *         or `false` otherwise
328  */
329 RapidContext.Util.isHTML = function (obj) {
330     var ns = RapidContext.Util.NS.HTML;
331     return RapidContext.Util.isDOM(obj) &&
332            MochiKit.Base.findIdentical(ns, obj.namespaceURI) >= 0;
333 };

354 /**
355  * Returns an immediate child node from a parent DOM node. If a numeric index
356  * is provided, the index will be range checked and any matching child DOM
357  * node will be returned. Otherwise, the DOM tree is traversed to find the
358  * immediate child that corresponds to the specified node.
359  *
360  * @param {Node} parent the parent HTML DOM node
361  * @param {Number/Node} indexOrNode the child index or a descendant node
362  *
363  * @return {Node} the child HTML DOM node, or
364  *         `null` if no matching node was found
365  *
366  * @example
367  * var child = RapidContext.Util.childNode(parent, 2);
368  * ==> parent.childNodes[2] or null
369  *
370  * @example
371  * var child = RapidContext.Util.childNode(node, evt.target());
372  * ==> child DOM node if descendant or null otherwise
373  */
374 RapidContext.Util.childNode = function (parent, indexOrNode) {
375     parent = MochiKit.DOM.getElement(parent);
376     if (typeof(indexOrNode) == "number") {
377         if (indexOrNode < 0 || indexOrNode >= parent.childNodes.length) {
378             return null;
379         } else {
380             return parent.childNodes[indexOrNode];
381         }
382     } else {
383         var node = MochiKit.DOM.getElement(indexOrNode);
384         while (node != null && node !== parent && node.parentNode !== parent) {
385             node = node.parentNode;
386         }
387         return (node == null || node === parent) ? null : node;
388     }
389 };

391 /**
392  * Creates a DOM node with a namespace.
393  *
394  * @param {String} ns the DOM namespace
395  * @param {String} tag the DOM tag name
396  * @param {Object} [attrs] the node attributes, or null for none
397  * @param {Object} [...] the nodes or text to add as children
398  *
399  * @return {Object} the DOM node created
400  *
401  * @example
402  * RapidContext.Util.createDOMExt("http://www.w3.org/2000/svg", "g");
403  * ==> an SVG <g> element
404  */
405 RapidContext.Util.createDOMExt = function (ns, tag, attrs/*, ...*/) {
406     var doc = MochiKit.DOM.currentDocument();
407     var node = (ns) ? doc.createElementNS(ns, tag) : doc.createElement(tag);
408     MochiKit.DOM.updateNodeAttributes(node, attrs);
409     var children = MochiKit.Base.extend([], arguments, 3);
410     MochiKit.DOM.appendChildNodes(node, children);
411     return node;
412 };

427 /**
428  * Returns a function for creating a specific kind of DOM nodes. The returned
429  * function will optionally require a sequence of non-null arguments that will
430  * be added as attributes to the node creation. The returned function will
431  * otherwise work similar to the `createDOMExt()` function, taking attributes
432  * and child nodes.
433  *
434  * @param {String} ns the DOM namespace, or `null` for HTML
435  * @param {String} tag the DOM tag name
436  * @param {Array} [args] the array with required arguments, or `null` for no
437  *            required arguments
438  * @param {Object} [attrs] the default node attributes, or `null` for none
439  * @param {Object} [...] the default nodes or text to add as children
440  *
441  * @return {Function} the function that creates the DOM nodes
442  */
443 RapidContext.Util.createDOMFuncExt = function (ns, tag, args, attrs/*, ...*/) {
444     args = args || [];
445     attrs = attrs || {};
446     var children = MochiKit.Base.extend([], arguments, 4);
447     return function (/*arg1, ..., argN, attrs, ...*/) {
448         var myAttrs = MochiKit.Base.update({}, attrs);
449         for (var pos = 0; pos < args.length; pos++) {
450             if (arguments[pos] == null) {
451                 throw new Error("Argument '" + args[pos] + "' cannot be null");
452             }
453             myAttrs[args[pos]] = arguments[pos];
454         }
455         MochiKit.Base.update(myAttrs, arguments[args.length]);
456         var myChildren = MochiKit.Base.extend([], children);
457         MochiKit.Base.extend(myChildren, arguments, args.length + 1);
458         return RapidContext.Util.createDOMExt(ns, tag, myAttrs, myChildren);
459     }
460 };

463  * Blurs (unfocuses) a specified DOM node and all relevant child nodes. This
464  * function will recursively blur all `<a>`, `<button>`, `<input>`,
465  * `<textarea>` and `<select>` child nodes found.

480 /**
481  * Registers algebraic constraints for the element width, height and/or aspect
482  * ratio. The constraints may either be fixed numeric values, functions or
483  * algebraic formulas (in a string).
484  *
485  * The formulas will be converted to JavaScript functions, replacing any "%"
486  * character with a reference to the corresponding parent dimension value (i.e.
487  * the parent element width, height or aspect ratio as a percentage). It is
488  * also possible to directly reference the parent values as `w` or `h`.
489  *
490  * Constraint functions must take two arguments (parent width and height) and
491  * return a number. The returned number is set as the new element width or
492  * height (in pixels). Any returned value will also be bounded by the parent
493  * element size to avoid calculation errors.
494  *
495  * @param {Object} node the HTML DOM node
496  * @param {Number/Function/String} [width] the width constraint
497  * @param {Number/Function/String} [height] the height constraint
498  * @param {Number/Function/String} [aspect] the aspect ratio constraint
499  *
500  * @see RapidContext.Util.resizeElements
501  *
502  * @example
503  * RapidContext.Util.registerSizeConstraints(node, "50%-20", "100%");
504  * ==> Sets width to 50%-20 px and height to 100% of parent dimension
505  *
506  * @example
507  * RapidContext.Util.registerSizeConstraints(otherNode, null, null, 1.0);
508  * ==> Ensures a square aspect ratio
509  *
510  * @example
511  * RapidContext.Util.resizeElements(node, otherNode);
512  * ==> Evaluates the size constraints for both nodes
513  */
514 RapidContext.Util.registerSizeConstraints = function (node, width, height, aspect) {
515     node = MochiKit.DOM.getElement(node);
516     var sc = node.sizeConstraints = { w: null, h: null, a: null };
517     if (typeof(width) == "number") {
518         sc.w = function (w, h) { return width; }
519     } else if (typeof(width) == "function") {
520         sc.w = width;
521     } else if (typeof(width) == "string") {
522         var code = "return " + width.replace(/%/g, "*0.01*w") + ";";
523         sc.w = new Function("w", "h", code);
524     }
525     if (typeof(height) == "number") {
526         sc.h = function (w, h) { return height; }
527     } else if (typeof(height) == "function") {
528         sc.h = height;
529     } else if (typeof(height) == "string") {
530         var code = "return " + height.replace(/%/g, "*0.01*h") + ";";
531         sc.h = new Function("w", "h", code);
532     }
533     if (typeof(aspect) == "number") {
534         sc.a = function (w, h) { return aspect; }
535     } else if (typeof(aspect) == "function") {
536         sc.a = aspect;
537     } else if (typeof(aspect) == "string") {
538         var code = "return " + aspect.replace(/%/g, "*0.01*w/h") + ";";
539         sc.a = new Function("w", "h", code);
540     }
541 };

543 /**
544  * Resizes one or more DOM nodes using their registered size constraints and
545  * their parent element sizes. The resize operation will only modify those
546  * elements that have constraints, but will perform a depth-first recursion
547  * over all element child nodes as well.
548  *
549  * Partial constraints are accepted, in which case only the width or the height
550  * is modified. Aspect ratio constraints are applied after the width and height
551  * constraints. The result will always be bounded by the parent element width
552  * or height.
553  *
554  * The recursive descent of this function can be limited by adding a
555  * `resizeContent` function to a DOM node. Such a function will be called to
556  * handle all subnode resizing, making it possible to limit or omitting the
557  * DOM tree traversal.
558  *
559  * @param {Object} [...] the HTML DOM nodes to resize
560  *
561  * @see RapidContext.Util.registerSizeConstraints
562  *
563  * @example
564  * RapidContext.Util.resizeElements(node);
565  * ==> Evaluates the size constraints for a node and all child nodes
566  *
567  * @example
568  * elem.resizeContent = MochiKit.Base.noop;
569  * ==> Assigns a no-op child resize handler to elem
570  */
571 RapidContext.Util.resizeElements = function (/* ... */) {
572     var args = MochiKit.Base.flattenArray(arguments);
573     for (var i = 0; i < args.length; i++) {
574         var node = MochiKit.DOM.getElement(args[i]);
575         if (node != null && node.nodeType === 1 && // Node.ELEMENT_NODE
576             node.parentNode != null && node.sizeConstraints != null) {
577
578             var ref = { w: node.parentNode.w, h: node.parentNode.h };
579             if (ref.w == null && ref.h == null) {
580                 ref = MochiKit.Style.getElementDimensions(node.parentNode, true);
581             }
582             var dim = RapidContext.Util._evalConstraints(node.sizeConstraints, ref);
583             MochiKit.Style.setElementDimensions(node, dim);
584             node.w = dim.w;
585             node.h = dim.h;
586         }
587         if (node != null && typeof(node.resizeContent) == "function") {
588             try {
589                 node.resizeContent();
590             } catch (e) {
591                 RapidContext.Log.error("Error in resizeContent()", node, e);
592             }
593         } else {
594             node = node.firstChild;
595             while (node != null) {
596                 if (node.nodeType === 1) { // Node.ELEMENT_NODE
597                     RapidContext.Util.resizeElements(node);
598                 }
599                 node = node.nextSibling;
600             }
601         }
602     }
603 };

605 /**
606  * Evaluates the size constraint functions with a refeence dimension
607  * object. This is an internal function used to encapsulate the
608  * function calls and provide logging on errors.
609  *
610  * @param {Object} sc the size constraints object
611  * @param {Object} ref the MochiKit.Style.Dimensions maximum
612  *            reference values
613  *
614  * @return {Object} the MochiKit.Style.Dimensions with evaluated size
615  *         constraint values (some may be null)
616  */
617 RapidContext.Util._evalConstraints = function (sc, ref) {
618     var log = MochiKit.Logging.logError;
619     var w, h, a;
620     if (typeof(sc.w) == "function") {
621         try {
622             w = Math.max(0, Math.min(ref.w, sc.w(ref.w, ref.h)));
623         } catch (e) {
624             log("Error evaluating width size constraint; " +
625                 "w: " + ref.w + ", h: " + ref.h, e);
626         }
627     }
628     if (typeof(sc.h) == "function") {
629         try {
630             h = Math.max(0, Math.min(ref.h, sc.h(ref.w, ref.h)));
631         } catch (e) {
632             log("Error evaluating height size constraint; " +
633                 "w: " + ref.w + ", h: " + ref.h, e);
634         }
635     }
636     if (typeof(sc.a) == "function") {
637         try {
638             a = sc.a(ref.w, ref.h);
639             w = w || ref.w;
640             h = h || ref.h;
641             if (h * a > ref.w) {
642                 h = ref.w / a;
643             }
644             if (w / a > ref.h) {
645                 w = ref.h * a;
646             }
647             if (w > h * a) {
648                 w = h * a;
649             } else {
650                 h = w / a;
651             }
652         } catch (e) {
653             log("Error evaluating aspect size constraint; " +
654                 "w: " + ref.w + ", h: " + ref.h, e);
655         }
656     }
657     if (w != null) {
658         w = Math.floor(w);
659     }
660     if (h != null) {
661         h = Math.floor(h);
662     }
663     return new MochiKit.Style.Dimensions(w, h);
664 };
