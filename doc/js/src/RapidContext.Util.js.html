 34 /**
 35  * Creates a dictionary object from a list of keys and values. Optionally a
 36  * list of key-value pairs can be provided instead. As a third option, a single
 37  * (non-array) value can be assigned to all the keys.
 38  *
 39  * If a key is specified twice, only the last value will be used. Note that
 40  * this function is the reverse of `MochiKit.Base.items()`,
 41  * `MochiKit.Base.keys()` and `MochiKit.Base.values()`.
 42  *
 43  * @param {Array} itemsOrKeys the list of keys or items
 44  * @param {Array} [values] the list of values (optional if key-value
 45  *            pairs are specified in first argument)
 46  *
 47  * @return {Object} an object with properties for each key-value pair
 48  *
 49  * @example
 50  * RapidContext.Util.dict(['a','b'], [1, 2])
 51  * ==> { a: 1, b: 2 }
 52  *
 53  * @example
 54  * RapidContext.Util.dict([['a', 1], ['b', 2]])
 55  * ==> { a: 1, b: 2 }
 56  *
 57  * @example
 58  * RapidContext.Util.dict(['a','b'], true)
 59  * ==> { a: true, b: true }
 60  */
 61 RapidContext.Util.dict = function (itemsOrKeys, values) {
 62     var o = {};
 63     if (!MochiKit.Base.isArrayLike(itemsOrKeys)) {
 64         throw new TypeError("First argument must be array-like");
 65     }
 66     if (MochiKit.Base.isArrayLike(values) && itemsOrKeys.length !== values.length) {
 67         throw new TypeError("Both arrays must be of same length");
 68     }
 69     for (var i = 0; i < itemsOrKeys.length; i++) {
 70         var k = itemsOrKeys[i];
 71         if (k === null || k === undefined) {
 72             throw new TypeError("Key at index " + i + " is null or undefined");
 73         } else if (MochiKit.Base.isArrayLike(k)) {
 74             o[k[0]] = k[1];
 75         } else if (MochiKit.Base.isArrayLike(values)) {
 76             o[k] = values[i];
 77         } else {
 78             o[k] = values;
 79         }
 80     }
 81     return o;
 82 };

 84 /**
 85  * Filters an object by removing a list of keys. A list of key names (or an
 86  * object whose property names will be used as keys) must be specified as an
 87  * argument. A new object containing the source object values for the specified
 88  * keys will be returned. The source object will be modified by removing all
 89  * the specified keys.
 90  *
 91  * @param {Object} src the source object to select and modify
 92  * @param {Array/Object} keys the list of keys to remove, or an
 93  *            object with the keys to remove
 94  *
 95  * @return {Object} a new object containing the matching keys and
 96  *             values found in the source object
 97  *
 98  * @deprecated This function will be removed in the future.
 99  *
100  * @example
101  * var o = { a: 1, b: 2 };
102  * RapidContext.Util.mask(o, ['a', 'c']);
103  * ==> { a: 1 } and modifies o to { b: 2 }
104  *
105  * @example
106  * var o = { a: 1, b: 2 };
107  * RapidContext.Util.mask(o, { a: null, c: null });
108  * ==> { a: 1 } and modifies o to { b: 2 }
109  */
110 RapidContext.Util.mask = function (src, keys) {
111     var res = {};
112     if (!MochiKit.Base.isArrayLike(keys)) {
113         keys = MochiKit.Base.keys(keys);
114     }
115     for (var i = 0; i < keys.length; i++) {
116         var k = keys[i];
117         if (k in src) {
118             res[k] = src[k];
119             delete src[k];
120         }
121     }
122     return res;
123 };

125 /**
126  * Converts a string to a title-cased string. All word boundaries are replaced
127  * with a single space and the subsequent character is capitalized.
128  *
129  * All underscore ("_"), hyphen ("-") and lower-upper character pairs are
130  * recognized as word boundaries. Note that this function does not change the
131  * capitalization of other characters in the string.
132  *
133  * @param {String} str the string to convert
134  *
135  * @return {String} the converted string
136  *
137  * @example
138  * RapidContext.Util.toTitleCase("a short heading")
139  * ==> "A Short Heading"
140  *
141  * @example
142  * RapidContext.Util.toTitleCase("camelCase")
143  * ==> "Camel Case"
144  *
145  * @example
146  * RapidContext.Util.toTitleCase("bounding-box")
147  * ==> "Bounding Box"
148  *
149  * @example
150  * RapidContext.Util.toTitleCase("UPPER_CASE_VALUE")
151  * ==> "UPPER CASE VALUE"
152  */
153 RapidContext.Util.toTitleCase = function (str) {
154     str = str.replace(/[._-]+/g, " ").trim();
155     str = str.replace(/[a-z][A-Z]/g, function (match) {
156         return match.charAt(0) + " " + match.charAt(1);
157     });
158     str = str.replace(/(^|\s)[a-z]/g, function (match) {
159         return match.toUpperCase();
160     });
161     return str;
162 };

164 /**
165  * Returns the name of a function. If the function is anonymous (i.e. the
166  * `name` property is `undefined` or an empty string), the value of the
167  * `displayName` property is returned instead.
168  *
169  * @param {Function} func the function to check
170  *
171  * @return {String} the function name, or `undefined` if not available
172  *
173  * @example
174  * var o = { test: function () {} };
175  * RapidContext.Util.registerFunctionNames(o, "o");
176  * RapidContext.Util.functionName(o.test)
177  * ==> "o.test"
178  *
179  * @see RapidContext.Util.registerFunctionNames
180  */
181 RapidContext.Util.functionName = function (func) {
182     if (func == null) {
183         return null;
184     } else if (func.name != null && func.name != "") {
185         return func.name;
186     } else {
187         return func.displayName;
188     }
189 };

191 /**
192  * Registers function names for anonymous functions. This is useful when
193  * debugging code in Firebug or similar tools, as readable stack traces can be
194  * provided.
195  *
196  * This function will add a `displayName` property to all functions without a
197  * `name` property. Both the `obj` properties and `obj.prototype` properties
198  * are processed recursively, using the base name as a namespace (i.e.
199  * `[name].[property]` or `[name].prototype.[property]`).
200  *
201  * @param {Object} obj the function or object to process
202  * @param {String} [name] the function or object (class) name
203  *
204  * @example
205  * var o = { name: "MyObject", test: function () {} };
206  * RapidContext.Util.registerFunctionNames(o);
207  * o.test.displayName
208  * ==> "MyObject.test"
209  *
210  * @see RapidContext.Util.functionName
211  */
212 RapidContext.Util.registerFunctionNames = function (obj, name) {
213     function worker(o, name, stack) {
214         var isObj = (o != null && typeof(o) === "object");
215         var isFunc = (typeof(o) === "function");
216         var isAnon = isFunc && (o.name == null || o.name == "");
217         var isProto = (o === Object.prototype || o === Function.prototype);
218         var isNode = isObj && (typeof(o.nodeType) === "number");
219         var isVisited = stack.includes(o);
220         if (isFunc && isAnon && !o.displayName) {
221             o.displayName = name;
222         }
223         if ((isObj || isFunc) && !isProto && !isNode && !isVisited) {
224             stack.push(o);
225             for (var prop in o) {
226                 if (Object.prototype.hasOwnProperty.call(o, prop)) {
227                     worker(o[prop], name + "." + prop, stack);
228                 }
229             }
230             worker(o.prototype, name + ".prototype", stack);
231             stack.pop();
232         }
233     }
234     worker(obj, name || obj.name || obj.displayName || obj.NAME || "", []);
235 };

237 /**
238  * Resolves a relative URI to an absolute URI. This function will return
239  * absolute URI:s directly and traverse any "../" directory paths in the
240  * specified URI. The base URI provided must be absolute.
241  *
242  * @param {String} uri the relative URI to resolve
243  * @param {String} [base] the absolute base URI, defaults to the
244  *            the current document base URI
245  *
246  * @return {String} the resolved absolute URI
247  *
248  * @deprecated This function will be removed and/or renamed in the future.
249  *     Better solutions for handling URL:s is to use a URL-parsing library such
250  *     as URL.js.
251  */
252 RapidContext.Util.resolveURI = function (uri, base) {
253     var pos;
254     base = base || document.baseURI || document.getElementsByTagName("base")[0].href;
255     if (uri.includes(":")) {
256         return uri;
257     } else if (uri.startsWith("#")) {
258         pos = base.lastIndexOf("#");
259         if (pos >= 0) {
260             base = base.substring(0, pos);
261         }
262         return base + uri;
263     } else if (uri.startsWith("/")) {
264         pos = base.indexOf("/", base.indexOf("://") + 3);
265         base = base.substring(0, pos);
266         return base + uri;
267     } else if (uri.startsWith("../")) {
268         pos = base.lastIndexOf("/");
269         base = base.substring(0, pos);
270         uri = uri.substring(3);
271         return RapidContext.Util.resolveURI(uri, base);
272     } else {
273         pos = base.lastIndexOf("/");
274         base = base.substring(0, pos + 1);
275         return base + uri;
276     }
277 };

290 /**
291  * Returns `true` if the specified object looks like a DOM node. Otherwise,
292  * `false` will be returned. Any non-null object with a `nodeType` > 0 will be
293  * considered a DOM node by this function.
294  *
295  * @param {Object} obj the object to check
296  *
297  * @return {Boolean} `true` if the object looks like a DOM node, or
298  *         `false` otherwise
299  */
300 RapidContext.Util.isDOM = function (obj) {
301     return obj != null &&
302            typeof(obj.nodeType) === "number" &&
303            obj.nodeType > 0;
304 };

306 /**
307  * Returns `true` if the specified object looks like an HTML or XHTML DOM node.
308  * Otherwise, `false` will be returned. Any non-null object with a
309  * `nodeType` > 0 will be considered a DOM node, but only those with a matching
310  * `namespaceURI` will be considered HTML DOM nodes.
311  *
312  * @param {Object} obj the object to check
313  *
314  * @return {Boolean} `true` if the object looks like an HTML DOM node,
315  *         or `false` otherwise
316  */
317 RapidContext.Util.isHTML = function (obj) {
318     return RapidContext.Util.isDOM(obj) &&
319            RapidContext.Util.NS.HTML.includes(obj.namespaceURI);
320 };

322 /**
323  * Returns an array with DOM node attribute name and value pairs. The name and
324  * value pairs are also stored in arrays with two elements.
325  *
326  * @param {Object} node the HTML DOM node
327  *
328  * @return {Array} an array containing attribute name and value
329  *             pairs (as arrays)

331 RapidContext.Util.attributeArray = function (node) {
332     var res = [];
333     node = MochiKit.DOM.getElement(node);
334     for (var i = 0; node != null && i < node.attributes.length; i++) {
335         var a = node.attributes[i];
336         res.push([a.name, a.value]);

341 /**
342  * Returns an immediate child node from a parent DOM node. If a numeric index
343  * is provided, the index will be range checked and any matching child DOM
344  * node will be returned. Otherwise, the DOM tree is traversed to find the
345  * immediate child that corresponds to the specified node.
346  *
347  * @param {Node} parent the parent HTML DOM node
348  * @param {Number/Node} indexOrNode the child index or a descendant node
349  *
350  * @return {Node} the child HTML DOM node, or
351  *         `null` if no matching node was found
352  *
353  * @example
354  * var child = RapidContext.Util.childNode(parent, 2);
355  * ==> parent.childNodes[2] or null
356  *
357  * @example
358  * var child = RapidContext.Util.childNode(node, evt.target());
359  * ==> child DOM node if descendant or null otherwise
360  */
361 RapidContext.Util.childNode = function (parent, indexOrNode) {
362     parent = MochiKit.DOM.getElement(parent);
363     if (typeof(indexOrNode) == "number") {
364         if (indexOrNode < 0 || indexOrNode >= parent.childNodes.length) {
365             return null;
366         } else {
367             return parent.childNodes[indexOrNode];
368         }
369     } else {
370         var node = MochiKit.DOM.getElement(indexOrNode);
371         while (node != null && node !== parent && node.parentNode !== parent) {
372             node = node.parentNode;
373         }
374         return (node == null || node === parent) ? null : node;
375     }
376 };

378 /**
379  * Creates a DOM node with a namespace.
380  *
381  * @param {String} ns the DOM namespace
382  * @param {String} tag the DOM tag name
383  * @param {Object} [attrs] the node attributes, or null for none
384  * @param {Object} [...] the nodes or text to add as children
385  *
386  * @return {Object} the DOM node created
387  *
388  * @example
389  * RapidContext.Util.createDOMExt("http://www.w3.org/2000/svg", "g");
390  * ==> an SVG <g> element
391  */
392 RapidContext.Util.createDOMExt = function (ns, tag, attrs/*, ...*/) {
393     var doc = MochiKit.DOM.currentDocument();
394     var node = (ns) ? doc.createElementNS(ns, tag) : doc.createElement(tag);
395     MochiKit.DOM.updateNodeAttributes(node, attrs);
396     var children = Array.from(arguments).slice(3);
397     MochiKit.DOM.appendChildNodes(node, children);
398     return node;
399 };

414 /**
415  * Returns a function for creating a specific kind of DOM nodes. The returned
416  * function will optionally require a sequence of non-null arguments that will
417  * be added as attributes to the node creation. The returned function will
418  * otherwise work similar to the `createDOMExt()` function, taking attributes
419  * and child nodes.
420  *
421  * @param {String} ns the DOM namespace, or `null` for HTML
422  * @param {String} tag the DOM tag name
423  * @param {Array} [args] the array with required arguments, or `null` for no
424  *            required arguments
425  * @param {Object} [attrs] the default node attributes, or `null` for none
426  * @param {Object} [...] the default nodes or text to add as children
427  *
428  * @return {Function} the function that creates the DOM nodes
429  */
430 RapidContext.Util.createDOMFuncExt = function (ns, tag, args, attrs/*, ...*/) {
431     args = args || [];
432     attrs = attrs || {};
433     var children = Array.from(arguments).slice(4);
434     return function (/*arg1, ..., argN, attrs, ...*/) {
435         var myArgs = Array.from(arguments);
436         var myAttrs = Object.assign({}, attrs);
437         args.forEach(function (key, idx) {
438             if (myArgs[idx] == null) {
439                 throw new Error("Argument '" + key + "' cannot be null");
440             }
441             myAttrs[key] = myArgs[idx];
442         });
443         Object.assign(myAttrs, myArgs[args.length]);
444         var myChildren = [].concat(children, myArgs.slice(args.length + 1));
445         return RapidContext.Util.createDOMExt(ns, tag, myAttrs, myChildren);
446     };
447 };

449 /**
450  * Blurs (unfocuses) a specified DOM node and all relevant child nodes. This
451  * function will recursively blur all `<a>`, `<button>`, `<input>`,
452  * `<textarea>` and `<select>` child nodes found.
453  *
454  * @param {Object} node the HTML DOM node
455  */
456 RapidContext.Util.blurAll = function (node) {
457     node.blur();
458     var tags = ["A", "BUTTON", "INPUT", "TEXTAREA", "SELECT"];
459     for (var i = 0; i < tags.length; i++) {
460         var nodes = node.getElementsByTagName(tags[i]);
461         for (var j = 0; j < nodes.length; j++) {
462             nodes[j].blur();
463         }
464     }
465 };

467 /**
468  * Registers size constraints for the element width and/or height. The
469  * constraints may either be fixed numeric values or simple arithmetic (in a
470  * string). The formulas will be converted to CSS calc() expressions.
471  *
472  * Legacy constraint functions are still supported and must take two arguments
473  * (parent width and height) and should return a number. The returned number is
474  * set as the new element width or height (in pixels). Any returned value will
475  * also be bounded by the parent element size to avoid calculation errors.
476  *
477  * @param {Object} node the HTML DOM node
478  * @param {Number/String/Function} [width] the width constraint
479  * @param {Number/String/Function} [height] the height constraint
480  *
481  * @see RapidContext.Util.resizeElements
482  *
483  * @example
484  * RapidContext.Util.registerSizeConstraints(node, "50%-20", "100%");
485  * ==> Sets width to 50%-20 px and height to 100% of parent dimension
486  *
487  * @example
488  * RapidContext.Util.resizeElements(node, otherNode);
489  * ==> Evaluates the size constraints for both nodes
490  */
491 RapidContext.Util.registerSizeConstraints = function (node, width, height) {
492     function toCSS(val) {
493         if (/[+-]/.test(val)) {
494             val = "calc( " + val.replace(/[+-]/g, " $& ") + " )";
495         }
496         val = val.replace(/(\d)( |$)/g, "$1px$2");
497         return val;
498     }
499     node = MochiKit.DOM.getElement(node);
500     if (typeof(width) == "number" || typeof(width) == "string") {
501         node.style.width = toCSS(String(width));
502     } else if (typeof(width) == "function") {
503         console.info("registerSizeConstraints: width function support will be removed", node);
504         node.sizeConstraints = node.sizeConstraints || { w: null, h: null };
505         node.sizeConstraints.w = width;
506     }
507     if (typeof(height) == "number" || typeof(height) == "string") {
508         node.style.height = toCSS(String(height));
509     } else if (typeof(height) == "function") {
510         console.info("registerSizeConstraints: height function support will be removed", node);
511         node.sizeConstraints = node.sizeConstraints || { w: null, h: null };
512         node.sizeConstraints.h = height;
513     }
514 };

516 /**
517  * Resizes one or more DOM nodes using their registered size constraints and
518  * their parent element sizes. The resize operation will only modify those
519  * elements that have constraints, but will perform a depth-first recursion
520  * over all element child nodes as well.
521  *
522  * Partial constraints are accepted, in which case only the width or the height
523  * is modified. Aspect ratio constraints are applied after the width and height
524  * constraints. The result will always be bounded by the parent element width
525  * or height.
526  *
527  * The recursive descent of this function can be limited by adding a
528  * `resizeContent` function to a DOM node. Such a function will be called to
529  * handle all subnode resizing, making it possible to limit or omitting the
530  * DOM tree traversal.
531  *
532  * @param {Object} [...] the HTML DOM nodes to resize
533  *
534  * @see RapidContext.Util.registerSizeConstraints
535  *
536  * @example
537  * RapidContext.Util.resizeElements(node);
538  * ==> Evaluates the size constraints for a node and all child nodes
539  *
540  * @example
541  * elem.resizeContent = MochiKit.Base.noop;
542  * ==> Assigns a no-op child resize handler to elem
543  */
544 RapidContext.Util.resizeElements = function (/* ... */) {
545     Array.from(arguments).forEach(function (arg) {
546         var node = MochiKit.DOM.getElement(arg);
547         if (node && node.nodeType === 1 && node.parentNode && node.sizeConstraints) {
548             var ref = { w: node.parentNode.w, h: node.parentNode.h };
549             if (ref.w == null && ref.h == null) {
550                 ref = MochiKit.Style.getElementDimensions(node.parentNode, true);
551             }
552             var dim = RapidContext.Util._evalConstraints(node.sizeConstraints, ref);
553             MochiKit.Style.setElementDimensions(node, dim);
554             node.w = dim.w;
555             node.h = dim.h;
556         }
557         if (node && typeof(node.resizeContent) == "function") {
558             try {
559                 node.resizeContent();
560             } catch (e) {
561                 console.error("Error in resizeContent()", node, e);
562             }
563         } else if (node && node.childNodes) {
564             Array.from(node.childNodes).forEach(function (child) {
565                 if (child.nodeType === 1) {
566                     RapidContext.Util.resizeElements(child);
567                 }
568             });
569         }
570     });
571 };

573 /**
574  * Evaluates the size constraint functions with a refeence dimension
575  * object. This is an internal function used to encapsulate the
576  * function calls and provide logging on errors.
577  *
578  * @param {Object} sc the size constraints object
579  * @param {Object} ref the MochiKit.Style.Dimensions maximum
580  *            reference values
581  *
582  * @return {Object} the MochiKit.Style.Dimensions with evaluated size
583  *         constraint values (some may be null)
584  */
585 RapidContext.Util._evalConstraints = function (sc, ref) {
586     var w, h;
587     if (typeof(sc.w) == "function") {
588         try {
589             w = Math.max(0, Math.min(ref.w, sc.w(ref.w, ref.h)));
590         } catch (e) {
591             console.error("Error evaluating width size constraint; " +
592                           "w: " + ref.w + ", h: " + ref.h, e);
593         }
594     }
595     if (typeof(sc.h) == "function") {
596         try {
597             h = Math.max(0, Math.min(ref.h, sc.h(ref.w, ref.h)));
598         } catch (e) {
599             console.error("Error evaluating height size constraint; " +
600                           "w: " + ref.w + ", h: " + ref.h, e);
601         }
602     }
603     if (w != null) {
604         w = Math.floor(w);
605     }
606     if (h != null) {
607         h = Math.floor(h);
608     }
609     return new MochiKit.Style.Dimensions(w, h);
610 };
