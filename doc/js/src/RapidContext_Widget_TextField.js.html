 21 /**
 22  * Creates a new text field widget.
 23  *
 24  * @constructor
 25  * @param {Object} attrs the widget and node attributes
 26  * @param {String} [attrs.name] the form field name
 27  * @param {String} [attrs.value] the field value, defaults to ""
 28  * @param {String} [attrs.helpText] the help text when empty (deprecated)
 29  * @param {Boolean} [attrs.disabled] the disabled widget flag, defaults to
 30  *            false
 31  * @param {Boolean} [attrs.hidden] the hidden widget flag, defaults to false
 32  * @param {Object} [...] the initial text content
 33  *
 34  * @return {Widget} the widget DOM node
 35  *
 36  * @class The text field widget class. Used to provide a text input field for a
 37  *     single line, using the `<input>` HTML element. The text field may also
 38  *     be connected to a popup (for auto-complete or similar).
 39  * @property {Boolean} disabled The read-only widget disabled flag.
 40  * @property {String} defaultValue The value to use on form reset.
 41  * @extends RapidContext.Widget
 42  *
 43  * @example {JavaScript}
 44  * var attrs = { name: "name", placeholder: "Your Name Here" };
 45  * var field = RapidContext.Widget.TextField(attrs);
 46  *
 47  * @example {User Interface XML}
 48  * <TextField name="name" placeholder="Your Name Here" />
 49  */
 50 RapidContext.Widget.TextField = function (attrs/*, ...*/) {
 51     function scrape(val) {
 52         return String(val && val.textContent || val || "");
 53     }
 54     var type = (attrs && attrs.type) || "text";
 55     var text = (attrs && attrs.value) || "";
 56     text += Array.prototype.slice.call(arguments, 1).map(scrape).join("");
 57     var o = MochiKit.DOM.INPUT({ type: type, value: text });
 58     RapidContext.Widget._widgetMixin(o, RapidContext.Widget.TextField);
 59     o.addClass("widgetTextField");
 60     o._popupCreated = false;
 61     o.setAttrs(MochiKit.Base.update({}, attrs, { value: text }));
 62     o.addEventListener("input", o._handleChange);
 63     return o;
 64 };

 73  * data and is triggered on blur. The synthetic onchange events all
 74  * contain an 'event.detail' object with 'before', 'after' and 'cause'

 88 /**
 89  * Destroys this widget.
 90  */
 91 RapidContext.Widget.TextField.prototype.destroy = function () {
 92     // FIXME: Use AbortSignal instead to disconnect
 93     this.removeEventListener("input", this._handleChange);
 94     this.removeEventListener("blur", this._handleBlur);
 95 };

 97 /**
 98  * Updates the widget or HTML DOM node attributes.
 99  *
100  * @param {Object} attrs the widget and node attributes to set
101  * @param {String} [attrs.name] the form field name
102  * @param {String} [attrs.value] the field value
103  * @param {String} [attrs.helpText] the help text when empty (deprecated)
104  * @param {Boolean} [attrs.disabled] the disabled widget flag
105  * @param {Boolean} [attrs.hidden] the hidden widget flag
106  */
107 RapidContext.Widget.TextField.prototype.setAttrs = function (attrs) {
108     attrs = MochiKit.Base.update({}, attrs);
109     var locals = RapidContext.Util.mask(attrs, ["helpText", "value"]);
110     if ("helpText" in locals) {
111         attrs.placeholder = attrs.placeholder || locals.helpText;
112     }
113     if ("value" in locals) {
114         this.value = locals.value || "";
115         this._handleChange(null);
116     }
117     this.__setAttrs(attrs);
118 };
119
120 /**
121  * Resets the text area form value to the initial value.
122  */
123 RapidContext.Widget.TextField.prototype.reset = function () {
124     this.setAttrs({ value: this.defaultValue });
125 };

136 /**
137  * Returns (or creates) a popup for this text field. The popup will
138  * not be shown by this method, only returned as-is. If the create
139  * flag is specified, a new popup will be created if none has been
140  * created previously.
141  *
142  * @param {Boolean} create the create popup flag
143  *
144  * @return {Widget} the popup widget, or
145  *         null if none existed or was created
146  */
147 RapidContext.Widget.TextField.prototype.popup = function (create) {
148     if (!this._popupCreated && create) {
149         this.autocomplete = "off";
150         this._popupCreated = true;
151         var dim = MochiKit.Style.getElementDimensions(this);
152         var style = { "max-height": "300px", "width": Math.max(dim.w - 5, 300) + "px" };
153         var popup = RapidContext.Widget.Popup({ style: style });
154         MochiKit.DOM.insertSiblingNodesAfter(this, popup);
155         MochiKit.Style.makePositioned(this.parentNode);
156         MochiKit.Signal.connect(this, "onkeydown", this, "_handleKeyDown");
157         MochiKit.Signal.connect(popup, "onclick", this, "_handleClick");
158         this.addEventListener("blur", this._handleBlur);
159     }
160     return (this._popupCreated) ? this.nextSibling : null;
161 };

163 /**
164  * Shows a popup for the text field containing the specified items.
165  * The items specified may be either a list of HTML DOM nodes or
166  * text strings.
167  *
168  * @param {Object} [attrs] the popup attributes to set
169  * @param {Number} [attrs.delay] the popup auto-hide delay, defaults
170  *            to 30 seconds
171  * @param {Array} [items] the items to show, or null to keep the
172  *            previous popup content
173  *
174  * @example
175  * var items = ["Cat", "Dog", "Elephant", "Zebra"];
176  * field.showPopup({}, items);
177  */
178 RapidContext.Widget.TextField.prototype.showPopup = function (attrs, items) {
179     var popup = this.popup(true);
180     if (items) {
181         popup.hide();
182         MochiKit.DOM.replaceChildNodes(popup);
183         for (var i = 0; i < items.length; i++) {
184             if (typeof(items[i]) == "string") {
185                 var node = MochiKit.DOM.DIV({ "class": "widgetPopupItem" }, items[i]);
186                 node.value = items[i];
187                 popup.appendChild(node);
188             } else {
189                 MochiKit.DOM.appendChildNodes(popup, items[i]);
190             }
191         }
192     }
193     if (popup.childNodes.length > 0) {
194         var pos = {
195             x: this.offsetLeft + 1,
196             y: this.offsetTop + this.offsetHeight + 1
197         };
198         MochiKit.Style.setElementPosition(popup, pos);
199         popup.setAttrs(MochiKit.Base.update({ delay: 30000 }, attrs));
200         popup.show();
201         if (items && items.length == 1) {
202             popup.selectChild(0);
203         }
204     }
205 };

207 /**
208  * Handles input events for this this widget.
209  *
210  * @param {Event} [evt] the DOM Event object or null for manual
211  */
212 RapidContext.Widget.TextField.prototype._handleChange = function (evt) {
213     var cause = (evt && evt.inputType) || "set";
214     var detail = { before: this.storedValue || "", after: this.value, cause: cause };
215     this._dispatch("change", { detail: detail, bubbles: true });
216     this.storedValue = this.value;
217 };
218
219 /**
220  * Handles the key down event for the text field.
221  *
222  * @param {Event} evt the `MochiKit.Signal.Event` object
223  */
224 RapidContext.Widget.TextField.prototype._handleKeyDown = function (evt) {
225     var popup = this.popup(false);
226     if (popup != null) {
227         popup.resetDelay();
228         if (popup.isHidden()) {
229             switch (evt.key().string) {
230             case "KEY_ESCAPE":
231                 evt.stop();
232                 break;
233             case "KEY_ARROW_UP":
234             case "KEY_ARROW_DOWN":
235                 this.showPopup();
236                 popup.selectChild(0);
237                 evt.stop();
238                 break;
239             }
240         } else {
241             switch (evt.key().string) {
242             case "KEY_TAB":
243             case "KEY_ENTER":
244                 popup.hide();
245                 evt.stop();
246                 if (popup.selectedChild() != null) {
247                     this._dispatch("dataavailable");
248                 }
249                 break;
250             case "KEY_ESCAPE":
251                 popup.hide();
252                 evt.stop();
253                 break;
254             case "KEY_ARROW_UP":
255             case "KEY_ARROW_DOWN":
256                 popup.selectMove(evt.key().string == "KEY_ARROW_UP" ? -1 : 1);
257                 evt.stop();
258                 break;
259             }
260         }
261     }
262 };
263
264 /**
265  * Handles the mouse click event on the popup.
266  *
267  * @param evt the `MochiKit.Signal.Event` object
268  */
269 RapidContext.Widget.TextField.prototype._handleClick = function (evt) {
270     this.blur();
271     this.focus();
272     this._dispatch("dataavailable");
273 };
274
275 /**
276  * Handles blur events for this widget (if popup attached).
277  *
278  * @param evt the DOM Event object
279  */
280 RapidContext.Widget.TextField.prototype._handleBlur = function (evt) {
281     var popup = this.popup();
282     if (popup && !popup.isHidden()) {
283         popup.setAttrs({ delay: 250 });
284     }
285 };
