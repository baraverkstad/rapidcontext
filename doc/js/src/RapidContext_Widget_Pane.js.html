 21 /**
 22  * Creates a new pane widget.
 23  *
 24  * @constructor
 25  * @param {Object} attrs the widget and node attributes
 26  * @param {String} [attrs.pageTitle] the page title used when inside
 27  *            a page container, defaults to "Page"
 28  * @param {String/Object} [attrs.pageStatus] the page status used
 29  *            when inside a page container, use one of the predefined
 30  *            status constants in this class, defaults to `ANY`
 31  * @param {Boolean} [attrs.pageCloseable] the page closeable flag
 32  *            used when inside some page containers, defaults to
 33  *            `false`
 34  * @param {Boolean} [attrs.hidden] the hidden widget flag, defaults to false
 35  * @param {Object} [...] the child widgets or DOM nodes
 36  *
 37  * @return {Widget} the widget DOM node
 38  *
 39  * @class The pane widget class. Used to create the simplest form of element
 40  *     container. It is also used inside various types of paged containers,
 41  *     such as a `TabContainer`, a `Wizard` or similar. A pane only uses a
 42  *     `<div>` HTML element, and supports being hidden and shown according to
 43  *     any page transitions required by a parent container.
 44  * @property {String} pageTitle [read-only] The current page title.
 45  * @property {Object} pageStatus [read-only] The current page status.
 46  * @property {Boolean} pageCloseable [read-only] The current page
 47  *               closeable flag value.
 48  * @extends RapidContext.Widget
 49  *
 50  * @example {JavaScript}
 51  * var h1 = MochiKit.DOM.H1({}, "Hello, world!");
 52  * var helloPane = RapidContext.Widget.Pane({}, h1);
 53  *
 54  * @example {User Interface XML}
 55  * <Pane id="helloPane">
 56  *   <h1>Hello, world!</h1>
 57  * </Pane>
 58  */
 59 RapidContext.Widget.Pane = function (attrs/*, ... */) {
 60     var o = MochiKit.DOM.DIV();
 61     RapidContext.Widget._widgetMixin(o, arguments.callee);
 62     o.addClass("widgetPane");
 63     o.setAttrs(MochiKit.Base.update({ pageTitle: "Page", pageStatus: "ANY", pageCloseable: false }, attrs));
 64     o.addAll(MochiKit.Base.extend(null, arguments, 1));
 65     return o;
 66 };

136 /**
137  * Updates the widget or HTML DOM node attributes.
138  *
139  * @param {Object} attrs the widget and node attributes to set
140  * @param {String} [attrs.pageTitle] the page title used when inside
141  *            a page container
142  * @param {String/Object} [attrs.pageStatus] the page status used
143  *            when inside a page container, use one of the predefined
144  *            status constants in this class
145  * @param {Boolean} [attrs.pageCloseable] the page closeable flag
146  *            used when inside some page containers
147  * @param {Boolean} [attrs.hidden] the hidden widget flag
148  */
149 RapidContext.Widget.Pane.prototype.setAttrs = function (attrs) {
150     attrs = MochiKit.Base.update({}, attrs);
151     var locals = RapidContext.Util.mask(attrs, ["pageTitle", "pageStatus", "pageCloseable"]);
152     var modified = false;
153     if (typeof(locals.pageTitle) != "undefined") {
154         this.pageTitle = locals.pageTitle;
155         modified = true;
156     }
157     if (typeof(locals.pageStatus) != "undefined") {
158         if (typeof(locals.pageStatus) == "string") {
159             locals.pageStatus = RapidContext.Widget.Pane[locals.pageStatus];
160         }
161         this.pageStatus = locals.pageStatus;
162         modified = true;
163     }
164     if (typeof(locals.pageCloseable) != "undefined") {
165         this.pageCloseable = MochiKit.Base.bool(locals.pageCloseable);
166         modified = true;
167     }
168     if (modified && this.parentNode &&
169         typeof(this.parentNode._updateStatus) == "function") {
170         this.parentNode._updateStatus();
171     }
172     this.__setAttrs(attrs);
173 };

175 /**
176  * Handles the page enter event. This method is called by a parent
177  * page container widget, such as a `TabContainer` or a `Wizard`. It will
178  * reset the form validations (optional), show the pane (optional),
179  * and finally trigger the "onenter" event.
180  *
181  * @param {Object} opts the page transition options
182  * @param {Boolean} [opts.show] the show pane flag, defaults to `true`
183  * @param {Boolean} [opts.validateReset] the form validation reset
184  *            flag, used to clear all form validations in the pane
185  */
186 RapidContext.Widget.Pane.prototype._handleEnter = function (opts) {
187     opts = MochiKit.Base.update({ show: true, validateReset: false }, opts);
188     if (MochiKit.Base.bool(opts.validateReset)) {
189         var forms = this.getElementsByTagName("FORM");
190         for (var i = 0; i < forms.length; i++) {
191             if (typeof(forms[i].validateReset) == "function") {
192                 forms[i].validateReset();
193             }
194         }
195     }
196     if (MochiKit.Base.bool(opts.show)) {
197         this.show();
198         RapidContext.Util.resizeElements(this);
199     }
200     RapidContext.Widget.emitSignal(this, "onenter");
201 };
202
203 /**
204  * Handles the page exit event. This method is called by a parent
205  * page container widget, such as a `TabContainer` or a `Wizard`. It will
206  * validate the form (optional), unfocus all form fields, hide the
207  * pane (optional), and finally trigger the "onexit" event.
208  *
209  * @param {Object} opts the page transition options
210  * @param {Boolean} [opts.hide] the hide pane flag, defaults to `true`
211  * @param {Boolean} [opts.validate] the form validation flag, used to
212  *            check all forms in the page for valid entries before
213  *            proceeding, defaults to `false`
214  *
215  * @return {Boolean} `true` if the page exit event completed, or
216  *         `false` if it was cancelled (due to validation errors)
217  */
218 RapidContext.Widget.Pane.prototype._handleExit = function (opts) {
219     opts = MochiKit.Base.update({ hide: true, validate: false }, opts);
220     if (MochiKit.Base.bool(opts.validate)) {
221         var forms = this.getElementsByTagName("FORM");
222         for (var i = 0; i < forms.length; i++) {
223             if (typeof(forms[i].validate) == "function") {
224                 var res = forms[i].validate();
225                 // TODO: handle MochiKit.Async.Deferred?
226                 if (!res) {
227                     return false;
228                 }
229             }
230         }
231     }
232     this.blurAll();
233     if (MochiKit.Base.bool(opts.hide)) {
234         this.hide();
235     }
236     RapidContext.Widget.emitSignal(this, "onexit");
237     return true;
238 };
