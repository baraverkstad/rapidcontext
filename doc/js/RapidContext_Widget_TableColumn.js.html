21/**
22 * Creates a new data table column widget.
23 *
24 * @constructor
25 * @param {Object} attrs the widget and node attributes
26 * @param {string} attrs.title the column title
27 * @param {string} attrs.field the data property name
28 * @param {string} [attrs.type] the data property type, one of
29 *            "string", "number", "date", "time", "datetime",
30 *            "boolean" or "object" (defaults to "string")
31 * @param {string} [attrs.sort] the sort direction, one of "asc",
32 *            "desc", "none" (disabled) or null (unsorted)
33 * @param {number} [attrs.maxLength] the maximum data length,
34 *            overflow will be displayed as a tooltip, only used by
35 *            the default renderer
36 * @param {boolean} [attrs.key] the unique key value flag, only to be
37 *            set for a single column per table
38 * @param {string} [attrs.tooltip] the tooltip text to display on the
39 *            column header
40 * @param {string} [attrs.cellStyle] the CSS styles or class names to set on
41 *            the rendered cells
42 * @param {function} [attrs.renderer] the function that renders the converted
43 *            data value into a table cell, called as
44 *            `renderer(<td>, value, data)` with the DOM node, field value and
45 *            data object as arguments
46 *
47 * @return {Widget} the widget DOM node
48 *
49 * @class The table column widget class. Used to provide a sortable data table
50 *     column, using a `<th>` HTML element for the header (and rendering data
51 *     to `<td>` HTML elements).
52 * @extends RapidContext.Widget
53 *
54 * @example <caption>JavaScript</caption>
55 * var attrs1 = { field: "id", title: "Identifier", key: true, type: "number" };
56 * var attrs2 = { field: "name", title: "Name", maxLength: 50, sort: "asc" };
57 * var attrs3 = { field: "modified", title: "Last Modified", type: "datetime" };
58 * var col1 = RapidContext.Widget.TableColumn(attrs1);
59 * var col2 = RapidContext.Widget.TableColumn(attrs2);
60 * var col3 = RapidContext.Widget.TableColumn(attrs3);
61 * var exampleTable = RapidContext.Widget.Table({}, col1, col2, col3);
62 * RapidContext.Util.registerSizeConstraints(exampleTable, "50%", "100%");
63 *
64 * @example <caption>User Interface XML</caption>
65 * <Table id="exampleTable" w="50%" h="100%">
66 *   <TableColumn field="id" title="Identifier" key="true" type="number" />
67 *   <TableColumn field="name" title="Name" maxLength="50" sort="asc" />
68 *   <TableColumn field="modified" title="Last Modified" type="datetime" />
69 * </Table>
70 */
71RapidContext.Widget.TableColumn = function (attrs) {
72    if (attrs.field == null) {
73        throw new Error("The 'field' attribute cannot be null for a TableColumn");
74    }
75    var o = document.createElement("th");
76    RapidContext.Widget._widgetMixin(o, RapidContext.Widget.TableColumn);
77    o.addClass("widgetTableColumn");
78    o.setAttrs(Object.assign({ title: attrs.field, type: "string", key: false }, attrs));
79    o.addEventListener("click", o._handleClick);
80    return o;
81};

103/**
104 * Updates the widget or HTML DOM node attributes. Note that some
105 * updates will not take effect until the parent table is cleared
106 * or data is reloaded.
107 *
108 * @param {Object} attrs the widget and node attributes to set
109 * @param {string} [attrs.title] the column title
110 * @param {string} [attrs.field] the data property name
111 * @param {string} [attrs.type] the data property type, one of
112 *            "string", "number", "date", "time", "datetime",
113 *            "boolean" or "object"
114 * @param {string} [attrs.sort] the sort direction, one of "asc",
115 *            "desc", "none" (disabled) or null (unsorted)
116 * @param {number} [attrs.maxLength] the maximum data length,
117 *            overflow will be displayed as a tooltip, only used by
118 *            the default renderer
119 * @param {boolean} [attrs.key] the unique key value flag, only to be
120 *            set for a single column per table
121 * @param {string} [attrs.tooltip] the tooltip text to display on the
122 *            column header
123 * @param {string} [attrs.cellStyle] the CSS styles or class names to set on
124 *            the rendered cells
125 * @param {function} [attrs.renderer] the function that renders the converted
126 *            data value into a table cell, called as
127 *            `renderer(<td>, value, data)` with the DOM node, field value and
128 *            data object as arguments
129 */
130RapidContext.Widget.TableColumn.prototype.setAttrs = function (attrs) {
131    attrs = Object.assign({}, attrs);
132    var locals = RapidContext.Util.mask(attrs, [
133        "title", "field", "type", "sort", "maxLength", "key", "tooltip",
134        "cellStyle", "renderer"
135    ]);
136    if (typeof(locals.title) !== "undefined") {
137        this.innerText = locals.title;
138    }
139    if (typeof(locals.field) !== "undefined") {
140        this.field = locals.field;
141    }
142    if (typeof(locals.type) !== "undefined") {
143        this.type = locals.type;
144    }
145    if (typeof(locals.sort) !== "undefined") {
146        this.sort = locals.sort;
147        this.classList.toggle("sortNone", locals.sort === "none");
148        this.classList.toggle("sortDesc", locals.sort === "desc");
149        this.classList.toggle("sortAsc", locals.sort === "asc");
150    }
151    if (typeof(locals.maxLength) !== "undefined") {
152        this.maxLength = parseInt(locals.maxLength);
153    }
154    if (typeof(locals.key) !== "undefined") {
155        this.key = MochiKit.Base.bool(locals.key);
156    }
157    if (typeof(locals.tooltip) !== "undefined") {
158        this.title = locals.tooltip;
159    }
160    if (typeof(locals.cellStyle) !== "undefined") {
161        this.cellStyle = locals.cellStyle;
162    }
163    if (typeof(locals.renderer) === "function") {
164        this.renderer = locals.renderer;
165    }
166    this.__setAttrs(attrs);
167};

169/**
170 * Maps the column field from one object onto another. This method
171 * will also convert the data depending on the column data type.
172 *
173 * @param src                the source object (containing the field)
174 * @param dst                the destination object
175 */
176RapidContext.Widget.TableColumn.prototype._map = function (src, dst) {
177    var value = src[this.field];
178    if (value != null) {
179        if (this.key) {
180            dst.$id = value;
181        }
182        switch (this.type) {
183        case "number":
184            if (value instanceof Number) {
185                value = value.valueOf();
186            } else if (typeof(value) != "number") {
187                value = parseFloat(value);
188            }
189            break;
190        case "date":
191            if (/^@\d+$/.test(value)) {
192                value = new Date(+value.substr(1));
193            }
194            if (value instanceof Date) {
195                value = MochiKit.DateTime.toISODate(value);
196            } else {
197                value = MochiKit.Text.truncate(value, 10);
198            }
199            break;
200        case "datetime":
201            if (/^@\d+$/.test(value)) {
202                value = new Date(+value.substr(1));
203            }
204            if (value instanceof Date) {
205                value = MochiKit.DateTime.toISOTimestamp(value);
206            } else {
207                value = MochiKit.Text.truncate(value, 19);
208            }
209            break;
210        case "time":
211            if (/^@\d+$/.test(value)) {
212                value = new Date(+value.substr(1));
213            }
214            if (value instanceof Date) {
215                value = MochiKit.DateTime.toISOTime(value);
216            } else {
217                if (typeof(value) !== "string") {
218                    value = value.toString();
219                }
220                if (value.length > 8) {
221                    value = value.substring(value.length - 8);
222                }
223            }
224            break;
225        case "boolean":
226            if (typeof(value) !== "boolean") {
227                value = MochiKit.Base.bool(value);
228            }
229            break;
230        case "string":
231            if (typeof(value) !== "string") {
232                value = value.toString();
233            }
234            break;
235        }
236    }
237    dst[this.field] = value;
238};

240/**
241 * Renders the column field value into a table cell.
242 *
243 * @param obj                the data object (containing the field)
244 *
245 * @return the table cell DOM node
246 */
247RapidContext.Widget.TableColumn.prototype._render = function (obj) {
248    var td = document.createElement("td");
249    var value = obj[this.field];
250    if (typeof(this.cellStyle) === "string" && this.cellStyle.includes(":")) {
251        td.style = this.cellStyle;
252    } else if (typeof(this.cellStyle) === "string") {
253        td.className = this.cellStyle;
254    }
255    if (typeof(this.renderer) === "function") {
256        try {
257            this.renderer(td, value, obj.$data);
258        } catch (e) {
259            td.append(e.toString());
260        }
261    } else if (typeof(value) == "boolean") {
262        var css = value ? "fa fa-check-square-o" : "fa fa-square-o";
263        td.append(RapidContext.Widget.Icon(css));
264    } else {
265        if (value == null || (typeof(value) == "number" && isNaN(value))) {
266            value = "";
267        } else if (typeof(value) != "string") {
268            value = value.toString();
269        }
270        if (this.maxLength && this.maxLength < value.length) {
271            td.title = value;
272            value = MochiKit.Text.truncate(value, this.maxLength, "...");
273        }
274        td.append(value);
275    }
276    return td;
277};

279/**
280 * Handles click events on the column header.
281 */
282RapidContext.Widget.TableColumn.prototype._handleClick = function () {
283    let table = this.closest(".widget.widgetTable");
284    let dir = (this.sort == "asc") ? "desc" : "asc";
285    table && table.sortData(this.field, dir);
286};
