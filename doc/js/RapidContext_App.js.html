32/**
33 * Initializes the platform, API:s and RapidContext UI. If an app
34 * identifier is provided, the default platform UI will not be
35 * created. Instead the app will be launched with the root document
36 * as its UI container.
37 *
38 * @param {string|Object} [app] the app id or class name to start
39 *
40 * @return {Promise} a promise that will resolve when initialization has
41 *         either completed or failed
42 */
43RapidContext.App.init = function (app) {
44    // Setup libraries
45    RapidContext.Log.context("RapidContext.App.init()");
46    console.info("Initializing RapidContext");
47    if (!RapidContext.Browser.isSupported()) {
48        var el = document.getElementById("unsupported-browser");
49        el.className = el.className.replace("hidden", "");
50        return Promise.reject(new Error("Unsupported browser"));
51    }
52
53    // Setup UI
54    RapidContext.Util.registerSizeConstraints(document.body, "100%-20", "100%-20");
55    var resizer = () => RapidContext.Util.resizeElements(document.body);
56    MochiKit.Signal.connect(window, "onresize", resizer);
57    document.body.innerHTML = "";
58    document.body.append(new RapidContext.Widget.Overlay({ message: "Loading..." }));
59
60    // Load platform data (into cache)
61    var cachedData = [
62        RapidContext.App.callProc("system/status"),
63        RapidContext.App.callProc("system/session/current"),
64        RapidContext.App.callProc("system/app/list")
65    ];
66
67    // Launch app
68    return Promise.all(cachedData)
69        .then(function () {
70            const hourly = 60 * 60 * 1000;
71            setInterval(() => RapidContext.App.callProc("system/session/current"), hourly);
72            if (app && app !== "start") {
73                return RapidContext.App.startApp(app || "start").catch(function (err) {
74                    RapidContext.UI.showError(err);
75                    return RapidContext.App.startApp("start");
76                });
77            } else {
78                return RapidContext.App.startApp("start");
79            }
80        })
81        .catch(function (err) {
82            RapidContext.UI.showError(err);
83            return Promise.reject(err);
84        })
85        .finally(function () {
86            RapidContext.Log.context(null);
87        });
88};

137/**
138 * Finds the app launcher from an app instance, class name or
139 * launcher. In the last case, the matching cached launcher will be
140 * returned.
141 *
142 * @param {string|Object} app the app id, instance, class name or
143 *        launcher
144 *
145 * @return {Object} the read-only app launcher, or
146 *         `null` if not found
147 */
148RapidContext.App.findApp = function (app) {
149    if (app == null) {
150        return null;
151    }
152    var apps = RapidContext.App.apps();
153    for (var i = 0; i < apps.length; i++) {
154        var l = apps[i];
155        if (l.id == app || l.className == app || l.id == app.id) {
156            return l;
157        }
158    }
159    return null;
160};

162/**
163 * Creates and starts an app instance. Normally apps are started in the default
164 * app tab container or in a provided widget DOM node. By specifying a newly
165 * created window object as the parent container, apps can also be launched
166 * into separate windows.
167 *
168 * Note that when a new window is used, the returned deferred will callback
169 * immediately with a `null` app instance (normal app communication is not
170 * possible cross-window).
171 *
172 * @param {string|Object} app the app id, class name or launcher
173 * @param {Widget/Window} [container] the app container widget or
174 *            window, defaults to create a new pane in the app tab
175 *            container
176 *
177 * @return {Promise} a `RapidContext.Async` promise that will
178 *         resolve when the app has launched
179 *
180 * @example
181 * // Starts the help app in a new tab
182 * RapidContext.App.startApp('help');
183 *
184 * // Starts the help app in a new window
185 * RapidContext.App.startApp('help', window.open());
186 */
187RapidContext.App.startApp = function (app, container) {
188    function loadResource(launcher, res) {
189        if (res.type == "code") {
190            return RapidContext.App.loadScript(res.url);
191        } else if (res.type == "module") {
192            return import(RapidContext.Util.resolveURI(res.url)).then(function (mod) {
193                launcher.creator = mod["default"] || mod["create"];
194            });
195        } else if (res.type == "style") {
196            return RapidContext.App.loadStyles(res.url);
197        } else if (res.type == "ui") {
198            return RapidContext.App.loadXML(res.url).then(function (node) {
199                launcher.ui = node;
200            });
201        } else if (res.type == "json" && res.id != null) {
202            return RapidContext.App.loadJSON(res.url).then(function (data) {
203                launcher.resource[res.id] = data;
204            });
205        } else {
206            if (res.id != null) {
207                launcher.resource[res.id] = res.url;
208            }
209            return Promise.resolve(res);
210        }
211    }
212    function load(launcher) {
213        console.info("Loading app/" + launcher.id + " resources", launcher);
214        launcher.resource = {};
215        var promises = launcher.resources.map((res) => loadResource(launcher, res));
216        return Promise.all(promises).then(function () {
217            launcher.creator = launcher.creator || window[launcher.className];
218            if (launcher.creator == null) {
219                console.error("App constructor " + launcher.className + " not defined", launcher);
220                throw new Error("App constructor " + launcher.className + " not defined");
221            }
222        });
223    }
224    function buildUI(parent, ids, ui) {
225        var root = ui.documentElement;
226        for (var i = 0; i < root.attributes.length; i++) {
227            var attr = root.attributes[i];
228            if (attr.specified && typeof(parent.setAttrs) === "function") {
229                parent.setAttrs({ [attr.name]: attr.value });
230            } else if (attr.specified && attr.name === "class") {
231                attr.value.split(/\s+/g).forEach(function (cls) {
232                    parent.classList.add(cls);
233                });
234            } else if (attr.specified) {
235                parent.setAttribute(attr.name, attr.value);
236            }
237        }
238        parent.append(...RapidContext.UI.buildUI(ui, ids));
239        RapidContext.Util.resizeElements(parent);
240    }
241    function launch(launcher, ui) {
242        RapidContext.Log.context("RapidContext.App.startApp(" + launcher.id + ")");
243        return RapidContext.Async.wait(0)
244            .then(() => launcher.creator ? true : load(launcher))
245            .then(function () {
246                console.info("Starting app/" + launcher.id, launcher);
247                /* eslint new-cap: "off" */
248                var instance = new launcher.creator();
249                launcher.instances.push(instance);
250                var props = MochiKit.Base.setdefault({ ui: ui }, launcher);
251                delete props.creator;
252                delete props.instances;
253                MochiKit.Base.setdefault(instance, props);
254                MochiKit.Signal.disconnectAll(ui.root, "onclose");
255                var halt = () => RapidContext.App.stopApp(instance);
256                MochiKit.Signal.connect(ui.root, "onclose", halt);
257                if (launcher.ui != null) {
258                    buildUI(ui.root, ui, launcher.ui);
259                }
260                ui.overlay.hide();
261                ui.overlay.setAttrs({ message: "Working..." });
262                return RapidContext.App.callApp(instance, "start");
263            })
264            .catch(function (e) {
265                console.error("Failed to start app", e);
266                MochiKit.Signal.disconnectAll(ui.root, "onclose");
267                let div = document.createElement("div");
268                div.append(String(e));
269                ui.root.append(div);
270                if (e.url) {
271                    let link = document.createElement("a");
272                    link.className = "block mt-2";
273                    link.setAttribute("href", e.url);
274                    link.append(e.url);
275                    ui.root.append(link);
276                }
277                ui.overlay.hide();
278                return Promise.reject(e);
279            })
280            .finally(function () {
281                RapidContext.Log.context(null);
282            });
283    }
284    function moveAppToStart(instance, elems) {
285        var start = RapidContext.App.findApp("start").instances[0];
286        var opts = { title: instance.name, closeable: false, background: true };
287        var ui = start.initAppPane(null, opts);
288        ui.root.removeAll();
289        ui.root.addAll(elems);
290        RapidContext.Util.resizeElements(ui.root);
291    }
292    return new RapidContext.Async(function (resolve, reject) {
293        var launcher = RapidContext.App.findApp(app);
294        if (launcher == null) {
295            var msg = "No matching app launcher found";
296            console.error(msg, app);
297            throw new Error([msg, ": ", app].join(""));
298        }
299        var instances = RapidContext.App._instances();
300        var start = RapidContext.App.findApp("start");
301        if ($.isWindow(container)) {
302            // Launch app into separate window/tab
303            var href = "rapidcontext/app/" + launcher.id;
304            container.location.href = RapidContext.Util.resolveURI(href);
305            resolve();
306        } else if (start && start.instances && start.instances.length > 0) {
307            // Launch app into start app tab
308            var paneOpts = { title: launcher.name, closeable: (launcher.launch != "once") };
309            var paneUi = start.instances[0].initAppPane(container, paneOpts);
310            resolve(launch(launcher, paneUi));
311        } else if (instances.length > 0) {
312            // Switch from single-app to multi-app mode
313            var elems = Array.from(document.body.childNodes);
314            var overlay = new RapidContext.Widget.Overlay({ message: "Loading..." });
315            document.body.insertBefore(overlay, document.body.childNodes[0]);
316            let ui = { root: document.body, overlay: overlay };
317            var move = () => moveAppToStart(instances[0], elems);
318            var recall = () => (launcher.id === "start") ? true : RapidContext.App.startApp(launcher.id);
319            resolve(launch(start, ui).then(move).then(recall));
320        } else {
321            // Launch single-app mode
322            var ui = { root: document.body, overlay: document.body.childNodes[0] };
323            resolve(launch(launcher, ui));
324        }
325    });
326};

328/**
329 * Stops an app instance. If only the class name or launcher is specified, the
330 * most recently created instance will be stopped.
331 *
332 * @param {string|Object} app the app id, instance, class name or launcher
333 *
334 * @return {Promise} a `RapidContext.Async` promise that will
335 *         resolve when the app has been stopped
336 */
337RapidContext.App.stopApp = function (app) {
338    return new RapidContext.Async(function (resolve, reject) {
339        var launcher = RapidContext.App.findApp(app);
340        if (!launcher || launcher.instances.length <= 0) {
341            var msg = "No running app instance found";
342            console.error(msg, app);
343            throw new Error([msg, ": ", app].join(""));
344        }
345        console.info("Stopping app " + launcher.name);
346        var pos = MochiKit.Base.findIdentical(launcher.instances, app);
347        if (pos < 0) {
348            app = launcher.instances.pop();
349        } else {
350            launcher.instances.splice(pos, 1);
351        }
352        app.stop();
353        if (app.ui.root != null) {
354            RapidContext.Widget.destroyWidget(app.ui.root);
355        }
356        for (var k in app.ui) {
357            delete app.ui[k];
358        }
359        for (var n in app) {
360            delete app[n];
361        }
362        MochiKit.Signal.disconnectAllTo(app);
363        resolve();
364    });
365};

367/**
368 * Performs an asynchronous call to a method in an app. If only the class name
369 * or launcher is specified, the most recently created instance will be used.
370 * If no instance is running, one will be started. Also, before calling the app
371 * method, the app UI will be focused.
372 *
373 * @param {string|Object} app the app id, instance, class name or launcher
374 * @param {string} method the app method name
375 * @param {Mixed} [args] additional parameters sent to method
376 *
377 * @return {Promise} a `RapidContext.Async` promise that will
378 *         resolve with the result of the call on success
379 */
380RapidContext.App.callApp = function (app, method) {
381    var args = Array.from(arguments).slice(2);
382    return new RapidContext.Async(function (resolve, reject) {
383        var launcher = RapidContext.App.findApp(app);
384        if (launcher == null) {
385            var msg = "No matching app launcher found";
386            console.error(msg, app);
387            throw new Error([msg, ": ", app].join(""));
388        }
389        var exists = launcher.instances.length > 0;
390        var promise = exists ? RapidContext.Async.wait(0) : RapidContext.App.startApp(app);
391        promise = promise.then(function () {
392            RapidContext.Log.context("RapidContext.App.callApp(" + launcher.id + "," + method + ")");
393            var pos = MochiKit.Base.findIdentical(launcher.instances, app);
394            var instance = (pos >= 0) ? app : launcher.instances[launcher.instances.length - 1];
395            var child = instance.ui.root;
396            var parent = child.parentNode.closest(".widget");
397            if (parent != null && typeof(parent.selectChild) == "function") {
398                parent.selectChild(child);
399            }
400            var methodName = launcher.className + "." + method;
401            if (instance[method] == null) {
402                var msg = "No app method " + methodName + " found";
403                console.error(msg);
404                throw new Error(msg);
405            }
406            console.log("Calling app method " + methodName, args);
407            try {
408                return instance[method].apply(instance, args);
409            } catch (e) {
410                var reason = "Caught error in " + methodName;
411                console.error(reason, e);
412                throw new Error(reason + ": " + e.toString(), { cause: e });
413            }
414        }).finally(function () {
415            RapidContext.Log.context(null);
416        });
417        resolve(promise);
418    });
419};

421/**
422 * Performs an asynchronous procedure call.
423 *
424 * @param {string} name the procedure name
425 * @param {Array} [args] the array of arguments, or `null`
426 *
427 * @return {Promise} a `RapidContext.Async` promise that will
428 *         resolve with the response data on success
429 */
430RapidContext.App.callProc = function (name, args) {
431    // TODO: remove this legacy name conversion
432    if (name.startsWith("RapidContext.")) {
433        name = "System" + name.substring(8);
434    }
435    console.log("Call request " + name, args);
436    var params = {};
437    for (var i = 0; args != null && i < args.length; i++) {
438        if (args[i] == null) {
439            params["arg" + i] = "null";
440        } else {
441            params["arg" + i] = RapidContext.Encode.toJSON(args[i]);
442        }
443    }
444    var logLevel = RapidContext.Log.level();
445    if (logLevel == "log" || logLevel == "all") {
446        params["system:trace"] = 1;
447    }
448    var url = "rapidcontext/procedure/" + name;
449    var options = { method: "POST", timeout: 60000 };
450    return RapidContext.App.loadJSON(url, params, options).then(function (res) {
451        if (res.trace != null) {
452            console.log(name + " trace:", res.trace);
453        }
454        if (res.error != null) {
455            console.info(name + " error:", res.error);
456            throw new Error(res.error);
457        } else {
458            console.log(name + " response:", res.data);
459        }
460        if (name.startsWith("system/") && !res.error) {
461            RapidContext.App._Cache.update(name, res.data);
462        }
463        return res.data;
464    });

467/**
468 * Performs an asynchronous login. If the current session is already bound to a
469 * user, that session will be terminated and a new one will be created. If an
470 * authentication token is specified, the login and password fields are not
471 * used (can be null).
472 *
473 * @param {string} login the user login name or email address
474 * @param {string} password the password to authenticate the user
475 * @param {string} [token] the authentication token to identify user/password
476 *
477 * @return {Promise} a `RapidContext.Async` promise that resolves when
478 *         the authentication has either succeeded or failed
479 */
480RapidContext.App.login = function (login, password, token) {
481    function searchLogin() {
482        var proc = "system/user/search";
483        return RapidContext.App.callProc(proc, [login]).then(function (user) {
484            if (user && user.id) {
485                return login = user.id;
486            } else {
487                throw new Error("no user with that email address");
488            }
489        });
490    }
491    function getNonce() {
492        var proc = "system/session/current";
493        return RapidContext.App.callProc(proc).then(function (session) {
494            return session.nonce;
495        });
496    }
497    function passwordAuth(nonce) {
498        var realm = RapidContext.App.status().realm;
499        var hash = CryptoJS.MD5(login + ":" + realm + ":" + password);
500        hash = CryptoJS.MD5(hash.toString() + ":" + nonce).toString();
501        var args = [login, nonce, hash];
502        return RapidContext.App.callProc("system/session/authenticate", args);
503    }
504    function tokenAuth() {
505        return RapidContext.App.callProc("system/session/authenticateToken", [token]);
506    }
507    function verifyAuth(res) {
508        if (!res.success || res.error) {
509            console.info("login failed", login, res.error);
510            throw new Error(res.error || "authentication failed");
511        }
512    }
513    var promise = RapidContext.Async.wait(0);
514    var user = RapidContext.App.user();
515    if (user && user.id) {
516        promise = promise.then(RapidContext.App.logout);
517    }
518    if (token) {
519        return promise.then(tokenAuth).then(verifyAuth);
520    } else {
521        if (/@/.test(login)) {
522            promise = promise.then(searchLogin);
523        }
524        return promise.then(getNonce).then(passwordAuth).then(verifyAuth);
525    }
526};

528/**
529 * Performs an asyncronous logout. This function terminates the current
530 * session and either reloads the browser window or returns a deferred object
531 * that will produce either a `callback` or an `errback` response.
532 *
533 * @param {boolean} [reload=true] the reload browser flag
534 *
535 * @return {Promise} a `RapidContext.Async` promise that will
536 *         resolve when user is logged out
537 */
538RapidContext.App.logout = function (reload) {
539    var promise = RapidContext.App.callProc("system/session/terminate", [null]);
540    if (reload !== false) {
541        promise.then(() => window.location.reload());
542    }
543    return promise;
544};

547 * Performs an asynchronous HTTP request and parses the JSON response. The
548 * request parameters are automatically encoded to query string or JSON format,
549 * depending on the `Content-Type` header. The parameters will be sent either
550 * in the URL or as the request payload (depending on the HTTP `method`).
551 *
552 * @param {string} url the URL to request
553 * @param {Object} [params] the request parameters, or `null`
554 * @param {Object} [options] the request options, or `null`
555 * @config {string} [method] the HTTP method, default is `GET`
556 * @config {number} [timeout] the timeout in milliseconds, default is 30s
557 * @config {Object} [headers] the specific HTTP headers to use
558 *
559 * @return {Promise} a `RapidContext.Async` promise that will
560 *         resolve with the parsed response JSON on success
561 */
562RapidContext.App.loadJSON = function (url, params, options) {
563    var opts = Object.assign({ responseType: "json" }, options);
564    return RapidContext.App.loadXHR(url, params, opts).then(function (xhr) {
565        return xhr.response;
566    });

615/**
616 * Performs an asynchronous HTTP request. The request parameters are
617 * automatically encoded to query string or JSON format, depending on the
618 * `Content-Type` header. The parameters will be sent either in the URL or as
619 * the request payload (depending on the HTTP `method`).
620 *
621 * @param {string} url the URL to request
622 * @param {Object} [params] the request parameters, or `null`
623 * @param {Object} [options] the request options, or `null`
624 * @config {string} [method] the HTTP method, default is `GET`
625 * @config {number} [timeout] the timeout in milliseconds, default is 30s
626 * @config {Object} [headers] the specific HTTP headers to use
627 *
628 * @return {Promise} a `RapidContext.Async` promise that will
629 *         resolve with the XMLHttpRequest instance on success
630 */
631RapidContext.App.loadXHR = function (url, params, options) {
632    var opts = Object.assign({ method: "GET", headers: {}, timeout: 30000 }, options);
633    opts.timeout = (opts.timeout < 1000) ? opts.timeout * 1000 : opts.timeout;
634    var hasBody = params && ["PATCH", "POST", "PUT"].includes(opts.method);
635    var hasJsonBody = opts.headers["Content-Type"] === "application/json";
636    if (!hasBody) {
637        url += params ? "?" + RapidContext.Encode.toUrlQuery(params) : "";
638    } else if (params && hasBody && hasJsonBody) {
639        opts.body = RapidContext.Encode.toJSON(params);
640    } else if (params && hasBody) {
641        opts.headers["Content-Type"] = "application/x-www-form-urlencoded";
642        opts.body = RapidContext.Encode.toUrlQuery(params);
643    }
644    console.log("Starting XHR loading", url, opts);
645    return RapidContext.Async.xhr(url, opts).then(
646        function (res) {
647            console.log("Completed XHR loading", url);
648            return res;
649        },
650        function (err) {
651            console.warn("Failed XHR loading", url, err);
652            return Promise.reject(err);
653        }
654    );
655};

657/**
658 * Loads a JavaScript to the the current page. The script is loaded by
659 * inserting a `<script>` tag in the document `<head>`. Function definitions
660 * and values must therefore be stored to global variables by the script to
661 * become accessible after loading. If the script is already loaded, the
662 * promise will resolve immediately.
663 *
664 * @param {string} url the URL to the script
665 *
666 * @return {Promise} a `RapidContext.Async` promise that will
667 *         resolve when the script has loaded
668 *
669 * @see Use dynamic `import()` instead, if supported by the environment.
670 */
671RapidContext.App.loadScript = function (url) {
672    var selector = ["script[src*='", url, "']"].join("");
673    if (document.querySelectorAll(selector).length > 0) {
674        console.log("script already loaded, skipping", url);
675        return RapidContext.Async.wait(0);
676    } else {
677        console.log("loading script", url);
678        return RapidContext.Async.script(url);
679    }
680};

703/**
704 * Downloads a file to the user desktop. This works by creating a new
705 * window or an inner frame which downloads the file from the server.
706 * Due to `Content-Disposition` headers being set on the server, the
707 * web browser will popup a dialog for the user to save the file.
708 * This function can also be used for saving a file that doesn't
709 * exist by first posting the file (text) content to the server.
710 *
711 * @param {string} url the URL or filename to download
712 * @param {string} [data] the optional file data (if not available
713 *            on the server-side)
714 */
715RapidContext.App.downloadFile = function (url, data) {
716    if (data == null) {
717        url = url + (url.includes("?") ? "&" : "?") + "download";
718        var attrs = {
719            src: url,
720            border: "0",
721            frameborder: "0",
722            height: "0",
723            width: "0"
724        };
725        var iframe = MochiKit.DOM.createDOM("iframe", attrs);
726        document.body.append(iframe);
727    } else {
728        var name = MochiKit.DOM.INPUT({ name: "fileName", value: url });
729        var file = MochiKit.DOM.INPUT({ name: "fileData", value: data });
730        var flag = MochiKit.DOM.INPUT({ name: "download", value: "1" });
731        var formAttrs = {
732            action: "rapidcontext/download",
733            method: "POST",
734            target: "_blank",
735            style: { display: "none" }
736        };
737        var form = MochiKit.DOM.FORM(formAttrs, name, file, flag);
738        document.body.append(form);
739        form.submit();
740    }
741};

743/**
744 * Uploads a file to the server. The upload is handled by an
745 * asynchronous HTTP request.
746 *
747 * @param {string} id the upload identifier to use
748 * @param {File} file the file object to upload
749 * @param {function} [onProgress] the progress event handler
750 */
751RapidContext.App.uploadFile = function (id, file, onProgress) {

754    var opts = {

757        timeout: 300000,
758        progress: onProgress,
759        log: id + " upload"

761    return RapidContext.Async.xhr("rapidcontext/upload/" + id, opts);

764/**
765 * The application data cache. Contains the most recently retrieved
766 * data for some commonly used objects in the execution environment.
767 */
768RapidContext.App._Cache = {
769    version: new Date().getTime().toString(16).slice(-8),
770    status: null,
771    user: null,
772    apps: {},
773
774    // Normalizes an app manifest and its resources
775    _normalizeApp: function (app) {
776        function toType(type, url) {
777            var isJs = !type && /\.js$/i.test(url);
778            var isCss = !type && /\.css$/i.test(url);
779            if (["code", "js", "javascript"].includes(type) || isJs) {
780                return "code";
781            } else if (!type && /\.mjs$/i.test(url)) {
782                return "module";
783            } else if (["style", "css"].includes(type) || isCss) {
784                return "style";
785            } else if (!type && /\.json$/i.test(url)) {
786                return "json";
787            } else if (!type && /ui\.xml$/i.test(url)) {
788                return "ui";
789            } else if (!type && !app.icon && /\.(gif|jpg|jpeg|png|svg)$/i.test(url)) {
790                return "icon";
791            } else {
792                return type;
793            }
794        }
795        function toIcon(res) {
796            if (res.url) {
797                return $("<img/>").attr({ src: res.url }).addClass("-app-icon").get(0);
798            } else if (res.html) {
799                var node = $("<span/>").html(res.html).addClass("-app-icon").get(0);
800                return node.childNodes.length === 1 ? node.childNodes[0] : node;
801            } else if (res["class"]) {
802                return $("<i/>").addClass(res["class"]).addClass("-app-icon").get(0);
803            } else {
804                return null;
805            }
806        }
807        function toResource(res) {
808            res = (typeof(res) === "string") ? { url: res } : res;
809            res.type = toType(res.type, res.url);
810            if (res.url && !res.url.includes(":") && !res.url.startsWith("/")) {
811                var ver = app.version || RapidContext.App._Cache.version;
812                res.url = "rapidcontext/files@" + ver + "/" + res.url;
813            }
814            if (!app.icon && res.type === "icon") {
815                app.icon = toIcon(res);
816            }
817            return res;
818        }
819        app = Object.assign({}, app);
820        app.launch = app.launch || "manual";
821        app.resources = [].concat(app.resources).filter(Boolean).map(toResource);
822        if (!app.icon) {
823            app.icon = toIcon({ "class": "fa fa-4x fa-question-circle unimportant" });
824        }
825        return app;
826    },
827
828    // Updates the cache data with the results from a procedure.
829    update: function (proc, data) {
830        switch (proc) {
831        case "system/status":
832            this.status = Object.assign({}, data);
833            console.log("Updated cached status", this.status);
834            break;
835        case "system/session/current":
836            if (this.user && this.user.id !== (data && data.user && data.user.id)) {
837                RapidContext.UI.Msg.error.loggedOut();
838                RapidContext.App.callProc = () => Promise.reject(new Error("logged out"));
839            } else if (data && data.user && !this.user) {
840                // TODO: use deep clone
841                data = Object.assign({}, data.user);
842                data.longName = data.name ? `${data.name} (${data.id})` : data.id;
843                this.user = data;
844                console.log("Updated cached user", this.user);
845            }
846            break;
847        case "system/app/list":
848            for (var i = 0; data && i < data.length; i++) {
849                var launcher = this._normalizeApp(data[i]);
850                launcher.instances = (this.apps[launcher.id] || {}).instances || [];
851                this.apps[launcher.id] = Object.assign(this.apps[launcher.id] || {}, launcher);
852            }
853            console.log("Updated cached apps", this.apps);
854            break;
855        }
856    }
857};
